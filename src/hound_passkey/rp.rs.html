1//! The sans-IO relying party: a thin wrapper over `webauthn_rs::Webauthn`
2//! exposing the four ceremony calls plus the RP-config accessors the
3//! `.well-known` endpoints read. It owns NO state beyond
4//! the configured `Webauthn` — the in-flight ceremony state (`PasskeyRegistration`
5//! / `PasskeyAuthentication`) is returned to and stored by the caller, and the
6//! `Passkey` credentials live in the caller's database.
7//!
8//! We wrap `webauthn-rs`; we do not reimplement any crypto.
9
10use uuid::Uuid;
11use webauthn_rs::prelude::{Url, Webauthn, WebauthnBuilder};
12
13use crate::error::PasskeyError;
14use crate::proto::{
15    CreationChallengeResponse, PublicKeyCredential, RegisterPublicKeyCredential,
16    RequestChallengeResponse,
17};
18
19// The relying-party crypto types the caller must store across a ceremony or in its
20// database. Re-exported so the backend imports them from `hound-passkey` rather
21// than depending on `webauthn-rs` directly.
22pub use webauthn_rs::prelude::{
23    AuthenticationResult, Passkey, PasskeyAuthentication, PasskeyRegistration,
24};
25
26/// A configured WebAuthn relying party. Sans-IO: every method is synchronous and
27/// pure over the wrapped `Webauthn`.
28#[derive(Clone, Debug)]
29pub struct RelyingParty {
30    rp_id: String,
31    webauthn: Webauthn,
32}
33
34impl RelyingParty {
35    /// Build the relying party for `rp_id` / `rp_origin`, appending any
36    /// `extra_origins` (e.g. dev origins, native app origins).
37    pub fn new(rp_id: &str, rp_origin: &Url, extra_origins: &[Url]) -> Result<Self, PasskeyError> {
38        let mut builder = WebauthnBuilder::new(rp_id, rp_origin)?;
39        for origin in extra_origins {
40            builder = builder.append_allowed_origin(origin);
41        }
42        Ok(Self {
43            rp_id: rp_id.to_owned(),
44            webauthn: builder.build()?,
45        })
46    }
47
48    /// Begin registration for a (provisional) user. Returns the creation challenge
49    /// to send to the client and the [`PasskeyRegistration`] state the caller must
50    /// persist until `finish_registration`.
51    pub fn start_registration(
52        &self,
53        user_id: Uuid,
54        email: &str,
55        display_name: &str,
56    ) -> Result<(CreationChallengeResponse, PasskeyRegistration), PasskeyError> {
57        Ok(self
58            .webauthn
59            .start_passkey_registration(user_id, email, display_name, None)?)
60    }
61
62    /// Complete registration: verify the client's created credential against the
63    /// stored [`PasskeyRegistration`], yielding the [`Passkey`] to persist.
64    pub fn finish_registration(
65        &self,
66        credential: &RegisterPublicKeyCredential,
67        state: &PasskeyRegistration,
68    ) -> Result<Passkey, PasskeyError> {
69        Ok(self
70            .webauthn
71            .finish_passkey_registration(credential, state)?)
72    }
73
74    /// Begin authentication against the user's known passkeys. Returns the request
75    /// challenge to send to the client and the [`PasskeyAuthentication`] state the
76    /// caller must persist until `finish_authentication`.
77    pub fn start_authentication(
78        &self,
79        passkeys: &[Passkey],
80    ) -> Result<(RequestChallengeResponse, PasskeyAuthentication), PasskeyError> {
81        Ok(self.webauthn.start_passkey_authentication(passkeys)?)
82    }
83
84    /// Complete authentication: verify the client's assertion against the stored
85    /// [`PasskeyAuthentication`], yielding the [`AuthenticationResult`] (which the
86    /// caller inspects for `needs_update()` to roll the credential counter).
87    pub fn finish_authentication(
88        &self,
89        credential: &PublicKeyCredential,
90        state: &PasskeyAuthentication,
91    ) -> Result<AuthenticationResult, PasskeyError> {
92        Ok(self
93            .webauthn
94            .finish_passkey_authentication(credential, state)?)
95    }
96
97    /// The configured relying-party id (`rp_id`). Served by the `/webauthn`
98    /// RP-config endpoint.
99    pub fn rp_id(&self) -> &str {
100        &self.rp_id
101    }
102
103    /// The allowed origins. Served (as bare `scheme://host[:port]`) by the
104    /// `/webauthn` RP-config endpoint.
105    pub fn get_allowed_origins(&self) -> &[Url] {
106        self.webauthn.get_allowed_origins()
107    }
108}
