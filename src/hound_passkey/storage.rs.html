1//! Platform-keyring persistence for the session token pair — the high-level
2//! storage API the app hosts consume (the FRB layer adapts it onto
3//! `hound-client`'s `SessionStore` seam).
4//!
5//! The token pair is persisted as ONE JSON blob under a single credential, so a
6//! refresh (which rotates BOTH tokens) can never leave a torn pair. Backend per
7//! platform is selected at compile time:
8//!   - **macOS** — the login Keychain (`apple-native-keyring-store/keychain`).
9//!   - **Linux** — the freedesktop Secret Service (gnome-keyring / KWallet) over
10//!     pure-Rust zbus (`zbus-secret-service-keyring-store`), no system
11//!     libdbus/openssl link.
12//!   - **elsewhere** (Windows/mobile, not yet brought online) — the in-memory
13//!     mock: builds everywhere, persists nothing.
14//!
15//! **Linux blocking hazard**: the secret-service store drives zbus through its
16//! *blocking* API, and zbus panics if a blocking call is made from inside an
17//! async runtime — which is where app hosts call this (FRB commands run on FRB's
18//! tokio runtime). [`run_blocking`] hops every keyring operation onto a scratch
19//! thread on Linux (the sanctioned keyring-rs pattern for async apps); the macOS
20//! keychain is plain sync FFI and is called in place.
21
22use crate::error::PasskeyError;
23
24/// The keychain "service" attribute all hound credentials live under (the app
25/// bundle id, so every hound build reads the same credential).
26const SERVICE: &str = "chat.hound.app";
27/// The credential account name for the persisted session (one entry).
28const SESSION_KEY: &str = "session-tokens";
29
30/// The persisted access + refresh token pair.
31#[derive(Clone, Debug, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
32pub struct SessionTokens {
33    pub access_token: String,
34    pub refresh_token: String,
35}
36
37/// The session-token store over the OS credential store (via `keyring-core`).
38/// Construction registers the platform backend as `keyring-core`'s process-wide
39/// default exactly once (idempotent).
40pub struct SessionTokenStore {
41    _priv: (),
42}
43
44impl SessionTokenStore {
45    /// Build the store, registering the platform keyring backend as the default
46    /// if nothing has registered one yet. Construction may already talk to the
47    /// backend (the secret-service store opens its D-Bus session), so a headless
48    /// Linux box with no secret service errors HERE — loudly, not by silently
49    /// not persisting.
50    pub fn new() -> Result<Self, PasskeyError> {
51        run_blocking(|| {
52            if keyring_core::get_default_store().is_none() {
53                keyring_core::set_default_store(platform_store()?);
54            }
55            Ok(Self { _priv: () })
56        })
57    }
58
59    /// Load the persisted session, if one was saved. `Ok(None)` ⇒ signed out.
60    pub fn load(&self) -> Result<Option<SessionTokens>, PasskeyError> {
61        match run_blocking(|| entry().map(|e| e.get_password()))? {
62            Ok(raw) => {
63                tracing::info!("keyring: loaded persisted session");
64                let tokens: SessionTokens =
65                    serde_json::from_str(&raw).map_err(|e| storage_err(&e))?;
66                Ok(Some(tokens))
67            }
68            Err(keyring_core::Error::NoEntry) => {
69                tracing::info!("keyring: no stored session entry");
70                Ok(None)
71            }
72            // A read failure here (vs no-entry) is the dev-build tell: an
73            // ad-hoc-signed rebuild can't read the prior build's keychain item
74            // (ACL is bound to the code signature), so persistence appears to
75            // "not work" across rebuilds.
76            Err(e) => {
77                tracing::warn!(error = %e, "keyring: read failed (often a dev re-sign / ACL mismatch)");
78                Err(keyring_err(e))
79            }
80        }
81    }
82
83    /// Persist the token pair (called after login + every refresh). Atomic: one
84    /// credential holds both tokens.
85    pub fn save(&self, access: &str, refresh: &str) -> Result<(), PasskeyError> {
86        let raw = serde_json::to_string(&SessionTokens {
87            access_token: access.to_owned(),
88            refresh_token: refresh.to_owned(),
89        })
90        .map_err(|e| storage_err(&e))?;
91        run_blocking(|| entry()?.set_password(&raw).map_err(keyring_err))
92    }
93
94    /// Drop the persisted session (sign-out). Idempotent — clearing an empty
95    /// store succeeds.
96    pub fn clear(&self) -> Result<(), PasskeyError> {
97        run_blocking(|| match entry()?.delete_credential() {
98            Ok(()) | Err(keyring_core::Error::NoEntry) => Ok(()),
99            Err(e) => Err(keyring_err(e)),
100        })
101    }
102}
103
104fn entry() -> Result<keyring_core::Entry, PasskeyError> {
105    keyring_core::Entry::new(SERVICE, SESSION_KEY).map_err(keyring_err)
106}
107
108fn keyring_err(e: keyring_core::Error) -> PasskeyError {
109    PasskeyError::Storage(format!("keyring: {e}"))
110}
111
112fn storage_err(e: &dyn std::fmt::Display) -> PasskeyError {
113    PasskeyError::Storage(format!("keyring: {e}"))
114}
115
116/// Run a keyring operation somewhere it is safe to block. On Linux the
117/// secret-service backend's blocking zbus calls panic when made from a tokio
118/// worker thread — so hop onto a scratch thread (scoped, so closures can
119/// borrow). Keyring ops are rare (login / refresh / logout) and fast; briefly
120/// parking one async worker is fine. On macOS the keychain is synchronous FFI
121/// with no runtime detection: call in place.
122#[cfg(target_os = "linux")]
123fn run_blocking<T: Send>(f: impl FnOnce() -> T + Send) -> T {
124    std::thread::scope(|s| {
125        s.spawn(f)
126            .join()
127            .unwrap_or_else(|_| panic!("keyring operation panicked"))
128    })
129}
130
131#[cfg(not(target_os = "linux"))]
132fn run_blocking<T: Send>(f: impl FnOnce() -> T + Send) -> T {
133    f()
134}

136/// The `keyring-core` store for this OS (compile-time selected).
137#[cfg(target_os = "macos")]
138fn platform_store()
139-> Result<std::sync::Arc<apple_native_keyring_store::keychain::Store>, PasskeyError> {
140    apple_native_keyring_store::keychain::Store::new().map_err(keyring_err)
141}
142
143#[cfg(target_os = "linux")]
144fn platform_store() -> Result<std::sync::Arc<zbus_secret_service_keyring_store::Store>, PasskeyError>
145{
146    // See the module doc: blocking API, must run off any async runtime (which
147    // `run_blocking` guarantees for every caller in this module).
148    zbus_secret_service_keyring_store::Store::new().map_err(keyring_err)
149}
150
151#[cfg(not(any(target_os = "macos", target_os = "linux")))]
152fn platform_store() -> Result<std::sync::Arc<keyring_core::mock::Store>, PasskeyError> {
153    // No platform backend wired for this target yet; the crate must still build.
154    // A real backend (windows-native/apple-protected) is added in a parallel
155    // `[target.*]` table when that platform is brought online.
156    keyring_core::mock::Store::new().map_err(keyring_err)
157}

159#[cfg(test)]
160mod tests {
161    use super::*;
162
163    /// One sequential test: the default store is process-wide and every
164    /// [`SessionTokenStore`] reads the same credential, so parallel `#[test]`s
165    /// over it would race each other. Forces the mock backend so the round-trip
166    /// is hermetic on every dev/CI OS (never touches a real keychain).
167    #[test]
168    fn round_trips_the_token_pair_and_clear_is_idempotent() {
169        if keyring_core::get_default_store().is_none() {
170            keyring_core::set_default_store(keyring_core::mock::Store::new().unwrap());
171        }
172        let store = SessionTokenStore::new().unwrap();
173
174        assert!(store.load().unwrap().is_none());
175        store.save("acc", "ref").unwrap();
176        let t = store.load().unwrap().unwrap();
177        assert_eq!(t.access_token, "acc");
178        assert_eq!(t.refresh_token, "ref");
179
180        store.clear().unwrap();
181        assert!(store.load().unwrap().is_none());
182        // Clearing an already-empty store still succeeds.
183        store.clear().unwrap();
184        assert!(store.load().unwrap().is_none());
185    }
186}
