1//! The error surface for the passkey core. Wraps `webauthn-rs`'s `WebauthnError`
2//! (every ceremony failure — only under the `rp` feature, since the relying-party
3//! state machine is what produces it) plus a generic variant for the authenticator
4//! seam and anything that doesn't originate in the crypto machine.
5
6/// A passkey-ceremony failure. The backend lifts the [`WebAuthn`](Self::WebAuthn)
7/// variant straight into its own `ApiError::WebAuthn`, so the HTTP status mapping
8/// is preserved.
9#[derive(Debug, thiserror::Error)]
10pub enum PasskeyError {
11    /// A failure inside the `webauthn-rs` relying-party state machine (challenge
12    /// generation, credential verification, counter checks, …). Only present with
13    /// the `rp` feature (the client never runs the state machine).
14    #[cfg(feature = "rp")]
15    #[error("webauthn: {0}")]
16    WebAuthn(#[from] webauthn_rs::prelude::WebauthnError),
17
18    /// A session-token persistence failure in the platform keyring (only under
19    /// the `storage` feature). A host/environment fault — the keychain refused
20    /// the read/write — not a ceremony failure.
21    #[cfg(feature = "storage")]
22    #[error("storage: {0}")]
23    Storage(String),
24
25    /// Anything else — the authenticator seam, malformed inputs the wrapper
26    /// rejects before reaching the state machine, etc.
27    #[error("passkey: {0}")]
28    Other(String),
29}
