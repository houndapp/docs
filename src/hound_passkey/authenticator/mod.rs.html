1//! The CTAP2 authenticator seam.
2//!
3//! Only the trait lives here — the relying-party side (the backend) never
4//! implements it; client-side authenticator cores do. It is the dual of
5//! [`RelyingParty`](crate::rp::RelyingParty): given the server's challenge, an
6//! [`Authenticator`] produces the credential/assertion the relying party then
7//! verifies.
8//!
9//! [`ctap2`] (feature `ctap2`) is Phase 2's implementor: a physical USB security
10//! key over CTAP2.
11
12use crate::error::PasskeyError;
13use crate::proto::{
14    CreationChallengeResponse, PublicKeyCredential, RegisterPublicKeyCredential,
15    RequestChallengeResponse,
16};
17
18#[cfg(feature = "ctap2")]
19pub mod ctap2;
20
21/// A WebAuthn authenticator: the device-side counterpart of the relying party.
22///
23/// Implementors hold the platform credential store / CTAP2 transport. Both calls
24/// are `&mut self` because a real authenticator mutates device state (signature
25/// counters, resident-credential storage) per ceremony.
26pub trait Authenticator {
27    /// Respond to a registration challenge ([`CreationChallengeResponse`], from
28    /// `register/start`) by creating a credential — the `navigator.credentials
29    /// .create()` step — yielding the [`RegisterPublicKeyCredential`] to send to
30    /// `register/finish`.
31    fn make_credential(
32        &mut self,
33        options: CreationChallengeResponse,
34    ) -> Result<RegisterPublicKeyCredential, PasskeyError>;
35
36    /// Respond to an authentication challenge ([`RequestChallengeResponse`], from
37    /// `login/start`) by signing an assertion — the `navigator.credentials.get()`
38    /// step — yielding the [`PublicKeyCredential`] to send to `login/finish`.
39    fn get_assertion(
40        &mut self,
41        options: RequestChallengeResponse,
42    ) -> Result<PublicKeyCredential, PasskeyError>;
43}
