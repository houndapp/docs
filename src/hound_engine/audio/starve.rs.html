1//! # Per-slot playout starvation classifier (pure, sans-IO)
2//!
3//! The mix bus pulls each peer's jitter buffer once per 10 ms tick. When a pull comes
4//! up short the bus must decide: is this an **audible underflow** (real voice stutter
5//! — the "audio is sacred" violation we must surface and alarm on) or **expected
6//! silence** (the peer is quiet, so Opus stopped sending / sends sparse comfort-noise
7//! frames — a dry carry that produces no audible gap)?
8//!
9//! Counting silence as stutter is not cosmetic: it makes the metric cry wolf (the live
10//! reports showed a permanently-silent peer logging ~5 "starves" on *every* DTX gap,
11//! drowning out real glitches) and makes the invariant un-measurable. This module is
12//! the single discriminator, extracted from the bus so the policy is unit-tested in
13//! isolation against scripted tick sequences.
14//!
15//! ## The discriminator: recently-released energy
16//!
17//! A dry carry is only audible if real audio was playing *just before* it. So the
18//! tracker watches the energy of the audio the slot releases: a peak above
19//! [`SILENCE_FLOOR`] marks the slot **active**; quiet output (comfort noise / the tail
20//! of speech) lets it decay back to **silent** after [`ACTIVE_DECAY_TICKS`]. A short
21//! carry while active counts; while silent it does not. This is the right rule
22//! regardless of *why* the carry is dry (DTX, a lost run of silence, the peer left):
23//! the test is "would a listener hear a gap", and the answer is no when the recent
24//! signal was silence.
25//!
26//! A genuinely abrupt stop (loud frame → immediate silence) still counts the first
27//! [`STARVE_IDLE_TICKS`] of the gap — the transient when someone stops mid-word — then
28//! falls silent. That is a real, brief, audible edge and counting it once is correct;
29//! the pathology this fixes is counting *every* gap of a *continuously* silent peer.
30
31/// Peak `|sample|` (i16 scale, 0..=32767) below which released audio is treated as
32/// silence / comfort noise rather than speech. ~328 ≈ 1% of full scale ≈ −40 dBFS:
33/// well above comfort-noise / dither, well below any intelligible speech peak.
34pub const SILENCE_FLOOR: i32 = 328;
35
36/// Ticks of quiet output after which a slot is considered silent (no recent real
37/// audio), so a dry carry reads as DTX rather than stutter. 8 ticks ≈ 80 ms at the
38/// 10 ms mix cadence. Must be ≥ [`STARVE_IDLE_TICKS`] so a *real* mid-speech gap is
39/// still counted for its full onset window before the slot is deemed silent.
40pub const ACTIVE_DECAY_TICKS: u32 = 8;
41
42/// Empty-streak length past which even an active-looking gap stops counting: a peer
43/// that has been dry this long has stopped, not stuttered. 5 ticks ≈ 50 ms — long
44/// enough to register a real mid-speech dropout, short enough not to climb forever.
45pub const STARVE_IDLE_TICKS: u32 = 5;
46
47/// One mix tick's drain result, fed to [`StarveTracker::observe`].
48#[derive(Debug, Clone, Copy)]
49pub struct Tick {
50    /// Samples this slot actually contributed to the mix this tick.
51    pub contributed: usize,
52    /// Samples the bus asked for (a full mix chunk).
53    pub want: usize,
54    /// Peak `|sample|` among the samples contributed this tick (0 if none).
55    pub peak_abs: i32,
56}
57
58/// Per-slot starvation state, advanced once per mix tick on the RT bus thread. Plain
59/// (single-threaded by construction); the owning slot mirrors the cumulative count
60/// into an atomic for the cross-thread stats reader.
61#[derive(Debug, Default, Clone, Copy)]
62pub struct StarveTracker {
63    /// Set once the slot has ever contributed audio — starvation is meaningless before
64    /// the first frame (a fresh slot is legitimately empty during warmup).
65    ever_fed: bool,
66    /// Consecutive fully-empty ticks (carry dry).
67    empty_streak: u32,
68    /// Ticks of "recently active" credit remaining: refilled to [`ACTIVE_DECAY_TICKS`]
69    /// whenever above-[`SILENCE_FLOOR`] audio is released, decremented otherwise.
70    /// **Starts at 0** (a never-yet-loud slot is silent), so a stream that is quiet
71    /// from the first frame — comfort noise / a muted peer — is never "active" and its
72    /// dry/short carries never count. `> 0` ⇒ real audio played within the decay window
73    /// ⇒ a gap is audible.
74    active_credit: u32,
75}
76
77impl StarveTracker {
78    pub fn new() -> Self {
79        Self::default()
80    }
81
82    /// Classify one tick. Returns `true` iff it is an AUDIBLE underflow that should be
83    /// counted as playout stutter. Advances the internal state.
84    pub fn observe(&mut self, tick: Tick) -> bool {
85        let had_energy = tick.contributed > 0 && tick.peak_abs > SILENCE_FLOOR;
86        if tick.contributed > 0 {
87            self.ever_fed = true;
88        }
89        if had_energy {
90            self.active_credit = ACTIVE_DECAY_TICKS;
91        } else {
92            self.active_credit = self.active_credit.saturating_sub(1);
93        }
94
95        // Warmup: nothing has ever played, so an empty carry is not a gap.
96        if !self.ever_fed {
97            return false;
98        }
99        let recently_active = self.active_credit > 0;
100
101        if tick.contributed >= tick.want {
102            self.empty_streak = 0;
103            return false; // a full chunk — no gap
104        }
105        if tick.contributed == 0 {
106            self.empty_streak += 1;
107            // Count the onset of a gap only while the slot was recently producing real
108            // audio AND we are within the early-gap window; sustained dryness is the
109            // peer being silent (DTX), not stutter.
110            return recently_active && self.empty_streak <= STARVE_IDLE_TICKS;
111        }
112        // Partial drain (0 < contributed < want): fed-but-short. Audible only if the
113        // slot was recently active (a short drain of silence is inaudible).
114        self.empty_streak = 0;
115        recently_active
116    }
117
118    /// Whether the slot has ever been fed (exposed for the bus's per-slot gauges).
119    pub fn ever_fed(&self) -> bool {
120        self.ever_fed
121    }
122}

124#[cfg(test)]
125mod tests {
126    use super::*;
127
128    const WANT: usize = 480; // a 10 ms mix chunk @ 48 kHz
129    const LOUD: i32 = 10_000; // a clear speech-level peak
130    const QUIET: i32 = 100; // below SILENCE_FLOOR — comfort noise / silence
131
132    fn full(peak: i32) -> Tick {
133        Tick { contributed: WANT, want: WANT, peak_abs: peak }
134    }
135    fn empty() -> Tick {
136        Tick { contributed: 0, want: WANT, peak_abs: 0 }
137    }
138    fn partial(n: usize, peak: i32) -> Tick {
139        Tick { contributed: n, want: WANT, peak_abs: peak }
140    }
141
142    #[test]
143    fn warmup_before_first_feed_never_counts() {
144        let mut t = StarveTracker::new();
145        for _ in 0..20 {
146            assert!(!t.observe(empty()), "pre-first-frame empty is not a gap");
147        }
148    }
149
150    #[test]
151    fn steady_full_feed_never_counts() {
152        let mut t = StarveTracker::new();
153        for _ in 0..100 {
154            assert!(!t.observe(full(LOUD)));
155        }
156    }
157
158    #[test]
159    fn active_speech_gap_counts_onset_then_idles() {
160        // Loud audio then a sustained gap: the onset counts up to STARVE_IDLE_TICKS,
161        // then the slot is treated as having stopped (no runaway count).
162        let mut t = StarveTracker::new();
163        assert!(!t.observe(full(LOUD)), "the priming full chunk is not a gap");
164        let counted: u32 = (0..50).map(|_| t.observe(empty()) as u32).sum();
165        assert_eq!(
166            counted, STARVE_IDLE_TICKS,
167            "a real mid-speech gap counts its onset only"
168        );
169    }
170
171    #[test]
172    fn continuously_silent_peer_never_counts() {
173        // The live pathology: a peer that is silent throughout (comfort noise then DTX
174        // gaps) must register ZERO stutter. Feed quiet output until silent, then gaps.
175        let mut t = StarveTracker::new();
176        for _ in 0..ACTIVE_DECAY_TICKS + 2 {
177            assert!(!t.observe(full(QUIET)), "quiet full feed is never a gap");
178        }
179        let counted: u32 = (0..100).map(|_| t.observe(empty()) as u32).sum();
180        assert_eq!(counted, 0, "a continuously-silent peer's DTX gaps are not stutter");
181    }
182
183    #[test]
184    fn partial_drain_of_active_audio_counts() {
185        // A pump that can't keep up during speech (half-chunk drains) is real stutter.
186        let mut t = StarveTracker::new();
187        let counted: u32 = (0..10).map(|_| t.observe(partial(WANT / 2, LOUD)) as u32).sum();
188        assert_eq!(counted, 10, "each short drain of active audio counts");
189    }
190
191    #[test]
192    fn partial_drain_of_silence_does_not_count() {
193        // Half-chunk drains of comfort noise are inaudible — not stutter.
194        let mut t = StarveTracker::new();
195        let counted: u32 = (0..10).map(|_| t.observe(partial(WANT / 2, QUIET)) as u32).sum();
196        assert_eq!(counted, 0, "short drains of silence are not stutter");
197    }
198
199    #[test]
200    fn recovery_resets_the_gap_onset_window() {
201        // A gap, then audio recovers, then another gap: the second gap gets its OWN
202        // full onset window (the metric is per-gap, not cumulative).
203        let mut t = StarveTracker::new();
204        assert!(!t.observe(full(LOUD)));
205        let first: u32 = (0..10).map(|_| t.observe(empty()) as u32).sum();
206        assert_eq!(first, STARVE_IDLE_TICKS);
207        // Audio returns for a few ticks (resets the streak + activity).
208        for _ in 0..3 {
209            assert!(!t.observe(full(LOUD)));
210        }
211        let second: u32 = (0..10).map(|_| t.observe(empty()) as u32).sum();
212        assert_eq!(second, STARVE_IDLE_TICKS, "the second gap counts fresh");
213    }
214
215    #[test]
216    fn abrupt_stop_counts_once_then_silent() {
217        // Loud → immediate silence (no quiet tail): the first gap's onset counts, but a
218        // SECOND gap after the slot has gone silent does not (it's now DTX).
219        let mut t = StarveTracker::new();
220        assert!(!t.observe(full(LOUD)));
221        let first: u32 = (0..20).map(|_| t.observe(empty()) as u32).sum();
222        assert_eq!(first, STARVE_IDLE_TICKS);
223        // Still silent; a comfort-noise blip then another gap counts nothing.
224        let _ = t.observe(full(QUIET));
225        let second: u32 = (0..20).map(|_| t.observe(empty()) as u32).sum();
226        assert_eq!(second, 0, "a silent peer's later gaps are not stutter");
227    }
