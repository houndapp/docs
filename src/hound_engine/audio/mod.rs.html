1//! # Portable audio engine services
2//!
3//! The pieces of the audio stack that are pure engine logic — built ON TOP of
4//! the platform backends, not part of them:
5//!
6//! * [`apm`] — the full-duplex audio-processing module (AEC3 + NS + AGC2,
7//!   pure-Rust `sonora`), shared between the mic capture path
8//!   (`process_capture`) and the playout bus (`process_render`).
9//! * [`mix`] — the mixed-playout bus: ONE output device stream per call and
10//!   ONE coherent AEC render reference, summing every remote peer's due PCM.
11//! * [`level_from_rms`] — the perceptual speaking-level mapping for UI meters.
12//!
13//! All of it is platform-agnostic (the only device contact is through the
14//! injected output sink), which is why a new platform backend gets working
15//! multi-party audio + AEC for free.
16

21/// One-shot reopen delay for audio endpoints on a non-48 kHz device (a Bluetooth
22/// headset in HFP/hands-free mode). The SCO link is frequently still establishing
23/// when a stream first opens; the stream then runs SILENTLY — no error fires, so
24/// the dead-stream reopen never triggers. A single reopen this long after open
25/// lands on the settled route. Applies to BOTH directions (playout in the mix
26/// bus, capture in the client mic loop); 48 kHz devices never arm it.
27pub const ROUTE_WARMUP_DELAY: std::time::Duration = std::time::Duration::from_millis(1200);

32/// Promote the CURRENT thread to real-time audio scheduling and hold the policy
33/// for the thread's whole lifetime (the handle is stashed in a thread-local).
34///
35/// Intended as a `tokio::runtime::Builder::on_thread_start` hook for the dedicated
36/// audio-receive/fill runtime: each per-peer pump decodes Opus and fills its mix
37/// jitter buffer, and if a CPU-heavy screen-share decode/scale/render load
38/// deschedules that fill, the RT mix thread finds the buffer dry at its pull tick
39/// and the call audibly stutters (`mix_starve`). The mix bus (the playout side) is
40/// already RT; this extends the same guarantee to the producer side so video load
41/// can't starve voice — "audio is sacred". The work is bounded and mostly parked on
42/// the network, so RT promotion is safe (no busy-loop to starve the system).
43///
44/// Best-effort: on failure (no entitlement, unsupported platform) it logs and the
45/// thread runs at normal priority — never a hard error. Idempotent per thread.
46///
47/// NOTE: this protects the decode+fill half of the audio receive path. The QUIC
48/// socket reads feeding it run on the shared connection runtime; if THAT is the
49/// descheduled link under load, this alone won't fully cure `mix_starve`.
50pub fn promote_current_thread_for_audio_fill() {
51    use crate::types::SAMPLE_RATE;
52    use std::cell::RefCell;
53
54    thread_local! {
55        // Holds the RT policy handle for the thread's lifetime. Keeping it alive is
56        // belt-and-suspenders (the mach backend has no demote-on-drop, but Windows
57        // MMCSS frees the task when its handle drops), and makes the promotion
58        // outlive this function call.
59        static RT_HANDLE: RefCell<Option<audio_thread_priority::RtPriorityHandle>> =
60            const { RefCell::new(None) };
61    }
62
63    // A 10 ms-equivalent audio quantum (matches the mix tick cadence) as the RT
64    // period hint — the fill produces 20 ms Opus frames, but a tighter period biases
65    // the scheduler toward running it promptly when a frame lands.
66    let frames_hint = (SAMPLE_RATE.hz() / 100).max(1);
67    match audio_thread_priority::promote_current_thread_to_real_time(frames_hint, SAMPLE_RATE.hz())
68    {
69        Ok(handle) => {
70            RT_HANDLE.with_borrow_mut(|slot| *slot = Some(handle));
71            tracing::info!("voice: audio-rx fill thread promoted to real-time priority");
72        }
73        Err(e) => {
74            tracing::warn!("voice: audio-rx RT priority unavailable ({e:?}) — normal priority");
75        }
76    }
77}

79/// Map a linear RMS amplitude (`0.0..=1.0`, as [`crate::types::PcmFrame::rms`]
80/// returns — i16 full-scale = 1.0) to a perceptual `0.0..=1.0` meter level for
81/// speaking-level UI.
82///
83/// ## Mapping chosen: dBFS-windowed
84///
85/// We convert to dBFS (`20·log10(rms)`) and linearly map a **−60 dB … 0 dB**
86/// window onto `0.0..=1.0`: `level = (dbfs + 60) / 60`, clamped. This is the
87/// right mapping for a voice meter: loudness perception is logarithmic, so a
88/// dB window gives a meter where normal speech sits in the upper-middle and
89/// quiet-room noise sits near the bottom — unlike a raw-linear meter where
90/// everything below a shout looks dead. −60 dBFS is the practical noise floor
91/// (below it a quiet room and silence are indistinguishable); 0 dBFS is full
92/// scale. A true-zero RMS maps to exactly 0.0 (silence reads empty, not −∞).
93///
94/// Pure and total: no NaN/Inf escapes (silence short-circuits, the result is
95/// always clamped to `0.0..=1.0`).
96pub fn level_from_rms(rms: f32) -> f32 {
97    if rms <= 0.0 {
98        return 0.0;
99    }
100    let dbfs = 20.0 * rms.log10();
101    ((dbfs + 60.0) / 60.0).clamp(0.0, 1.0)
102}

104#[cfg(test)]
105mod tests {
106    use super::*;
107
108    /// Silence (and any non-positive RMS) reads as an empty meter — never −∞ dB.
109    #[test]
110    fn level_silence_is_zero() {
111        assert_eq!(level_from_rms(0.0), 0.0);
112        assert_eq!(
113            level_from_rms(-0.5),
114            0.0,
115            "guard against a bogus negative rms"
116        );
117    }
118
119    /// Full-scale RMS (1.0) maps to a full meter (0 dBFS → 1.0).
120    #[test]
121    fn level_full_scale_is_one() {
122        assert_eq!(level_from_rms(1.0), 1.0);
123    }
124
125    /// The window floor: −60 dBFS (rms = 1e-3) maps to ~0.0, and anything quieter
126    /// clamps to 0.0 rather than going negative.
127    #[test]
128    fn level_floor_clamps() {
129        // 10^(-60/20) = 1e-3 → dbfs == -60 → (−60+60)/60 == 0.
130        let at_floor = level_from_rms(1e-3);
131        assert!(
132            at_floor.abs() < 1e-4,
133            "−60 dBFS sits at the meter floor, got {at_floor}"
134        );
135        // Quieter than the floor clamps, never goes negative.
136        assert_eq!(level_from_rms(1e-5), 0.0);
137    }
138
139    /// The mapping is monotonic: louder input always reads as a higher (or equal,
140    /// at the clamp) meter level — the property a level meter must have.
141    #[test]
142    fn level_is_monotonic() {
143        let samples = [0.0f32, 1e-4, 1e-3, 1e-2, 0.05, 0.1, 0.3, 0.5, 0.8, 1.0];
144        let mut prev = -1.0f32;
145        for &rms in &samples {
146            let lvl = level_from_rms(rms);
147            assert!(
148                lvl >= prev,
149                "level must be non-decreasing in rms: rms={rms} gave {lvl} < prev {prev}"
150            );
151            assert!((0.0..=1.0).contains(&lvl), "level out of range: {lvl}");
152            prev = lvl;
153        }
154    }
155
156    /// A mid-level speech RMS (~−26 dBFS, rms ≈ 0.05) lands in the upper-middle of
157    /// the meter — the perceptual win over a raw-linear meter (where 0.05 would
158    /// read as a near-dead 5%).
159    #[test]
160    fn level_mid_speech_is_visible() {
161        let lvl = level_from_rms(0.05);
162        assert!(
163            lvl > 0.45 && lvl < 0.7,
164            "mid speech should read upper-middle, got {lvl}"
165        );
166    }
