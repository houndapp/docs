1//! # The high-level, platform-agnostic engine API
2//!
3//! This is the only surface the app consumes (together with
4//! [`crate::types`]): open devices by typed id, list the device inventory,
5//! build codecs by typed configuration, and query what the running platform
6//! can do. No platform type (cpal, AVFoundation, VideoToolbox, …) is
7//! reachable from here — the handles below wrap the compile-time-selected
8//! backend and expose only engine vocabulary.
9//!
10//! ## Zero-cost facade
11//!
12//! [`Engine`] is a zero-sized value; every handle is a thin newtype over the
13//! platform backend chosen by `cfg` at compile time. There is no `dyn` on the
14//! per-frame hot paths (the one boxed seam is the playout bus's output sink,
15//! which is injected for testability and called 100×/s, not per sample).
16//!
17//! ## Capability, not panic
18//!
19//! Platforms differ (no camera backend on a server build, no HW HEVC off Apple).
20//! Codec differences are *queried* at runtime via [`Engine::encode_caps`] /
21//! [`Engine::decode_caps`]; every other miss surfaces as the typed
22//! [`CaptureError::Unsupported`] a constructor returns when a caller asks anyway
23//! — never a panic, never a silently-dead pipeline.
24

264    /// Open a screen/window capture per `config`, stamping frames on `clock`.
265    ///
266    /// Open is construct: the returned [`ScreenShare`] is already delivering
267    /// I420 frames scaled to `config.dimensions` (aspect-preserved), and dropping
268    /// it stops the session. When `config.capture_audio` is set **and** the
269    /// platform supports the source's audio scope
270    /// ([`Capabilities::app_audio_capture`]; window sources capture the app's
271    /// own audio on macOS 13+ and Windows, display sources additionally capture
272    /// system audio on Windows), the returned tuple's second element is
273    /// `Some(`[`ScreenAudio`]`)` delivering that audio as 48 kHz mono 20 ms
274    /// frames (the mic contract); otherwise it is `None`.
275    ///
276    /// A Screen Recording TCC denial surfaces [`CaptureError::PermissionDenied`];
277    /// a platform without a screen backend returns [`CaptureError::Unsupported`].
278    /// The handle is `!Send` (the SCStream is tied to its setup context):
279    /// construct, drive, and drop it on one dedicated capture thread.
280    pub fn open_screen_capture(
281        self,
282        source: &crate::types::ScreenSource,
283        config: ScreenCaptureConfig,
284        clock: hound_media::CaptureClock,
285    ) -> Result<(ScreenShare, Option<ScreenAudio>), CaptureError> {
286        // Camera-as-screen: open the chosen camera and stream it onto a screen
287        // tile. Works wherever a camera backend exists (independent of the
288        // platform SCREEN backend); video only, so no app-audio leg.
289        if let crate::types::ScreenSource::Camera(id) = source {
290            let cam = self.open_camera(
291                &DeviceChoice::Named(id.clone()),
292                config.dimensions,
293                config.fps,
294                clock,
295            )?;
296            return Ok((ScreenShare(ScreenShareInner::Camera(cam)), None));
297        }
298
299        // Every platform opens through the same SPI; `ScreenBackend` is the
300        // platform-selected alias (SCK on Apple, WGC on Windows, portal+PipeWire
301        // on Linux, the uninhabited stub — a typed `Unsupported` — elsewhere).
302        // Audio is `Some` only where the backend yields an audio leg (Apple
303        // window sources; Windows window/display sources; Linux display sources
304        // — Wayland's portal hides a window's owning app, so Linux window
305        // shares carry no auto audio and `os::linux` offers app-scoped capture).
306        let (video, audio) = ScreenBackend::open(source, config, clock)?;
307        Ok((
308            ScreenShare(ScreenShareInner::Platform(video)),
309            audio.map(ScreenAudio),
310        ))
311    }

313    /// Build a real-time H.264 encoder for `dims`@`fps` targeting `bitrate`,
314    /// preferring **hardware** (VideoToolbox on Apple) and falling back to
315    /// **software** (openh264, the Cisco WebRTC baseline codec) where no HW
316    /// encoder exists — i.e. every Linux/Windows desktop today. The
317    /// geometry/rate preconditions are carried by the parameter types, so an
318    /// invalid configuration cannot be expressed.
319    ///
320    /// This always succeeds in producing an H.264 encoder (HW or SW) on every
321    /// supported platform: H.264 is the all-platform wire floor, so any client
322    /// MUST be able to encode it. The returned [`VideoEncoder`] hides which path
323    /// was selected; the codec on the wire is H.264 either way. Selection order is
324    /// HW > SW (see [`Capabilities::hw_h264_encode`] / [`Capabilities::sw_h264`]).
325    ///
326    /// Thin wrapper over [`Self::video_encoder`]`(VideoCodec::H264, …)`.
327    /// Not `Send`: lives on the capture/encode thread.
328    pub fn h264_encoder(
329        self,
330        dims: Dimensions,
331        fps: Fps,
332        bitrate: Bitrate,
333        cadence: KeyframeCadence,
334    ) -> Result<VideoEncoder, CaptureError> {
335        self.video_encoder(VideoCodec::H264, dims, fps, bitrate, cadence)
336    }

338    /// Build a real-time video encoder for a specific `codec`, preferring hardware:
339    ///
340    /// * **H.264** — HW VideoToolbox on Apple, else SW openh264. Always succeeds
341    ///   (the universal floor); a HW failure falls through to SW.
342    /// * **HEVC (H.265)** — HW VideoToolbox only. `Unsupported` on silicon without
343    ///   a HW HEVC block and everywhere off Apple (there is no SW HEVC encoder).
344    /// * **AV1** — HW VideoToolbox on M5 Pro/Max + modern discrete GPUs. Not yet
345    ///   wired (the VideoToolbox AV1 encoder awaits validation on AV1-encode HW),
346    ///   so it currently returns `Unsupported`; [`Self::can_encode`] reports the
347    ///   same, so the codec ladder never selects AV1 for encode until it lands.
348    ///
349    /// The codec ladder ([`crate::types::pick_encode_codec`]) chooses which `codec`
350    /// to pass here; callers that just want "an H.264 encoder" use
351    /// [`Self::h264_encoder`]. Not `Send`: lives on the capture/encode thread.
352    ///
353    /// Platform dispatch (HW VideoToolbox vs SW openh264, per-codec availability)
354    /// lives in [`crate::platform::VideoEncoderBackend`] — this is pure delegation.
355    pub fn video_encoder(
356        self,
357        codec: VideoCodec,
358        dims: Dimensions,
359        fps: Fps,
360        bitrate: Bitrate,
361        cadence: KeyframeCadence,
362    ) -> Result<VideoEncoder, CaptureError> {
363        crate::platform::VideoEncoderBackend::open(codec, dims, fps, bitrate, cadence)
364            .map(VideoEncoder)
365    }

369    /// (which prefers HW); this is for when the SW path is specifically required.
370    pub fn sw_h264_encoder(
371        self,
372        dims: Dimensions,
373        fps: Fps,
374        bitrate: Bitrate,
375        cadence: KeyframeCadence,
376    ) -> Result<impl VideoEncode, CaptureError> {

378    }

430/// A running camera: a started, platform-selected video capture delivering
431/// I420 frames at the requested encode geometry. Obtained from

436/// The concrete capture behind a [`Camera`]: the compile-time-selected platform
437/// backend, or (behind `synthetic`) the camera-free test source. An enum rather
438/// than `Box<dyn VideoCapture>` — the arm set is closed and known at compile
439/// time, so dispatch stays static and every existing [`Camera`] call site (which
440/// only ever goes through its inherent methods, never `.0`) is unaffected.
441enum CameraInner {
442    Platform(CameraBackend),
443    #[cfg(feature = "synthetic")]
444    Synthetic(crate::platform::generic::synthetic::SyntheticCapturer),
445}

447impl Camera {
448    /// Block until the next frame is available. `Err` when the stream ends or
449    /// the device is lost (the caller re-opens; a vanished named device falls
450    /// back to default inside the open).
451    pub fn next_frame(&mut self) -> Result<RawFrame, CaptureError> {
452        match &mut self.0 {
453            CameraInner::Platform(c) => c.next_frame(),
454            #[cfg(feature = "synthetic")]
455            CameraInner::Synthetic(c) => c.next_frame(),
456        }
457    }
458
459    /// The geometry + pixel format delivered frames carry (the encode target,
460    /// post any internal scale) — always equal to the `target` it was opened
461    /// with.
462    pub fn negotiated(&self) -> (Dimensions, PixelFormat) {
463        match &self.0 {
464            CameraInner::Platform(c) => c.negotiated(),
465            #[cfg(feature = "synthetic")]
466            CameraInner::Synthetic(c) => c.negotiated(),
467        }
468    }

471/// A running screen share: either a platform screen/window capture, or a
472/// **camera shared as a screen** (the picker's "Devices" tab — a webcam,
473/// capture card, or virtual camera streamed onto a screen tile). Obtained from
474/// [`Engine::open_screen_capture`]; dropping it stops the session and (with it)
475/// any paired [`ScreenAudio`] leg. Not `Send`: drive and drop on the one
476/// capture thread (the `!Send` contract).
477///
478/// Both arms produce the same I420/NV12 [`RawFrame`]s, so the publish pipeline
479/// is identical — the only differences are that a camera is never damage-driven
480/// (it always delivers) and never resizes, so its `next_frame_timeout` always
481/// blocks for a frame and its `content_size` is `None`.
482pub struct ScreenShare(ScreenShareInner);
483
484enum ScreenShareInner {
485    /// A platform display/window capture (SCK / WGC / PipeWire).
486    Platform(ScreenBackend),
487    /// A camera streamed as a screen. Continuous (not damage-driven).
488    Camera(Camera),
489}
490
491impl ScreenShare {
492    /// Block until the next frame is available. `Err` when the share ends — the
493    /// window closed, the user stopped it via the OS overlay, the camera was
494    /// lost, or the session was lost. The caller treats that as a clean capture
495    /// end (stop publishing the screen track), not a crash.
496    pub fn next_frame(&mut self) -> Result<RawFrame, CaptureError> {
497        match &mut self.0 {
498            ScreenShareInner::Platform(s) => s.next_frame(),
499            ScreenShareInner::Camera(c) => c.next_frame(),
500        }
501    }
502
503    /// Bounded-wait pull: `Ok(None)` when `timeout` elapses with no frame.
504    /// Screen capture is damage-driven — a static source produces no frames —
505    /// so a consumer that must stay live (preview, keyframe-on-demand for new
506    /// subscribers) polls with this and re-emits its cached last frame on
507    /// `None`. A camera is continuous, so it just blocks for the next frame
508    /// (the `timeout` is irrelevant — a frame is always coming).
509    pub fn next_frame_timeout(
510        &mut self,
511        timeout: std::time::Duration,
512    ) -> Result<Option<RawFrame>, CaptureError> {
513        match &mut self.0 {
514            ScreenShareInner::Platform(s) => s.next_frame_timeout(timeout),
515            ScreenShareInner::Camera(c) => c.next_frame().map(Some),
516        }
517    }
518
519    /// The source's current content size (points) observed on delivered
520    /// frames, where the platform reports it. A running capture's pixel
521    /// geometry is fixed at open, so a change here means the shared window was
522    /// resized — restart the share to track it. `None` = unknown (always for a
523    /// camera — it has a fixed negotiated size and never resizes).
524    pub fn content_size(&self) -> Option<(u32, u32)> {
525        match &self.0 {
526            ScreenShareInner::Platform(s) => s.content_size(),
527            ScreenShareInner::Camera(_) => None,
528        }
529    }
530
531    /// Whether the captured SOURCE still exists (see
532    /// [`crate::backend::ScreenCapture::source_alive`]). A camera-as-screen share
533    /// reports alive — its device loss surfaces through `next_frame` errors.
534    /// Blocking (~tens of ms on SCK) — call from the capture thread while idle.
535    pub fn source_alive(&self) -> bool {
536        match &self.0 {
537            ScreenShareInner::Platform(s) => s.source_alive(),
538            ScreenShareInner::Camera(_) => true,
539        }
540    }
541
542    /// The geometry + pixel format delivered frames carry (the encode target,
543    /// post the native→target scale) — what an encoder built for this share
544    /// should be configured with.
545    pub fn negotiated(&self) -> (Dimensions, PixelFormat) {
546        match &self.0 {
547            ScreenShareInner::Platform(s) => s.negotiated(),
548            ScreenShareInner::Camera(c) => c.negotiated(),
549        }
550    }
551}

561impl ScreenAudio {

566    }

574    }

607/// A running microphone: a started, platform-selected audio input delivering
608/// fixed 20 ms 48 kHz mono PCM frames. Obtained from
609/// [`Engine::open_microphone`]; dropping it stops capture and releases the
610/// device (on the owning thread — the `!Send` contract).
611pub struct Microphone(MicrophoneInner);
612
613/// The concrete audio input behind a [`Microphone`]: the platform cpal capturer,
614/// or (behind `synthetic`) the mic-free test tone. Mirrors [`CameraInner`].
615enum MicrophoneInner {
616    // Boxed: `CpalCapturer` is far larger than the synthetic arm (rings +
617    // resampler state), and clippy flags the resulting size skew
618    // (`large_enum_variant`) once the `synthetic` feature adds a second arm.
619    Platform(Box<cpal_io::CpalCapturer>),
620    #[cfg(feature = "synthetic")]
621    Synthetic(crate::platform::generic::synthetic_audio::SyntheticAudioInput),
622}
623
624impl Microphone {
625    /// The opened device's native sample rate (Hz). A non-48 kHz rate means a
626    /// resampled open — in practice a Bluetooth headset in HFP mode, whose SCO
627    /// route may not be live yet at open (see
628    /// [`crate::audio::ROUTE_WARMUP_DELAY`]); the capture owner uses this to arm
629    /// the one-shot warmup reopen. Always 48 kHz for the synthetic source (it
630    /// generates at the engine rate directly), so it never arms that warmup.
631    pub fn device_rate(&self) -> u32 {
632        use crate::backend::AudioInput;
633        match &self.0 {
634            MicrophoneInner::Platform(m) => m.negotiated().0.hz(),
635            #[cfg(feature = "synthetic")]
636            MicrophoneInner::Synthetic(m) => m.negotiated().0.hz(),
637        }
638    }
639
640    /// Block until the next fixed-size PCM frame is available. `Err` when the
641    /// stream ends or the device is lost (the caller re-opens — a vanished
642    /// named device falls back to default inside the open).
643    pub fn next_frame(&mut self) -> Result<PcmFrame, AudioError> {
644        match &mut self.0 {
645            MicrophoneInner::Platform(m) => m.next_frame(),
646            #[cfg(feature = "synthetic")]
647            MicrophoneInner::Synthetic(m) => m.next_frame(),
648        }
649    }
650
651    /// The (rate, channels) the device actually negotiated — the device truth
652    /// for diagnostics. Delivery is always 48 kHz mono regardless.
653    pub fn negotiated(&self) -> (SampleRate, ChannelCount) {
654        match &self.0 {
655            MicrophoneInner::Platform(m) => m.negotiated(),
656            #[cfg(feature = "synthetic")]
657            MicrophoneInner::Synthetic(m) => m.negotiated(),
658        }
659    }
660
661    /// The name of the device actually opened (post-fallback), for logging
662    /// what a hot-swap landed on. `Some("synthetic")` for the test source.
663    pub fn opened_name(&self) -> Option<&str> {
664        match &self.0 {
665            MicrophoneInner::Platform(m) => m.opened_name(),
666            #[cfg(feature = "synthetic")]
667            MicrophoneInner::Synthetic(_) => Some("synthetic"),
668        }
669    }
670}
