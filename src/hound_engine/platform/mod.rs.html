1//! # Per-platform backends
2//!
3//! The concrete implementations of the [`crate::backend`] SPI, selected at
4//! **compile time** per target. Nothing in here is visible outside the crate
5//! (`pub(crate)`): the app reaches devices and codecs only through
6//! [`crate::api`], which wraps these in platform-agnostic handles.
7//!
8//! * [`generic`] — the portable pieces that already build on every desktop
9//!   target: cpal audio I/O (CoreAudio / WASAPI / ALSA-Pulse / AAudio),
10//!   device-rate resampling, audio device enumeration, the SW H.264 codec
11//!   (openh264, both ways), the synthetic test source, and pixel-format
12//!   conversion.
13//! * `apple` — the Apple-only camera + HW codec backend (AVFoundation
14//!   capture, VideoToolbox H.264/HEVC/AV1, and the safe sys layer under them).
15//! * `linux` — the Linux camera backend (V4L2 via the `v4l` crate). Video codecs
16//!   default to the portable SW openh264 H.264 floor; a VAAPI HW backend
17//!   (cros-libva / cros-codecs: HW H.264 encode, H.264 + HEVC decode) is gated
18//!   behind the off-by-default `linux-vaapi` feature (cros-libva's system-libva
19//!   build coupling — see the feature note). HEVC encode is unwired regardless.
20//! * `windows` — the Windows camera backend (Media Foundation via the official
21//!   `windows` crate, `IMFSourceReader` synchronous pull) plus the Media
22//!   Foundation Transform HW video codecs: HW H.264/HEVC encode (async MFT) and
23//!   H.264/HEVC/AV1 decode (DXVA), with the SW openh264 H.264 floor retained.
24//! * [`stub`] — the uninhabited per-role backends for capabilities a target
25//!   lacks (a typed `Unsupported`, never a panic).
26//!
27//! ## Backend selection lives HERE
28//!
29//! The per-role backend **types** are resolved by `cfg` in this module
30//! (`CameraBackend`, `ScreenBackend`, `AppAudioBackend`, `VideoDecoderBackend`),
31//! so [`crate::api`] stays platform-agnostic — it names `platform::CameraBackend`
32//! and never carries a `#[cfg(target_os = …)]` of its own. A capability a target
33//! lacks resolves to the matching [`stub`] type.
34//!
35//! ## Adding a backend (the Windows/Linux parity path)
36//!
37//! Add a `windows`/`linux` module gated on its `target_os`, implement the
38//! [`crate::backend`] traits for the native capture/codec APIs, then point the
39//! `cfg`-resolved alias below at it (and, for a new HW codec, wire the runtime
40//! [`crate::api::Engine::encode_caps`] / [`crate::api::Engine::decode_caps`]
41//! probes). Platform deps go in this crate's `[target.'cfg(...)'.dependencies]`
42//! — the app crate never gains one.
43
44#[cfg(any(target_os = "macos", target_os = "ios"))]
45pub(crate) mod apple;
46pub(crate) mod generic;
47#[cfg(target_os = "linux")]
48pub(crate) mod linux;
49pub(crate) mod stub;
50#[cfg(target_os = "windows")]
51pub(crate) mod windows;

57/// The camera capture backend: AVFoundation (Apple), V4L2 (Linux), Media

73/// The video **encoder** backend, by platform:
74/// * **Apple** — HW VideoToolbox (H.264 / HEVC) + SW openh264 floor.
75/// * **Windows** — HW async Media Foundation Transform (H.264 / HEVC) + SW
76///   openh264 floor.
77/// * **Linux** — SW openh264 floor by default; HW VAAPI (H.264, cros-codecs)
78///   when built with the `linux-vaapi` feature (see below).
79/// * **else** — SW openh264 only (H.264).
80///
81/// All expose the uniform `open(codec, dims, fps, bitrate)` / `caps()` the facade
82/// dispatches through.
83///
84/// NOTE (Linux): the VAAPI HW path ([`linux::video_codec::LinuxVideoEncoder`],
85/// H.264 + SW floor) is gated behind the off-by-default `linux-vaapi` feature.
86/// cros-libva bindgens the system libva headers and only compiles against the
87/// libva generation it was written for, so the default Linux build stays on the
88/// always-buildable SW floor; enable `linux-vaapi` on a box whose libva matches
89/// (or after vendoring a patched cros-libva). HEVC encode is unwired regardless
90/// (no VAAPI HEVC encoder upstream).
91#[cfg(any(target_os = "macos", target_os = "ios"))]
92pub(crate) type VideoEncoderBackend = apple::video_codec::AppleVideoEncoder;
93#[cfg(target_os = "windows")]
94pub(crate) type VideoEncoderBackend = windows::video_codec::WindowsVideoEncoder;
95#[cfg(all(target_os = "linux", feature = "linux-vaapi"))]
96pub(crate) type VideoEncoderBackend = linux::video_codec::LinuxVideoEncoder;
97#[cfg(not(any(
98    target_os = "macos",
99    target_os = "ios",
100    target_os = "windows",
101    all(target_os = "linux", feature = "linux-vaapi")
102)))]
103pub(crate) type VideoEncoderBackend = generic::sw_codec::SwVideoEncoder;
104
105/// The video **decoder** backend, by platform:
106/// * **Apple** — HW VideoToolbox (H.264 / HEVC / AV1).
107/// * **Windows** — HW Media Foundation Transform (H.264 / HEVC via DXVA) with a
108///   SW openh264 fallback for H.264.
109/// * **Linux** — SW openh264 (H.264) by default; HW VAAPI (H.264 / HEVC via
110///   cros-codecs) with a SW H.264 fallback when built with the `linux-vaapi`
111///   feature (see the encoder note above for why it is off by default).
112/// * **else** — SW openh264 (H.264 only).
113///
114/// All expose the uniform `open(codec)` / `caps()` the facade dispatches through.
115#[cfg(any(target_os = "macos", target_os = "ios"))]
116pub(crate) type VideoDecoderBackend = apple::vt_decoder::VideoToolboxDecoder;
117#[cfg(target_os = "windows")]
118pub(crate) type VideoDecoderBackend = windows::video_codec::WindowsVideoDecoder;
119#[cfg(all(target_os = "linux", feature = "linux-vaapi"))]
120pub(crate) type VideoDecoderBackend = linux::video_codec::LinuxVideoDecoder;
121#[cfg(not(any(
122    target_os = "macos",
123    target_os = "ios",
124    target_os = "windows",
125    all(target_os = "linux", feature = "linux-vaapi")
126)))]
127pub(crate) type VideoDecoderBackend = generic::sw_codec::SwVideoDecoder;

179        linux::camera::enumerate_cameras()
180    }

184    }
185    #[cfg(not(any(
186        target_os = "macos",
187        target_os = "ios",
188        target_os = "linux",
189        target_os = "windows"
190    )))]
191    {
192        Vec::new()
193    }
194}

196/// Enumerate shareable screen sources: ScreenCaptureKit (Apple) and WGC
197/// (Windows) list concrete displays + windows; the Linux portal backend lists
198/// one entry per castable *kind* (the compositor's dialog picks the concrete
199/// source, by Wayland design); `Unsupported` where there is no backend.
200pub(crate) fn enumerate_screen_sources()
201-> Result<Vec<crate::types::ScreenSourceInfo>, crate::error::CaptureError> {
202    #[cfg(any(target_os = "macos", target_os = "ios"))]
203    {
204        apple::screen::enumerate_sources()
205    }
206    #[cfg(target_os = "windows")]
207    {
208        windows::wgc::enumerate_sources()
209    }
210    #[cfg(target_os = "linux")]
211    {
212        linux::portal_screen::enumerate_sources()
213    }
214    #[cfg(not(any(
215        target_os = "macos",
216        target_os = "ios",
217        target_os = "windows",
218        target_os = "linux"
219    )))]
220    {
221        Err(crate::error::CaptureError::Unsupported {
222            what: "screen capture",
223        })
224    }
225}

227/// One-shot screen-source thumbnail for the share picker: ScreenCaptureKit
228/// (Apple, macOS 14+) or WGC (Windows); `Unsupported` on Linux (capturing
229/// would raise the portal dialog per row) and where there is no backend. A
230/// failure on any platform is non-fatal — the picker degrades to a title-only
231/// card.
232#[cfg_attr(
233    not(any(
234        target_os = "macos",
235        target_os = "ios",
236        target_os = "windows",
237        target_os = "linux"
238    )),
239    allow(unused_variables)
240)]
241pub(crate) fn screen_thumbnail(
242    source: &crate::types::ScreenSource,
243    max_w: u32,
244    max_h: u32,
245) -> Result<crate::types::ThumbnailImage, crate::error::CaptureError> {
246    #[cfg(any(target_os = "macos", target_os = "ios"))]
247    {
248        apple::screen::capture_thumbnail(source, max_w, max_h)
249    }
250    #[cfg(target_os = "windows")]
251    {
252        windows::wgc::capture_thumbnail(source, max_w, max_h)
253    }
254    #[cfg(target_os = "linux")]
255    {
256        linux::portal_screen::capture_thumbnail(source, max_w, max_h)
257    }
258    #[cfg(not(any(
259        target_os = "macos",
260        target_os = "ios",
261        target_os = "windows",
262        target_os = "linux"
263    )))]
264    {
265        Err(crate::error::CaptureError::Unsupported {
266            what: "screen thumbnails",
267        })
268    }
