1//! # Portal + PipeWire screen/window capture (Linux)
2//!
3//! The Linux [`ScreenCapture`] backend, driven natively like its siblings
4//! (macOS ScreenCaptureKit, Windows WGC): an xdg-desktop-portal ScreenCast
5//! session ([`super::portal`]) selects the source and hands over a PipeWire
6//! video node; this module consumes that node directly with a libpipewire
7//! capture stream and delivers scaled, tightly-packed I420 [`RawFrame`]s
8//! stamped on the shared capture clock.
9//!
10//! ## Source model (Wayland's rules)
11//!
12//! Applications cannot enumerate windows/displays on Wayland — the compositor's
13//! portal dialog owns selection. So [`enumerate_sources`] returns one entry per
14//! *castable kind* (screen / window, probed from the portal) and the real pick
15//! happens in the system dialog at open. The picked kind still drives the
16//! share-audio scope exactly like Windows: a **display** share carries
17//! default-sink (whole-system) audio, a **window** share carries **no** audio —
18//! the portal never reveals the window's owning application, and widening a
19//! window share to system audio would leak other apps (the privacy rule shared
20//! with the WASAPI process-loopback path). App-scoped audio by explicit
21//! identity is [`crate::os::linux`]'s escape hatch.
22//!
23//! ## Pipeline & threading
24//!
25//! The portal thread (zbus) lives in [`super::portal`]. Here, a **bridge
26//! thread** owns every PipeWire object: connect over the portal fd, one video
27//! capture stream on the cast node negotiated down to **BGRx/BGRA over
28//! shared memory** (no dmabuf modifiers are offered, so compositors fall back
29//! to memfd — the zero-copy dmabuf→VAAPI import is the documented follow-up),
30//! `process` → stride-aware BGRA→I420 ([`convert::bgra_to_i420`]) → aspect-fit
31//! scale to the encode target → drop-oldest [`BoundedRing`]. The consumer-side
32//! contract (`next_frame_timeout` `Idle`/`Ended`, damage-driven delivery,
33//! `content_size` resize signal, `source_alive`) matches the other platforms:
34//! PipeWire renegotiates the stream format when the shared window resizes
35//! (a new `param_changed`), which is surfaced through [`ScreenCapture::content_size`],
36//! and the compositor ending the share (user hits "stop", window closes)
37//! destroys the node — observed as the stream erroring/unconnecting → a typed
38//! capture end, never a hang.
39

61/// Frame ring capacity (whole frames) — small; frames are large, a deep queue
62/// is just latency. Overflow drops the OLDEST (the encoder is behind).
63const RING_FRAMES: usize = 4;
64
65/// Bridge loop iteration timeout (frame delivery is event-driven; this only
66/// bounds stop-flag latency).
67const LOOP_TICK: Duration = Duration::from_millis(50);
68
69/// How long the open waits for the compositor to negotiate the video format
70/// after the portal handshake (the dialog has already been answered by then).
71const FORMAT_WAIT: Duration = Duration::from_secs(10);

73/// The sentinel ids [`enumerate_sources`] hands out. The portal dialog does the
74/// real selection, so the id only encodes which *kind* the picker is
75/// restricted to; a stale/unknown id of the right variant opens the same way.
76const PORTAL_SCREEN_ID: &str = "portal-screen";
77const PORTAL_WINDOW_ID: &str = "portal-window";

79/// List the castable source kinds as picker entries — one "Screen" and/or one
80/// "Window" row (the compositor's dialog picks the concrete source at open).
81/// An absent portal surfaces [`CaptureError::Unsupported`] so the UI degrades
82/// the same way as a platform without a screen backend.
83pub(crate) fn enumerate_sources() -> Result<Vec<ScreenSourceInfo>, CaptureError> {
84    let kinds = portal::available_kinds()?;
85    if kinds.is_empty() {
86        return Err(CaptureError::Unsupported {
87            what: "screen capture (portal offers no castable sources)",
88        });
89    }
90    // Placeholder geometry: the true size is known only after the portal
91    // dialog; the share opens at the source's native size regardless.
92    let placeholder = Dimensions::even_clamped(1920, 1080);
93    Ok(kinds
94        .into_iter()
95        .map(|kind| match kind {
96            portal::SourceKinds::Monitor => ScreenSourceInfo {
97                source: ScreenSource::Display(DisplayId::new(PORTAL_SCREEN_ID.to_string())),
98                title: "Screen (choose in system dialog)".to_string(),
99                app_name: None,
100                dimensions: placeholder,
101                is_on_screen: true,
102            },
103            portal::SourceKinds::Window => ScreenSourceInfo {
104                source: ScreenSource::Window(WindowId::new(PORTAL_WINDOW_ID.to_string())),
105                title: "Window (choose in system dialog)".to_string(),
106                app_name: None,
107                dimensions: placeholder,
108                is_on_screen: true,
109            },
110        })
111        .collect())
112}

127/// A running portal screen capture: the portal session guard keeps the cast
128/// alive, the bridge thread fills the ring, this handle pulls I420 frames.
129/// `!Send`-by-use (drive and drop on one capture thread, like every backend).
130pub(crate) struct PortalScreenCapturer {
131    ring: Arc<BoundedRing<RawFrame>>,
132    stop: Arc<AtomicBool>,
133    bridge: Option<JoinHandle<()>>,
134    encode_target: Dimensions,
135    /// Latest negotiated source size (points), packed `(w << 32) | h`; a
136    /// change means the shared window resized (the session restarts the share).
137    content_size: Arc<AtomicU64>,
138    /// Cleared when the stream errors/disconnects or the cast node vanishes.
139    alive: Arc<AtomicBool>,
140    /// Held for the share's lifetime; dropping it closes the portal session.
141    _portal: portal::PortalSession,
142}
143
144impl PortalScreenCapturer {
145    /// Open a share for `source`'s kind (the portal dialog picks the concrete
146    /// target) and begin delivery. The encode target is the source's native
147    /// size aspect-fit into `config.dimensions` (a 1080p-class cap).
148    ///
149    /// The audio leg follows the picked kind when `config.capture_audio`:
150    /// display → whole-system (default-sink monitor); window → `None` (no
151    /// owner identity on Wayland — never silently widened). An audio-leg
152    /// failure degrades to video-only, parity with the other platforms.
153    pub(crate) fn open(
154        source: &ScreenSource,
155        config: ScreenCaptureConfig,
156        clock: hound_media::CaptureClock,
157    ) -> Result<(Self, Option<pw_audio::PipeWireAppAudio>), CaptureError> {
158        let kinds = match source {
159            ScreenSource::Display(_) => portal::SourceKinds::Monitor,
160            ScreenSource::Window(_) => portal::SourceKinds::Window,
161            // Camera sources are opened as cameras (see `Engine::open_screen_capture`).
162            ScreenSource::Camera(_) => {
163                return Err(CaptureError::Unsupported {
164                    what: "camera source is not a portal screen target",
165                });
166            }
167        };
168
169        let (session, handoff) = portal::open(kinds)?;
170        let picked = handoff.kind;
171
172        let ring = BoundedRing::<RawFrame>::new(RING_FRAMES);
173        let stop = Arc::new(AtomicBool::new(false));
174        let alive = Arc::new(AtomicBool::new(true));
175        let content_size = Arc::new(AtomicU64::new(0));
176        let cap = config.dimensions;
177        let (ready_tx, ready_rx) = std::sync::mpsc::channel::<Result<Dimensions, CaptureError>>();
178
179        let bridge = std::thread::Builder::new()
180            .name("pw-screen-bridge".into())
181            .spawn({
182                let ring = Arc::clone(&ring);
183                let stop = Arc::clone(&stop);
184                let alive = Arc::clone(&alive);
185                let content_size = Arc::clone(&content_size);
186                move || {
187                    bridge_loop(
188                        handoff,
189                        cap,
190                        clock,
191                        ring,
192                        stop,
193                        alive,
194                        content_size,
195                        ready_tx,
196                    )
197                }
198            })
199            .map_err(|e| CaptureError::Device(format!("pw screen thread spawn failed: {e}")))?;
200
201        // Wait (bounded) for the format negotiation → the encode target. The
202        // dialog is already answered, so this is compositor latency only.
203        let encode_target = match ready_rx.recv_timeout(FORMAT_WAIT) {
204            Ok(Ok(dims)) => dims,
205            Ok(Err(e)) => {
206                stop.store(true, Ordering::Relaxed);
207                let _ = bridge.join();
208                return Err(e);
209            }
210            Err(_) => {
211                stop.store(true, Ordering::Relaxed);
212                let _ = bridge.join();
213                return Err(CaptureError::Device(
214                    "portal cast delivered no video format".into(),
215                ));
216            }
217        };
218
219        // The audio leg, scoped by what was actually picked. Failure degrades
220        // to video-only — the same posture as the WGC/WASAPI pairing.
221        let audio = if config.capture_audio {
222            match picked {
223                portal::PickedKind::Monitor | portal::PickedKind::Virtual => {
224                    match pw_audio::PipeWireAppAudio::open(
225                        clock,
226                        pw_audio::AudioScope::SystemMonitor,
227                    ) {
228                        Ok(leg) => Some(leg),
229                        Err(e) => {
230                            tracing::warn!(
231                                target: "portal_screen",
232                                "display-share audio leg unavailable ({e}); sharing video only"
233                            );
234                            None
235                        }
236                    }
237                }
238                portal::PickedKind::Window => {
239                    tracing::info!(
240                        target: "portal_screen",
241                        "window share: portal exposes no owning app, so no app-audio leg \
242                         (app-scoped capture is available via os::linux::screen_share_app_audio)"
243                    );
244                    None
245                }
246            }
247        } else {
248            None
249        };
250
251        Ok((
252            Self {
253                ring,
254                stop,
255                bridge: Some(bridge),
256                encode_target,
257                content_size,
258                alive,
259                _portal: session,
260            },
261            audio,
262        ))
263    }
264}

266/// Per-stream negotiation state shared between callbacks and the loop body
267/// (all on the bridge thread).
268struct VideoState {
269    /// Negotiated (format, width, height); `None` until the first Format param.
270    info: Option<(spa::param::video::VideoFormat, u32, u32)>,
271    /// The encode target derived from the FIRST negotiation (frames are always
272    /// delivered at this geometry; a renegotiation only signals resize).
273    encode_target: Option<Dimensions>,
274    /// Stream reached an unrecoverable state.
275    dead: bool,
276}
277
278/// Bridge-thread body: connect the portal fd, negotiate BGRx/BGRA, report the
279/// encode target, then pump frames → I420 → scaled → ring until stopped or the
280/// cast ends. Every PipeWire object lives and dies on this thread.
281#[allow(clippy::too_many_arguments)]
282fn bridge_loop(
283    handoff: portal::CastHandoff,
284    cap: Dimensions,
285    clock: hound_media::CaptureClock,
286    ring: Arc<BoundedRing<RawFrame>>,
287    stop: Arc<AtomicBool>,
288    alive: Arc<AtomicBool>,
289    content_size: Arc<AtomicU64>,
290    ready_tx: std::sync::mpsc::Sender<Result<Dimensions, CaptureError>>,
291) {
292    super::pw_init_once();
293
294    let fail = |ready_tx: &std::sync::mpsc::Sender<Result<Dimensions, CaptureError>>,
295                alive: &AtomicBool,
296                msg: String| {
297        alive.store(false, Ordering::Relaxed);
298        let _ = ready_tx.send(Err(CaptureError::Device(msg)));
299    };
300
301    let mainloop = match pw::main_loop::MainLoopRc::new(None) {
302        Ok(l) => l,
303        Err(e) => return fail(&ready_tx, &alive, format!("pw main loop: {e}")),
304    };
305    let context = match pw::context::ContextRc::new(&mainloop, None) {
306        Ok(c) => c,
307        Err(e) => return fail(&ready_tx, &alive, format!("pw context: {e}")),
308    };
309    let core = match context.connect_fd_rc(handoff.fd, None) {
310        Ok(c) => c,
311        Err(e) => return fail(&ready_tx, &alive, format!("pw connect (portal fd): {e}")),
312    };
313
314    let state = Rc::new(RefCell::new(VideoState {
315        info: None,
316        encode_target: None,
317        dead: false,
318    }));
319
320    let props = pw::properties::properties! {
321        *pw::keys::MEDIA_TYPE => "Video",
322        *pw::keys::MEDIA_CATEGORY => "Capture",
323        *pw::keys::MEDIA_ROLE => "Screen",
324        *pw::keys::NODE_NAME => "hound-screen-capture",
325    };
326    let stream = match pw::stream::StreamRc::new(core.clone(), "hound-screen-capture", props) {
327        Ok(s) => s,
328        Err(e) => return fail(&ready_tx, &alive, format!("pw stream: {e}")),
329    };
330
331    let _listener = match stream
332        .add_local_listener::<()>()
333        .state_changed({
334            let state = Rc::clone(&state);
335            move |_, _, _old, new| match new {
336                pw::stream::StreamState::Error(e) => {
337                    tracing::warn!(target: "portal_screen", "video stream error: {e}");
338                    state.borrow_mut().dead = true;
339                }
340                // The cast node was destroyed (share stopped / window closed):
341                // the stream falls back to Unconnected after having run.
342                pw::stream::StreamState::Unconnected => {
343                    state.borrow_mut().dead = true;
344                }
345                _ => {}
346            }
347        })
348        .param_changed({
349            let state = Rc::clone(&state);
350            let content_size = Arc::clone(&content_size);
351            move |_, _, id, param| {
352                let Some(param) = param else { return };
353                if id != spa::param::ParamType::Format.as_raw() {
354                    return;
355                }
356                let Ok((mt, mst)) = spa::param::format_utils::parse_format(param) else {
357                    return;
358                };
359                if mt != spa::param::format::MediaType::Video
360                    || mst != spa::param::format::MediaSubtype::Raw
361                {
362                    return;
363                }
364                let mut info = spa::param::video::VideoInfoRaw::default();
365                if info.parse(param).is_err() {
366                    return;
367                }
368                let (w, h) = (info.size().width, info.size().height);
369                if w == 0 || h == 0 {
370                    return;
371                }
372                // Even-floor so the resize comparison is stable against the
373                // 1px jitter an odd size would force downstream.
374                content_size.store(
375                    (u64::from(w & !1) << 32) | u64::from(h & !1),
376                    Ordering::Relaxed,
377                );
378                state.borrow_mut().info = Some((info.format(), w, h));
379                tracing::debug!(
380                    target: "portal_screen",
381                    format = ?info.format(), w, h, "cast video format negotiated"
382                );
383            }
384        })
385        .process({
386            let state = Rc::clone(&state);
387            let ring = Arc::clone(&ring);
388            move |stream, _| {
389                let Some(mut buffer) = stream.dequeue_buffer() else {
390                    return;
391                };
392                let s = state.borrow();
393                let (Some((format, w, h)), Some(target)) = (s.info, s.encode_target) else {
394                    return; // pre-negotiation buffer; nothing to interpret
395                };
396                drop(s);
397                let datas = buffer.datas_mut();
398                let Some(d) = datas.first_mut() else { return };
399                let chunk_offset = d.chunk().offset() as usize;
400                let chunk_size = d.chunk().size() as usize;
401                let stride = d.chunk().stride() as usize;
402                let Some(bytes) = d.data() else {
403                    // A dmabuf-backed buffer would land here (no mapped data);
404                    // the format offer excludes modifiers so this is abnormal.
405                    let n = ring.record_bad_frame();
406                    if n <= 3 || n.is_power_of_two() {
407                        tracing::warn!(
408                            target: "portal_screen",
409                            "cast buffer had no mapped data (dmabuf?); dropped (bad={n})"
410                        );
411                    }
412                    return;
413                };
414                let end = (chunk_offset + chunk_size).min(bytes.len());
415                if chunk_offset >= end {
416                    return;
417                }
418                if let Some(frame) =
419                    convert_frame(&bytes[chunk_offset..end], format, w, h, stride, target)
420                {
421                    ring.push(RawFrame {
422                        dims: target,
423                        format: PixelFormat::I420,
424                        timestamp_us: TimestampUs::from_micros(clock.now()),
425                        captured_at: Instant::now(),
426                        data: frame,
427                        surface: None,
428                    });
429                } else {
430                    let n = ring.record_bad_frame();
431                    if n <= 3 || n.is_power_of_two() {
432                        tracing::warn!(
433                            target: "portal_screen",
434                            "cast frame conversion failed ({w}x{h}, stride {stride}); dropped (bad={n})"
435                        );
436                    }
437                }
438            }
439        })
440        .register()
441    {
442        Ok(l) => l,
443        Err(e) => return fail(&ready_tx, &alive, format!("pw stream listener: {e}")),
444    };
445
446    let pod_bytes = video_format_pod(handoff.size);
447    let Some(pod) = spa::pod::Pod::from_bytes(&pod_bytes) else {
448        return fail(&ready_tx, &alive, "video format pod build failed".into());
449    };
450    if let Err(e) = stream.connect(
451        spa::utils::Direction::Input,
452        Some(handoff.node_id),
453        pw::stream::StreamFlags::AUTOCONNECT | pw::stream::StreamFlags::MAP_BUFFERS,
454        &mut [pod],
455    ) {
456        return fail(&ready_tx, &alive, format!("pw stream connect: {e}"));
457    }
458
459    // Pump until the format lands, report the encode target, then keep pumping
460    // frames until stopped or the cast dies.
461    let mut reported = false;
462    while !stop.load(Ordering::Relaxed) {
463        mainloop
464            .loop_()
465            .iterate(pw::loop_::Timeout::Finite(LOOP_TICK));
466        let mut s = state.borrow_mut();
467        if s.dead {
468            break;
469        }
470        if !reported && let Some((_, w, h)) = s.info {
471            let (tw, th) = screen_filter::fit_even(w.max(2), h.max(2), cap.width(), cap.height());
472            let target = Dimensions::even_clamped(tw, th);
473            s.encode_target = Some(target);
474            reported = true;
475            if ready_tx.send(Ok(target)).is_err() {
476                break; // opener gave up
477            }
478        }
479    }
480
481    alive.store(false, Ordering::Relaxed);
482    let _ = stream.disconnect();
483    ring.stop();
484}

486/// One negotiated cast frame → tightly-packed I420 at the encode target.
487/// BGRx/BGRA take the alpha-ignoring converter; RGBx/RGBA go through the same
488/// converter after a channel swap. Stride-aware; `None` on a short buffer.
489/// Pure — unit-tested below.
490fn convert_frame(
491    bytes: &[u8],
492    format: spa::param::video::VideoFormat,
493    w: u32,
494    h: u32,
495    stride: usize,
496    target: Dimensions,
497) -> Option<Vec<u8>> {
498    use spa::param::video::VideoFormat;
499
500    let (w, h) = ((w & !1) as usize, (h & !1) as usize);
501    if w == 0 || h == 0 {
502        return None;
503    }
504    let stride = if stride == 0 { w * 4 } else { stride };
505
506    let i420 = match format {
507        VideoFormat::BGRx | VideoFormat::BGRA => convert::bgra_to_i420(bytes, w, h, stride)?,
508        VideoFormat::RGBx | VideoFormat::RGBA => {
509            // Swap R↔B per row into a scratch BGRA image, then share the
510            // converter (screen content; not a per-pixel hot path worth a
511            // second SIMD-shaped converter).
512            let mut bgra = vec![0u8; w * h * 4];
513            for row in 0..h {
514                let src = bytes.get(row * stride..row * stride + w * 4)?;
515                let dst = &mut bgra[row * w * 4..(row + 1) * w * 4];
516                for (s, d) in src.chunks_exact(4).zip(dst.chunks_exact_mut(4)) {
517                    d[0] = s[2];
518                    d[1] = s[1];
519                    d[2] = s[0];
520                    d[3] = s[3];
521                }
522            }
523            convert::bgra_to_i420(&bgra, w, h, w * 4)?
524        }
525        _ => return None,
526    };
527
528    if (w as u32, h as u32) == (target.width(), target.height()) {
529        Some(i420)
530    } else {
531        convert::scale_i420(
532            &i420,
533            w,
534            h,
535            target.width() as usize,
536            target.height() as usize,
537        )
538    }
539}

541/// The stream's format offer: raw video, BGRx or BGRA (every compositor's
542/// shared-memory screencast format), any size (default = the portal-reported
543/// source size), any framerate. No dmabuf modifier property is offered, which
544/// steers negotiation to shared memory — mappable buffers the CPU convert can
545/// read. Pure (a serialized `EnumFormat` pod), unit-tested below.
546fn video_format_pod(source_size: Option<(u32, u32)>) -> Vec<u8> {
547    let (dw, dh) = source_size.unwrap_or((1920, 1080));
548    let obj = spa::pod::object!(
549        spa::utils::SpaTypes::ObjectParamFormat,
550        spa::param::ParamType::EnumFormat,
551        spa::pod::property!(
552            spa::param::format::FormatProperties::MediaType,
553            Id,
554            spa::param::format::MediaType::Video
555        ),
556        spa::pod::property!(
557            spa::param::format::FormatProperties::MediaSubtype,
558            Id,
559            spa::param::format::MediaSubtype::Raw
560        ),
561        spa::pod::property!(
562            spa::param::format::FormatProperties::VideoFormat,
563            Choice,
564            Enum,
565            Id,
566            spa::param::video::VideoFormat::BGRx,
567            spa::param::video::VideoFormat::BGRx,
568            spa::param::video::VideoFormat::BGRA,
569            spa::param::video::VideoFormat::RGBx,
570            spa::param::video::VideoFormat::RGBA
571        ),
572        spa::pod::property!(
573            spa::param::format::FormatProperties::VideoSize,
574            Choice,
575            Range,
576            Rectangle,
577            spa::utils::Rectangle {
578                width: dw.max(2),
579                height: dh.max(2)
580            },
581            spa::utils::Rectangle {
582                width: 1,
583                height: 1
584            },
585            spa::utils::Rectangle {
586                width: 16384,
587                height: 16384
588            }
589        ),
590        spa::pod::property!(
591            spa::param::format::FormatProperties::VideoFramerate,
592            Choice,
593            Range,
594            Fraction,
595            spa::utils::Fraction { num: 30, denom: 1 },
596            spa::utils::Fraction { num: 0, denom: 1 },
597            spa::utils::Fraction { num: 360, denom: 1 }
598        ),
599    );
600    spa::pod::serialize::PodSerializer::serialize(
601        std::io::Cursor::new(Vec::new()),
602        &spa::pod::Value::Object(obj),
603    )
604    .map(|(c, _)| c.into_inner())
605    .unwrap_or_default()
606}

610impl ScreenCapture for PortalScreenCapturer {
611    fn next_frame(&mut self) -> Result<RawFrame, CaptureError> {
612        self.ring.pop_blocking().ok_or(CaptureError::SourceEnded)
613    }
614
615    fn next_frame_timeout(&mut self, timeout: Duration) -> Result<Option<RawFrame>, CaptureError> {
616        match self.ring.pop_timeout(timeout) {
617            PopWait::Frame(f) => Ok(Some(f)),
618            PopWait::Idle => Ok(None),
619            PopWait::Ended => Err(CaptureError::SourceEnded),
620        }
621    }
622
623    fn source_alive(&self) -> bool {
624        self.alive.load(Ordering::Relaxed)
625    }
626
627    fn content_size(&self) -> Option<(u32, u32)> {
628        let packed = self.content_size.load(Ordering::Relaxed);
629        (packed != 0).then_some(((packed >> 32) as u32, packed as u32))
630    }
631
632    fn negotiated(&self) -> (Dimensions, PixelFormat) {
633        (self.encode_target, PixelFormat::I420)
634    }
635}
636
637impl Drop for PortalScreenCapturer {
638    fn drop(&mut self) {
639        // Stop the bridge (tears down the stream on its owning thread), then
640        // the portal session guard closes the cast itself.
641        self.stop.store(true, Ordering::Relaxed);
642        self.ring.stop();
643        if let Some(h) = self.bridge.take() {
644            let _ = h.join();
645        }
646    }
647}

720    /// THE live end-to-end proof: portal handshake → PipeWire fd → format
721    /// negotiation → shared-memory frames → I420 at the encode target, plus
722    /// the display-share system-audio leg. `#[ignore]`: needs a GPU-backed
723    /// Wayland desktop with a real ScreenCast portal backend (Mutter/KWin, or
724    /// xdg-desktop-portal-wlr on a compositor with a DRM renderer) and either a
725    /// human answering the picker or a portal configured with
726    /// `chooser_type=none`. A headless WSL2 sway session cannot satisfy it: its
727    /// wlroots headless backend has no DRM renderer, so the portal never
728    /// completes a screencast (the D-Bus error mapping — cancelled →
729    /// `PermissionDenied`, absent interface → `Unsupported` — is what's
730    /// exercised headlessly instead).
731    #[test]
732    #[ignore = "needs a GPU-backed Wayland desktop with a real ScreenCast portal"]
733    fn portal_cast_delivers_frames() {
734        use crate::types::Fps;
735
736        let sources = enumerate_sources().expect("portal reachable");
737        let screen = sources
738            .iter()
739            .find(|s| matches!(s.source, ScreenSource::Display(_)))
740            .expect("a castable screen source")
741            .source
742            .clone();
743
744        let clock = hound_media::CaptureClock::start();
745        let config = ScreenCaptureConfig {
746            dimensions: Dimensions::even_clamped(1280, 720),
747            fps: Fps::new(30),
748            capture_audio: true,
749        };
750        let (mut share, audio) =
751            PortalScreenCapturer::open(&screen, config, clock).expect("cast opens");
752        assert!(
753            audio.is_some(),
754            "a display share on a PipeWire host must carry the system-audio leg"
755        );
756        let (dims, fmt) = share.negotiated();
757        assert_eq!(fmt, PixelFormat::I420);
758
759        let deadline = Instant::now() + Duration::from_secs(10);
760        let frame = loop {
761            assert!(Instant::now() < deadline, "no frame within 10 s");
762            match share.next_frame_timeout(Duration::from_millis(500)) {
763                Ok(Some(f)) => break f,
764                Ok(None) => continue,
765                Err(e) => panic!("cast ended early: {e}"),
766            }
767        };
768        assert_eq!(frame.dims, dims, "frames arrive at the negotiated target");
769        assert_eq!(
770            frame.data.len(),
771            dims.width() as usize * dims.height() as usize * 3 / 2,
772            "tightly-packed I420"
773        );
774        assert!(share.source_alive(), "cast is live while frames flow");
775        assert!(
776            share.content_size().is_some(),
777            "the negotiated source size is the resize signal"
778        );
779    }
