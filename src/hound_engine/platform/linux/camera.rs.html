1//! # V4L2 camera capturer (Linux)
2//!
3//! The camera [`VideoCapture`] for Linux, built on **V4L2** via the [`v4l`]
4//! crate (safe Rust over the kernel ioctl ABI). It mirrors the AVFoundation
5//! capturer's contract exactly so the engine facade treats the two identically:
6//! construct-is-start, `next_frame` blocks for the next frame, drop stops the
7//! stream and releases the device.
8//!
9//! ## Pipeline
10//!
11//! ```text
12//! /dev/videoN ─► Device::with_path ─► enum_formats ─► rank (NV12>YUYV>MJPEG)
13//!        │                                                   │
14//!        │            set_format(best @ native size) ◄───────┘
15//!        │            set_params(1/fps)
16//!        ▼
17//!   MmapStream::with_buffers (REQBUFS/QBUF) ── kernel mmap ring
18//!        │
19//!   next_frame(): DQBUF ─► convert (YUYV/NV12 → tightly-packed I420)
20//!                          ─► scale native → encode target (scale_i420)
21//!                          ─► stamp on CaptureClock ─► RawFrame ── QBUF (recycle)
22//! ```
23//!
24//! Unlike AVFoundation (whose delegate fires on an Apple-owned dispatch queue,
25//! forcing a bounded ring + panic containment across an FFI callback boundary),
26//! V4L2's `MmapStream::next()` is a **blocking pull on our own thread** — there
27//! is no foreign callback thread, so `next_frame` simply dequeues, converts, and
28//! requeues inline. The kernel's mmap buffer ring (REQBUFS count) is the
29//! backpressure: if we fall behind, the kernel overwrites the oldest buffer
30//! (the V4L2 analogue of the AVFoundation drop-oldest policy).
31//!
32//! ## Format negotiation (prefer raw, avoid a JPEG dep)
33//! We rank the device's offered formats with the shared, pure
34//! [`select_capture_format`] (NV12 > YUYV > MJPEG). NV12 is a plane
35//! de-interleave to I420; YUYV is a cheap repack + chroma average — both need no
36//! extra dependency. A camera that offers **only MJPEG** would need a JPEG
37//! decoder we deliberately do not pull in, so it surfaces a typed
38//! [`CaptureError::UnsupportedFormat`] naming MJPEG as the follow-up rather than
39//! silently pulling a dep. (Most UVC webcams offer YUYV alongside MJPEG, so this
40//! is rare.)
41//!
42//! ## Typed errors (never a silent zero stream)
43//! * `EACCES` opening the device ⇒ [`CaptureError::PermissionDenied`] (the user
44//!   is not in the `video` group / lacks device perms) — distinct from absence.
45//! * No `/dev/video*` / a vanished named device with no default ⇒
46//!   [`CaptureError::NoDevice`].
47//! * No convertible format ⇒ [`CaptureError::UnsupportedFormat`].
48//! * Any other ioctl failure ⇒ [`CaptureError::Device`] with the OS error text.
49//!
50//! ## Threading
51//! Like the other capturers it is `!Send` in our contract: construct, drive
52//! (`next_frame`), and drop on the one dedicated capture thread. Drop stops the
53//! stream (the `v4l` `Stream`'s own Drop issues STREAMOFF and unmaps the
54//! buffers — RAII), so there is no explicit teardown to forget.
55

77/// How many mmap buffers to request from the kernel. A few frames of slack so a
78/// brief consumer hiccup does not stall capture, but small (video frames are
79/// large): a deep queue here is just latency. Matches the AVFoundation ring
80/// depth (4).
81const MMAP_BUFFER_COUNT: u32 = 4;
82
83/// V4L2 camera backend (Linux). Holds the open device + the running mmap stream;
84/// pulls a frame per [`VideoCapture::next_frame`], converts it to I420, and
85/// scales it to the encode target. Not `Send` (the V4L2 stream/device are tied
86/// to this capture thread). Open is construct ([`Self::open`]); dropping it
87/// stops the stream (STREAMOFF + unmap, via the `v4l` `Stream`'s Drop).
88pub(crate) struct V4l2Camera {
89    /// The running mmap capture stream. Held alongside the device; its Drop does
90    /// the STREAMOFF + unmap.
91    stream: MmapStream<'static>,
92    /// The device, kept alive for the stream's lifetime. `MmapStream` borrows the
93    /// device only at construction (it dups the fd internally), so holding the
94    /// device here is for the lifetime/ownership tie, not an active borrow.
95    _device: Device,
96    /// The format actually negotiated (post `set_format`) — which raw layout the
97    /// dequeued buffers carry, and at what native geometry/stride.
98    capture_format: CaptureFormat,
99    /// Native capture geometry the device negotiated (what the mmap buffers
100    /// carry, before the scale to the encode target).
101    capture_native: Dimensions,
102    /// The bytes-per-row the driver reported for the (Y/luma) plane. Used by the
103    /// NV12 de-interleave to drop row padding; YUYV uses width*2 directly.
104    native_stride: usize,
105    /// The ENCODE-target geometry the caller requested. Every emitted frame is
106    /// scaled to this so the encoder (built from this size) always agrees.
107    encode_target: Dimensions,
108    /// Actual negotiated fps (from the stream params), falling back to the
109    /// request when the driver doesn't report a usable interval.
110    neg_fps: Fps,
111    /// Stamps each frame on the session timeline (A/V sync), shared with audio.
112    clock: hound_media::CaptureClock,
113}

115impl V4l2Camera {
116    /// The camera's actual native resolution — what V4L2 negotiated, before the
117    /// scale-to-encode-target in `next_frame`. Diagnostic accessor kept for API
118    /// parity with the sibling capturers (apple/windows expose the same trio);
119    /// only [`Self::is_scaling`] has a lib caller today.
120    #[allow(dead_code)]
121    pub(crate) fn capture_native(&self) -> Dimensions {
122        self.capture_native
123    }
124
125    /// The fps the driver actually negotiated (from the stream params), not the
126    /// request. Diagnostic accessor — see [`Self::capture_native`].
127    #[allow(dead_code)]
128    pub(crate) fn negotiated_fps(&self) -> Fps {
129        self.neg_fps
130    }
131
132    /// Whether each frame is resized between the camera's native resolution and
133    /// the encode target (so encode timing honestly includes the scale).
134    pub(crate) fn is_scaling(&self) -> bool {
135        self.capture_native != self.encode_target
136    }
137
138    /// Open the chosen camera and begin capture at as close to `target`@`fps` as
139    /// the device allows. The actual geometry is read back and frames are scaled
140    /// to `target`; frames are stamped on `clock`. Open is construct: the
141    /// returned value is already delivering; dropping it stops the stream.
142    pub(crate) fn open(
143        choice: &DeviceChoice<CameraId>,
144        target: Dimensions,
145        fps: Fps,
146        clock: hound_media::CaptureClock,
147    ) -> Result<Self, CaptureError> {
148        // 1) Resolve the device path. A named device that has vanished falls back
149        //    to the default (logged) — the mid-call unplug / hot-swap path, same
150        //    as the AVFoundation capturer.
151        let path = match choice {
152            DeviceChoice::Named(id) => {
153                let p = id.as_str().to_string();
154                if Path::new(&p).exists() {
155                    p
156                } else {
157                    tracing::warn!(
158                        target: "v4l2_capture",
159                        requested = %p,
160                        "requested camera path not found — falling back to default camera"
161                    );
162                    default_device_path()
163                        .ok_or_else(|| CaptureError::NoDevice("no /dev/video* device".into()))?
164                }
165            }
166            DeviceChoice::Default => default_device_path()
167                .ok_or_else(|| CaptureError::NoDevice("no /dev/video* device".into()))?,
168        };
169
170        // 2) Open the device, mapping EACCES to a distinct PermissionDenied (the
171        //    user is not in the `video` group), absence to NoDevice.
172        let device = Device::with_path(&path).map_err(|e| map_open_error(&path, e))?;
173
174        // 3) Negotiate a format: enumerate what the device offers, rank with the
175        //    shared pure ranker (NV12 > YUYV > MJPEG), refuse if nothing
176        //    convertible (or only MJPEG — the JPEG-decode follow-up).
177        let offered: Vec<[u8; 4]> = device
178            .enum_formats()
179            .map_err(|e| CaptureError::Device(format!("VIDIOC_ENUM_FMT failed: {e}")))?
180            .into_iter()
181            .map(|d| d.fourcc.repr)
182            .collect();
183        let chosen = select_capture_format(&offered).ok_or_else(|| {
184            CaptureError::UnsupportedFormat(format!(
185                "camera offers no convertible format (NV12/YUYV); offered {:?}",
186                offered
187                    .iter()
188                    .map(|r| String::from_utf8_lossy(r).into_owned())
189                    .collect::<Vec<_>>()
190            ))
191        })?;
192        if chosen.needs_decoder() {
193            // Only MJPEG on offer — we don't carry a JPEG decoder (deliberate, to
194            // avoid the dep). Surface it as the documented follow-up rather than
195            // pulling a decoder in.
196            return Err(CaptureError::UnsupportedFormat(
197                "camera offers only MJPEG; raw NV12/YUYV not available. MJPEG decode is a \
198                 documented follow-up (no JPEG dependency is pulled in)"
199                    .into(),
200            ));
201        }
202
203        // 4) Request the format at the requested geometry. The driver returns the
204        //    geometry/stride it actually set (cameras rarely honour the exact
205        //    ask), which we record and later scale from.
206        let requested = Format::new(
207            target.width(),
208            target.height(),
209            FourCC::new(&chosen.fourcc()),
210        );
211        let actual = device
212            .set_format(&requested)
213            .map_err(|e| CaptureError::Device(format!("VIDIOC_S_FMT failed: {e}")))?;
214
215        // The driver may have substituted a different FourCC than we asked for
216        // (e.g. refused NV12, kept YUYV); re-derive which converter to use from
217        // what it actually set, so the pixel handling always matches the buffers.
218        let capture_format = CaptureFormat::from_fourcc(actual.fourcc.repr).ok_or_else(|| {
219            CaptureError::UnsupportedFormat(format!(
220                "driver set an unconvertible format {}",
221                actual.fourcc
222            ))
223        })?;
224        let capture_native = Dimensions::try_new(actual.width, actual.height).ok_or_else(|| {
225            CaptureError::UnsupportedFormat(format!(
226                "driver set an odd/zero geometry {}x{}",
227                actual.width, actual.height
228            ))
229        })?;
230        let native_stride = actual.stride.max(actual.width) as usize;
231
232        // 5) Request the frame rate (best-effort; many cams ignore it). Read back
233        //    what the driver actually applied.
234        let neg_fps = match device.set_params(&v4l::video::capture::Parameters::with_fps(fps.get()))
235        {
236            Ok(p) => fps_from_interval(p.interval.numerator, p.interval.denominator)
237                .and_then(Fps::try_new)
238                .unwrap_or(fps),
239            Err(_) => fps,
240        };
241
242        // 6) Start streaming: REQBUFS + QBUF the mmap ring (STREAMON happens on
243        //    the first `next()`). The stream clones the device's refcounted handle
244        //    (it does not borrow the device value), so we then hold both for the
245        //    capturer's life; see [`build_stream`] for why no `unsafe` is needed.
246        let stream = build_stream(&device)?;
247
248        let capturer = Self {
249            stream,
250            _device: device,
251            capture_format,
252            capture_native,
253            native_stride,
254            encode_target: target,
255            neg_fps,
256            clock,
257        };
258
259        tracing::info!(
260            target: "v4l2_capture",
261            path = %path,
262            requested = format!("{target}@{}", fps.get()),
263            capture_format = ?capturer.capture_format,
264            capture_native = %capturer.capture_native,
265            native_stride = capturer.native_stride,
266            encode_target = %capturer.encode_target,
267            negotiated_fps = capturer.neg_fps.get(),
268            scaling = capturer.is_scaling(),
269            "V4L2 camera started (→ I420); reporting actual negotiated format, \
270             scaling native → encode target on the capture thread"
271        );
272
273        Ok(capturer)
274    }
275}

277/// Convert one raw V4L2 buffer (in the negotiated `format`) to a tightly-packed
278/// I420 buffer at the `native` geometry. A free function over snapshotted config
279/// (not a `&self` method) so it does not borrow the capturer while the mmap
280/// buffer slice from `stream.next()` is still alive (that slice mutably borrows
281/// `self.stream`, so any concurrent `&self` borrow would not type-check).
282/// `None` (a skippable frame) on a malformed/short buffer rather than a panic.
283fn convert_native(
284    buf: &[u8],
285    format: CaptureFormat,
286    native: Dimensions,
287    native_stride: usize,
288) -> Option<Vec<u8>> {
289    let (w, h) = (native.width() as usize, native.height() as usize);
290    match format {
291        CaptureFormat::Yuyv => convert::yuyv_to_i420(buf, w, h),
292        CaptureFormat::Nv12 => convert::nv12_bytes_to_i420(buf, w, h, native_stride),
293        // MJPEG is rejected at open() — never reached here.
294        CaptureFormat::Mjpeg => None,
295    }
296}

300impl VideoCapture for V4l2Camera {
301    fn negotiated(&self) -> (Dimensions, PixelFormat) {
302        // The encoder is built from the encode target; that is the geometry the
303        // emitted (scaled) frames carry, so report it — not the camera-native
304        // size (`capture_native` exposes that separately).
305        (self.encode_target, PixelFormat::I420)
306    }
307
308    fn next_frame(&mut self) -> Result<RawFrame, CaptureError> {
309        // Snapshot the immutable config into locals BEFORE `stream.next()` — the
310        // dequeued buffer slice mutably borrows `self.stream` for its whole scope,
311        // so we must not touch `&self` again while it is alive.
312        let format = self.capture_format;
313        let native = self.capture_native;
314        let native_stride = self.native_stride;
315        let target = self.encode_target;
316        // Stamp before the (short) conversion so the timestamp reflects capture,
317        // and so we don't read `self.clock` while `buf` borrows `self.stream`.
318        let timestamp_us = TimestampUs::from_micros(self.clock.now());
319        let captured_at = Instant::now();
320
321        // Blocking DQBUF on our own thread (no foreign callback). The kernel hands
322        // back the next filled mmap buffer; `next` re-queues the previous one.
323        let (buf, _meta) = self
324            .stream
325            .next()
326            .map_err(|e| CaptureError::Device(format!("VIDIOC_DQBUF failed: {e}")))?;
327
328        // Convert native raw → tightly-packed I420 (native geometry). This owns
329        // its output, so the `buf` borrow of `self.stream` ends here.
330        let native_i420 = convert_native(buf, format, native, native_stride).ok_or_else(|| {
331            CaptureError::Device(format!(
332                "malformed {} frame at {native} (stride {native_stride}): conversion produced no I420",
333                format_name(format),
334            ))
335        })?;
336
337        // Scale native → encode target on THIS (capture) thread, mirroring the
338        // AVFoundation path. No-op fast path when sizes already match.
339        if native == target {
340            return Ok(RawFrame {
341                dims: target,
342                format: PixelFormat::I420,
343                timestamp_us,
344                captured_at,
345                data: native_i420,
346                surface: None,
347            });
348        }
349        match convert::scale_i420(
350            &native_i420,
351            native.width() as usize,
352            native.height() as usize,
353            target.width() as usize,
354            target.height() as usize,
355        ) {
356            Some(scaled) => Ok(RawFrame {
357                dims: target,
358                format: PixelFormat::I420,
359                timestamp_us,
360                captured_at,
361                data: scaled,
362                surface: None,
363            }),
364            None => Err(CaptureError::Device(format!(
365                "frame scale {native} → {target} failed (bad plane sizes)"
366            ))),
367        }
368    }
369}

371/// Build the mmap capture stream for `device`.
372///
373/// `MmapStream<'a>`'s lifetime parameter is **not** a borrow of the `Device`:
374/// `with_buffers(dev: &Device, …)` clones the device's refcounted `Arc<Handle>`
375/// (it does not retain the `&Device`), and the `'a` tags the stream's own mmap'd
376/// buffer memory (kernel-owned, freed by the stream's Drop), which is unrelated
377/// to the device value's location. The `'a` is therefore caller-chosen and free
378/// to be `'static` — no `unsafe`, no lifetime transmute. We still hold the
379/// `Device` in [`V4l2Camera`] so its fd outlives the stream (drop order: `stream`
380/// before `_device`), which is what keeps the cloned handle's fd valid.
381fn build_stream(device: &Device) -> Result<MmapStream<'static>, CaptureError> {
382    MmapStream::with_buffers(device, BufferType::VideoCapture, MMAP_BUFFER_COUNT)
383        .map_err(|e| CaptureError::Device(format!("mmap stream setup failed (REQBUFS): {e}")))
384}

386/// Map a device-open `io::Error` to a typed [`CaptureError`], via the shared
387/// (host-independent, unit-tested) [`classify_open_errno`]: EACCES ⇒
388/// PermissionDenied (not in the `video` group), ENOENT/ENODEV ⇒ NoDevice,
389/// anything else ⇒ Device with the OS text.
390fn map_open_error(path: &str, e: io::Error) -> CaptureError {
391    match classify_open_errno(e.raw_os_error()) {
392        OpenErrorKind::PermissionDenied => CaptureError::PermissionDenied {
393            kind: DeviceKind::Camera,
394        },
395        OpenErrorKind::NoDevice => CaptureError::NoDevice(format!("{path}: {e}")),
396        OpenErrorKind::Other => CaptureError::Device(format!("open {path}: {e}")),
397    }
398}

400/// The default camera device path: the lowest-numbered `/dev/video*` that is a
401/// capture device. Uses the `v4l` context enumeration so it matches what the
402/// kernel reports, falling back to a `/dev/video*` scan if that yields nothing.
403fn default_device_path() -> Option<String> {
404    let mut nodes: Vec<(usize, String)> = v4l::context::enum_devices()
405        .into_iter()
406        .map(|n| (n.index(), n.path().to_string_lossy().into_owned()))
407        .collect();
408    nodes.sort_by_key(|(idx, _)| *idx);
409    nodes.into_iter().map(|(_, p)| p).next()
410}

412/// Human-readable name for a capture format, for error/log text.
413fn format_name(fmt: CaptureFormat) -> &'static str {
414    match fmt {
415        CaptureFormat::Nv12 => "NV12",
416        CaptureFormat::Yuyv => "YUYV",
417        CaptureFormat::Mjpeg => "MJPEG",
418    }
419}

421/// Enumerate the connected V4L2 cameras (`/dev/video*`). Best-effort: returns an
422/// empty list if nothing is connected; never panics, never opens a stream (so it
423/// is safe from a blocking pool). The typed [`CameraId`] is the device PATH
424/// (stable for a fixed USB port; the human name comes from the driver). Only
425/// devices that report a video-capture capability are listed (a `/dev/videoN`
426/// can be a metadata or output node).
427pub(crate) fn enumerate_cameras() -> Vec<DeviceDescriptor<CameraId>> {
428    let default = default_device_path();
429    let mut out = Vec::new();
430    let mut nodes: Vec<_> = v4l::context::enum_devices();
431    nodes.sort_by_key(|n| n.index());
432    for node in nodes {
433        let path = node.path().to_string_lossy().into_owned();
434        // Only list nodes that can actually capture video (filter out output /
435        // metadata nodes). Opening read-only to query caps is cheap and does not
436        // start a stream.
437        let Ok(device) = Device::with_path(&path) else {
438            continue;
439        };
440        // One query: filter out output/metadata nodes (must report VIDEO_CAPTURE)
441        // and take the driver's card name in the same call. Opening read-only to
442        // query caps is cheap and does not start a stream.
443        let Ok(caps) = device.query_caps() else {
444            continue;
445        };
446        if !caps
447            .capabilities
448            .contains(v4l::capability::Flags::VIDEO_CAPTURE)
449        {
450            continue;
451        }
452        // Prefer the driver-reported card name; fall back to the node name/path.
453        let name = Some(caps.card)
454            .filter(|c| !c.is_empty())
455            .or_else(|| node.name())
456            .unwrap_or_else(|| path.clone());
457        let is_default = default.as_deref() == Some(path.as_str());
458        out.push(DeviceDescriptor {
459            id: CameraId::new(path),
460            name,
461            is_default,
462        });
463    }
464    out
465}

467// NOTE: the pure, host-independent logic this module relies on — format-rank
468// (`select_capture_format`), fps-from-interval, open-errno classification, and
469// the YUYV/NV12→I420 conversions — is unit-tested in `platform::generic`
470// (`camera_format` + `convert`), so it runs on every host INCLUDING the macOS
471// dev box where this V4L2 module cannot compile. The tests below cover only the
472// thin Linux-module glue that maps those into the engine's typed errors; they
473// run on a Linux host / CI.
474#[cfg(test)]
475mod tests {
476    use super::*;
477
478    /// `map_open_error` threads the shared errno classification into the engine's
479    /// typed errors: EACCES ⇒ PermissionDenied(Camera) (denial is never a silent
480    /// zero stream), ENOENT ⇒ NoDevice, else ⇒ Device.
481    #[test]
482    fn map_open_error_is_typed() {
483        let eacces = map_open_error("/dev/video0", io::Error::from_raw_os_error(13));
484        assert!(matches!(
485            eacces,
486            CaptureError::PermissionDenied {
487                kind: DeviceKind::Camera
488            }
489        ));
490        let enoent = map_open_error("/dev/video9", io::Error::from_raw_os_error(2));
491        assert!(matches!(enoent, CaptureError::NoDevice(_)));
492        let other = map_open_error("/dev/video0", io::Error::from_raw_os_error(5));
493        assert!(matches!(other, CaptureError::Device(_)));
494    }
495
496    /// The format-name helper covers every variant (no `_ =>` fallthrough that
497    /// could silently mislabel a new format).
498    #[test]
499    fn format_name_covers_all() {
500        assert_eq!(format_name(CaptureFormat::Nv12), "NV12");
501        assert_eq!(format_name(CaptureFormat::Yuyv), "YUYV");
502        assert_eq!(format_name(CaptureFormat::Mjpeg), "MJPEG");
503    }
