1//! # cpal microphone capture + speaker playback (CoreAudio on macOS)
2//!
3//! The runnable device I/O for the audio engine. [`CpalCapturer`] opens an
4//! input device and bridges its **real-time callback** to the blocking-pull
5//! [`AudioInput`] trait through a bounded ring buffer; [`CpalPlayback`] opens
6//! an output device and drains a ring buffer the [`AudioOutput`] trait pushes
7//! into. cpal's CoreAudio backend is the macOS/iOS path; the same code is the
8//! Linux (ALSA/Pulse) / Windows (WASAPI) / Android (AAudio) path behind these
9//! traits — which is why this lives in `platform::generic`.
10//!
11//! ## Lifecycle
12//!
13//! Open **is** construct ([`CpalCapturer::open`] / [`CpalPlayback::open`])
14//! and stop is `Drop` — a value that exists is streaming, so a
15//! "capturer not started" runtime error class is unrepresentable.
16//!
17//! ## The real-time contract (the one thing that can go wrong here)
18//!
19//! cpal invokes the data callback on a **CoreAudio real-time thread**. That
20//! callback:
21//! * MUST NOT allocate, lock unboundedly, or block — an xrun is audible.
22//! * Here it only does a bounded `push`/`pop` over an SPSC ring
23//!   ([`rtrb`]-style, but we use a `Mutex<VecDeque>` with a *try* discipline:
24//!   the lock is held for O(burst) memcpy only, never across a syscall, and
25//!   the callback never waits on the consumer). A true lock-free SPSC ring is
26//!   the noted production refinement.
27//!
28//! ## Format negotiation
29//!
30//! We ASK for 48 kHz mono i16. CoreAudio's default device config is often
31//! 48 kHz already, but the channel count and sample FORMAT are
32//! device-dependent (built-in mics frequently expose f32, and some expose
33//! stereo). So:
34//! * We open with the device's **default config's sample format** and convert
35//!   whatever it delivers (f32/i16/u16/…) to i16 in the callback via cpal's
36//!   `Sample`/`FromSample` traits.
37//! * If the device delivers multi-channel, we **downmix to mono** by
38//!   averaging channels in the callback before buffering.
39//! * The device keeps its native RATE; we **resample** to/from the engine's
40//!   48 kHz at the trait boundary (capture `next_frame` / playback `play`,
41//!   via [`super::resample::MonoResampler`]) on worker threads, never in the
42//!   RT callback. AirPods mics (24 kHz HFP) and 44.1 kHz outputs are common;
43//!   mislabelling instead of resampling plays every voice pitch-shifted
44//!   (chipmunk/deep-voice). [`AudioInput::negotiated`] still reports the
45//!   device truth.
46//!
47//! ## Permission / honest-verdict
48//!
49//! On CoreAudio a denied microphone fails at `build_input_stream` / `play`
50//! time. cpal 0.18 maps that to [`cpal::error::ErrorKind::PermissionDenied`]
51//! (and absence to `DeviceNotAvailable`), which we translate to the typed
52//! [`AudioError::PermissionDenied`] / [`AudioError::NoDevice`] — a DISTINCT
53//! error, never a silent stream of zero-RMS frames.
54

69/// Ring-buffer capacity (in mono i16 samples). Sized to ~10 frames of 20 ms
70/// (≈200 ms) so a brief consumer hiccup does not drop data, but the buffer can
71/// never grow unbounded: on overflow the oldest samples are dropped (capture)
72/// or the producer drops the frame (playback).
73const RING_CAPACITY_SAMPLES: usize = SAMPLES_PER_FRAME * 10;
74
75/// Linux only: the shared sound-server PCM — `pipewire`, else `pulse`. These
76/// PCMs **multiplex** (non-exclusive access), so opening one gives the shared
77/// access users expect and follows the server's selected default sink/source.
78///
79/// A raw `hw:` / `default` ALSA PCM is **exclusive**: `snd_pcm_open` returns
80/// `EBUSY` ("device temporarily busy") whenever the sound server — or another
81/// app, or a stale process — already holds the card. That is the "no audio at
82/// all" failure on a Linux box whose `default` does not itself route through the
83/// server. Routing through `pipewire`/`pulse` sidesteps it entirely.
84#[cfg(target_os = "linux")]
85fn shared_server_device(host: &cpal::Host, input: bool) -> Option<cpal::Device> {
86    let devices: Vec<cpal::Device> = if input {
87        host.input_devices().ok()?.collect()
88    } else {
89        host.output_devices().ok()?.collect()
90    };
91    for pref in ["pipewire", "pulse"] {
92        if let Some(dev) = devices.iter().find(|d| d.to_string() == pref) {
93            return Some(dev.clone());
94        }
95    }
96    None
97}

99/// Ordered **input** devices to try opening, best first. We try each in turn so a
100/// busy/exclusive primary still yields audio rather than failing the call:
101/// 1. the user's explicit named choice (if present),
102/// 2. **(Linux)** the shared `pipewire`/`pulse` PCM — non-exclusive access (for a
103///    `Default` choice this is the preferred device; for a named choice the
104///    fallback when the named device is busy/exclusive),
105/// 3. the host default PCM.
106fn input_candidates(host: &cpal::Host, choice: &DeviceChoice<MicId>) -> Vec<cpal::Device> {
107    let mut out: Vec<cpal::Device> = Vec::new();
108    if let DeviceChoice::Named(want) = choice
109        && let Ok(devices) = host.input_devices()
110        && let Some(dev) = devices.into_iter().find(|d| d.to_string() == want.as_str())
111    {
112        out.push(dev);
113    }
114    #[cfg(target_os = "linux")]
115    if let Some(dev) = shared_server_device(host, true) {
116        out.push(dev);
117    }
118    if let Some(dev) = host.default_input_device() {
119        out.push(dev);
120    }
121    out
122}
123
124/// Ordered **output** devices to try opening, best first. Mirror of
125/// [`input_candidates`] (named → Linux shared server PCM → host default).
126fn output_candidates(host: &cpal::Host, choice: &DeviceChoice<SpeakerId>) -> Vec<cpal::Device> {
127    let mut out: Vec<cpal::Device> = Vec::new();
128    if let DeviceChoice::Named(want) = choice
129        && let Ok(devices) = host.output_devices()
130        && let Some(dev) = devices.into_iter().find(|d| d.to_string() == want.as_str())
131    {
132        out.push(dev);
133    }
134    #[cfg(target_os = "linux")]
135    if let Some(dev) = shared_server_device(host, false) {
136        out.push(dev);
137    }
138    if let Some(dev) = host.default_output_device() {
139        out.push(dev);
140    }
141    out
142}

144/// Map a cpal stream-build / control error to our taxonomy, preserving the
145/// permission-vs-absence distinction that the honest-verdict UI depends on.
146/// `kind` names which device direction was being opened (typed permission).
147fn map_cpal_err(context: &str, kind: DeviceKind, e: cpal::Error) -> AudioError {
148    use cpal::ErrorKind;
149    let msg = format!("{context}: {e}");
150    match e.kind() {
151        ErrorKind::PermissionDenied => AudioError::PermissionDenied { kind },
152        ErrorKind::DeviceNotAvailable | ErrorKind::HostUnavailable => AudioError::NoDevice(msg),
153        ErrorKind::UnsupportedConfig
154        | ErrorKind::InvalidInput
155        | ErrorKind::UnsupportedOperation => AudioError::UnsupportedFormat(msg),
156        _ => AudioError::Device(msg),
157    }
158}

161// Shared ring buffer between the RT callback and the worker thread.
162// ---------------------------------------------------------------------------
163
164/// A bounded mono-i16 ring shared between a cpal RT callback (one side) and
165/// the trait's blocking-pull / non-blocking-push API (the other). The mutex is
166/// only ever held for a bounded memcpy, never across a syscall, and the
167/// callback never *waits* on it (it would rather drop than block the audio
168/// thread).
169struct Ring {
170    buf: Mutex<VecDeque<i16>>,
171    /// Signalled by the capture callback when it has pushed samples, so a
172    /// blocked `next_frame` wakes promptly instead of polling.
173    not_empty: Condvar,
174    /// Cleared on stop so a blocked consumer unblocks and returns.
175    running: AtomicBool,
176    /// Whether samples were dropped on overflow (capture side) — a
177    /// diagnostic; an overflowing capture ring means the consumer is too slow.
178    overflowed: AtomicBool,
179}
180
181impl Ring {
182    fn new() -> Arc<Self> {
183        Arc::new(Self {
184            buf: Mutex::new(VecDeque::with_capacity(RING_CAPACITY_SAMPLES)),
185            not_empty: Condvar::new(),
186            running: AtomicBool::new(true),
187            overflowed: AtomicBool::new(false),
188        })
189    }
190
191    /// Producer (capture RT callback): push mono i16 samples, dropping the
192    /// OLDEST on overflow so the buffer is bounded and the callback never
193    /// blocks.
194    fn push_capture(&self, samples: &[i16]) {
195        if let Ok(mut q) = self.buf.lock() {
196            // Bulk-append, then trim the oldest past capacity in one drain (drop-oldest,
197            // never blocks) — a single memcpy instead of a per-sample push_back.
198            q.extend(samples.iter().copied());
199            if q.len() > RING_CAPACITY_SAMPLES {
200                let excess = q.len() - RING_CAPACITY_SAMPLES;
201                q.drain(..excess);
202                self.overflowed.store(true, Ordering::Relaxed);
203            }
204            drop(q);
205            self.not_empty.notify_one();
206        }
207    }
208
209    /// Consumer (worker thread): block until `n` samples are available or the
210    /// stream stops; returns `None` if stopped before enough data.
211    fn pop_blocking(&self, n: usize) -> Option<Vec<i16>> {
212        let mut q = self.buf.lock().ok()?;
213        while q.len() < n {
214            if !self.running.load(Ordering::Relaxed) {
215                return None;
216            }
217            // Wake periodically so a stop that races the notify is still seen.
218            let (g, _timeout) = self
219                .not_empty
220                .wait_timeout(q, std::time::Duration::from_millis(100))
221                .ok()?;
222            q = g;
223            if !self.running.load(Ordering::Relaxed) && q.len() < n {
224                return None;
225            }
226        }
227        Some(q.drain(..n).collect())
228    }
229
230    fn stop(&self) {
231        self.running.store(false, Ordering::Relaxed);
232        self.not_empty.notify_all();
233    }
234}

236// ---------------------------------------------------------------------------
237// Playback ring: wait-free SPSC.
238//
239// The PLAYBACK path has exactly one producer (the mix thread, via `play`) and one
240// consumer (the CoreAudio RT output callback), so it uses a wait-free SPSC ring
241// (`rtrb`) rather than the mutex-guarded `Ring` above. The RT output callback then
242// NEVER takes a lock — a mutex-guarded ring could block it if the mix thread were
243// preempted mid-push, a priority inversion whenever the mix thread is not real-time
244// (RT promotion is best-effort and fails with no audio entitlement). "Audio is
245// sacred": the device thread always makes progress, padding silence on a dry ring
246// instead of waiting. Capture keeps the mutex `Ring` (its consumer blocks on a condvar,
247// which SPSC can't express, and it is not the latency-critical output path).
248// ---------------------------------------------------------------------------
249
250/// Diagnostics shared across the playback ring's three reachers: the mix thread
251/// (owner), the RT output callback (counts underrun), and the stream error callback
252/// (marks death). Plain atomics behind an `Arc`.
253struct PlaybackShared {
254    /// Cumulative output samples the RT callback silence-padded because the ring ran
255    /// dry — the definitive end-of-pipeline AUDIBLE GAP. Steady state is ZERO; a rising
256    /// count IS the stutter, made measurable. Read+reset via [`PlaybackProducer::take_output_underruns`].
257    output_underrun_samples: AtomicU64,
258    /// Set by the OUTPUT stream's error callback (e.g. "default output device changed"
259    /// when AirPods drop to the built-in speaker): the cpal stream is then dead but the
260    /// `CpalPlayback` lives on, so the mix bus polls [`PlaybackProducer::is_output_dead`]
261    /// each tick and reopens the device.
262    output_dead: AtomicBool,
263}
264
265/// The mix-thread (sole producer) half of the playback ring + shared diagnostics.
266struct PlaybackProducer {
267    producer: rtrb::Producer<i16>,
268    capacity: usize,
269    shared: Arc<PlaybackShared>,
270}
271
272/// The RT-output-callback (sole consumer) half + shared diagnostics.
273struct PlaybackConsumer {
274    consumer: rtrb::Consumer<i16>,
275    shared: Arc<PlaybackShared>,
276}
277
278/// Build a wait-free SPSC playback ring of `capacity` mono samples.
279fn playback_ring(capacity: usize) -> (PlaybackProducer, PlaybackConsumer) {
280    let (producer, consumer) = rtrb::RingBuffer::<i16>::new(capacity);
281    let shared = Arc::new(PlaybackShared {
282        output_underrun_samples: AtomicU64::new(0),
283        output_dead: AtomicBool::new(false),
284    });
285    (
286        PlaybackProducer {
287            producer,
288            capacity,
289            shared: Arc::clone(&shared),
290        },
291        PlaybackConsumer { consumer, shared },
292    )
293}
294
295impl PlaybackProducer {
296    /// Mix thread: push mono samples for the RT callback to drain. Wait-free; on a full
297    /// ring (the device stalled and we outran it — rare, the ring is rate-matched and
298    /// cushioned) the newest samples are dropped rather than blocking.
299    fn push_playback(&mut self, samples: &[i16]) {
300        for &s in samples {
301            if self.producer.push(s).is_err() {
302                break; // full: drop the rest (overproduction, self-corrects)
303            }
304        }
305    }
306
307    /// Buffered (unplayed) samples = capacity − free write slots. Coarse (the consumer
308    /// may pop concurrently); the mix bus uses it only for the ms-scale output cushion.
309    fn buffered(&self) -> usize {
310        self.capacity.saturating_sub(self.producer.slots())
311    }
312
313    /// Read and reset the underrun-sample counter (the silence the RT callback inserted
314    /// on a dry ring). Polled by the bus each tick for the live "playout underruns" stat.
315    fn take_output_underruns(&self) -> u64 {
316        self.shared
317            .output_underrun_samples
318            .swap(0, Ordering::Relaxed)
319    }
320
321    /// Whether the output stream has errored (device changed/disconnected) → reopen.
322    fn is_output_dead(&self) -> bool {
323        self.shared.output_dead.load(Ordering::Relaxed)
324    }
325}
326
327impl PlaybackConsumer {
328    /// RT output callback: pop up to `out.len()` mono samples (front-filled), silence-pad
329    /// the remainder on underrun, and count the gap. Wait-free — never locks, never blocks.
330    fn pop_playback(&mut self, out: &mut [i16]) {
331        let mut n = 0;
332        while n < out.len() {
333            match self.consumer.pop() {
334                Ok(s) => {
335                    out[n] = s;
336                    n += 1;
337                }
338                Err(_) => break, // ring dry: the rest is silence
339            }
340        }
341        for slot in &mut out[n..] {
342            *slot = 0;
343        }
344        let padded = out.len() - n;
345        if padded > 0 {
346            self.shared
347                .output_underrun_samples
348                .fetch_add(padded as u64, Ordering::Relaxed);
349        }
350    }
351}

353/// Convert one cpal callback buffer of arbitrary sample type `T` and channel
354/// count to mono i16, appending into `scratch`. Downmix = average of channels.
355fn to_mono_i16<T>(data: &[T], channels: usize, scratch: &mut Vec<i16>)
356where
357    T: Sample,
358    i16: cpal::FromSample<T>,
359{
360    scratch.clear();
361    if channels <= 1 {
362        scratch.extend(data.iter().map(|&s| i16::from_sample(s)));
363        return;
364    }
365    for frame in data.chunks_exact(channels) {
366        // Average in i32 to avoid i16 overflow, then clamp.
367        let sum: i32 = frame.iter().map(|&s| i16::from_sample(s) as i32).sum();
368        scratch.push((sum / channels as i32) as i16);
369    }
370}

376/// Microphone capture via cpal. The cpal `Stream` is `!Send` on CoreAudio, so
377/// it is created and held on the thread that calls [`Self::open`]; the ring it
378/// feeds is `Send`/`Sync` and is what [`AudioInput::next_frame`] reads.
379/// Dropping the capturer stops the stream and releases the device (on the
380/// owning thread — the `!Send` contract).
381pub(crate) struct CpalCapturer {
382    ring: Arc<Ring>,
383    /// Held to keep the device stream alive; dropped (released) with `self`.
384    _stream: cpal::Stream,
385    clock: hound_media::CaptureClock,
386    /// The device truth: what the device negotiated (we DELIVER 48 k mono).
387    negotiated: (SampleRate, ChannelCount),
388    /// The name of the device actually opened, for the hot-swap path to log
389    /// what it landed on. `None` if the device name could not be read.
390    opened_name: Option<String>,
391    /// Device rate → 48 kHz conversion (pass-through when the device is 48 k).
392    /// Runs on the consumer thread in [`AudioInput::next_frame`], NEVER in the
393    /// RT callback.
394    resampler: super::resample::MonoResampler,
395    /// Converted 48 kHz samples awaiting frame assembly (FIFO tail of the
396    /// resampler output that did not fill a whole 20 ms frame yet).
397    pending_48k: Vec<i16>,
398}
399
400impl CpalCapturer {
401    /// Open the best available input device and begin capture, stamping frames on
402    /// `clock`. Tries [`input_candidates`] in order (named → Linux shared server
403    /// PCM → host default), so a busy/exclusive or vanished device falls through
404    /// to the next rather than failing the call — non-exclusive access by default.
405    pub(crate) fn open(
406        choice: &DeviceChoice<MicId>,
407        clock: hound_media::CaptureClock,
408    ) -> Result<Self, AudioError> {
409        let host = cpal::default_host();
410        let candidates = input_candidates(&host, choice);
411        if candidates.is_empty() {
412            return Err(AudioError::NoDevice("no input device".into()));
413        }
414        let mut last_err: Option<AudioError> = None;
415        for device in &candidates {
416            match Self::open_device(device, clock) {
417                Ok(cap) => return Ok(cap),
418                Err(e) => {
419                    tracing::warn!(
420                        target: "audio_endpoint",
421                        device = %device.to_string(),
422                        "input open failed ({e}); trying next candidate"
423                    );
424                    last_err = Some(e);
425                }
426            }
427        }
428        Err(last_err.unwrap_or_else(|| AudioError::NoDevice("no input device".into())))
429    }
430
431    /// Open one specific device at its native format. See [`Self::open`] for the
432    /// candidate-selection wrapper.
433    fn open_device(
434        device: &cpal::Device,
435        clock: hound_media::CaptureClock,
436    ) -> Result<Self, AudioError> {
437        let opened_name = Some(device.to_string());
438
439        // Use the device's default config for the sample FORMAT (built-in mics
440        // are often f32), but request mono and 48 kHz where possible.
441        let default_cfg = device
442            .default_input_config()
443            .map_err(|e| map_cpal_err("query default input config", DeviceKind::Microphone, e))?;
444        let sample_format = default_cfg.sample_format();
445        let dev_channels = default_cfg.channels();
446        let dev_rate = default_cfg.sample_rate();
447
448        // We keep the device's native channel count and rate (downmix to mono
449        // in the RT callback), then RESAMPLE to the engine's 48 kHz on the
450        // consumer thread in `next_frame`. Asking for a hard mono/48k config
451        // that the device does not support would fail the stream build
452        // (AirPods mics are 24 kHz, many defaults 44.1 kHz); mislabelling the
453        // rate instead is the chipmunk/deep-voice bug. The delivered contract
454        // is always 48 k mono.
455        let config = cpal::StreamConfig {
456            channels: dev_channels,
457            sample_rate: dev_rate,
458            buffer_size: cpal::BufferSize::Default,
459        };
460        let negotiated_rate = SampleRate::try_new(dev_rate)
461            .ok_or_else(|| AudioError::UnsupportedFormat("device reports 0 Hz".into()))?;
462        let resampler = super::resample::MonoResampler::new(dev_rate, SAMPLE_RATE.hz())?;
463        if dev_rate != SAMPLE_RATE.hz() {
464            tracing::info!(
465                target: "audio_endpoint",
466                device_rate = dev_rate,
467                "capture device is not 48 kHz — resampling to the engine rate"
468            );
469        }
470
471        let ring = Ring::new();
472        let channels = dev_channels as usize;
473
474        let err_ring = Arc::clone(&ring);
475        let error_cb = move |e: cpal::Error| {
476            tracing::error!(target: "audio_endpoint", "cpal input stream error: {e}");
477            // A device-lost mid-stream unblocks the consumer (returns Err on
478            // the next next_frame via the stopped ring).
479            err_ring.stop();
480        };
481
482        // Build the stream for the device's actual sample format, converting
483        // to mono i16 in the RT callback. Each arm reuses a per-callback
484        // scratch Vec captured by the closure — allocated ONCE at build time
485        // and reused; the only growth is bounded by the burst size.
486        let mut scratch: Vec<i16> = Vec::with_capacity(RING_CAPACITY_SAMPLES);
487        let cb_ring = Arc::clone(&ring);
488        let stream = match sample_format {
489            SampleFormat::F32 => device.build_input_stream(
490                config,
491                move |data: &[f32], _| {
492                    to_mono_i16(data, channels, &mut scratch);
493                    cb_ring.push_capture(&scratch);
494                },
495                error_cb,
496                None,
497            ),
498            SampleFormat::I16 => device.build_input_stream(
499                config,
500                move |data: &[i16], _| {
501                    to_mono_i16(data, channels, &mut scratch);
502                    cb_ring.push_capture(&scratch);
503                },
504                error_cb,
505                None,
506            ),
507            SampleFormat::U16 => device.build_input_stream(
508                config,
509                move |data: &[u16], _| {
510                    to_mono_i16(data, channels, &mut scratch);
511                    cb_ring.push_capture(&scratch);
512                },
513                error_cb,
514                None,
515            ),
516            other => {
517                return Err(AudioError::UnsupportedFormat(format!(
518                    "unsupported input sample format {other:?}"
519                )));
520            }
521        }
522        .map_err(|e| map_cpal_err("build input stream", DeviceKind::Microphone, e))?;
523
524        stream
525            .play()
526            .map_err(|e| map_cpal_err("start input stream", DeviceKind::Microphone, e))?;
527
528        Ok(Self {
529            ring,
530            _stream: stream,
531            clock,
532            negotiated: (negotiated_rate, CHANNELS), // device truth; we DELIVER 48k mono
533            opened_name,
534            resampler,
535            pending_48k: Vec::new(),
536        })
537    }
538
539    /// The name of the input device actually opened. `None` if the device
540    /// name could not be read.
541    pub(crate) fn opened_name(&self) -> Option<&str> {
542        self.opened_name.as_deref()
543    }
544}

548impl AudioInput for CpalCapturer {
549    fn next_frame(&mut self) -> Result<PcmFrame, AudioError> {
550        // Pull device-rate samples and convert until one full 48 kHz frame is
551        // pending. Pass-through devices pull exactly one frame; resampled
552        // devices pull in the resampler's 20 ms input blocks.
553        while self.pending_48k.len() < SAMPLES_PER_FRAME {
554            let pull = if self.resampler.is_passthrough() {
555                SAMPLES_PER_FRAME
556            } else {
557                self.resampler.chunk_in()
558            };
559            let samples = self
560                .ring
561                .pop_blocking(pull)
562                .ok_or_else(|| AudioError::Device("capture stream stopped".into()))?;
563            self.resampler
564                .process_into(&samples, &mut self.pending_48k)?;
565        }
566        let samples: Vec<i16> = self.pending_48k.drain(..SAMPLES_PER_FRAME).collect();
567        // Stamp at assembly time (within one frame interval of the device
568        // callback; a future refinement carries the callback's host time).
569        Ok(PcmFrame::at_engine_rate(
570            samples,
571            TimestampUs::from_micros(self.clock.now()),
572        ))
573    }
574
575    fn negotiated(&self) -> (SampleRate, ChannelCount) {
576        self.negotiated
577    }
578}

580impl Drop for CpalCapturer {
581    /// Wake any blocked consumer, then release the device (the stream drops
582    /// with `self`, on the thread that built it — the `!Send` contract).
583    fn drop(&mut self) {
584        self.ring.stop();
585    }
586}

592/// Speaker playback via cpal. The output callback drains the ring; the
593/// trait's [`AudioOutput::play`] pushes into it (non-blocking, bounded).
594/// Dropping the sink stops the stream and releases the device.
595pub(crate) struct CpalPlayback {
596    /// Producer half of the wait-free playback ring (the RT callback owns the consumer).
597    out: PlaybackProducer,
598    /// Held to keep the device stream alive; dropped (released) with `self`.
599    _stream: cpal::Stream,
600    /// 48 kHz → device-rate conversion (pass-through on 48 k devices). Runs in
601    /// [`AudioOutput::play`] on the caller's thread, NEVER in the RT callback
602    /// (the ring stays device-rate mono).
603    resampler: super::resample::MonoResampler,
604    /// Reused conversion output buffer (no per-frame allocation).
605    scratch: Vec<i16>,
606    opened_name: Option<String>,
607    /// The device's sample rate — to convert the ring's buffered sample count to
608    /// milliseconds for the mix bus's output-cushion controller.
609    dev_rate: u32,
610}
611
612impl CpalPlayback {
613    /// Open the best available output device at the engine format (resampling to
614    /// the device rate as needed). Tries [`output_candidates`] in order (named →
615    /// Linux shared server PCM → host default), so a busy/exclusive or vanished
616    /// device falls through rather than failing — non-exclusive access by default.
617    pub(crate) fn open(choice: &DeviceChoice<SpeakerId>) -> Result<Self, AudioError> {
618        let host = cpal::default_host();
619        let candidates = output_candidates(&host, choice);
620        if candidates.is_empty() {
621            return Err(AudioError::NoDevice("no output device".into()));
622        }
623        let mut last_err: Option<AudioError> = None;
624        for device in &candidates {
625            match Self::open_device(device) {
626                Ok(pb) => return Ok(pb),
627                Err(e) => {
628                    tracing::warn!(
629                        target: "audio_endpoint",
630                        device = %device.to_string(),
631                        "output open failed ({e}); trying next candidate"
632                    );
633                    last_err = Some(e);
634                }
635            }
636        }
637        Err(last_err.unwrap_or_else(|| AudioError::NoDevice("no output device".into())))
638    }
639
640    /// Open one specific output device at its native format. See [`Self::open`]
641    /// for the candidate-selection wrapper.
642    fn open_device(device: &cpal::Device) -> Result<Self, AudioError> {
643        let opened_name = Some(device.to_string());
644
645        let default_cfg = device
646            .default_output_config()
647            .map_err(|e| map_cpal_err("query default output config", DeviceKind::Speaker, e))?;
648        let sample_format = default_cfg.sample_format();
649        let dev_channels = default_cfg.channels();
650        let dev_rate = default_cfg.sample_rate();
651
652        let config = cpal::StreamConfig {
653            channels: dev_channels,
654            sample_rate: dev_rate,
655            buffer_size: cpal::BufferSize::Default,
656        };
657        // Engine frames are 48 kHz; the ring (and the RT callback that drains
658        // it) runs at the DEVICE rate. `play` converts on the way in — a
659        // 44.1 k output device otherwise plays 48 k samples slow ("deep
660        // voice").
661        let resampler = super::resample::MonoResampler::new(SAMPLE_RATE.hz(), dev_rate)?;
662        if dev_rate != SAMPLE_RATE.hz() {
663            tracing::info!(
664                target: "audio_endpoint",
665                device_rate = dev_rate,
666                "output device is not 48 kHz — resampling from the engine rate"
667            );
668        }
669
670        let (producer, mut consumer) = playback_ring(RING_CAPACITY_SAMPLES);
671        let channels = dev_channels as usize;
672
673        // Mark the stream dead on error so the mix bus reopens the device (a "default
674        // output device changed" leaves the stream dead but this object alive — without
675        // this, output goes silent until a manual hot-swap).
676        let err_shared = Arc::clone(&producer.shared);
677        let error_cb = move |e: cpal::Error| {
678            tracing::error!(target: "audio_endpoint", "cpal output stream error: {e}");
679            err_shared.output_dead.store(true, Ordering::Relaxed);
680        };
681
682        // The output callback wants `channels`-interleaved samples; our ring
683        // is mono, so each mono sample is duplicated across the device's
684        // channels. A per-callback mono scratch (allocated once, reused) feeds
685        // the upmix. The consumer is moved into the (single) matched callback.
686        let mut mono_scratch: Vec<i16> = Vec::with_capacity(RING_CAPACITY_SAMPLES);
687        let stream = match sample_format {
688            SampleFormat::F32 => device.build_output_stream(
689                config,
690                move |out: &mut [f32], _| {
691                    let frames = out.len() / channels.max(1);
692                    mono_scratch.resize(frames, 0);
693                    consumer.pop_playback(&mut mono_scratch);
694                    for (frame, &m) in out.chunks_mut(channels).zip(mono_scratch.iter()) {
695                        let v = f32::from_sample(m);
696                        frame.iter_mut().for_each(|s| *s = v);
697                    }
698                },
699                error_cb,
700                None,
701            ),
702            SampleFormat::I16 => device.build_output_stream(
703                config,
704                move |out: &mut [i16], _| {
705                    let frames = out.len() / channels.max(1);
706                    mono_scratch.resize(frames, 0);
707                    consumer.pop_playback(&mut mono_scratch);
708                    for (frame, &m) in out.chunks_mut(channels).zip(mono_scratch.iter()) {
709                        frame.iter_mut().for_each(|s| *s = m);
710                    }
711                },
712                error_cb,
713                None,
714            ),
715            SampleFormat::U16 => device.build_output_stream(
716                config,
717                move |out: &mut [u16], _| {
718                    let frames = out.len() / channels.max(1);
719                    mono_scratch.resize(frames, 0);
720                    consumer.pop_playback(&mut mono_scratch);
721                    for (frame, &m) in out.chunks_mut(channels).zip(mono_scratch.iter()) {
722                        let v = u16::from_sample(m);
723                        frame.iter_mut().for_each(|s| *s = v);
724                    }
725                },
726                error_cb,
727                None,
728            ),
729            other => {
730                return Err(AudioError::UnsupportedFormat(format!(
731                    "unsupported output sample format {other:?}"
732                )));
733            }
734        }
735        .map_err(|e| map_cpal_err("build output stream", DeviceKind::Speaker, e))?;
736
737        stream
738            .play()
739            .map_err(|e| map_cpal_err("start output stream", DeviceKind::Speaker, e))?;
740
741        Ok(Self {
742            out: producer,
743            _stream: stream,
744            resampler,
745            scratch: Vec::new(),
746            opened_name,
747            dev_rate,
748        })
749    }
750
751    /// The name of the output device actually opened.
752    pub(crate) fn opened_name(&self) -> Option<&str> {
753        self.opened_name.as_deref()
754    }
755}
756
757impl sealed::Sealed for CpalPlayback {}
758
759impl AudioOutput for CpalPlayback {
760    fn take_output_underruns(&self) -> u64 {
761        self.out.take_output_underruns()
762    }
763
764    fn buffered_ms(&self) -> u32 {
765        let rate = self.dev_rate.max(1);
766        (self.out.buffered() as u64 * 1000 / rate as u64) as u32
767    }
768
769    fn is_dead(&self) -> bool {
770        self.out.is_output_dead()
771    }
772
773    fn device_rate(&self) -> u32 {
774        self.dev_rate
775    }
776
777    fn play(&mut self, frame: &PcmFrame) -> Result<(), AudioError> {
778        // Convert the 48 kHz engine frame to the device rate (pass-through on
779        // 48 k devices), then non-blocking push into the bounded ring; the RT
780        // output callback drains it. Mono in, mono ring (upmixed to device
781        // channels in the callback).
782        if self.resampler.is_passthrough() {
783            self.out.push_playback(&frame.samples);
784        } else {
785            self.scratch.clear();
786            self.resampler
787                .process_into(&frame.samples, &mut self.scratch)?;
788            self.out.push_playback(&self.scratch);
789        }
790        Ok(())
791    }
792}

797    /// the SPSC ring has no blocked consumer to wake (unlike the capture ring's condvar).
798    fn drop(&mut self) {
799        // `_stream` drop stops playback + releases the device.
800    }
801}

803#[cfg(test)]
804mod ring_underrun_tests {
805    use super::{RING_CAPACITY_SAMPLES, playback_ring};
806
807    /// When the playback RT callback asks for more samples than the ring holds, the
808    /// missing tail is silence-padded AND counted — the definitive end-of-pipeline
809    /// audible-gap signal. A fully-fed pop counts zero. The producer counter and the
810    /// consumer's pad share one atomic, so `take_output_underruns` sees the gap.
811    #[test]
812    fn pop_playback_counts_only_the_silence_padded_tail() {
813        let (mut prod, mut cons) = playback_ring(RING_CAPACITY_SAMPLES);
814        prod.push_playback(&[1, 2, 3, 4]);
815
816        // Ask for 4 with 4 available → no underrun.
817        let mut out = [0i16; 4];
818        cons.pop_playback(&mut out);
819        assert_eq!(out, [1, 2, 3, 4]);
820        assert_eq!(
821            prod.take_output_underruns(),
822            0,
823            "exact fill is not an underrun"
824        );
825
826        // Ask for 6 with 2 available → 4 silence-padded samples counted.
827        prod.push_playback(&[5, 6]);
828        let mut out6 = [0i16; 6];
829        cons.pop_playback(&mut out6);
830        assert_eq!(out6, [5, 6, 0, 0, 0, 0], "tail is silence");
831        assert_eq!(
832            prod.take_output_underruns(),
833            4,
834            "the 4 missing samples are the gap"
835        );
836        // take_* resets, so a subsequent fed pop is clean.
837        prod.push_playback(&[7]);
838        let mut out1 = [0i16; 1];
839        cons.pop_playback(&mut out1);
840        assert_eq!(prod.take_output_underruns(), 0, "counter reset on read");
841    }
842
843    /// A completely empty ring drained for a full callback is all-underrun (the
844    /// startup / device-stalled case): every requested sample is silence.
845    #[test]
846    fn empty_ring_pop_is_all_underrun() {
847        let (prod, mut cons) = playback_ring(RING_CAPACITY_SAMPLES);
848        let mut out = [0i16; 480];
849        cons.pop_playback(&mut out);
850        assert!(out.iter().all(|&s| s == 0));
851        assert_eq!(prod.take_output_underruns(), 480);
852    }
