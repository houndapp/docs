1//! # Sans-io 20 ms PCM framer (portable)
2//!
3//! The shared assembler that turns a ragged stream of **48 kHz mono i16** samples
4//! — however the OS capture callback batches them — into the engine's fixed
5//! [`crate::types::SAMPLES_PER_FRAME`]-sample (20 ms) [`PcmFrame`]s, each carrying
6//! a **unique, monotonically advancing** capture-clock timestamp.
7//!
8//! It is pure (`now_us` is passed in; no clock, no I/O, no locks), so every
9//! cadence rule below is exercised by the unit tests with plain integers — no
10//! audio device required. Both screen-share audio backends feed it: macOS
11//! ScreenCaptureKit ([`crate::platform::apple::screen`]) and Windows WASAPI
12//! loopback ([`crate::platform::windows::wasapi_loopback`]); the microphone path
13//! frames at the device boundary instead. One copy of the cadence math, one set
14//! of tests.
15//!
16//! ## Why the timestamp cadence is not just `clock.now()`
17//!
18//! A capture callback does not hand you 20 ms at a time — it delivers whatever
19//! the OS batched (often several frames' worth at once). Stamping every block
20//! carved out of one callback with the *same* `now()` produces duplicate
21//! timestamps, which the receiver's playout buffer discards as duplicates →
22//! choppy audio. So the framer advances an internal cursor by exactly one frame
23//! (20 ms) per emitted block: a single fat callback yields evenly spaced, unique
24//! timestamps. The cursor **re-anchors** to `now_us` on the first block and
25//! whenever it has drifted more than [`REANCHOR_US`] from wall-clock (a capture
26//! stall/pause), so the cadence tracks real time without lagging forever.
27
28use crate::types::{PcmFrame, SAMPLE_RATE, SAMPLES_PER_FRAME, TimestampUs};
29
30/// Wall-clock span of one 20 ms frame, in microseconds (960 / 48 000 s).
31const FRAME_US: i64 = (SAMPLES_PER_FRAME as i64 * 1_000_000) / SAMPLE_RATE.hz() as i64;
32
33/// Re-anchor the cadence cursor to `now_us` once it drifts this far from
34/// wall-clock — a capture stall/pause. Keeps timestamps tracking real time
35/// while a few-frame batch still gets a smooth 20 ms cadence.
36const REANCHOR_US: i64 = 200_000;
37
38/// Accumulates 48 kHz mono i16 samples and emits fixed 20 ms [`PcmFrame`]s with a
39/// monotonic, re-anchoring timestamp cadence. Sans-io: drive it with
40/// [`Self::push`], passing the current capture-clock time; it never blocks, never
41/// allocates beyond the pending tail, and holds no OS handle.
42pub(crate) struct PcmFramer {
43    /// Samples carried over from the previous push (< one full frame).
44    pending: Vec<i16>,
45    /// Playout timestamp (µs) for the NEXT block; `None` until the first emit and
46    /// after a re-anchor decision. See the module docs.
47    next_ts_us: Option<i64>,
48}
49
50impl PcmFramer {
51    /// A framer with an empty accumulator and an unanchored cadence.
52    pub(crate) fn new() -> Self {
53        Self {
54            pending: Vec::with_capacity(SAMPLES_PER_FRAME * 2),
55            next_ts_us: None,
56        }
57    }
58
59    /// Append `mono` (48 kHz mono i16) and emit every complete 20 ms frame now
60    /// ready into `out` (appended, not cleared), each stamped on the cadence
61    /// anchored to `now_us`. A partial tail is retained for the next push.
62    ///
63    /// `out` is a caller-owned scratch buffer so the steady state allocates
64    /// nothing per callback beyond the frames themselves.
65    pub(crate) fn push(&mut self, mono: &[i16], now_us: i64, out: &mut Vec<PcmFrame>) {
66        if mono.is_empty() && self.pending.len() < SAMPLES_PER_FRAME {
67            return;
68        }
69        self.pending.extend_from_slice(mono);
70        while self.pending.len() >= SAMPLES_PER_FRAME {
71            let block: Vec<i16> = self.pending.drain(..SAMPLES_PER_FRAME).collect();
72            // Continue the running cadence while it tracks wall-clock; otherwise
73            // (first block, or after a gap) re-anchor to `now_us`. Either way the
74            // block gets a UNIQUE timestamp 20 ms after the previous one.
75            let ts = match self.next_ts_us {
76                Some(t) if (now_us - t).abs() <= REANCHOR_US => t,
77                _ => now_us,
78            };
79            self.next_ts_us = Some(ts + FRAME_US);
80            out.push(PcmFrame::at_engine_rate(
81                block,
82                TimestampUs::from_micros(ts),
83            ));
84        }
85    }
86
87    /// Samples currently held back awaiting a full frame (diagnostics/tests).
88    #[cfg(test)]
89    pub(crate) fn pending_len(&self) -> usize {
90        self.pending.len()
91    }
92}
93
94impl Default for PcmFramer {
95    fn default() -> Self {
96        Self::new()
97    }
98}

100#[cfg(test)]
101mod tests {
102    use super::*;
103
104    /// Distinct, deterministic samples so we can assert framing boundaries.
105    fn ramp(n: usize, start: i32) -> Vec<i16> {
106        (0..n).map(|i| ((start + i as i32) % 1000) as i16).collect()
107    }
108
109    #[test]
110    fn exact_block_emits_one_frame_at_now() {
111        let mut f = PcmFramer::new();
112        let mut out = Vec::new();
113        f.push(&ramp(SAMPLES_PER_FRAME, 0), 5_000, &mut out);
114        assert_eq!(out.len(), 1);
115        assert_eq!(out[0].timestamp_us.micros(), 5_000);
116        assert_eq!(f.pending_len(), 0);
117    }
118
119    #[test]
120    fn partial_accumulates_across_pushes() {
121        let mut f = PcmFramer::new();
122        let mut out = Vec::new();
123        let half = SAMPLES_PER_FRAME / 2;
124        f.push(&ramp(half, 0), 0, &mut out);
125        assert!(out.is_empty(), "half a frame must not emit");
126        assert_eq!(f.pending_len(), half);
127        f.push(
128            &ramp(SAMPLES_PER_FRAME - half, half as i32),
129            1_000,
130            &mut out,
131        );
132        assert_eq!(out.len(), 1, "the completing push emits exactly one frame");
133        assert_eq!(f.pending_len(), 0);
134    }
135
136    #[test]
137    fn fat_callback_yields_unique_evenly_spaced_timestamps() {
138        // One callback carrying 3 frames' worth must NOT stamp them all with the
139        // same `now` — they must be 20 ms apart and strictly increasing.
140        let mut f = PcmFramer::new();
141        let mut out = Vec::new();
142        f.push(&ramp(SAMPLES_PER_FRAME * 3, 0), 100_000, &mut out);
143        assert_eq!(out.len(), 3);
144        let ts: Vec<i64> = out.iter().map(|f| f.timestamp_us.micros()).collect();
145        assert_eq!(
146            ts,
147            vec![100_000, 100_000 + FRAME_US, 100_000 + 2 * FRAME_US]
148        );
149        // Strictly monotonic (no duplicates the receiver would drop).
150        assert!(ts.windows(2).all(|w| w[1] > w[0]));
151    }
152
153    #[test]
154    fn steady_cadence_ignores_now_jitter_within_tolerance() {
155        // Across many callbacks the cadence stays a clean 20 ms grid even when
156        // `now_us` jitters, because each block continues the running cursor.
157        let mut f = PcmFramer::new();
158        let mut out = Vec::new();
159        f.push(&ramp(SAMPLES_PER_FRAME, 0), 0, &mut out);
160        // now jitters by a few ms (well under REANCHOR) on the next callbacks.
161        f.push(&ramp(SAMPLES_PER_FRAME, 0), FRAME_US + 3_000, &mut out);
162        f.push(&ramp(SAMPLES_PER_FRAME, 0), 2 * FRAME_US - 2_000, &mut out);
163        let ts: Vec<i64> = out.iter().map(|f| f.timestamp_us.micros()).collect();
164        assert_eq!(
165            ts,
166            vec![0, FRAME_US, 2 * FRAME_US],
167            "cursor-driven, not now-driven"
168        );
169    }
170
171    #[test]
172    fn reanchors_after_a_large_gap() {
173        // A capture stall: `now_us` jumps far past the cursor → the next block
174        // re-anchors to the new now rather than back-dating a stale cursor.
175        let mut f = PcmFramer::new();
176        let mut out = Vec::new();
177        f.push(&ramp(SAMPLES_PER_FRAME, 0), 0, &mut out);
178        let gap = REANCHOR_US * 2;
179        f.push(&ramp(SAMPLES_PER_FRAME, 0), gap, &mut out);
180        let ts: Vec<i64> = out.iter().map(|f| f.timestamp_us.micros()).collect();
181        assert_eq!(ts[0], 0);
182        assert_eq!(ts[1], gap, "re-anchored to the post-gap now");
183    }
184
185    #[test]
186    fn empty_push_is_a_noop() {
187        let mut f = PcmFramer::new();
188        let mut out = Vec::new();
189        f.push(&[], 0, &mut out);
190        assert!(out.is_empty());
191        assert_eq!(f.pending_len(), 0);
192    }
193
194    #[test]
195    fn every_frame_has_full_length() {
196        let mut f = PcmFramer::new();
197        let mut out = Vec::new();
198        // Ragged inputs that sum to several frames plus a tail.
199        f.push(&ramp(700, 0), 0, &mut out);
200        f.push(&ramp(900, 0), FRAME_US, &mut out);
201        f.push(&ramp(800, 0), 2 * FRAME_US, &mut out);
202        for frame in &out {
203            assert_eq!(frame.samples.len(), SAMPLES_PER_FRAME);
204        }
205        assert_eq!(f.pending_len(), (700 + 900 + 800) % SAMPLES_PER_FRAME);
206    }
