1//! # Synthetic video source (camera-free test source)
2//!
3//! A [`VideoCapturer`] that fabricates timestamped I420 frames at the requested
4//! resolution/fps **without any camera**. It lets the encode path be exercised and
5//! benchmarked (e.g. *can this codec encode {1280×720, 1920×1080} @ {30, 60} in
6//! real time within the CPU/thermal budget?*) on any machine, regardless of whether
7//! a camera is present or the capture backend works — so it is the default source
8//! for codec/perf tests.
9//!
10//! ## Why a *moving* pattern, not a flat frame
11//! A static frame compresses to almost nothing after the first keyframe, so the
12//! encoder would do unrepresentative work and the real-time verdict would be a
13//! lie. This source therefore produces genuinely varying content every frame:
14//!   * a diagonal luma gradient that **scrolls** (one column/frame), giving the
15//!     encoder spatial detail plus global motion;
16//!   * a solid block that **bounces** around the frame, giving localised motion
17//!     vectors / residuals;
18//!   * a slowly cycling chroma background so the U/V planes are not constant.
19//!
20//! That keeps the per-frame encode cost in the same ballpark as real camera
21//! content (high-frequency detail + motion), which is what the measurement needs.
22//!
23//! ## Timing contract
24//! [`Self::next_frame`] **paces itself** to the requested fps by sleeping the
25//! remainder of the frame interval — so `captureFps` reflects the *target* rate
26//! and the harness measures whether encode keeps up with that cadence (exactly
27//! the camera's role). If the downstream encode is slower than the interval, no
28//! sleep happens and frames are produced back-to-back (the honest "can't keep
29//! up" signal). Frames are stamped on the shared [`hound_media::CaptureClock`]
30//! like any real capturer.
31//!
32//! ## Real-time-safety
33//! All buffers are sized exactly `w*h + 2*(w/2)*(h/2)` and written by index into
34//! a pre-validated `Vec` — no `unwrap`, no unchecked slicing, nothing that can
35//! panic on the capture thread. Generation is plain integer arithmetic.
36

50/// The static test-card frame: fixed checkerboard background (distinctive,
51/// unmistakable for a "real" capture) plus a centered solid block "stable
52/// pattern" — both pure functions of `(w, h)`, never `frame_idx`, so every
53/// call with the same geometry produces byte-identical output. Chroma is
54/// fixed too (no cycling), unlike the moving pattern's.
55fn render_testcard(w: usize, h: usize) -> Vec<u8> {
56    let cw = w / 2;
57    let ch = h / 2;
58    let y_size = w * h;
59    let c_size = cw * ch;
60    let mut data = vec![0u8; y_size + 2 * c_size];
61    let (y_plane, uv) = data.split_at_mut(y_size);
62    let (u_plane, v_plane) = uv.split_at_mut(c_size);
63
64    // Checkerboard: 32px squares alternating a dark/light luma pair — coarse
65    // enough to survive lossy encode + scaling down to a rendered tile size
66    // without aliasing into noise.
67    const SQUARE: usize = 32;
68    for j in 0..h {
69        for i in 0..w {
70            let on = ((i / SQUARE) + (j / SQUARE)).is_multiple_of(2);
71            y_plane[j * w + i] = if on { 200 } else { 40 };
72        }
73    }
74    // Centered solid block — the "stable pattern" a diff can anchor on even
75    // if the checkerboard phase were ever to change.
76    let (bw, bh) = ((w / 4).max(2), (h / 4).max(2));
77    let (bx, by) = ((w.saturating_sub(bw)) / 2, (h.saturating_sub(bh)) / 2);
78    for j in by..(by + bh).min(h) {
79        for i in bx..(bx + bw).min(w) {
80            y_plane[j * w + i] = 128;
81        }
82    }
83
84    // Fixed, non-cycling chroma — a distinctive teal-ish cast so the card
85    // reads as synthetic at a glance, not just a grayscale ramp.
86    u_plane.fill(170);
87    v_plane.fill(90);
88
89    data
90}

92/// Generates a moving test pattern as I420 frames at a fixed geometry/fps.
93/// Open is construct ([`Self::open`]): a value that exists is producing, so
94/// a "synthetic source not started" error class is unrepresentable.
95pub(crate) struct SyntheticCapturer {
96    /// Encode-target geometry. The source produces I420 directly at this
97    /// size, so there is never a capture-vs-encode mismatch and no scale step
98    /// — `capture_native == encode_target` and [`Self::is_scaling`] is false.
99    dims: Dimensions,
100    fps: Fps,
101    clock: hound_media::CaptureClock,
102    /// Monotonic frame index, drives the animation phase.
103    frame_idx: u64,
104    /// When the previous frame was emitted, for fps pacing. `None` before the
105    /// first frame.
106    last_emit: Option<Instant>,
107    /// See [`testcard_enabled`] — cached at open time, not re-read per frame.
108    testcard: bool,
109}
110
111impl SyntheticCapturer {
112    /// Begin producing frames at as close to `width`×`height` as I420 allows
113    /// (odd requests are even-clamped, matching real capturers) at `fps`,
114    /// stamped on `clock`. Infallible in practice — kept `Result` to mirror
115    /// the real device openers it stands in for.
116    /// `testcard` switches from the moving pattern to the static test card:
117    /// no per-frame motion, pure function of geometry — for visual snapshot
118    /// testing (ui/integration_test/visual_test.dart + crates/snapdiff), where
119    /// a rendered peer tile must be pixel-identical run over run. The moving
120    /// pattern stays the norm everywhere else: its motion is what keeps the
121    /// encoder producing real bitrate, which the call e2e suite's
122    /// video-flow assertions depend on.
123    pub(crate) fn open(
124        width: u32,
125        height: u32,
126        fps: Fps,
127        clock: hound_media::CaptureClock,
128        testcard: bool,
129    ) -> Result<Self, CaptureError> {
130        // Even-clamp (I420 chroma is half-res) — match the encoder-side clamp
131        // so an encoder built from the same request always agrees.
132        let dims = Dimensions::even_clamped(width, height);
133        if testcard {
134            tracing::info!(
135                width = dims.width(),
136                height = dims.height(),
137                fps = fps.get(),
138                "synthetic source started: STATIC test card, \
139                 I420 at encode target — no camera, no motion"
140            );
141        } else {
142            tracing::info!(
143                width = dims.width(),
144                height = dims.height(),
145                fps = fps.get(),
146                "synthetic source started: moving test pattern (scrolling gradient + \
147                 bouncing block + cycling chroma), I420 at encode target — no camera"
148            );
149        }
150        Ok(Self {
151            dims,
152            fps,
153            clock,
154            frame_idx: 0,
155            last_emit: None,
156            testcard,
157        })
158    }
159
160    /// The fps the source honours — always the request, exactly. Diagnostic;
161    /// only the tests read it (a live join reads geometry off [`Self::negotiated`]).
162    #[cfg_attr(not(test), allow(dead_code))]
163    pub(crate) fn negotiated_fps(&self) -> Fps {
164        self.fps
165    }
166
167    /// The synthetic source generates at the encode target, so the "native"
168    /// capture resolution equals the encode resolution. Diagnostic; test-only caller.
169    #[cfg_attr(not(test), allow(dead_code))]
170    pub(crate) fn capture_native(&self) -> Dimensions {
171        self.dims
172    }
173
174    /// Always `false`: the source produces frames at the encode target directly.
175    /// Diagnostic; test-only caller.
176    #[cfg_attr(not(test), allow(dead_code))]
177    pub(crate) fn is_scaling(&self) -> bool {
178        false
179    }
180
181    /// Build one I420 frame for the current `frame_idx`. Pure CPU, fallible-free:
182    /// every write is to a pre-sized buffer at an in-bounds index.
183    fn render(&self, w: usize, h: usize) -> Vec<u8> {
184        if self.testcard {
185            return render_testcard(w, h);
186        }
187        let cw = w / 2;
188        let ch = h / 2;
189        let y_size = w * h;
190        let c_size = cw * ch;
191        let mut data = vec![0u8; y_size + 2 * c_size];
192        let (y_plane, uv) = data.split_at_mut(y_size);
193        let (u_plane, v_plane) = uv.split_at_mut(c_size);
194
195        let t = self.frame_idx;
196        // Scrolling diagonal luma gradient: detail + global motion. The phase
197        // advances one pixel/frame so successive frames differ everywhere.
198        let scroll = (t % w.max(1) as u64) as usize;
199        for j in 0..h {
200            let row = j * w;
201            for i in 0..w {
202                let x = (i + scroll) % w;
203                // 0..=235-ish luma ramp (kept in limited range) varying in x+y.
204                let v = 16 + (((x + j) * 219 / (w + h).max(1)) % 220) as u8;
205                y_plane[row + i] = v;
206            }
207        }
208
209        // A bouncing solid block of bright luma — localised motion the encoder
210        // must track with motion vectors / residuals.
211        let bw = (w / 6).max(2);
212        let bh = (h / 6).max(2);
213        let span_x = (w.saturating_sub(bw)).max(1);
214        let span_y = (h.saturating_sub(bh)).max(1);
215        // Triangle-wave position so it bounces rather than wraps.
216        let bx = triangle(t, span_x as u64) as usize;
217        let by = triangle(t / 2, span_y as u64) as usize;
218        for j in by..(by + bh).min(h) {
219            let row = j * w;
220            for i in bx..(bx + bw).min(w) {
221                y_plane[row + i] = 235;
222            }
223        }
224
225        // Slowly cycling chroma so U/V are not constant (keeps the chroma planes
226        // doing real work). Phase differs between U and V for colour movement.
227        let u_val = (128 + ((t / 3) % 64) as i32 - 32).clamp(16, 240) as u8;
228        let v_val = (128 + ((t / 5) % 64) as i32 - 32).clamp(16, 240) as u8;
229        for px in u_plane.iter_mut() {
230            *px = u_val;
231        }
232        for px in v_plane.iter_mut() {
233            *px = v_val;
234        }
235
236        data
237    }
238}
239
240/// Triangle wave in `0..=span`, period `2*span`, from a monotonic counter. Used
241/// to bounce the block between the frame edges instead of wrapping.
242fn triangle(t: u64, span: u64) -> u64 {
243    if span == 0 {
244        return 0;
245    }
246    let period = 2 * span;
247    let phase = t % period;
248    if phase <= span { phase } else { period - phase }
249}

251impl sealed::Sealed for SyntheticCapturer {}
252
253impl VideoCapture for SyntheticCapturer {
254    fn negotiated(&self) -> (Dimensions, PixelFormat) {
255        (self.dims, PixelFormat::I420)
256    }
257
258    fn next_frame(&mut self) -> Result<RawFrame, CaptureError> {
259        let (w, h) = (self.dims.width() as usize, self.dims.height() as usize);
260
261        // Pace to the target fps: sleep whatever remains of the frame interval
262        // since the last emit. If encode already overran the interval, there's
263        // nothing to sleep and frames flow back-to-back (the honest "can't keep
264        // up" signal). This makes the capture fps reflect the target cadence the
265        // way a real camera's fixed frame clock would.
266        let interval = Duration::from_secs_f64(1.0 / self.fps.get() as f64);
267        if let Some(prev) = self.last_emit
268            && let Some(remaining) = interval.checked_sub(prev.elapsed())
269        {
270            std::thread::sleep(remaining);
271        }
272
273        let data = self.render(w, h);
274        let captured_at = Instant::now();
275        let timestamp_us = TimestampUs::from_micros(self.clock.stamp(captured_at));
276        self.last_emit = Some(captured_at);
277        self.frame_idx = self.frame_idx.wrapping_add(1);
278
279        Ok(RawFrame {
280            dims: self.dims,
281            format: PixelFormat::I420,
282            timestamp_us,
283            captured_at,
284            data,
285            surface: None,
286        })
287    }
288}

290#[cfg(test)]
291mod tests {
292    use super::*;
293
294    fn started(w: u32, h: u32, fps: u32) -> SyntheticCapturer {
295        SyntheticCapturer::open(
296            w,
297            h,
298            Fps::new(fps),
299            hound_media::CaptureClock::start(),
300            false,
301        )
302        .expect("open")
303    }
304
305    #[test]
306    fn emits_correctly_sized_i420() {
307        let mut c = started(1280, 720, 30);
308        let f = c.next_frame().expect("frame");
309        let (w, h) = (1280usize, 720usize);
310        assert_eq!(f.data.len(), w * h + 2 * (w / 2) * (h / 2));
311        assert_eq!(f.dims.wh(), (1280, 720));
312        assert_eq!(f.format, PixelFormat::I420);
313    }
314
315    #[test]
316    fn odd_dimensions_are_even_clamped() {
317        let c = started(641, 361, 30);
318        // Even-clamped to 640x360; never produces an odd plane geometry.
319        assert_eq!(
320            c.negotiated(),
321            (Dimensions::new(640, 360), PixelFormat::I420)
322        );
323    }
324
325    #[test]
326    fn successive_frames_differ() {
327        // The whole point: content must vary so the encoder does real work.
328        let mut c = started(320, 240, 60);
329        let a = c.next_frame().expect("a");
330        let b = c.next_frame().expect("b");
331        assert_ne!(a.data, b.data, "synthetic frames must not be static");
332    }
333
334    // Test-card assertions exercise `render_testcard` directly — a pure
335    // function of geometry, so no capturer needs opening.
336
337    #[test]
338    fn testcard_is_correctly_sized_i420() {
339        let (w, h) = (320usize, 240usize);
340        let data = render_testcard(w, h);
341        assert_eq!(data.len(), w * h + 2 * (w / 2) * (h / 2));
342    }
343
344    #[test]
345    fn testcard_frames_are_pixel_identical_across_calls() {
346        // No motion: two independent renders at the same geometry must be
347        // byte-identical (what makes a rendered peer tile diff-stable).
348        let a = render_testcard(640, 360);
349        let b = render_testcard(640, 360);
350        assert_eq!(a, b, "test card must not vary run to run");
351    }
352
353    #[test]
354    fn testcard_is_not_a_flat_frame() {
355        // Distinctive, not a single solid color — the checkerboard + centered
356        // block must produce more than one luma value.
357        let data = render_testcard(320, 240);
358        let y_plane = &data[..320 * 240];
359        let distinct: std::collections::HashSet<u8> = y_plane.iter().copied().collect();
360        assert!(
361            distinct.len() >= 3,
362            "expected checkerboard + center block luma levels, got {distinct:?}"
363        );
364    }
365
366    #[test]
367    fn reports_encode_target_as_capture_native_no_scaling() {
368        let c = started(1920, 1080, 60);
369        assert_eq!(c.capture_native(), Dimensions::new(1920, 1080));
370        assert!(!c.is_scaling());
371        assert_eq!(c.negotiated_fps(), Fps::new(60));
372    }
373
374    #[test]
375    fn triangle_bounces_within_span() {
376        for t in 0..1000u64 {
377            assert!(triangle(t, 100) <= 100);
378        }
379        assert_eq!(triangle(0, 100), 0);
380        assert_eq!(triangle(100, 100), 100);
381        assert_eq!(triangle(200, 100), 0);
382    }
