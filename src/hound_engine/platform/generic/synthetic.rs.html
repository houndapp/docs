1//! # Synthetic video source (camera-free test source)
2//!
3//! A [`VideoCapturer`] that fabricates timestamped I420 frames at the requested
4//! resolution/fps **without any camera**. It lets the encode path be exercised and
5//! benchmarked (e.g. *can this codec encode {1280×720, 1920×1080} @ {30, 60} in
6//! real time within the CPU/thermal budget?*) on any machine, regardless of whether
7//! a camera is present or the capture backend works — so it is the default source
8//! for codec/perf tests.
9//!
10//! ## Why a *moving* pattern, not a flat frame
11//! A static frame compresses to almost nothing after the first keyframe, so the
12//! encoder would do unrepresentative work and the real-time verdict would be a
13//! lie. This source therefore produces genuinely varying content every frame:
14//!   * a diagonal luma gradient that **scrolls** (one column/frame), giving the
15//!     encoder spatial detail plus global motion;
16//!   * a solid block that **bounces** around the frame, giving localised motion
17//!     vectors / residuals;
18//!   * a slowly cycling chroma background so the U/V planes are not constant.
19//!
20//! That keeps the per-frame encode cost in the same ballpark as real camera
21//! content (high-frequency detail + motion), which is what the measurement needs.
22//!
23//! ## Timing contract
24//! [`Self::next_frame`] **paces itself** to the requested fps by sleeping the
25//! remainder of the frame interval — so `captureFps` reflects the *target* rate
26//! and the harness measures whether encode keeps up with that cadence (exactly
27//! the camera's role). If the downstream encode is slower than the interval, no
28//! sleep happens and frames are produced back-to-back (the honest "can't keep
29//! up" signal). Frames are stamped on the shared [`hound_media::CaptureClock`]
30//! like any real capturer.
31//!
32//! ## Real-time-safety
33//! All buffers are sized exactly `w*h + 2*(w/2)*(h/2)` and written by index into
34//! a pre-validated `Vec` — no `unwrap`, no unchecked slicing, nothing that can
35//! panic on the capture thread. Generation is plain integer arithmetic.
36

50/// Generates a moving test pattern as I420 frames at a fixed geometry/fps.
51/// Open is construct ([`Self::open`]): a value that exists is producing, so
52/// a "synthetic source not started" error class is unrepresentable.
53pub(crate) struct SyntheticCapturer {
54    /// Encode-target geometry. The source produces I420 directly at this
55    /// size, so there is never a capture-vs-encode mismatch and no scale step
56    /// — `capture_native == encode_target` and [`Self::is_scaling`] is false.
57    dims: Dimensions,
58    fps: Fps,
59    clock: hound_media::CaptureClock,
60    /// Monotonic frame index, drives the animation phase.
61    frame_idx: u64,
62    /// When the previous frame was emitted, for fps pacing. `None` before the
63    /// first frame.
64    last_emit: Option<Instant>,
65}
66
67impl SyntheticCapturer {
68    /// Begin producing frames at as close to `width`×`height` as I420 allows
69    /// (odd requests are even-clamped, matching real capturers) at `fps`,
70    /// stamped on `clock`. Infallible in practice — kept `Result` to mirror
71    /// the real device openers it stands in for.
72    pub(crate) fn open(
73        width: u32,
74        height: u32,
75        fps: Fps,
76        clock: hound_media::CaptureClock,
77    ) -> Result<Self, CaptureError> {
78        // Even-clamp (I420 chroma is half-res) — match the encoder-side clamp
79        // so an encoder built from the same request always agrees.
80        let dims = Dimensions::even_clamped(width, height);
81        tracing::info!(
82            width = dims.width(),
83            height = dims.height(),
84            fps = fps.get(),
85            "synthetic source started: moving test pattern (scrolling gradient + \
86             bouncing block + cycling chroma), I420 at encode target — no camera"
87        );
88        Ok(Self {
89            dims,
90            fps,
91            clock,
92            frame_idx: 0,
93            last_emit: None,
94        })
95    }
96
97    /// The fps the source honours — always the request, exactly. Diagnostic;
98    /// only the tests read it (a live join reads geometry off [`Self::negotiated`]).
99    #[cfg_attr(not(test), allow(dead_code))]
100    pub(crate) fn negotiated_fps(&self) -> Fps {
101        self.fps
102    }
103
104    /// The synthetic source generates at the encode target, so the "native"
105    /// capture resolution equals the encode resolution. Diagnostic; test-only caller.
106    #[cfg_attr(not(test), allow(dead_code))]
107    pub(crate) fn capture_native(&self) -> Dimensions {
108        self.dims
109    }
110
111    /// Always `false`: the source produces frames at the encode target directly.
112    /// Diagnostic; test-only caller.
113    #[cfg_attr(not(test), allow(dead_code))]
114    pub(crate) fn is_scaling(&self) -> bool {
115        false
116    }
117
118    /// Build one I420 frame for the current `frame_idx`. Pure CPU, fallible-free:
119    /// every write is to a pre-sized buffer at an in-bounds index.
120    fn render(&self, w: usize, h: usize) -> Vec<u8> {
121        let cw = w / 2;
122        let ch = h / 2;
123        let y_size = w * h;
124        let c_size = cw * ch;
125        let mut data = vec![0u8; y_size + 2 * c_size];
126        let (y_plane, uv) = data.split_at_mut(y_size);
127        let (u_plane, v_plane) = uv.split_at_mut(c_size);
128
129        let t = self.frame_idx;
130        // Scrolling diagonal luma gradient: detail + global motion. The phase
131        // advances one pixel/frame so successive frames differ everywhere.
132        let scroll = (t % w.max(1) as u64) as usize;
133        for j in 0..h {
134            let row = j * w;
135            for i in 0..w {
136                let x = (i + scroll) % w;
137                // 0..=235-ish luma ramp (kept in limited range) varying in x+y.
138                let v = 16 + (((x + j) * 219 / (w + h).max(1)) % 220) as u8;
139                y_plane[row + i] = v;
140            }
141        }
142
143        // A bouncing solid block of bright luma — localised motion the encoder
144        // must track with motion vectors / residuals.
145        let bw = (w / 6).max(2);
146        let bh = (h / 6).max(2);
147        let span_x = (w.saturating_sub(bw)).max(1);
148        let span_y = (h.saturating_sub(bh)).max(1);
149        // Triangle-wave position so it bounces rather than wraps.
150        let bx = triangle(t, span_x as u64) as usize;
151        let by = triangle(t / 2, span_y as u64) as usize;
152        for j in by..(by + bh).min(h) {
153            let row = j * w;
154            for i in bx..(bx + bw).min(w) {
155                y_plane[row + i] = 235;
156            }
157        }
158
159        // Slowly cycling chroma so U/V are not constant (keeps the chroma planes
160        // doing real work). Phase differs between U and V for colour movement.
161        let u_val = (128 + ((t / 3) % 64) as i32 - 32).clamp(16, 240) as u8;
162        let v_val = (128 + ((t / 5) % 64) as i32 - 32).clamp(16, 240) as u8;
163        for px in u_plane.iter_mut() {
164            *px = u_val;
165        }
166        for px in v_plane.iter_mut() {
167            *px = v_val;
168        }
169
170        data
171    }
172}
173
174/// Triangle wave in `0..=span`, period `2*span`, from a monotonic counter. Used
175/// to bounce the block between the frame edges instead of wrapping.
176fn triangle(t: u64, span: u64) -> u64 {
177    if span == 0 {
178        return 0;
179    }
180    let period = 2 * span;
181    let phase = t % period;
182    if phase <= span { phase } else { period - phase }
183}
184
185impl sealed::Sealed for SyntheticCapturer {}
186
187impl VideoCapture for SyntheticCapturer {
188    fn negotiated(&self) -> (Dimensions, PixelFormat) {
189        (self.dims, PixelFormat::I420)
190    }
191
192    fn next_frame(&mut self) -> Result<RawFrame, CaptureError> {
193        let (w, h) = (self.dims.width() as usize, self.dims.height() as usize);
194
195        // Pace to the target fps: sleep whatever remains of the frame interval
196        // since the last emit. If encode already overran the interval, there's
197        // nothing to sleep and frames flow back-to-back (the honest "can't keep
198        // up" signal). This makes the capture fps reflect the target cadence the
199        // way a real camera's fixed frame clock would.
200        let interval = Duration::from_secs_f64(1.0 / self.fps.get() as f64);
201        if let Some(prev) = self.last_emit
202            && let Some(remaining) = interval.checked_sub(prev.elapsed())
203        {
204            std::thread::sleep(remaining);
205        }
206
207        let data = self.render(w, h);
208        let captured_at = Instant::now();
209        let timestamp_us = TimestampUs::from_micros(self.clock.stamp(captured_at));
210        self.last_emit = Some(captured_at);
211        self.frame_idx = self.frame_idx.wrapping_add(1);
212
213        Ok(RawFrame {
214            dims: self.dims,
215            format: PixelFormat::I420,
216            timestamp_us,
217            captured_at,
218            data,
219            surface: None,
220        })
221    }
222}

224#[cfg(test)]
225mod tests {
226    use super::*;
227
228    fn started(w: u32, h: u32, fps: u32) -> SyntheticCapturer {
229        SyntheticCapturer::open(w, h, Fps::new(fps), hound_media::CaptureClock::start())
230            .expect("open")
231    }
232
233    #[test]
234    fn emits_correctly_sized_i420() {
235        let mut c = started(1280, 720, 30);
236        let f = c.next_frame().expect("frame");
237        let (w, h) = (1280usize, 720usize);
238        assert_eq!(f.data.len(), w * h + 2 * (w / 2) * (h / 2));
239        assert_eq!(f.dims.wh(), (1280, 720));
240        assert_eq!(f.format, PixelFormat::I420);
241    }
242
243    #[test]
244    fn odd_dimensions_are_even_clamped() {
245        let c = started(641, 361, 30);
246        // Even-clamped to 640x360; never produces an odd plane geometry.
247        assert_eq!(
248            c.negotiated(),
249            (Dimensions::new(640, 360), PixelFormat::I420)
250        );
251    }
252
253    #[test]
254    fn successive_frames_differ() {
255        // The whole point: content must vary so the encoder does real work.
256        let mut c = started(320, 240, 60);
257        let a = c.next_frame().expect("a");
258        let b = c.next_frame().expect("b");
259        assert_ne!(a.data, b.data, "synthetic frames must not be static");
260    }
261
262    #[test]
263    fn reports_encode_target_as_capture_native_no_scaling() {
264        let c = started(1920, 1080, 60);
265        assert_eq!(c.capture_native(), Dimensions::new(1920, 1080));
266        assert!(!c.is_scaling());
267        assert_eq!(c.negotiated_fps(), Fps::new(60));
268    }
269
270    #[test]
271    fn triangle_bounces_within_span() {
272        for t in 0..1000u64 {
273            assert!(triangle(t, 100) <= 100);
274        }
275        assert_eq!(triangle(0, 100), 0);
276        assert_eq!(triangle(100, 100), 100);
277        assert_eq!(triangle(200, 100), 0);
278    }
