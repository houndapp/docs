1//! # scap screen/window capture (Linux)
2//!
3//! The Linux screen-share [`ScreenCapture`] backend: PipeWire +
4//! xdg-desktop-portal, via `scap`. **Video only** — the scap path has no audio
5//! leg. The other platforms have bespoke engine backends with more surface:
6//! macOS ScreenCaptureKit ([`super::super::apple`]: per-app audio, content-rect
7//! resize tracking, one-shot thumbnails) and Windows native WGC
8//! ([`super::super::windows::wgc`]: zero-copy D3D11 + a WASAPI system-audio leg).
9//!
10//! ## Why scap and not hand-rolled PipeWire
11//!
12//! PipeWire + the portal is the correct, modern capture path on Linux — but it
13//! is hundreds of lines of unsafe FFI + lifecycle. `scap` wraps exactly that,
14//! so we get the right native path behind one small, maintained, type-safe
15//! surface.
16//!
17//! ## Pipeline & threading
18//!
19//! `scap`'s `Capturer` is built and driven on a dedicated **bridge thread** (it
20//! owns a platform capture engine that must not move once started; building it
21//! on the bridge avoids any `Send` question). The bridge loops on the blocking
22//! `get_next_frame()`, converts each BGRA frame to I420 ([`convert::bgra_to_i420`]),
23//! scales it to the encode target ([`convert::scale_i420`]), stamps it on the
24//! shared [`CaptureClock`], and pushes it into a [`BoundedRing`] (drop-oldest).
25//! The [`ScreenCapture`] handle on the capture/encode thread just pulls from the
26//! ring — so `next_frame_timeout`'s `Idle`/`Ended` semantics (damage-driven
27//! liveness) work identically to the Apple path.
28//!
29//! Capture is at the source's native size (`Resolution::Captured`); we own the
30//! aspect-fit downscale to the 1080p-class cap, matching the SCK backend's
31//! geometry contract.
32

51/// Frame ring capacity (whole frames) — small; frames are large, a deep queue
52/// is just latency. Overflow drops the OLDEST (the encoder is behind).
53const RING_FRAMES: usize = 4;

55/// The app-audio leg type for this backend: none. scap is video-only on Linux (the
56/// only platform that still uses it — Windows is on the native WGC backend, which

60/// Enumerate shareable displays + windows via scap's target list.
61///
62/// `is_supported()` gates the platform; an ungranted screen-capture permission
63/// surfaces [`CaptureError::PermissionDenied`] (Linux portal / Windows privacy)
64/// so the UI shows the same "grant access" hint as macOS. Per-target native
65/// geometry is not exposed by scap 0.0.8, so [`ScreenSourceInfo::dimensions`]
66/// is a 1080p placeholder; the true size is learned at open via
67/// `get_output_frame_size()`.
68pub(crate) fn enumerate_sources() -> Result<Vec<ScreenSourceInfo>, CaptureError> {
69    if !scap::is_supported() {
70        return Err(CaptureError::Unsupported {
71            what: "screen capture",
72        });
73    }
74    if !scap::has_permission() {
75        return Err(CaptureError::PermissionDenied {
76            kind: crate::types::DeviceKind::Screen,
77        });
78    }
79
80    // Placeholder geometry: scap 0.0.8 exposes no per-target size publicly.
81    let placeholder = Dimensions::even_clamped(1920, 1080);
82    let mut out = Vec::new();
83    for target in scap::get_all_targets() {
84        let info = match target {
85            scap::Target::Display(d) => ScreenSourceInfo {
86                source: ScreenSource::Display(DisplayId::new(d.id.to_string())),
87                title: if d.title.is_empty() {
88                    format!("Display {}", d.id)
89                } else {
90                    d.title
91                },
92                app_name: None,
93                dimensions: placeholder,
94                is_on_screen: true,
95            },
96            scap::Target::Window(w) => ScreenSourceInfo {
97                source: ScreenSource::Window(WindowId::new(w.id.to_string())),
98                title: if w.title.is_empty() {
99                    format!("Window {}", w.id)
100                } else {
101                    w.title
102                },
103                app_name: None,
104                dimensions: placeholder,
105                is_on_screen: true,
106            },
107        };
108        out.push(info);
109    }
110    Ok(out)
111}

113/// One-shot RGBA thumbnail of a source for the share picker, aspect-fit within
114/// (`max_w`, `max_h`). The scap analogue of the SCK `capture_thumbnail`: open the
115/// target at a low fps, pull the first delivered frame, stop. Cross-platform via
116/// the same scap surface the live share uses — so Windows (WGC) gets dialog-free
117/// previews like macOS.
118///
119/// NOTE (Linux): scap routes through xdg-desktop-portal, which owns source
120/// selection — a per-source thumbnail open may prompt, so callers treat any error
121/// here as "no thumbnail" and degrade to a title-only card (the picker already
122/// does). Best-effort by design.
123///
124/// Blocking (drives scap's `get_next_frame`); call off the async runtime. Bounds
125/// the open to a short window so a source that never delivers a frame can't hang
126/// the picker — it returns a `Device` error instead.
127pub(crate) fn capture_thumbnail(
128    source: &ScreenSource,
129    max_w: u32,
130    max_h: u32,
131) -> Result<ThumbnailImage, CaptureError> {
132    if !scap::is_supported() {
133        return Err(CaptureError::Unsupported {
134            what: "screen capture",
135        });
136    }
137    if !scap::has_permission() {
138        return Err(CaptureError::PermissionDenied {
139            kind: crate::types::DeviceKind::Screen,
140        });
141    }
142    let target = resolve_target(source)?;
143    let options = Options {
144        fps: 10, // a few fps so the first frame lands quickly; we take one.
145        target: Some(target),
146        show_cursor: false,
147        show_highlight: false,
148        output_type: FrameType::BGRAFrame,
149        output_resolution: Resolution::Captured, // native; we fit/scale below
150        ..Default::default()
151    };
152    let mut capturer = Capturer::build(options).map_err(map_build_error)?;
153    capturer.start_capture();
154
155    // Take the first BGRA-family frame; skip the occasional non-BGRA frame, and
156    // bound the wait so an undeliverable source fails fast rather than hanging.
157    let deadline = Instant::now() + Duration::from_secs(3);
158    let thumb = loop {
159        if Instant::now() >= deadline {
160            break Err(CaptureError::Device(
161                "screen thumbnail timed out (no frame delivered)".into(),
162            ));
163        }
164        let frame = match capturer.get_next_frame() {
165            Ok(f) => f,
166            Err(_) => {
167                break Err(CaptureError::Device(
168                    "screen thumbnail capture ended before a frame".into(),
169                ));
170            }
171        };
172        let (data, w, h): (Vec<u8>, i32, i32) = match frame {
173            Frame::BGRA(f) => (f.data, f.width, f.height),
174            Frame::BGRx(f) => (f.data, f.width, f.height),
175            Frame::BGR0(f) => (f.data, f.width, f.height),
176            _ => continue,
177        };
178        if w <= 0 || h <= 0 {
179            continue;
180        }
181        match bgra_to_rgba_thumbnail(&data, w as usize, h as usize, max_w, max_h) {
182            Some(t) => break Ok(t),
183            None => continue,
184        }
185    };
186
187    capturer.stop_capture();
188    thumb
189}

191/// BGRA(/BGRx/BGR0) → opaque RGBA, bilinear-downscaled to aspect-fit
192/// (`max_w`, `max_h`). Resizes as 4-channel (order-agnostic) then swaps B↔R and
193/// forces alpha opaque. Returns `None` on a malformed/short buffer.
194fn bgra_to_rgba_thumbnail(
195    bgra: &[u8],
196    w: usize,
197    h: usize,
198    max_w: u32,
199    max_h: u32,
200) -> Option<ThumbnailImage> {
201    use fast_image_resize::images::Image;
202    use fast_image_resize::{PixelType, ResizeAlg, ResizeOptions, Resizer};
203
204    if w == 0 || h == 0 || bgra.len() < w * h * 4 {
205        return None;
206    }
207    let (dw, dh) = screen_filter::fit_even(w as u32, h as u32, max_w.max(2), max_h.max(2));
208    let (dw, dh) = (dw.max(2), dh.max(2));
209
210    let mut src_buf = bgra[..w * h * 4].to_vec();
211    let src = Image::from_slice_u8(w as u32, h as u32, &mut src_buf, PixelType::U8x4).ok()?;
212    let mut dst = Image::new(dw, dh, PixelType::U8x4);
213    Resizer::new()
214        .resize(
215            &src,
216            &mut dst,
217            &ResizeOptions::new().resize_alg(ResizeAlg::Convolution(
218                fast_image_resize::FilterType::Bilinear,
219            )),
220        )
221        .ok()?;
222
223    let mut rgba = dst.into_vec();
224    for px in rgba.chunks_exact_mut(4) {
225        px.swap(0, 2); // BGRA → RGBA
226        px[3] = 0xFF; // opaque (BGRx/BGR0 carry no real alpha)
227    }
228    Some(ThumbnailImage {
229        width: dw,
230        height: dh,
231        rgba,
232    })
233}

235/// Resolve our typed [`ScreenSource`] token (the stringified scap id) back to a
236/// live scap [`Target`](scap::Target) by re-enumerating and matching the id.
237fn resolve_target(source: &ScreenSource) -> Result<scap::Target, CaptureError> {
238    let want: &str = match source {
239        ScreenSource::Display(id) => id.as_str(),
240        ScreenSource::Window(id) => id.as_str(),
241        // A camera source is opened as a camera (see `Engine::open_screen_capture`),
242        // never resolved to a scap target.
243        ScreenSource::Camera(_) => {
244            return Err(CaptureError::Unsupported {
245                what: "camera source is not a scap screen target",
246            });
247        }
248    };
249    let is_display = matches!(source, ScreenSource::Display(_));
250    scap::get_all_targets()
251        .into_iter()
252        .find(|t| match (t, is_display) {
253            (scap::Target::Display(d), true) => d.id.to_string() == want,
254            (scap::Target::Window(w), false) => w.id.to_string() == want,
255            _ => false,
256        })
257        .ok_or_else(|| {
258            CaptureError::NoDevice(format!("screen source {want} is no longer available"))
259        })
260}

262/// A running scap screen capture: a bridge thread owns the scap `Capturer` and
263/// fills the ring; this handle pulls I420 frames from the ring. `!Send`-by-use
264/// (drive and drop on one capture/encode thread, like every backend).
265pub(crate) struct ScapScreenCapturer {
266    ring: Arc<BoundedRing<RawFrame>>,
267    stop: Arc<AtomicBool>,
268    bridge: Option<JoinHandle<()>>,
269    encode_target: Dimensions,
270    /// Latest delivered frame size (points), packed `(w << 32) | h`; the resize
271    /// signal — a running scap stream keeps one geometry, so a change means the
272    /// window/display was resized (the session restarts the share).
273    content_size: Arc<AtomicU64>,
274}
275
276impl ScapScreenCapturer {
277    /// Open `source` and begin delivery. The encode target is the source's
278    /// native size aspect-fit into `config.dimensions` (a 1080p-class cap).
279    /// Returns `(capturer, None)` — video only (Linux).
280    pub(crate) fn open(
281        source: &ScreenSource,
282        config: ScreenCaptureConfig,
283        clock: hound_media::CaptureClock,
284    ) -> Result<(Self, Option<ScapAppAudio>), CaptureError> {
285        let ring = BoundedRing::<RawFrame>::new(RING_FRAMES);
286        let stop = Arc::new(AtomicBool::new(false));
287        let content_size = Arc::new(AtomicU64::new(0));
288        let cap = config.dimensions;
289        // Only `Send` data crosses to the bridge thread — the scap `Target` (and
290        // the `Capturer` built from it) are resolved and owned entirely there,
291        // so their platform handles never move across threads.
292        let source = source.clone();
293        let fps = config.fps.get();
294
295        // Build + run scap on the bridge thread. It reports the negotiated
296        // encode target (or the open / resolve error) back through `ready`.
297        let (ready_tx, ready_rx) = std::sync::mpsc::channel::<Result<Dimensions, CaptureError>>();
298        let bridge = std::thread::Builder::new()
299            .name("scap-screen-bridge".into())
300            .spawn({
301                let ring = Arc::clone(&ring);
302                let stop = Arc::clone(&stop);
303                let content_size = Arc::clone(&content_size);
304                move || bridge_loop(source, fps, cap, clock, ring, stop, content_size, ready_tx)
305            })
306            .map_err(|e| CaptureError::Device(format!("scap bridge thread spawn failed: {e}")))?;
307
308        // Wait for the open result. A dropped sender (thread panicked before
309        // reporting) is a clean device error.
310        let encode_target = match ready_rx.recv() {
311            Ok(Ok(dims)) => dims,
312            Ok(Err(e)) => {
313                let _ = bridge.join();
314                return Err(e);
315            }
316            Err(_) => {
317                let _ = bridge.join();
318                return Err(CaptureError::Device(
319                    "scap capture thread exited before reporting open".into(),
320                ));
321            }
322        };
323
324        Ok((
325            Self {
326                ring,
327                stop,
328                bridge: Some(bridge),
329                encode_target,
330                content_size,
331            },
332            None,
333        ))
334    }
335}

337/// The bridge thread body: resolve the target, build the capturer, report the
338/// encode target, then drain frames → I420 → scaled → ring until stopped or the
339/// source ends. Everything scap-owned lives on this thread.
340#[allow(clippy::too_many_arguments)]
341fn bridge_loop(
342    source: ScreenSource,
343    fps: u32,
344    cap: Dimensions,
345    clock: hound_media::CaptureClock,
346    ring: Arc<BoundedRing<RawFrame>>,
347    stop: Arc<AtomicBool>,
348    content_size: Arc<AtomicU64>,
349    ready_tx: std::sync::mpsc::Sender<Result<Dimensions, CaptureError>>,
350) {
351    let target = match resolve_target(&source) {
352        Ok(t) => t,
353        Err(e) => {
354            let _ = ready_tx.send(Err(e));
355            return;
356        }
357    };
358    let options = Options {
359        fps,
360        target: Some(target),
361        show_cursor: true,
362        show_highlight: false,
363        output_type: FrameType::BGRAFrame,
364        output_resolution: Resolution::Captured, // native; we fit/scale below
365        ..Default::default()
366    };
367    let mut capturer = match Capturer::build(options) {
368        Ok(c) => c,
369        Err(e) => {
370            let _ = ready_tx.send(Err(map_build_error(e)));
371            return;
372        }
373    };
374    capturer.start_capture();
375
376    // Native size → aspect-fit encode target (the cap is 1080p-class).
377    let [nw, nh] = capturer.get_output_frame_size();
378    let (tw, th) = screen_filter::fit_even(nw.max(2), nh.max(2), cap.width(), cap.height());
379    let encode_target = Dimensions::even_clamped(tw, th);
380    if ready_tx.send(Ok(encode_target)).is_err() {
381        // The opener gave up; nothing to capture for.
382        capturer.stop_capture();
383        return;
384    }
385
386    while !stop.load(Ordering::Relaxed) {
387        let frame = match capturer.get_next_frame() {
388            Ok(f) => f,
389            // The frame channel closed: scap stopped (window closed / stream
390            // ended). Clean capture end.
391            Err(_) => break,
392        };
393        if let Some(raw) = convert_frame(frame, encode_target, &clock, &content_size, &ring) {
394            ring.push(raw);
395        }
396    }
397
398    capturer.stop_capture();
399    ring.stop();
400}

402/// Convert one scap frame to a scaled I420 [`RawFrame`]. BGRA/BGRx/BGR0 take the
403/// alpha-ignoring path; anything else is counted as a bad frame and dropped.
404fn convert_frame(
405    frame: Frame,
406    encode_target: Dimensions,
407    clock: &hound_media::CaptureClock,
408    content_size: &AtomicU64,
409    ring: &BoundedRing<RawFrame>,
410) -> Option<RawFrame> {
411    // (data, width, height) for the BGRA-family variants; `None` otherwise.
412    let (data, w, h): (Vec<u8>, i32, i32) = match frame {
413        Frame::BGRA(f) => (f.data, f.width, f.height),
414        Frame::BGRx(f) => (f.data, f.width, f.height),
415        Frame::BGR0(f) => (f.data, f.width, f.height),
416        _ => {
417            let n = ring.record_bad_frame();
418            if n <= 3 || n.is_power_of_two() {
419                tracing::warn!(
420                    target: "scap_screen",
421                    "scap delivered an unexpected (non-BGRA) frame type; dropped (bad={n})"
422                );
423            }
424            return None;
425        }
426    };
427    if w <= 0 || h <= 0 {
428        return None;
429    }
430    let (w, h) = (w as usize, h as usize);
431    // Track the source size (resize signal). Even-floor so the comparison is
432    // stable against a 1px jitter that an odd size would force.
433    let (ew, eh) = (w as u32 & !1, h as u32 & !1);
434    content_size.store((u64::from(ew) << 32) | u64::from(eh), Ordering::Relaxed);
435
436    // BGRA → I420 at the source size (alpha ignored; tightly packed stride).
437    let i420 =
438        convert::bgra_to_i420(&data, ew as usize, eh as usize, ew as usize * 4).or_else(|| {
439            let n = ring.record_bad_frame();
440            if n <= 3 || n.is_power_of_two() {
441                tracing::warn!(
442                    target: "scap_screen",
443                    "scap BGRA→I420 conversion failed for {ew}x{eh}; dropped (bad={n})"
444                );
445            }
446            None
447        })?;
448
449    // Aspect-fit scale to the encode target (no-op when already equal).
450    let (data, dims) = if (ew, eh) == (encode_target.width(), encode_target.height()) {
451        (i420, encode_target)
452    } else {
453        match convert::scale_i420(
454            &i420,
455            ew as usize,
456            eh as usize,
457            encode_target.width() as usize,
458            encode_target.height() as usize,
459        ) {
460            Some(scaled) => (scaled, encode_target),
461            None => return None,
462        }
463    };
464
465    Some(RawFrame {
466        dims,
467        format: PixelFormat::I420,
468        timestamp_us: TimestampUs::from_micros(clock.now()),
469        captured_at: Instant::now(),
470        data,
471        surface: None,
472    })
473}

475/// Map scap's build error to a typed [`CaptureError`].
476fn map_build_error(e: scap::capturer::CapturerBuildError) -> CaptureError {
477    use scap::capturer::CapturerBuildError;
478    match e {
479        CapturerBuildError::PermissionNotGranted => CaptureError::PermissionDenied {
480            kind: crate::types::DeviceKind::Screen,
481        },
482        CapturerBuildError::NotSupported => CaptureError::Unsupported {
483            what: "screen capture",
484        },
485    }
486}

488impl sealed::Sealed for ScapScreenCapturer {}

490impl ScreenCapture for ScapScreenCapturer {
491    fn next_frame(&mut self) -> Result<RawFrame, CaptureError> {
492        self.ring
493            .pop_blocking()
494            .ok_or_else(|| CaptureError::Device("screen share ended (no more frames)".into()))
495    }
496
497    fn next_frame_timeout(&mut self, timeout: Duration) -> Result<Option<RawFrame>, CaptureError> {
498        match self.ring.pop_timeout(timeout) {
499            PopWait::Frame(f) => Ok(Some(f)),
500            PopWait::Idle => Ok(None),
501            PopWait::Ended => Err(CaptureError::Device(
502                "screen share ended (no more frames)".into(),
503            )),
504        }
505    }
506
507    fn content_size(&self) -> Option<(u32, u32)> {
508        let packed = self.content_size.load(Ordering::Relaxed);
509        (packed != 0).then(|| ((packed >> 32) as u32, packed as u32))
510    }
511
512    fn negotiated(&self) -> (Dimensions, PixelFormat) {
513        (self.encode_target, PixelFormat::I420)
514    }
515}
516
517impl Drop for ScapScreenCapturer {
518    fn drop(&mut self) {
519        // Signal the bridge to stop, wake any blocked pull, and join so the
520        // scap engine is fully torn down on its owning thread.
521        self.stop.store(true, Ordering::Relaxed);
522        self.ring.stop();
523        if let Some(h) = self.bridge.take() {
524            let _ = h.join();
525        }
526    }
