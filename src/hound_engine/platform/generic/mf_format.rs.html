1//! # Media Foundation capture-subtype selection (pure, platform-agnostic)
2//!
3//! The Windows analogue of [`crate::platform::generic::camera_format`]: the
4//! preference ordering the Media Foundation camera backend uses to pick which
5//! video subtype to negotiate on an `IMFSourceReader`, expressed purely over
6//! the subtype's 16-byte GUID **data** so the ranking — the part most likely to
7//! harbour a "we picked a format we can't convert" bug — is unit-tested on
8//! every host, INCLUDING this macOS dev box where the Media Foundation capturer
9//! itself cannot compile.
10//!
11//! ## Why GUID bytes, not the `windows` crate's `GUID` type
12//!
13//! Keeping this module free of any `windows` type is what lets it compile and
14//! be tested off-Windows. The Media Foundation video subtype GUIDs all share
15//! the fixed tail `00000000-0010-8000-00AA-00389B71` (the "MFVideoFormat"
16//! family — a FourCC packed into `Data1`), so a subtype is fully identified by
17//! its `Data1` FourCC. We match on the FourCC bytes here and the Windows module
18//! maps the chosen [`MfSubtype`] back to the concrete `windows` GUID constant
19//! (`MFVideoFormat_NV12`, …) at the one call site.
20//!
21//! ## Preference order (best first)
22//!
23//! 1. **NV12** — biplanar 4:2:0, the format MF camera capture and the MFT
24//!    H.264 encoder both speak natively; the cheapest path to tightly-packed
25//!    I420 (a plane de-interleave, no colour math), reusing the exact
26//!    `nv12_bytes_to_i420` the AVFoundation/V4L2 paths use.
27//! 2. **YUY2** — packed 4:2:2 (the Windows name for YUYV), the near-universal
28//!    UVC webcam format; a repack + chroma reduction to I420 with no decoder
29//!    (the same `yuyv_to_i420`).
30//!
31//! MJPEG and other compressed subtypes are intentionally NOT ranked: unlike the
32//! V4L2 path (where the kernel hands a `MmapStream` of whatever raw format), the
33//! MF source reader can be asked to deliver NV12/YUY2 directly and will insert
34//! its own decoder/converter MFT for most cameras, so we always request a raw
35//! subtype and let MF satisfy it. A camera MF cannot serve as NV12 *or* YUY2
36//! surfaces as a typed `UnsupportedFormat` at open, never a silent path.
37
38// Pure logic consumed by the Windows MF capturer (cfg(windows)); on non-Windows
39// the only consumer is the test module below, so the items read as dead — allow
40// it rather than gate, exactly so the ranking is provable on this dev box.
41#![cfg_attr(not(target_os = "windows"), allow(dead_code))]
42
43/// A Media Foundation video subtype the engine knows how to turn into I420,
44/// ordered by how cheaply it converts (the `Ord` derive ranks `Nv12 < Yuy2`,
45/// i.e. NV12 is "smallest"/best — preferred = the minimum, mirroring
46/// [`crate::platform::generic::camera_format::CaptureFormat`]).
47#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
48pub(crate) enum MfSubtype {
49    /// Biplanar YUV 4:2:0 (`MFVideoFormat_NV12`). Best: a plane de-interleave
50    /// to I420, no colour conversion — the same `nv12_bytes_to_i420` path the
51    /// AVFoundation and V4L2 capturers use.
52    Nv12,
53    /// Packed YUV 4:2:2 (`MFVideoFormat_YUY2`, the Windows name for YUYV). The
54    /// common UVC format: a repack + chroma reduction to I420 via `yuyv_to_i420`.
55    Yuy2,
56}
57
58impl MfSubtype {
59    /// The subtype's `Data1` FourCC — the 4 bytes (little-endian within the
60    /// GUID's `Data1`, i.e. the ASCII FourCC in memory order) that identify it
61    /// within the `MFVideoFormat` GUID family. The Windows module compares these
62    /// against an enumerated media type's subtype `Data1` to recognise it, and
63    /// maps the chosen variant to the concrete `windows` GUID constant.
64    pub(crate) const fn fourcc(self) -> [u8; 4] {
65        match self {
66            MfSubtype::Nv12 => *b"NV12",
67            MfSubtype::Yuy2 => *b"YUY2",
68        }
69    }
70
71    /// Recognise a subtype by its FourCC (the GUID `Data1`, read as 4 ASCII
72    /// bytes in memory order). `None` for subtypes the engine cannot convert
73    /// directly (so they are never selected — MF is asked for a raw subtype and
74    /// inserts its own decoder for compressed cameras).
75    pub(crate) fn from_fourcc(fourcc: [u8; 4]) -> Option<Self> {
76        match &fourcc {
77            b"NV12" => Some(MfSubtype::Nv12),
78            b"YUY2" => Some(MfSubtype::Yuy2),
79            _ => None,
80        }
81    }
82}
83
84/// Pick the best subtype to *request* from the source reader, applying the
85/// preference order (NV12 > YUY2). This is the request preference, independent
86/// of what the camera natively offers: MF will convert most cameras to the
87/// requested raw subtype via an inserted MFT. The Windows backend tries these
88/// in order and takes the first the reader accepts.
89///
90/// Returned as a fixed slice (best first) so the caller iterates request
91/// attempts deterministically.
92pub(crate) const fn request_order() -> [MfSubtype; 2] {
93    [MfSubtype::Nv12, MfSubtype::Yuy2]
94}
95
96/// Given the FourCCs a camera's native media types advertise, pick the best one
97/// we can convert — the fallback used when the reader will not transcode to a
98/// requested subtype and we must instead accept a native one. Mirrors
99/// [`crate::platform::generic::camera_format::select_capture_format`]: ignores
100/// unrecognised subtypes, `None` when nothing is convertible.
101pub(crate) fn select_native_subtype(offered: &[[u8; 4]]) -> Option<MfSubtype> {
102    offered
103        .iter()
104        .filter_map(|&fourcc| MfSubtype::from_fourcc(fourcc))
105        // `Ord` makes NV12 the minimum (best); pick the most-preferred offered.
106        .min()
107}
108
109#[cfg(test)]
110mod tests {
111    use super::*;
112
113    #[test]
114    fn request_order_is_nv12_then_yuy2() {
115        assert_eq!(request_order(), [MfSubtype::Nv12, MfSubtype::Yuy2]);
116    }
117
118    #[test]
119    fn fourcc_round_trips() {
120        for sub in [MfSubtype::Nv12, MfSubtype::Yuy2] {
121            assert_eq!(MfSubtype::from_fourcc(sub.fourcc()), Some(sub));
122        }
123    }
124
125    #[test]
126    fn nv12_outranks_yuy2() {
127        // The `Ord` derive: NV12 is the minimum (best).
128        assert!(MfSubtype::Nv12 < MfSubtype::Yuy2);
129    }
130
131    #[test]
132    fn select_native_prefers_nv12_then_yuy2_ignores_rest() {
133        // Both offered → NV12 wins.
134        assert_eq!(
135            select_native_subtype(&[*b"YUY2", *b"NV12"]),
136            Some(MfSubtype::Nv12)
137        );
138        // NV12 absent → YUY2.
139        assert_eq!(select_native_subtype(&[*b"YUY2"]), Some(MfSubtype::Yuy2));
140        // Unrecognised (e.g. MJPG / RGB24) ignored; nothing convertible → None.
141        assert_eq!(select_native_subtype(&[*b"MJPG", *b"RGB3"]), None);
142        // One convertible among junk → that one.
143        assert_eq!(
144            select_native_subtype(&[*b"MJPG", *b"YUY2", *b"H264"]),
145            Some(MfSubtype::Yuy2)
146        );
147        // Empty → None.
148        assert_eq!(select_native_subtype(&[]), None);
149    }
150}
