1//! # Annex-B parameter-set detection (H.264 / HEVC), pure
2//!
3//! The HW encoders must honour the wire contract the decoders depend on: **every
4//! keyframe carries its in-band parameter sets** (H.264 SPS+PPS, HEVC
5//! VPS+SPS+PPS), so a fresh receiver bootstraps its decoder from a keyframe alone
6//! (see `apple::vt_decoder` / `windows::mft` — a keyframe without parameter sets
7//! makes the decoder return `Ok(None)` forever → a permanent black tile).
8//!
9//! Some encoder Media Foundation Transforms omit the parameter sets on the IDR
10//! sample and expose them only once via `MF_MT_MPEG_SEQUENCE_HEADER`. This module
11//! is the pure detection that lets the encoder decide when to prepend its cached
12//! sequence header — unit-tested on every host, no platform dependency.
13
14// Only the Windows MFT encoder consumes this at runtime today; the Linux VAAPI
15// encoder is the intended second consumer. On Linux the module is compiled (and
16// its tests run) ahead of that, so it reads as dead to the lib-only build.
17#![cfg_attr(not(target_os = "windows"), allow(dead_code))]
18
19use crate::types::VideoCodec;

21/// Does the Annex-B `stream` begin with a start code (`00 00 01` or
22/// `00 00 00 01`)? Used to tell an Annex-B elementary stream from a
23/// length-prefixed (AVCC) one.
24pub(crate) fn starts_with_start_code(stream: &[u8]) -> bool {
25    stream.starts_with(&[0, 0, 0, 1]) || stream.starts_with(&[0, 0, 1])
26}
27
28/// Collect the NAL-unit **type** of each NAL in an Annex-B `stream`, in order.
29/// Handles both 3-byte (`00 00 01`) and 4-byte (`00 00 00 01`) start codes. The
30/// type encoding is codec-specific: H.264 is `nal[0] & 0x1f`; HEVC is the 6-bit
31/// `(nal[0] >> 1) & 0x3f`. AV1 is not an Annex-B codec and yields nothing.
32fn nal_types(stream: &[u8], codec: VideoCodec) -> Vec<u8> {
33    let mut types = Vec::new();
34    if matches!(codec, VideoCodec::Av1) {
35        return types;
36    }
37    let mut i = 0usize;
38    while i + 3 <= stream.len() {
39        let start_len = if stream[i] == 0 && stream[i + 1] == 0 && stream[i + 2] == 1 {
40            Some(3)
41        } else if i + 4 <= stream.len()
42            && stream[i] == 0
43            && stream[i + 1] == 0
44            && stream[i + 2] == 0
45            && stream[i + 3] == 1
46        {
47            Some(4)
48        } else {
49            None
50        };
51        match start_len {
52            Some(len) => {
53                if let Some(&first) = stream.get(i + len) {
54                    types.push(match codec {
55                        VideoCodec::H264 => first & 0x1f,
56                        VideoCodec::H265 => (first >> 1) & 0x3f,
57                        VideoCodec::Av1 => 0,
58                    });
59                }
60                i += len;
61            }
62            None => i += 1,
63        }
64    }
65    types
66}

68/// Does this Annex-B access unit already carry the codec's **sequence parameter
69/// set** (H.264 SPS = NAL 7; HEVC SPS = NAL 33)? Encoders emit SPS together with
70/// the other parameter sets (PPS, and VPS for HEVC), so the SPS is the proxy for
71/// "the whole parameter set is in-band here". AV1 carries its config differently
72/// (sequence-header OBU) and is reported as already-present (this path is unused).
73pub(crate) fn contains_parameter_sets(stream: &[u8], codec: VideoCodec) -> bool {
74    match codec {
75        VideoCodec::H264 => nal_types(stream, codec).contains(&7),
76        VideoCodec::H265 => nal_types(stream, codec).contains(&33),
77        VideoCodec::Av1 => true,
78    }
79}

81/// Is `nal_type` a VCL (slice / picture-data) NAL for `codec`? Parameter sets and
82/// SEI are non-VCL; everything that carries coded picture data is VCL.
83fn is_vcl(nal_type: u8, codec: VideoCodec) -> bool {
84    match codec {
85        // H.264: VCL is NAL types 1..=5 (non-IDR, partitions, IDR).
86        VideoCodec::H264 => (1..=5).contains(&nal_type),
87        // HEVC: VCL is NAL types 0..=31.
88        VideoCodec::H265 => nal_type <= 31,
89        VideoCodec::Av1 => false,
90    }
91}
92
93/// Byte offset of the start code of the first VCL NAL in an Annex-B `stream`, i.e.
94/// where the coded slice begins. Everything before it in an IDR access unit is the
95/// parameter-set prefix (VPS/SPS/PPS, plus any SEI) — used to cache the sequence
96/// header. `None` if there is no VCL NAL (a parameter-set-only AU).
97pub(crate) fn first_vcl_offset(stream: &[u8], codec: VideoCodec) -> Option<usize> {
98    if matches!(codec, VideoCodec::Av1) {
99        return None;
100    }
101    let mut i = 0usize;
102    while i + 3 <= stream.len() {
103        let start_len = if stream[i] == 0 && stream[i + 1] == 0 && stream[i + 2] == 1 {
104            Some(3)
105        } else if i + 4 <= stream.len()
106            && stream[i] == 0
107            && stream[i + 1] == 0
108            && stream[i + 2] == 0
109            && stream[i + 3] == 1
110        {
111            Some(4)
112        } else {
113            None
114        };
115        match start_len {
116            Some(len) => {
117                if let Some(&first) = stream.get(i + len) {
118                    let nal_type = match codec {
119                        VideoCodec::H264 => first & 0x1f,
120                        VideoCodec::H265 => (first >> 1) & 0x3f,
121                        VideoCodec::Av1 => 0,
122                    };
123                    if is_vcl(nal_type, codec) {
124                        return Some(i);
125                    }
126                }
127                i += len;
128            }
129            None => i += 1,
130        }
131    }
132    None
133}

135#[cfg(test)]
136mod tests {
137    use super::*;
138
139    #[test]
140    fn detects_start_codes() {
141        assert!(starts_with_start_code(&[0, 0, 0, 1, 0x67]));
142        assert!(starts_with_start_code(&[0, 0, 1, 0x67]));
143        assert!(!starts_with_start_code(&[0, 0, 0, 5, 0x67])); // AVCC length prefix
144        assert!(!starts_with_start_code(&[0x67]));
145    }
146
147    #[test]
148    fn h264_sps_detected_both_start_code_widths() {
149        // 4-byte start + SPS (NAL 7), then 3-byte start + IDR slice (NAL 5).
150        let au = [0, 0, 0, 1, 0x67, 0x42, 0, 0, 1, 0x65, 0xAA];
151        assert!(contains_parameter_sets(&au, VideoCodec::H264));
152    }
153
154    #[test]
155    fn h264_idr_without_sps_is_missing_params() {
156        // IDR slice (NAL 5) only — no SPS.
157        let au = [0, 0, 0, 1, 0x65, 0xAA, 0xBB];
158        assert!(!contains_parameter_sets(&au, VideoCodec::H264));
159    }
160
161    #[test]
162    fn hevc_sps_detected() {
163        // HEVC SPS = NAL type 33 → first header byte (33 << 1) = 0x42.
164        let au = [0, 0, 0, 1, 0x42, 0x01, 0, 0, 0, 1, 0x26, 0x01]; // SPS then IDR_W_RADL(19)→0x26
165        assert!(contains_parameter_sets(&au, VideoCodec::H265));
166    }
167
168    #[test]
169    fn hevc_vcl_only_is_missing_params() {
170        // HEVC IDR_W_RADL slice (NAL 19) → 0x26; no SPS (33).
171        let au = [0, 0, 0, 1, 0x26, 0x01, 0xAA];
172        assert!(!contains_parameter_sets(&au, VideoCodec::H265));
173    }
