1//! # AV1 OBU parsing + `av1C` configuration-record synthesis
2//!
3//! VideoToolbox can decode AV1 in hardware (A17 Pro / M3 and later), but unlike
4//! H.264/HEVC there is **no** `CMVideoFormatDescriptionCreateFromAV1ParameterSets`
5//! in objc2-core-media 0.3.2. The AV1 format description must instead be built
6//! with the generic `CMVideoFormatDescriptionCreate`, carrying an **`av1C`**
7//! configuration-record (the AV1 Codec ISO-BMFF binding, [`AV1CodecConfigurationRecord`])
8//! as a sample-description extension atom. This module is the codec-only half of
9//! that: it has **no** VideoToolbox / objc2 dependency, so it is unit-testable on
10//! any host and shared cleanly with the decoder.
11//!
12//! ## What we parse, and what we deliberately don't
13//! AV1's "low-overhead bitstream format" (what `rav1e` emits, and what goes on the
14//! MoQ wire) is a sequence of OBUs:
15//!
16//! ```text
17//!  obu_header + (optional) leb128 obu_size + payload
18//! ```
19//!
20//! To synthesise an `av1C` we need exactly the fields the config record
21//! carries, all of which live near the **front** of the sequence-header OBU:
22//! `seq_profile`, `seq_level_idx[0]`, `seq_tier[0]`, the bit-depth flags
23//! (`high_bitdepth`/`twelve_bit`), `mono_chrome`, the chroma subsampling flags, and
24//! `chroma_sample_position`. We parse just far enough to reach those — through the
25//! (skippable) `timing_info` / `decoder_model_info` / `operating_points` preamble —
26//! and stop. We do **not** decode frames; libaom/dav1d/VideoToolbox does that.
27//!
28//! ## `av1C` layout (AV1-ISOBMFF §2.3.3, `AV1CodecConfigurationRecord`)
29//! ```text
30//! marker(1)=1, version(7)=1
31//! seq_profile(3), seq_level_idx_0(5)
32//! seq_tier_0(1), high_bitdepth(1), twelve_bit(1), monochrome(1),
33//!   chroma_subsampling_x(1), chroma_subsampling_y(1), chroma_sample_position(2)
34//! reserved(3)=0, initial_presentation_delay_present(1)=0,
35//!   initial_presentation_delay_minus_one(4) OR reserved(4)=0
36//! configOBUs[]  // the sequence-header OBU, verbatim (header + size + payload)
37//! ```
38//! VideoToolbox wants this atom under the `"av1C"` key inside
39//! `kCMFormatDescriptionExtension_SampleDescriptionExtensionAtoms`.
40
41/// OBU type for the sequence header (`OBU_SEQUENCE_HEADER`).
42const OBU_SEQUENCE_HEADER: u8 = 1;
43/// OBU type for a temporal delimiter (`OBU_TEMPORAL_DELIMITER`) — zero-length
44/// payload, marks the start of a temporal unit.
45const OBU_TEMPORAL_DELIMITER: u8 = 2;
46
47/// One OBU located inside an AV1 bitstream, as a borrowed slice of the *whole*
48/// OBU (header + size field + payload) plus its decoded type. Keeping the whole
49/// OBU lets us copy the sequence header verbatim into the `av1C` `configOBUs`.
50#[derive(Debug, Clone, Copy)]
51pub struct Obu<'a> {
52    pub obu_type: u8,
53    /// The complete OBU bytes (header + leb128 size + payload), exactly as they
54    /// appear in the stream — what `av1C.configOBUs` must contain verbatim.
55    pub whole: &'a [u8],
56    /// Just the payload (after header + size field). Used to parse the sequence
57    /// header fields for the `av1C` record.
58    pub payload: &'a [u8],
59}
60
61/// The fields extracted from a sequence-header OBU that the `av1C` record needs.
62#[derive(Debug, Clone, Copy, PartialEq, Eq)]
63pub struct SeqHeaderConfig {
64    pub seq_profile: u8,
65    pub seq_level_idx_0: u8,
66    pub seq_tier_0: u8,
67    pub high_bitdepth: u8,
68    pub twelve_bit: u8,
69    pub monochrome: u8,
70    pub chroma_subsampling_x: u8,
71    pub chroma_subsampling_y: u8,
72    pub chroma_sample_position: u8,
73}

75/// A minimal MSB-first bit reader over a byte slice. AV1 syntax is big-endian
76/// bit-packed (`f(n)` reads n bits MSB-first); this reads exactly that. Every
77/// read is bounds-checked and returns `None` past the end rather than panicking
78/// — the parser runs on attacker-influenced wire data.
79struct BitReader<'a> {
80    data: &'a [u8],
81    bit_pos: usize,
82}
83
84impl<'a> BitReader<'a> {
85    fn new(data: &'a [u8]) -> Self {
86        Self { data, bit_pos: 0 }
87    }
88
89    /// Read a single bit (`f(1)`).
90    fn bit(&mut self) -> Option<u8> {
91        let byte = self.bit_pos / 8;
92        if byte >= self.data.len() {
93            return None;
94        }
95        let shift = 7 - (self.bit_pos % 8);
96        let v = (self.data[byte] >> shift) & 1;
97        self.bit_pos += 1;
98        Some(v)
99    }
100
101    /// Read `n` bits MSB-first into a u32 (`f(n)`, n ≤ 32).
102    fn f(&mut self, n: u32) -> Option<u32> {
103        let mut v = 0u32;
104        for _ in 0..n {
105            v = (v << 1) | self.bit()? as u32;
106        }
107        Some(v)
108    }
109
110    /// `uvlc()` — unsigned variable-length code (Exp-Golomb-ish), used by
111    /// `timing_info` for `num_units_in_display_tick` etc. We only need to *skip*
112    /// these, but reading them correctly keeps the bit cursor aligned.
113    fn uvlc(&mut self) -> Option<u32> {
114        let mut leading_zeros = 0u32;
115        loop {
116            let done = self.bit()?;
117            if done == 1 {
118                break;
119            }
120            leading_zeros += 1;
121            if leading_zeros >= 32 {
122                return Some(u32::MAX);
123            }
124        }
125        let value = self.f(leading_zeros)?;
126        Some(value + (1u32 << leading_zeros) - 1)
127    }
128}

130/// Decode an unsigned LEB128 value at `data[pos..]`, returning `(value, bytes_read)`.
131/// AV1 uses LEB128 for `obu_size` and the temporal-unit/frame-unit framing of the
132/// length-delimited "Annex-B"-ish variants. Bounded to 8 bytes per the spec.
133fn read_leb128(data: &[u8], pos: usize) -> Option<(u64, usize)> {
134    let mut value = 0u64;
135    let mut i = 0usize;
136    loop {
137        if i >= 8 || pos + i >= data.len() {
138            return None;
139        }
140        let b = data[pos + i];
141        value |= ((b & 0x7f) as u64) << (i * 7);
142        i += 1;
143        if b & 0x80 == 0 {
144            break;
145        }
146    }
147    Some((value, i))
148}

150/// Split an AV1 **low-overhead** OBU stream into its OBUs (the format `rav1e`
151/// emits and that travels on our wire). Each OBU is `obu_header` (1–2 bytes) +
152/// optional `leb128 obu_size` + payload. We require `obu_has_size_field` (rav1e
153/// sets it; the low-overhead format mandates it for all but the last), and parse
154/// defensively: a malformed length stops the walk rather than panicking.
155///
156/// Returns the OBUs in stream order. Temporal delimiters are included (the caller
157/// filters), because the *whole* OBU slices are what `av1C`/`configOBUs` and the
158/// CMSampleBuffer payload are built from.
159pub fn parse_obus(stream: &[u8]) -> Vec<Obu<'_>> {
160    let mut obus = Vec::new();
161    let mut pos = 0usize;
162    while pos < stream.len() {
163        let header_start = pos;
164        let first = stream[pos];
165        // obu_header: forbidden(1)=0, type(4), extension_flag(1), has_size_field(1),
166        // reserved(1).
167        let obu_type = (first >> 3) & 0x0f;
168        let extension_flag = (first >> 2) & 1;
169        let has_size_field = (first >> 1) & 1;
170        pos += 1;
171        if extension_flag == 1 {
172            // temporal_id(3), spatial_id(2), reserved(3)
173            if pos >= stream.len() {
174                break;
175            }
176            pos += 1;
177        }
178        let payload_len = if has_size_field == 1 {
179            let (size, n) = match read_leb128(stream, pos) {
180                Some(v) => v,
181                None => break,
182            };
183            pos += n;
184            size as usize
185        } else {
186            // No size field → this OBU runs to the end of the stream (only valid
187            // for the final OBU in low-overhead format).
188            stream.len() - pos
189        };
190        let payload_start = pos;
191        let payload_end = payload_start.checked_add(payload_len);
192        let Some(payload_end) = payload_end else {
193            break;
194        };
195        if payload_end > stream.len() {
196            break; // truncated/malformed — stop rather than read OOB
197        }
198        obus.push(Obu {
199            obu_type,
200            whole: &stream[header_start..payload_end],
201            payload: &stream[payload_start..payload_end],
202        });
203        pos = payload_end;
204    }
205    obus
206}

208/// Find the first sequence-header OBU in a stream, if present.
209pub fn find_sequence_header(stream: &[u8]) -> Option<Obu<'_>> {
210    parse_obus(stream)
211        .into_iter()
212        .find(|o| o.obu_type == OBU_SEQUENCE_HEADER)
213}

215/// Parse the fields the `av1C` record needs out of a sequence-header OBU payload.
216///
217/// This walks the `sequence_header_obu()` syntax (AV1 spec §5.5) only as far as
218/// the color config, skipping the timing/decoder-model/operating-points preamble
219/// correctly so the bit cursor lands on the right fields. Returns `None` on any
220/// truncation or an unsupported shape we can't align past (we then refuse AV1
221/// rather than emit a wrong `av1C`).
222pub fn parse_sequence_header(payload: &[u8]) -> Option<SeqHeaderConfig> {
223    let mut r = BitReader::new(payload);
224
225    let seq_profile = r.f(3)? as u8;
226    let _still_picture = r.bit()?;
227    let reduced_still_picture_header = r.bit()?;
228
229    let mut seq_level_idx_0 = 0u8;
230    let mut seq_tier_0 = 0u8;
231
232    if reduced_still_picture_header == 1 {
233        // timing_info_present=0, decoder_model_info=0, etc.
234        // seq_level_idx[0] = f(5); seq_tier[0] = 0.
235        seq_level_idx_0 = r.f(5)? as u8;
236        seq_tier_0 = 0;
237    } else {
238        let timing_info_present_flag = r.bit()?;
239        let mut decoder_model_info_present_flag = 0u8;
240        let mut buffer_delay_length_minus_1 = 0u32;
241        if timing_info_present_flag == 1 {
242            // timing_info(): num_units_in_display_tick f(32), time_scale f(32),
243            // equal_picture_interval f(1), [if 1] num_ticks_per_picture_minus_1 uvlc.
244            r.f(32)?; // num_units_in_display_tick
245            r.f(32)?; // time_scale
246            let equal_picture_interval = r.bit()?;
247            if equal_picture_interval == 1 {
248                r.uvlc()?; // num_ticks_per_picture_minus_1
249            }
250            decoder_model_info_present_flag = r.bit()?;
251            if decoder_model_info_present_flag == 1 {
252                // decoder_model_info(): buffer_delay_length_minus_1 f(5),
253                // num_units_in_decoding_tick f(32), buffer_removal_time_length_minus_1 f(5),
254                // frame_presentation_time_length_minus_1 f(5).
255                buffer_delay_length_minus_1 = r.f(5)?;
256                r.f(32)?;
257                r.f(5)?;
258                r.f(5)?;
259            }
260        }
261
262        let initial_display_delay_present_flag = r.bit()?;
263        let operating_points_cnt_minus_1 = r.f(5)?;
264        for i in 0..=operating_points_cnt_minus_1 {
265            r.f(12)?; // operating_point_idc[i]
266            let level = r.f(5)?; // seq_level_idx[i]
267            let mut tier = 0u32;
268            if level > 7 {
269                tier = r.bit()? as u32; // seq_tier[i]
270            }
271            if decoder_model_info_present_flag == 1 {
272                let decoder_model_present_for_this_op = r.bit()?;
273                if decoder_model_present_for_this_op == 1 {
274                    // operating_parameters_info(): decoder_buffer_delay,
275                    // encoder_buffer_delay each (buffer_delay_length_minus_1+1) bits,
276                    // then low_delay_mode_flag f(1).
277                    let n = buffer_delay_length_minus_1 + 1;
278                    r.f(n)?;
279                    r.f(n)?;
280                    r.bit()?;
281                }
282            }
283            if initial_display_delay_present_flag == 1 {
284                let initial_display_delay_present_for_this_op = r.bit()?;
285                if initial_display_delay_present_for_this_op == 1 {
286                    r.f(4)?; // initial_display_delay_minus_1[i]
287                }
288            }
289            if i == 0 {
290                seq_level_idx_0 = level as u8;
291                seq_tier_0 = tier as u8;
292            }
293        }
294    }
295
296    // frame_width_bits_minus_1 f(4), frame_height_bits_minus_1 f(4),
297    // max_frame_width_minus_1 f(n), max_frame_height_minus_1 f(n).
298    let frame_width_bits_minus_1 = r.f(4)?;
299    let frame_height_bits_minus_1 = r.f(4)?;
300    r.f(frame_width_bits_minus_1 + 1)?; // max_frame_width_minus_1
301    r.f(frame_height_bits_minus_1 + 1)?; // max_frame_height_minus_1
302
303    // frame_id_numbers_present_flag (0 unless reduced_still_picture_header).
304    let mut frame_id_numbers_present_flag = 0u8;
305    if reduced_still_picture_header == 0 {
306        frame_id_numbers_present_flag = r.bit()?;
307    }
308    if frame_id_numbers_present_flag == 1 {
309        r.f(4)?; // delta_frame_id_length_minus_2
310        r.f(3)?; // additional_frame_id_length_minus_1
311    }
312
313    r.bit()?; // use_128x128_superblock
314    r.bit()?; // enable_filter_intra
315    r.bit()?; // enable_intra_edge_filter
316
317    if reduced_still_picture_header == 1 {
318        // enable_* flags forced; seq_force_* set to defaults. Nothing more to read
319        // before the color config in this branch beyond what the spec fixes.
320    } else {
321        r.bit()?; // enable_interintra_compound
322        r.bit()?; // enable_masked_compound
323        r.bit()?; // enable_warped_motion
324        r.bit()?; // enable_dual_filter
325        let enable_order_hint = r.bit()?;
326        if enable_order_hint == 1 {
327            r.bit()?; // enable_jnt_comp
328            r.bit()?; // enable_ref_frame_mvs
329        }
330        let seq_choose_screen_content_tools = r.bit()?;
331        let seq_force_screen_content_tools = if seq_choose_screen_content_tools == 1 {
332            2u32 // SELECT_SCREEN_CONTENT_TOOLS
333        } else {
334            r.bit()? as u32
335        };
336        if seq_force_screen_content_tools > 0 {
337            let seq_choose_integer_mv = r.bit()?;
338            if seq_choose_integer_mv == 0 {
339                r.bit()?; // seq_force_integer_mv
340            }
341        }
342        if enable_order_hint == 1 {
343            r.f(3)?; // order_hint_bits_minus_1
344        }
345    }
346
347    r.bit()?; // enable_superres
348    r.bit()?; // enable_cdef
349    r.bit()?; // enable_restoration
350
351    // color_config():
352    let high_bitdepth = r.bit()?;
353    let mut twelve_bit = 0u8;
354    let bit_depth;
355    if seq_profile == 2 && high_bitdepth == 1 {
356        twelve_bit = r.bit()?;
357        bit_depth = if twelve_bit == 1 { 12 } else { 10 };
358    } else {
359        // profile <= 2: bit_depth = high_bitdepth ? 10 : 8
360        bit_depth = if high_bitdepth == 1 { 10 } else { 8 };
361    }
362    let _ = bit_depth;
363
364    let monochrome = if seq_profile == 1 {
365        0u8 // mono_chrome not coded for profile 1
366    } else {
367        r.bit()?
368    };
369
370    let color_description_present_flag = r.bit()?;
371    let (mut color_primaries, mut transfer_characteristics, mut matrix_coefficients) =
372        (2u32, 2u32, 2u32); // CP/TC/MC_UNSPECIFIED
373    if color_description_present_flag == 1 {
374        color_primaries = r.f(8)?;
375        transfer_characteristics = r.f(8)?;
376        matrix_coefficients = r.f(8)?;
377    }
378
379    let chroma_subsampling_x;
380    let chroma_subsampling_y;
381    let mut chroma_sample_position = 0u8; // CSP_UNKNOWN
382
383    if monochrome == 1 {
384        r.bit()?; // color_range
385        // subsampling 1,1; chroma_sample_position 0.
386        chroma_subsampling_x = 1;
387        chroma_subsampling_y = 1;
388    } else if color_primaries == 1 && transfer_characteristics == 13 && matrix_coefficients == 0 {
389        // sRGB: color_range=1, subsampling 0,0.
390        chroma_subsampling_x = 0;
391        chroma_subsampling_y = 0;
392    } else {
393        r.bit()?; // color_range
394        if seq_profile == 0 {
395            chroma_subsampling_x = 1;
396            chroma_subsampling_y = 1;
397        } else if seq_profile == 1 {
398            chroma_subsampling_x = 0;
399            chroma_subsampling_y = 0;
400        } else {
401            // profile 2
402            if bit_depth == 12 {
403                chroma_subsampling_x = r.bit()?;
404                if chroma_subsampling_x == 1 {
405                    chroma_subsampling_y = r.bit()?;
406                } else {
407                    chroma_subsampling_y = 0;
408                }
409            } else {
410                chroma_subsampling_x = 1;
411                chroma_subsampling_y = 0;
412            }
413        }
414        if chroma_subsampling_x == 1 && chroma_subsampling_y == 1 {
415            chroma_sample_position = r.f(2)? as u8;
416        }
417    }
418
419    Some(SeqHeaderConfig {
420        seq_profile,
421        seq_level_idx_0,
422        seq_tier_0,
423        high_bitdepth,
424        twelve_bit,
425        monochrome,
426        chroma_subsampling_x,
427        chroma_subsampling_y,
428        chroma_sample_position,
429    })
430}

432/// Build the `av1C` configuration-record bytes from the sequence-header OBU.
433///
434/// `seq_header_obu` is the **whole** sequence-header OBU (header + size +
435/// payload), copied verbatim into `configOBUs`; `cfg` is the parsed config. The
436/// returned bytes are exactly what goes under the `"av1C"` key in the format
437/// description's `SampleDescriptionExtensionAtoms`.
438pub fn build_av1c(cfg: &SeqHeaderConfig, seq_header_obu: &[u8]) -> Vec<u8> {
439    let mut out = Vec::with_capacity(4 + seq_header_obu.len());
440
441    // Byte 0: marker(1)=1, version(7)=1  →  0b1_0000001 = 0x81.
442    out.push(0x81);
443    // Byte 1: seq_profile(3), seq_level_idx_0(5).
444    out.push(((cfg.seq_profile & 0x07) << 5) | (cfg.seq_level_idx_0 & 0x1f));
445    // Byte 2: seq_tier_0(1), high_bitdepth(1), twelve_bit(1), monochrome(1),
446    //         chroma_subsampling_x(1), chroma_subsampling_y(1), chroma_sample_position(2).
447    out.push(
448        ((cfg.seq_tier_0 & 1) << 7)
449            | ((cfg.high_bitdepth & 1) << 6)
450            | ((cfg.twelve_bit & 1) << 5)
451            | ((cfg.monochrome & 1) << 4)
452            | ((cfg.chroma_subsampling_x & 1) << 3)
453            | ((cfg.chroma_subsampling_y & 1) << 2)
454            | (cfg.chroma_sample_position & 0x03),
455    );
456    // Byte 3: reserved(3)=0, initial_presentation_delay_present(1)=0,
457    //         initial_presentation_delay_minus_one(4) → all zero.
458    out.push(0x00);
459
460    // configOBUs: the sequence-header OBU verbatim.
461    out.extend_from_slice(seq_header_obu);
462    out
463}

465/// Convenience: from a full AV1 temporal-unit byte stream that contains a
466/// sequence-header OBU, synthesise the `av1C` record. Returns `None` if there is
467/// no sequence header or it can't be parsed.
468pub fn av1c_from_stream(stream: &[u8]) -> Option<Vec<u8>> {
469    let sh = find_sequence_header(stream)?;
470    let cfg = parse_sequence_header(sh.payload)?;
471    Some(build_av1c(&cfg, sh.whole))
472}

474/// True if a temporal-delimiter-only / empty stream (nothing to decode).
475pub fn is_temporal_delimiter_only(stream: &[u8]) -> bool {
476    let obus = parse_obus(stream);
477    !obus.is_empty() && obus.iter().all(|o| o.obu_type == OBU_TEMPORAL_DELIMITER)
478}

486        // 0x80 0x01 = 128; single byte 0x7f = 127.
487        assert_eq!(read_leb128(&[0x7f], 0), Some((127, 1)));
488        assert_eq!(read_leb128(&[0x80, 0x01], 0), Some((128, 2)));

490        // Truncated continuation byte → None, no panic.
491        assert_eq!(read_leb128(&[0x80], 0), None);
