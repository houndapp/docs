1//! # Typed, zero-cost media vocabulary
2//!
3//! The unit-bearing newtypes every engine API speaks in, plus the shared frame
4//! types the pipelines exchange. The point of this module is to move whole
5//! classes of errors to **compile time**:
6//!
7//! * a [`MicId`] cannot be passed where a [`SpeakerId`] or [`CameraId`] is
8//!   expected (the "opened the wrong device kind" class is unrepresentable);
9//! * a [`Dimensions`] is always even and non-zero (validated at construction),
10//!   so 4:2:0 plane math downstream can never see an odd geometry;
11//! * a [`SampleRate`]/[`ChannelCount`]/[`Fps`]/[`Bitrate`] can never be zero,
12//!   and can never be confused with each other or with a bare count;
13//! * a [`TimestampUs`] is explicitly microseconds — no more "is this ms or µs"
14//!   at a call site.
15//!
16//! All newtypes are `Copy` wrappers over `NonZero*`/primitive values:
17//! **zero-cost** — they compile to the underlying integer. Conversions are
18//! explicit (`.hz()`, `.bps()`, `.micros()`, …); there are no `Deref` tricks
19//! and no implicit numeric coercions.
20//!
21//! The literal-friendly constructors (`Dimensions::new`, `Fps::new`,
22//! `Bitrate::kbps`, …) are `const fn`s that **panic on invalid input** — in a
23//! `const` context that panic is a *compile error*, which is exactly where a
24//! bad hard-coded geometry should fail. Runtime-sourced values go through the
25//! fallible `try_new` constructors instead.
26
27use std::num::{NonZeroU16, NonZeroU32};
28use std::time::Instant;

34/// An audio sample rate in Hz. Never zero.
35#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]

37
38impl SampleRate {
39    /// 48 kHz — Opus's native internal rate and the engine's end-to-end rate.

41
42    /// Literal constructor: panics on 0 (a compile error in `const` contexts).
43    pub const fn new(hz: u32) -> Self {
44        match NonZeroU32::new(hz) {
45            Some(v) => Self(v),
46            None => panic!("sample rate must be non-zero"),
47        }
48    }

53            Some(v) => Some(Self(v)),
54            None => None,
55        }
56    }

60        self.0.get()
61    }

161/// How often a video encoder emits keyframes (IDRs) **on its own**, absent an
162/// explicit [`crate::backend::VideoEncode::request_keyframe`] demand (a PLI / a
163/// late joiner's request).
164///
165/// A keyframe is the only self-contained frame — every receiver bootstrap (join,
166/// loss recovery, mid-call codec switch) needs one — but it is one to two orders
167/// of magnitude larger than a delta frame, so the cadence is a **bandwidth policy
168/// per content type**, not a constant:
169///
170/// * [`Periodic`](Self::Periodic) — motion content (camera). The scene changes
171///   wholesale anyway (deltas are already large), so a frequent IDR is cheap
172///   insurance that recovery never waits long. [`Self::MOTION`] is the 4 s default.
173/// * [`OnDemand`](Self::OnDemand) — static-leaning content (screenshare). Screen
174///   frames are mostly identical, so a periodic IDR of an unchanged 1080p frame is
175///   pure redundancy — hundreds of kbps on an otherwise static share, the single
176///   biggest data cost of "sharing" a screen nobody is drawing on. IDRs flow only
177///   on demand, with a slow safety-net interval and a frame-count GOP bound (the
178///   caller sizes it to its transport's per-GOP limits).
179///
180/// The two mapping helpers are the single source of truth every encoder backend
181/// (VideoToolbox, openh264, MFT) derives its native GOP settings from.
182#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
183pub enum KeyframeCadence {
184    /// A self-refreshing IDR every `seconds` (and every `fps × seconds` frames).
185    Periodic {
186        /// Max seconds between IDRs. Clamped to ≥ 1 by the mapping helpers.
187        seconds: u32,
188    },
189    /// IDRs only when demanded (`request_keyframe`), plus a slow safety net so a
190    /// receiver that somehow missed every demand path still recovers eventually.
191    OnDemand {
192        /// Safety-net max seconds between IDRs. Clamped to ≥ 1.
193        safety_seconds: u32,
194        /// Hard cap on frames per GOP regardless of elapsed time, so sustained
195        /// high-fps content still bounds the GOP (transports that map one GOP to
196        /// one unit — e.g. a moq group — have per-unit object limits).
197        max_gop_frames: u32,
198    },
199}
200
201impl KeyframeCadence {
202    /// The motion-content default: a 4 s periodic IDR (the historical cadence).
203    pub const MOTION: Self = Self::Periodic { seconds: 4 };
204
205    /// Max frames between IDRs at `fps` — the frame-count GOP bound encoders that
206    /// count frames (VideoToolbox `MaxKeyFrameInterval`, openh264
207    /// `IntraFramePeriod`) configure. Never 0.
208    pub const fn max_interval_frames(self, fps: Fps) -> u32 {
209        let frames = match self {
210            Self::Periodic { seconds } => fps.get().saturating_mul(seconds),
211            Self::OnDemand { max_gop_frames, .. } => max_gop_frames,
212        };
213        if frames == 0 { 1 } else { frames }
214    }
215
216    /// Max seconds between IDRs — the wall-clock GOP bound for encoders that also
217    /// take a duration (VideoToolbox `MaxKeyFrameIntervalDuration`). Never 0.
218    pub const fn max_interval_seconds(self) -> u32 {
219        let secs = match self {
220            Self::Periodic { seconds } => seconds,
221            Self::OnDemand { safety_seconds, .. } => safety_seconds,
222        };
223        if secs == 0 { 1 } else { secs }
224    }
225}

227/// A validated video geometry: **even**, non-zero width and height.
228///
229/// Every pipeline here is 4:2:0 (I420/NV12 — chroma at half resolution), so an
230/// odd dimension is unrepresentable rather than a runtime "needs even,
231/// non-zero dimensions" error deep inside an encoder. Constructing a
232/// `Dimensions` is the proof the plane math downstream needs.
233#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
234pub struct Dimensions {
235    width: NonZeroU32,
236    height: NonZeroU32,
237}
238
239impl Dimensions {
240    /// Literal constructor: panics unless both dimensions are even and
241    /// non-zero (a compile error in `const` contexts — exactly where a bad
242    /// hard-coded geometry should fail).
243    pub const fn new(width: u32, height: u32) -> Self {
244        match Self::try_new(width, height) {
245            Some(d) => d,
246            None => panic!("video dimensions must be even and non-zero (4:2:0 chroma is half-res)"),
247        }
248    }
249
250    /// Fallible constructor for runtime-sourced geometry (decoded pictures,
251    /// negotiated camera formats). `None` unless both are even and non-zero.
252    pub const fn try_new(width: u32, height: u32) -> Option<Self> {
253        if width == 0 || height == 0 || !width.is_multiple_of(2) || !height.is_multiple_of(2) {
254            return None;
255        }
256        // The zero checks above make these `NonZeroU32::new`s infallible.
257        match (NonZeroU32::new(width), NonZeroU32::new(height)) {
258            (Some(width), Some(height)) => Some(Self { width, height }),
259            _ => None,
260        }
261    }
262
263    /// Clamp arbitrary requested dimensions to the nearest valid geometry:
264    /// floored to even, with a floor of 2×2. The synthetic-source/camera
265    /// "even-clamp" in one place.
266    pub const fn even_clamped(width: u32, height: u32) -> Self {
267        let w = width & !1;
268        let h = height & !1;
269        Self::new(if w < 2 { 2 } else { w }, if h < 2 { 2 } else { h })
270    }
271
272    pub const fn width(self) -> u32 {
273        self.width.get()
274    }
275
276    pub const fn height(self) -> u32 {
277        self.height.get()
278    }
279
280    /// `(width, height)` as a tuple, for display/compare ergonomics.
281    pub const fn wh(self) -> (u32, u32) {
282        (self.width.get(), self.height.get())
283    }
284
285    /// Byte length of a tightly-packed I420 buffer at this geometry
286    /// (full-res Y plane + two half-res chroma planes, stride == width).
287    pub const fn i420_len(self) -> usize {
288        let w = self.width.get() as usize;
289        let h = self.height.get() as usize;
290        let cw = w / 2;
291        let ch = h / 2;
292        w * h + 2 * (cw * ch)
293    }
294}
295
296impl std::fmt::Display for Dimensions {
297    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
298        write!(f, "{}x{}", self.width, self.height)
299    }
300}

352    ($(#[$doc:meta])* $name:ident, $kind:expr) => {
353        $(#[$doc])*
354        #[derive(Debug, Clone, PartialEq, Eq, Hash)]
355        pub struct $name(String);
356
357        impl $name {

363                &self.0

370            }
371        }
372

375                Self(id)
376            }
377        }
378
379        impl std::fmt::Display for $name {
380            fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
381                f.write_str(&self.0)
382            }
383        }
384    };

433/// Which device to open: the host default, or a specific one by typed id.
434///
435/// The id parameter is what moves the "wrong-device-kind" class to compile
436/// time: a `DeviceChoice<MicId>` cannot be handed to a speaker opener.
437/// Resolution is **late** (at open time): a named device that has vanished
438/// falls back to the host default rather than failing, because it may have
439/// been unplugged between selection and open (hot-swap / vanish). The
440/// fallback is logged, never silent.
441#[derive(Debug, Clone, Default, PartialEq, Eq)]
442pub enum DeviceChoice<Id> {
443    /// The host's current default device for this kind.
444    #[default]
445    Default,
446    /// A specific device by its typed id.
447    Named(Id),
448}
449
450impl<Id: DeviceId> DeviceChoice<Id> {
451    /// Build a choice from an optional raw id string (the wire shape the UI
452    /// sends): `None`/empty ⇒ default, else named.
453    pub fn from_id(id: Option<String>) -> Self {
454        match id {
455            Some(name) if !name.is_empty() => DeviceChoice::Named(Id::from(name)),
456            _ => DeviceChoice::Default,
457        }
458    }
459}

516///

567/// How to open a screen/window capture: the encode-target geometry, the frame
568/// rate, and whether to also capture the source's app audio.
569///
570/// Screen content is *not* camera content: it is captured at the source's
571/// native size and **scaled down** to `dimensions` preserving aspect (no
572/// mirroring, no crop — letterboxing is the caller's dimension choice). Text
573/// stays legible when the encode target is capped at a 1080p-class geometry
574/// (with the source's aspect) rather than encoding a 4K/5K source natively at
575/// screen-share bitrates.
576///
577/// `capture_audio` only has an effect on a platform with an app-audio backend
578/// (the `AppAudioBackend` is uninhabited elsewhere), and its scope follows the
579/// source: a [`ScreenSource::Window`] captures the shared **app's** audio
580/// (macOS 13+ via SCK; Windows via process-tree loopback), and on Windows a
581/// [`ScreenSource::Display`] additionally captures whole-system audio
582/// (everything audible is on the shared screen). Otherwise it is ignored and no
583/// [`crate::backend::AppAudio`] handle is produced.
584#[derive(Debug, Clone, Copy, PartialEq, Eq)]
585pub struct ScreenCaptureConfig {
586    /// The encode-target geometry. Capture is native; frames are scaled to this.
587    pub dimensions: Dimensions,
588    /// Frame rate. Screen sharing is typically 15-30 fps; the caller chooses.
589    pub fps: Fps,
590    /// Request the source's audio alongside video (see the scope note above).
591    pub capture_audio: bool,
592}

598/// Pixel layout of a [`RawFrame`].
599///
600/// The pipeline normalises every captured frame to [`PixelFormat::I420`]
601/// (planar YUV 4:2:0, 8-bit) because that is what the encoders consume and the
602/// render seam displays. The variants for the formats cameras actually *emit*
603/// (NV12, RGB) are kept so a future zero-copy path can hand a HW encoder the
604/// native buffer without a planar conversion.
605#[derive(Debug, Clone, Copy, PartialEq, Eq)]
606pub enum PixelFormat {
607    /// Planar Y, then U, then V; chroma at half width/height. Encoder input.
608    I420,
609    /// Planar Y followed by interleaved UV; common AVFoundation/MediaCodec
610    /// output. Reserved for the zero-copy HW path.
611    Nv12,
612    /// Packed 8-bit R,G,B. Reserved for capture paths that decode to RGB.
613    Rgb,
614}

616/// Which codec a video encoder produces / a video decoder consumes.
617#[derive(Debug, Clone, Copy, PartialEq, Eq)]
618pub enum VideoCodec {
619    /// AV1 — HW encode on M5 Pro/Max + modern discrete GPUs (VideoToolbox);
620    /// HW decode broadly (A17 Pro / M3+). No SW path — a device without any HW
621    /// video codec falls back to SW H.264, never SW AV1.
622    Av1,
623    /// H.265 / HEVC — HW encode+decode (VideoToolbox today; MediaCodec later).
624    H265,
625    /// H.264 — HW where present (VideoToolbox), SW (openh264) everywhere as the
626    /// universal floor.
627    H264,
628}
629
630impl VideoCodec {
631    /// Map to the wire codec enum in `hound-media` so encoded frames packetize
632    /// with the correct header byte.
633    pub fn to_wire(self) -> hound_media::Codec {
634        match self {
635            VideoCodec::Av1 => hound_media::Codec::Av1,
636            VideoCodec::H265 => hound_media::Codec::H265,
637            VideoCodec::H264 => hound_media::Codec::H264,
638        }
639    }
640
641    /// Map back from the wire codec, for receive-side decoder selection.
642    /// `None` for non-video wire codecs (Opus).
643    pub fn from_wire(codec: hound_media::Codec) -> Option<Self> {
644        match codec {
645            hound_media::Codec::Av1 => Some(VideoCodec::Av1),
646            hound_media::Codec::H265 => Some(VideoCodec::H265),
647            hound_media::Codec::H264 => Some(VideoCodec::H264),
648            _ => None,
649        }
650    }

653/// Which of the three video codecs a device supports, as a small copyable record.
654/// Used both as a device's local encode capabilities and as the running
655/// intersection of every receiver's decode capabilities — the two inputs to the
656/// codec ladder ([`pick_encode_codec`]).
657#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
658pub struct VideoCodecCaps {
659    pub av1: bool,
660    pub h265: bool,
661    pub h264: bool,
662}
663
664impl VideoCodecCaps {
665    /// Does this set include `codec`?
666    pub fn supports(self, codec: VideoCodec) -> bool {
667        match codec {
668            VideoCodec::Av1 => self.av1,
669            VideoCodec::H265 => self.h265,
670            VideoCodec::H264 => self.h264,
671        }
672    }
673
674    /// The codecs **both** sets support — used to fold each receiver's decode caps
675    /// into the common set a sender may target.
676    pub fn intersect(self, other: Self) -> Self {
677        Self {
678            av1: self.av1 && other.av1,
679            h265: self.h265 && other.h265,
680            h264: self.h264 && other.h264,
681        }
682    }
683}
684
685/// Codec preference order, highest quality first: AV1 → HEVC → H.264. The ladder
686/// queries encode/decode support in this order; H.264 is the universal floor.
687pub const ENCODE_PRIORITY: [VideoCodec; 3] = [VideoCodec::Av1, VideoCodec::H265, VideoCodec::H264];
688
689/// Choose which codec to **encode**: the highest-priority codec this device can
690/// encode (`local_encode`) AND every receiver can decode (`common_decode`).
691///
692/// Falls back to H.264 when no higher rung is common — H.264 is always encodable
693/// (HW or SW openh264) and decodable, so a call never loses video for lack of a
694/// shared codec. `common_decode` is the intersection of every current video
695/// receiver's advertised decode caps (see [`VideoCodecCaps::intersect`]); it
696/// shrinks as less-capable peers join and grows back as they leave, which is what
697/// drives re-negotiation.
698pub fn pick_encode_codec(
699    local_encode: VideoCodecCaps,
700    common_decode: VideoCodecCaps,
701) -> VideoCodec {
702    for codec in ENCODE_PRIORITY {
703        if local_encode.supports(codec) && common_decode.supports(codec) {
704            return codec;
705        }
706    }
707    VideoCodec::H264
708}

710/// Sample rate for the whole audio engine. 48 kHz is Opus's native internal
711/// rate (no internal resample) and the CoreAudio default on Macs — picking it
712/// end to end avoids a resample on either the capture or the playback side.
713pub const SAMPLE_RATE: SampleRate = SampleRate::HZ_48K;
714
715/// Channel count. Mono for voice — half the bitrate of stereo and what AEC
716/// expects on the capture side. If a device only offers stereo input we
717/// downmix to mono before encode.
718pub const CHANNELS: ChannelCount = ChannelCount::MONO;
719
720/// Opus frame duration. 20 ms is the voice sweet spot: 960 samples @ 48 kHz,
721/// low algorithmic delay, the default the WebRTC/Opus VoIP stacks use.
722pub const FRAME_MS: u32 = 20;
723
724/// Samples per channel in one [`PcmFrame`] / one Opus frame: 48000 × 20 / 1000
725/// = **960**. Opus only accepts exact frame sizes; the capture side must hand
726/// it exactly this.
727pub const SAMPLES_PER_FRAME: usize = (SAMPLE_RATE.hz() as usize * FRAME_MS as usize) / 1000;
728
729/// One block of captured/decoded PCM, stamped on the shared capture clock.
730///
731/// Interleaved `i16` samples (here always mono, so interleaving is a no-op).
732/// `samples.len()` is exactly [`SAMPLES_PER_FRAME`] × [`CHANNELS`] for a frame
733/// destined for the encoder.
734#[derive(Debug, Clone)]
735pub struct PcmFrame {
736    /// Interleaved i16 PCM. Length is `samples_per_channel × channels`.
737    pub samples: Vec<i16>,
738    pub sample_rate: SampleRate,
739    pub channels: ChannelCount,
740    /// Capture timestamp on the session's `CaptureClock` — the SAME timeline
741    /// as video ([`RawFrame::timestamp_us`]) for A/V sync.
742    pub timestamp_us: TimestampUs,
743    /// When this frame was assembled, for capture→playout latency measurement.
744    /// Local instrumentation only; not on the wire.
745    pub captured_at: Instant,
746}
747
748impl PcmFrame {
749    /// A frame at the engine's fixed format (48 kHz mono) — the shape every
750    /// internal producer (capture, decode, mix) hands around.
751    pub fn at_engine_rate(samples: Vec<i16>, timestamp_us: TimestampUs) -> Self {
752        Self {
753            samples,
754            sample_rate: SAMPLE_RATE,
755            channels: CHANNELS,
756            timestamp_us,
757            captured_at: Instant::now(),
758        }
759    }
760
761    /// Root-mean-square amplitude over the block, normalised to `0.0..=1.0`
762    /// (i16 full-scale = 1.0). Distinguishes "silent room" from "no mic /
763    /// permission denied" (the latter yields *no frames at all*, not a quiet
764    /// RMS).
765    pub fn rms(&self) -> f32 {
766        if self.samples.is_empty() {
767            return 0.0;
768        }
769        let sum_sq: f64 = self
770            .samples
771            .iter()
772            .map(|&s| {
773                let f = s as f64;
774                f * f
775            })
776            .sum();
777        let mean_sq = sum_sq / self.samples.len() as f64;
778        (mean_sq.sqrt() / i16::MAX as f64) as f32
779    }
780}

792/// One captured raw video frame, stamped on the shared capture clock.
793///
794/// `data` layout depends on `format`. For [`PixelFormat::I420`] it is
795/// `Y (w*h) || U (w/2*h/2) || V (w/2*h/2)` with no row padding (stride ==

921impl RawFrame {
922    /// Luma width in pixels (convenience for `dims.width()`).
923    pub const fn width(&self) -> u32 {
924        self.dims.width()
925    }
926
927    /// Luma height in pixels (convenience for `dims.height()`).
928    pub const fn height(&self) -> u32 {
929        self.dims.height()
930    }
931
932    /// This frame as planar I420 — borrowed when it already is I420 (the common
933    /// case, zero-copy), converted when it is NV12 or RGB.
934    ///
935    /// The capture fast path keeps Apple frames in NV12 all the way to the
936    /// hardware encoder; consumers that genuinely need I420 — the software
937    /// encoder and the self-preview (an I420 render channel) — call this
938    /// at their boundary so the NV12→I420 de-interleave happens *only* when
939    /// something downstream can't take NV12. `None` on a malformed buffer (a
940    /// skippable frame, never a panic).
941    pub fn as_i420(&self) -> Option<std::borrow::Cow<'_, RawFrame>> {
942        use crate::platform::generic::convert;
943        use std::borrow::Cow;
944        let (w, h) = (self.width() as usize, self.height() as usize);
945        let data = match &self.surface {
946            // Zero-copy capture frame: lock + convert the platform buffer.
947            Some(surface) => surface.to_i420()?,
948            None => match self.format {
949                PixelFormat::I420 => return Some(Cow::Borrowed(self)),
950                // Tightly-packed NV12 → I420 (Y stride == width).
951                PixelFormat::Nv12 => convert::nv12_bytes_to_i420(&self.data, w, h, w)?,
952                PixelFormat::Rgb => convert::rgb_to_i420(&self.data, w, h),
953            },
954        };
955        Some(Cow::Owned(RawFrame {
956            dims: self.dims,
957            format: PixelFormat::I420,
958            timestamp_us: self.timestamp_us,
959            captured_at: self.captured_at,
960            data,
961            surface: None,
962        }))
963    }
964
965    /// A copy of this frame scaled to `dims` as tightly-packed planar I420
966    /// (materializing a zero-copy surface first). `None` if `dims` are invalid (zero
967    /// or odd — 4:2:0 needs even extents) or the pixels can't be read. The result is
968    /// a portable byte frame (`surface: None`), so the zero-copy hardware-encode path
969    /// is traded for one software scale: callers use this only when the quality
970    /// ladder degrades resolution, and pass the frame through untouched at full res.
971    pub fn scaled_to(&self, dims: Dimensions) -> Option<RawFrame> {
972        let src = self.as_i420()?;
973        let scaled = crate::platform::generic::convert::scale_i420(
974            &src.data,
975            src.width() as usize,
976            src.height() as usize,
977            dims.width() as usize,
978            dims.height() as usize,
979        )?;
980        Some(RawFrame {
981            dims,
982            format: PixelFormat::I420,
983            timestamp_us: self.timestamp_us,
984            captured_at: self.captured_at,
985            data: scaled,
986            surface: None,
987        })
988    }
989}
