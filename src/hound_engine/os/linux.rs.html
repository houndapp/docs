1//! # Linux-only capture extensions
2//!
3//! App-scoped screen-share audio by **explicit application identity**.
4//!
5//! On macOS and Windows a *window* share carries the owning app's audio
6//! automatically, because those platforms can map a window to its process.
7//! Wayland cannot: the xdg-desktop-portal picker never reveals which
8//! application owns the selected window, so nothing on Linux can auto-scope
9//! audio to "the shared window's app" (OBS and Discord hit the same wall and
10//! make the user pick the app). This escape hatch is that pick: given an
11//! explicit identity — a root pid (matched with its whole process tree) and/or
12//! an application binary/name — it captures exactly that application's
13//! PipeWire output streams as a standard [`crate::api::ScreenAudio`] leg,
14//! tracked live as the app starts and stops streams. The scope never widens:
15//! an identity that matches nothing yields silence, not system audio.
16
17use crate::api::ScreenAudio;
18use crate::error::AudioError;
19use crate::platform::linux::pw_audio::{AppTarget, AudioScope, PipeWireAppAudio};
20
21/// The application whose audio an app-scoped share leg captures. At least one
22/// criterion must be set; they are OR'd (pid-tree first, then the
23/// case-insensitive binary / application-name fallbacks).
24#[derive(Clone, Debug, Default)]
25pub struct AppAudioIdentity {
26    /// Root process id; the capture includes its whole descendant tree (a
27    /// browser renders audio in child processes).
28    pub root_pid: Option<u32>,
29    /// PipeWire `application.process.binary` (case-insensitive).
30    pub binary: Option<String>,
31    /// PipeWire `application.name` (case-insensitive).
32    pub app_name: Option<String>,
33}
34
35/// Open an app-scoped share-audio leg for `identity`, delivering the standard
36/// 48 kHz mono 20 ms frames ([`ScreenAudio::next_frame`]). Pair it with a
37/// window share whose audio the portal could not scope automatically.
38///
39/// `Err` on an empty identity or an unreachable PipeWire daemon; an identity
40/// that currently matches no streams opens successfully and starts delivering
41/// when the app plays. `!Send` — drive and drop it on the share's capture
42/// thread, like every audio leg.
43pub fn screen_share_app_audio(
44    clock: hound_media::CaptureClock,
45    identity: AppAudioIdentity,
46) -> Result<ScreenAudio, AudioError> {
47    if identity.root_pid.is_none() && identity.binary.is_none() && identity.app_name.is_none() {
48        return Err(AudioError::Device(
49            "app audio identity is empty (set a pid, binary, or application name)".into(),
50        ));
51    }
52    let target = AppTarget {
53        root_pid: identity.root_pid,
54        binary: identity.binary,
55        app_name: identity.app_name,
56    };
57    PipeWireAppAudio::open(clock, AudioScope::App(target)).map(ScreenAudio::from_backend)
58}
