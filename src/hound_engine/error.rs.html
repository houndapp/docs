1//! # Engine error taxonomy
2//!
3//! Per-domain error enums with typed fields where structure exists. The two
4//! domains mirror the two pipelines: [`AudioError`] for the mic/speaker/codec
5//! path and [`CaptureError`] for the camera/encode/decode path.
6//!
7//! Design rules:
8//! * **Permission and absence are distinct, typed variants** — a denied
9//!   device must never read as a silent zero stream, and the
10//!   [`DeviceKind`] on `PermissionDenied` names which device was denied.
11//! * Whole error classes are **unrepresentable** by construction:
12//!   "capturer not started" cannot exist because open-is-construct (a handle
13//!   that exists is running), and "odd dimensions" cannot exist because
14//!   [`crate::types::Dimensions`] is validated at construction.
15//! * Device-layer strings (`Device`, `NoDevice`, …) stay strings on purpose:
16//!   they wrap opaque OS/driver error text that has no structure to type.
17//!
18//! The app maps these to its UI-facing error type at its own boundary; the
19//! serialized UI contract lives there, not here.
20
21use crate::types::DeviceKind;
22
23/// Errors from the audio capture/encode/decode/playback path.
24///
25/// The split between [`Self::NoDevice`]/[`Self::PermissionDenied`] and the
26/// rest is deliberate and load-bearing: a denied mic must surface a *distinct*
27/// error, never a silent stream of zero/0-RMS frames. On CoreAudio a denied
28/// mic and an absent device both fail at stream build; we report the build
29/// error verbatim under [`Self::Device`] when we cannot tell them apart.
30#[derive(Debug, thiserror::Error)]
31pub enum AudioError {
32    #[error("no audio device available: {0}")]
33    NoDevice(String),
34    #[error("audio device error: {0}")]
35    Device(String),
36    #[error("requested audio format not supported: {0}")]
37    UnsupportedFormat(String),
38    #[error("codec error: {0}")]
39    Codec(String),
40    #[error("{kind} permission denied (audio access not granted)")]
41    PermissionDenied { kind: DeviceKind },
42}
43
44/// Errors from the video capture/encode/decode path.
45#[derive(Debug, thiserror::Error)]
46pub enum CaptureError {
47    #[error("no capture device available: {0}")]
48    NoDevice(String),
49    #[error("capture device error: {0}")]
50    Device(String),
51    #[error("requested format/resolution not supported: {0}")]
52    UnsupportedFormat(String),
53    #[error("encoder error: {0}")]
54    Encoder(String),
55    #[error("decoder error: {0}")]
56    Decoder(String),
57    #[error("permission denied ({kind} access not granted)")]
58    PermissionDenied { kind: DeviceKind },
59    /// The capture SOURCE ended — the shared window closed / its app quit / the
60    /// OS stopped the share. A CLEAN end, not a fault: the consumer should end
61    /// the stream (unpublish, drop tiles), never retry/restart against a source
62    /// that no longer exists (a restart would either fail or freeze on the last
63    /// frame). Distinct from `Device` so callers can classify without string
64    /// matching.
65    #[error("capture source ended (window closed / share stopped)")]
66    SourceEnded,
67    /// This platform backend has no implementation for the requested
68    /// capability (e.g. camera capture or a HW codec on a platform whose
69    /// backend doesn't provide one). Queried via
70    /// [`crate::api::Engine::capabilities`]; surfaced — never a panic — when a
71    /// caller asks anyway.
72    #[error("{what} is not supported on this platform")]
73    Unsupported { what: &'static str },
74}

76#[cfg(test)]
77mod tests {
78    use super::*;
79
80    /// The permission variants carry the device kind, so a denied camera and a
81    /// denied microphone are distinguishable without string parsing.
82    #[test]
83    fn permission_denied_names_the_device_kind() {
84        let mic = AudioError::PermissionDenied {
85            kind: DeviceKind::Microphone,
86        };
87        assert!(mic.to_string().contains("microphone"));
88        let cam = CaptureError::PermissionDenied {
89            kind: DeviceKind::Camera,
90        };
91        assert!(cam.to_string().contains("camera"));
92    }
93
94    #[test]
95    fn unsupported_is_a_typed_capability_miss() {
96        let e = CaptureError::Unsupported {
97            what: "camera capture",
98        };
99        assert_eq!(
100            e.to_string(),
101            "camera capture is not supported on this platform"
102        );
103    }
104}
