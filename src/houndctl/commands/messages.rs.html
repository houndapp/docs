27/// One-line, ≤60-char table preview. Truncates on CHARACTER boundaries — a
28/// byte slice panics mid-way through multi-byte UTF-8 (emoji, accents).
29fn preview(contents: &str) -> String {
30    let flat = contents.replace(['\n', '\r'], " ");
31    match flat.char_indices().nth(60) {
32        Some((byte_index, _)) => format!("{}…", &flat[..byte_index]),
33        None => flat,
34    }
35}

37pub async fn run(args: MessagesArgs, api: ApiClient, mode: OutputMode) -> eyre::Result<()> {
38    match args.command {
39        MessagesCommand::List { space, page, limit } => {
40            let result = api.get_messages(space, page, limit).await?;
41
42            if mode.is_json() {
43                output::print_json(&result);
44                return Ok(());
45            }
46
47            let user_name: std::collections::HashMap<Uuid, String> = result
48                .users
49                .iter()
50                .map(|u| {
51                    (
52                        u.user_id,
53                        u.name.clone().unwrap_or_else(|| u.user_id.to_string()),
54                    )
55                })
56                .collect();
57
58            output::print_table(
59                &["ID", "User", "Contents", "Edited"],
60                result.messages.contents.iter().map(|m| {
61                    let author = user_name
62                        .get(&m.user_id)
63                        .cloned()
64                        .unwrap_or_else(|| m.user_id.to_string());
65                    let edited = if m.edited_at.is_some() { "yes" } else { "no" };
66                    vec![
67                        m.id.to_string(),
68                        author,
69                        preview(&m.contents),
70                        edited.to_string(),
71                    ]
72                }),
73            );
74
75            let total = result.messages.total;
76            let current = result.messages.current;
77            output::hint(&format!(
78                "page {}/{} · {}",
79                current + 1,
80                (total + limit - 1) / limit.max(1),
81                output::count(total as usize, "message total"),
82            ));
83        }
84    }
85    Ok(())
86}

94        // 61 two-byte chars: a byte-indexed slice at 60 would panic.
95        let contents = "é".repeat(61);
96        assert_eq!(preview(&contents), format!("{}…", "é".repeat(60)));
97        assert_eq!(preview("short"), "short");
98        assert_eq!(preview("two\nlines\r"), "two lines ");
