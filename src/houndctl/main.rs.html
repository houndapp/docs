8#[derive(Parser)]
9#[command(
10    name = "houndctl",
11    about = "Hound instance management CLI",
12    version,
13    propagate_version = true
14)]
15struct Cli {
16    /// Backend server base URL. With --cluster and no explicit value, it is
17    /// discovered from the backend release's Ingress host.
18    #[arg(long, env = "HOUND_SERVER", global = true)]
19    server: Option<String>,
20
21    /// Bearer token for API authentication
22    #[arg(long, env = "HOUND_TOKEN", global = true)]
23    token: Option<String>,
24
25    /// Output raw JSON (always on when stdout is not a TTY)
26    #[arg(long, global = true)]
27    json: bool,
28
29    #[command(flatten)]
30    cluster: commands::kube::ClusterArgs,
31
32    #[command(subcommand)]
33    command: Command,
34}

60#[tokio::main]
61async fn main() -> eyre::Result<()> {
62    let cli = Cli::parse();
63
64    let cfg = config::Config::load();
65    let token = cli.token.or_else(|| cfg.token.clone());
66    let mode = output::OutputMode::detect(cli.json);
67
68    // `db` talks to the database, not the API server — don't resolve (or
69    // cluster-discover) a server URL it won't use.
70    if let Command::Db(args) = cli.command {
71        return commands::db::run(args, &cli.cluster).await;
72    }
73
74    let server = resolve_server(&cli.server, &cli.cluster, &cfg).await?;
75    let api = client::ApiClient::new(server.clone(), token.clone());
76
77    match cli.command {
78        Command::Auth(args) => commands::auth::run(args, cfg, server, token, mode).await,
79        Command::Users(args) => commands::users::run(args, api, mode).await,
80        Command::Communities(args) => commands::communities::run(args, api, mode).await,
81        Command::Spaces(args) => commands::spaces::run(args, api, mode).await,
82        Command::Messages(args) => commands::messages::run(args, api, mode).await,
83        Command::Instance(args) => commands::instance::run(args, api, mode).await,
84        Command::Db(_) => unreachable!("handled above"),
85    }
86}

88/// The backend base URL: explicit flag/env, then cluster Ingress discovery
89/// (`--cluster`), then the saved config, then the local-dev default. Always
90/// announces the target + where it came from (one stderr line), so every API
91/// command states what it is talking to — the db commands' target-block twin.
92async fn resolve_server(
93    server: &Option<String>,
94    cluster: &commands::kube::ClusterArgs,
95    cfg: &config::Config,
96) -> eyre::Result<String> {
97    let (url, source) = if let Some(server) = server {
98        (server.clone(), "--server / HOUND_SERVER".to_owned())
99    } else if cluster.cluster {
100        let cluster = commands::kube::Cluster::connect(cluster).await?;
101        let url = cluster.api_url().await?;
102        let source = format!("ingress/{} · {}", cluster.release, cluster.context);
103        (url, source)
104    } else if let Some(server) = &cfg.server {
105        (server.clone(), "saved config".to_owned())
106    } else {
107        ("http://localhost:8080".to_owned(), "default".to_owned())
108    };
109    output::kv_stderr(&[("server", format!("{url}   ({source})"))]);
110    Ok(url)
111}
