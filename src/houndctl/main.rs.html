1mod client;

6use clap::{Parser, Subcommand};
7
8#[derive(Parser)]
9#[command(
10    name = "houndctl",
11    about = "Hound instance management CLI",
12    version,
13    propagate_version = true
14)]
15struct Cli {
16    /// Backend server base URL
17    #[arg(
18        long,
19        env = "HOUND_SERVER",
20        default_value = "http://localhost:8080",
21        global = true
22    )]
23    server: String,
24
25    /// Bearer token for API authentication
26    #[arg(long, env = "HOUND_TOKEN", global = true)]
27    token: Option<String>,
28
29    /// Output raw JSON (always on when stdout is not a TTY)
30    #[arg(long, global = true)]
31    json: bool,
32
33    #[command(subcommand)]
34    command: Command,
35}
36
37#[derive(Subcommand)]
38enum Command {
39    #[command(about = "Manage authentication credentials")]
40    Auth(commands::auth::AuthArgs),
41
42    #[command(about = "Inspect users")]
43    Users(commands::users::UsersArgs),
44
45    #[command(about = "Inspect and manage communities")]
46    Communities(commands::communities::CommunitiesArgs),
47
48    #[command(about = "Inspect spaces")]
49    Spaces(commands::spaces::SpacesArgs),
50
51    #[command(about = "Inspect messages")]
52    Messages(commands::messages::MessagesArgs),
53
54    #[command(about = "Manage instance configuration and permissions")]
55    Instance(commands::instance::InstanceArgs),
56
57    #[command(about = "Run database migrations directly")]
58    Db(commands::db::DbArgs),
59}
60
61#[tokio::main]
62async fn main() -> eyre::Result<()> {
63    let cli = Cli::parse();
64
65    let cfg = config::Config::load();
66    let token = cli.token.or_else(|| cfg.token.clone());
67    let mode = output::OutputMode::detect(cli.json);
68    let api = client::ApiClient::new(cli.server.clone(), token.clone());
69
70    match cli.command {
71        Command::Auth(args) => commands::auth::run(args, cfg, cli.server, token, mode).await,
72        Command::Users(args) => commands::users::run(args, api, mode).await,
73        Command::Communities(args) => commands::communities::run(args, api, mode).await,
74        Command::Spaces(args) => commands::spaces::run(args, api, mode).await,
75        Command::Messages(args) => commands::messages::run(args, api, mode).await,
76        Command::Instance(args) => commands::instance::run(args, api, mode).await,
77        Command::Db(args) => commands::db::run(args).await,
78    }
79}
