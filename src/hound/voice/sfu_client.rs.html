1//! # SFU client
2//!
3//! Manages the iroh QUIC connection from the Tauri app to the SFU service.
4//!
5//! ## Lifecycle
6//!
7//! 1. `SfuClient::connect()` — binds an iroh endpoint, dials the SFU,
8//!    performs the Hello/HelloAck handshake, and starts a background task
9//!    that reads incoming frames and emits Tauri events.
10//! 2. `SfuClient::send_frame()` — encodes and fragments a frame, then sends
11//!    it via QUIC datagrams (delta frames) or a unidirectional stream (keyframes).
12//! 3. Dropping `SfuClient` closes the connection cleanly.
13//!
14//! ## Frame wire format
15//!
16//! Each wire packet is `FrameHeader::BYTES || payload_fragment`. The SFU
17//! knows our peer ID (from the HelloAck) so it can route incoming frames.
18//! The Tauri event listener on the TypeScript side calls `deliverFrame(buf)`
19//! which passes the raw bytes to the MediaWorker for header parsing.
20

31/// ALPN protocol string — must match `hound_sfu::sfu::HOUND_ALPN`.
32pub const HOUND_ALPN: &[u8] = b"hound-sfu/1";
33
34/// Maximum UDP payload for a single frame fragment.
35const MAX_FRAG_PAYLOAD: usize = 1100;
36
37/// Fixed-size frame header written at the front of every wire packet.
38/// Layout matches `hound_sfu::wire::frame::FrameHeader`.
39///
40/// Byte layout (20 bytes):
41/// ```text
42/// [0..8]   peer_id      u64 LE
43/// [8..12]  stream_seq   u32 LE
44/// [12]     frag_index   u8
45/// [13]     frag_total   u8
46/// [14]     track_type   u8  (0=CamVideo, 1=CamAudio, 2=ScrVideo, 3=ScrAudio)
47/// [15]     flags        u8  (bit 0 = keyframe)
48/// [16..20] timestamp_us u32 LE  (truncated µs — enough for jitter buffer)
49/// ```
50const HEADER_SIZE: usize = 20;
51const FLAG_KEYFRAME: u8 = 0x01;
52
53/// Track type byte values matching the Rust SFU enum.
54#[repr(u8)]
55enum TrackTypeByte {
56    CamVideo = 0,
57    CamAudio = 1,
58    ScrVideo = 2,
59    ScrAudio = 3,
60}
61
62fn parse_track_type(s: &str) -> u8 {
63    match s {
64        "cam-video" => TrackTypeByte::CamVideo as u8,
65        "cam-audio" => TrackTypeByte::CamAudio as u8,
66        "scr-video" => TrackTypeByte::ScrVideo as u8,
67        "scr-audio" => TrackTypeByte::ScrAudio as u8,
68        _ => TrackTypeByte::CamVideo as u8,
69    }
70}

72/// An active connection to the SFU.
73///
74/// Holds the iroh endpoint and QUIC connection. Dropping this struct closes
75/// the connection (the iroh `Connection` handles cleanup on drop).
76pub struct SfuClient {
77    /// Kept alive so the iroh endpoint isn't closed while the connection lives.
78    _endpoint: Endpoint,
79    connection: iroh::endpoint::Connection,
80    /// Peer ID assigned by the SFU in the HelloAck.
81    pub peer_id: u64,
82    /// Monotonically increasing stream sequence counter.
83    stream_seq: Arc<AtomicU32>,
84}
85
86impl SfuClient {
87    /// Connect to the SFU, perform the Hello/HelloAck handshake, and return
88    /// a ready-to-use client.
89    ///
90    /// Spawns a background task that reads incoming frames from the SFU and
91    /// emits them as `"voice-frame"` Tauri events. The TypeScript listener in
92    /// `VoiceClient` picks these up and forwards them to the MediaWorker.
93    ///
94    /// # Parameters
95    ///
96    /// * `relay_url` — iroh relay URL (from the backend `voice/join` response)
97    /// * `sfu_node_id` — iroh node ID of the SFU (hex string, from `sfu_node_id`
98    ///   field in the join response)
99    /// * `relay_token` — HMAC token issued by the backend for this session
100    /// * `app` — Tauri app handle used to emit events to the frontend
101    pub async fn connect<R: tauri::Runtime>(
102        relay_url: RelayUrl,
103        sfu_node_id: &str,
104        relay_token: &str,
105        app: tauri::AppHandle<R>,
106    ) -> Result<Self, VoiceError> {
107        // Parse the SFU's node ID (PublicKey / EndpointId).
108        let sfu_id: EndpointId =
109            sfu_node_id
110                .parse()
111                .map_err(|_| VoiceError::InvalidRelayToken {
112                    token: sfu_node_id.to_owned(),
113                })?;
114
115        // Generate a fresh ephemeral key for this session.
116        let secret_key = {
117            use rand::RngCore;
118            let mut buf = [0u8; 32];
119            rand::rngs::OsRng.fill_bytes(&mut buf);
120            SecretKey::from_bytes(&buf)
121        };
122
123        // Build our iroh endpoint — client-side, no ALPN needed.
124        let endpoint = Endpoint::builder(presets::N0)
125            .relay_mode(RelayMode::Custom(RelayMap::from(relay_url.clone())))
126            .secret_key(secret_key)
127            .bind()
128            .await
129            .map_err(|e| VoiceError::SfuConnect {
130                reason: e.to_string(),
131            })?;
132
133        // Dial the SFU. iroh uses the relay for NAT traversal if needed.
134        let sfu_addr = EndpointAddr::new(sfu_id).with_relay_url(relay_url);
135        let connection =
136            endpoint
137                .connect(sfu_addr, HOUND_ALPN)
138                .await
139                .map_err(|e| VoiceError::SfuConnect {
140                    reason: e.to_string(),
141                })?;
142
143        // Open a bi-directional stream for the Hello handshake.
144        let (mut send, mut recv) =
145            connection
146                .open_bi()
147                .await
148                .map_err(|e| VoiceError::SfuConnect {
149                    reason: e.to_string(),
150                })?;
151
152        // Build and send the Hello message.
153        let hello = HelloMsg {
154            token: relay_token.to_owned(),
155            codecs: HelloCodecs {
156                cam_video: vec!["av1"],
157                scr_video: vec!["h264"],
158                audio: vec!["opus"],
159            },
160        };
161        let hello_bytes = encode_message(&hello).map_err(|e| VoiceError::SfuConnect {
162            reason: e.to_string(),
163        })?;
164        send.write_all(&hello_bytes)
165            .await
166            .map_err(|e| VoiceError::SfuConnect {
167                reason: e.to_string(),
168            })?;
169        send.finish().map_err(|e| VoiceError::SfuConnect {
170            reason: e.to_string(),
171        })?;
172
173        // Read the HelloAck (length-prefixed JSON).
174        let ack_bytes = read_message(&mut recv)
175            .await
176            .map_err(|e| VoiceError::SfuConnect {
177                reason: e.to_string(),
178            })?;
179
180        let server_msg: ServerHandshake =
181            serde_json::from_slice(&ack_bytes).map_err(|e| VoiceError::SfuConnect {
182                reason: e.to_string(),
183            })?;
184
185        let (peer_id, ack_peers) = match server_msg {
186            ServerHandshake::Ack { peer_id, peers } => (peer_id, peers),
187            ServerHandshake::Nack { reason } => {
188                return Err(VoiceError::SfuConnect {
189                    reason: format!("SFU rejected hello: {reason:?}"),
190                });
191            }
192        };
193
194        tracing::info!(peer_id, "SFU handshake complete");
195
196        // Notify the frontend of our peer ID and the current peer list.
197        let _ = app.emit(
198            "voice-sfu-ready",
199            SfuReadyPayload {
200                my_peer_id: peer_id,
201                peers: ack_peers,
202            },
203        );
204
205        // Spawn a background task that reads incoming datagrams and streams
206        // from the SFU and emits them as Tauri events for the frontend.
207        tokio::spawn(receive_loop(connection.clone(), app));
208
209        Ok(Self {
210            _endpoint: endpoint,
211            connection,
212            peer_id,
213            stream_seq: Arc::new(AtomicU32::new(0)),
214        })
215    }
216
217    /// Send an encoded media frame to the SFU.
218    ///
219    /// * Delta frames (`is_keyframe = false`) — sent as unreliable QUIC datagrams.
220    /// * Keyframes (`is_keyframe = true`) — sent as reliable unidirectional streams.
221    ///
222    /// Large payloads are fragmented into ≤`MAX_FRAG_PAYLOAD`-byte chunks, each
223    /// with its own frame header.
224    pub async fn send_frame(
225        &self,
226        payload: &[u8],
227        track: &str,
228        is_keyframe: bool,
229    ) -> Result<(), VoiceError> {
230        let seq = self.stream_seq.fetch_add(1, Ordering::Relaxed);
231        let track_byte = parse_track_type(track);
232        let flags = if is_keyframe { FLAG_KEYFRAME } else { 0 };
233
234        // Determine fragment count.
235        let chunks: Vec<&[u8]> = if payload.is_empty() {
236            vec![payload]
237        } else {
238            payload.chunks(MAX_FRAG_PAYLOAD).collect()
239        };
240        let frag_total = chunks.len().min(255) as u8;
241
242        if is_keyframe {
243            // Reliable delivery: one unidirectional stream per keyframe.
244            let mut stream = self
245                .connection
246                .open_uni()
247                .await
248                .map_err(|e| VoiceError::SfuSend {
249                    reason: e.to_string(),
250                })?;
251
252            for (idx, chunk) in chunks.iter().enumerate() {
253                let header =
254                    build_header(self.peer_id, seq, idx as u8, frag_total, track_byte, flags);
255                stream
256                    .write_all(&header)
257                    .await
258                    .map_err(|e| VoiceError::SfuSend {
259                        reason: e.to_string(),
260                    })?;
261                stream
262                    .write_all(chunk)
263                    .await
264                    .map_err(|e| VoiceError::SfuSend {
265                        reason: e.to_string(),
266                    })?;
267            }
268            stream.finish().map_err(|e| VoiceError::SfuSend {
269                reason: e.to_string(),
270            })?;
271        } else {
272            // Unreliable delivery: one datagram per fragment.
273            for (idx, chunk) in chunks.iter().enumerate() {
274                let header =
275                    build_header(self.peer_id, seq, idx as u8, frag_total, track_byte, flags);
276                let mut packet = Vec::with_capacity(HEADER_SIZE + chunk.len());
277                packet.extend_from_slice(&header);
278                packet.extend_from_slice(chunk);
279                self.connection
280                    .send_datagram(Bytes::from(packet))
281                    .map_err(|e| VoiceError::SfuSend {
282                        reason: e.to_string(),
283                    })?;
284            }
285        }
286
287        Ok(())
288    }
289}

291// ─── Header builder ───────────────────────────────────────────────────────────
292
293fn build_header(
294    peer_id: u64,
295    stream_seq: u32,
296    frag_index: u8,
297    frag_total: u8,
298    track_type: u8,
299    flags: u8,
300) -> [u8; HEADER_SIZE] {
301    let mut h = [0u8; HEADER_SIZE];
302    h[0..8].copy_from_slice(&peer_id.to_le_bytes());
303    h[8..12].copy_from_slice(&stream_seq.to_le_bytes());
304    h[12] = frag_index;
305    h[13] = frag_total;
306    h[14] = track_type;
307    h[15] = flags;
308    // timestamp_us: leave as zero — the SFU doesn't use it for routing
309    h
310}

314#[derive(serde::Serialize)]
315struct HelloMsg {
316    token: String,
317    codecs: HelloCodecs,
318}

320/// Codec preferences sent in the Hello message.
321/// Field names and values must match `ClientCodecs` / `VideoCodecId` /
322/// `AudioCodecId` in hound_sfu (snake_case enum variants).
323#[derive(serde::Serialize)]
324struct HelloCodecs {
325    cam_video: Vec<&'static str>,
326    scr_video: Vec<&'static str>,
327    audio: Vec<&'static str>,
328}
329
330#[derive(serde::Deserialize)]
331#[serde(tag = "type", rename_all = "snake_case")]
332enum ServerHandshake {
333    Ack {
334        peer_id: u64,
335        peers: Vec<AckPeer>,
336        // `codecs` field is present in the SFU response but not needed client-side.
337    },
338    Nack {
339        reason: serde_json::Value,
340    },
341}

343#[derive(serde::Deserialize, serde::Serialize, Clone)]
344pub(crate) struct AckPeer {
345    peer_id: u64,
346    user_id: uuid::Uuid,
347}

356// ─── Background receiver ──────────────────────────────────────────────────────
357
358/// Reads frames from the SFU connection and emits them as Tauri events.
359///
360/// Runs until the connection closes (peer left, network error, etc.).
361/// Frames arrive as QUIC datagrams (delta frames) or unidirectional streams
362/// (keyframes). Both are forwarded as raw bytes to the TypeScript frontend via
363/// the `"voice-frame"` event; the MediaWorker parses the header and decodes.
364async fn receive_loop<R: tauri::Runtime>(
365    connection: iroh::endpoint::Connection,
366    app: tauri::AppHandle<R>,
367) {
368    loop {
369        tokio::select! {
370            // Unreliable path: datagrams (delta video + audio frames).
371            result = connection.read_datagram() => {
372                match result {
373                    Ok(bytes) => emit_frame(&app, bytes),
374                    Err(e) => {
375                        tracing::debug!("SFU datagram read error: {e}");
376                        break;
377                    }
378                }
379            }
380            // Reliable path: unidirectional streams (keyframes).
381            result = connection.accept_uni() => {
382                match result {
383                    Ok(mut recv) => {
384                        // Read the full stream (frames are small enough to buffer).
385                        match recv.read_to_end(4 * 1024 * 1024).await {
386                            Ok(bytes) => emit_frame(&app, Bytes::from(bytes)),
387                            Err(e) => tracing::debug!("SFU stream read error: {e}"),
388                        }
389                    }
390                    Err(e) => {
391                        tracing::debug!("SFU accept_uni error: {e}");
392                        break;
393                    }
394                }
395            }
396        }
397    }
398    tracing::info!("SFU receive loop ended");
399}
400
401fn emit_frame<R: tauri::Runtime>(app: &tauri::AppHandle<R>, bytes: Bytes) {
402    // Emit the raw bytes as a vec of u8. Tauri serialises this as a JSON
403    // number array, which the TypeScript side converts to Uint8Array.
404    if let Err(e) = app.emit("voice-frame", bytes.to_vec()) {
405        tracing::debug!("emit voice-frame failed: {e}");
406    }
407}

409// ─── Length-prefixed JSON framing (mirrors hound_sfu::wire::hello) ───────────
410
411fn encode_message<T: serde::Serialize>(msg: &T) -> Result<Vec<u8>, serde_json::Error> {
412    let json = serde_json::to_vec(msg)?;
413    let len = json.len() as u32;
414    let mut out = Vec::with_capacity(4 + json.len());
415    out.extend_from_slice(&len.to_le_bytes());
416    out.extend_from_slice(&json);
417    Ok(out)
418}
419
420async fn read_message(
421    recv: &mut iroh::endpoint::RecvStream,
422) -> Result<Vec<u8>, Box<dyn std::error::Error + Send + Sync>> {
423    let mut len_buf = [0u8; 4];
424    recv.read_exact(&mut len_buf)
425        .await
426        .map_err(|e| e.to_string())?;
427    let len = u32::from_le_bytes(len_buf) as usize;
428    let mut buf = vec![0u8; len];
429    recv.read_exact(&mut buf).await.map_err(|e| e.to_string())?;
430    Ok(buf)
431}
