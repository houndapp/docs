1//! # Platform-agnostic screen capture
2#![allow(dead_code)]
3//!
4//! Screen capture works very differently on each OS:
5//! * macOS — ScreenCaptureKit (Swift/Obj-C, requires entitlements)
6//! * Windows — DXGI + WASAPI
7//! * Linux — PipeWire or X11 Shared Memory
8//!
9//! Rather than scattering `#[cfg(target_os = "...")]` throughout the codebase,
10//! we define a single [`ScreenCapturer`] trait here. Adding support for a new
11//! platform means adding a new `impl ScreenCapturer` in a new file — callers
12//! never change.
13//!
14//! ## Adding a new platform
15//!
16//! 1. Create `src/voice/screen/<platform>.rs`.
17//! 2. Implement `ScreenCapturer` for your type.
18//! 3. Add a `#[cfg(target_os = "<platform>")]` alias to `PlatformCapturer` at
19//!    the bottom of this file.
20//!
21//! That's it — `commands.rs` uses `PlatformCapturer` without knowing which
22//! implementation it got.
23
24use tokio::sync::mpsc;

26// ─── Raw frame types ──────────────────────────────────────────────────────────
27
28/// A single raw video frame as returned by the OS capture API.
29///
30/// Pixels are in BGRA format (4 bytes per pixel), which is the native format
31/// on macOS (ScreenCaptureKit) and Windows (DXGI). The encoder converts to
32/// the target codec's input format.
33#[derive(Debug)]
34pub struct RawVideoFrame {
35    /// Width in pixels.
36    pub width: u32,
37    /// Height in pixels.
38    pub height: u32,
39    /// Pixel data in BGRA order, `width * height * 4` bytes.
40    pub data: Vec<u8>,
41    /// Capture timestamp in microseconds (from the OS capture callback).
42    pub timestamp_us: i64,
43}
44
45/// A single raw audio frame as returned by the OS capture API.
46///
47/// Samples are 32-bit floats, interleaved by channel
48/// (stereo: L R L R …). Standard sample rate is 48 000 Hz.
49#[derive(Debug)]
50pub struct RawAudioFrame {
51    pub channels: u16,
52    pub sample_rate: u32,
53    /// Interleaved f32 samples.
54    pub data: Vec<f32>,
55    pub timestamp_us: i64,
56}
57
58/// A handle that keeps a screen capture session alive.
59///
60/// Dropping this handle stops capture. The capture stops even if the
61/// underlying OS stream is still running — the `Drop` impl sends a stop
62/// signal to the capture thread.
63pub struct CaptureHandle {
64    /// Sending on this channel signals the capture thread to stop.
65    pub(super) _stop_tx: tokio::sync::oneshot::Sender<()>,
66}
67
68impl CaptureHandle {
69    /// Signals the capture thread to stop cleanly.
70    ///
71    /// This is also called automatically when the handle is dropped.
72    pub fn stop(self) {
73        // Dropping `_stop_tx` closes the channel, which the capture thread
74        // detects and uses as its stop signal. Nothing to do explicitly here.
75        drop(self);
76    }
77}

79/// Identifies a capturable screen or window.
80#[derive(Debug, Clone, serde::Serialize)]
81pub struct CaptureTarget {
82    /// A platform-specific opaque ID (e.g. CGWindowID on macOS).
83    pub id: u64,
84    /// Human-readable name shown in the target picker.
85    pub name: String,
86    /// Whether this target is a full display (vs. a window).
87    pub is_display: bool,
88}
89
90/// Configuration for a capture session.
91#[derive(Debug, Clone)]
92pub struct CaptureConfig {
93    /// The screen or window to capture.
94    pub target: CaptureTarget,
95    /// Maximum frames per second. The OS may deliver fewer.
96    pub max_fps: u32,
97    /// Desired resolution. The OS may return a different size.
98    pub width: u32,
99    pub height: u32,
100}
101
102/// Errors from the screen capture layer.
103#[derive(Debug, thiserror::Error)]
104pub enum CaptureError {
105    /// The platform does not support screen capture.
106    #[error("screen capture is not supported on this platform")]
107    Unsupported,
108
109    /// The user denied the screen recording permission.
110    #[error("screen recording permission denied — the user must grant it in System Settings")]
111    PermissionDenied,
112
113    /// The requested capture target was not found (e.g. window closed).
114    #[error("capture target {target_id} not found")]
115    TargetNotFound { target_id: u64 },
116
117    /// The OS capture API returned an unexpected error.
118    #[error("OS capture error: {reason}")]
119    Os { reason: String },
120}

122// ─── The trait ───────────────────────────────────────────────────────────────
123
124/// A screen and audio capturer for the current platform.
125///
126/// Implement this trait to add support for a new platform. Callers use
127/// [`PlatformCapturer`] (a type alias) and never depend on the concrete type.
128pub trait ScreenCapturer: Send + 'static {
129    /// List all capturable screens and windows on this device.
130    ///
131    /// This is a synchronous call — it should return quickly (< 100 ms).
132    fn list_targets() -> Result<Vec<CaptureTarget>, CaptureError>;
133
134    /// Start capturing the given target and stream frames to the callers.
135    ///
136    /// Returns a [`CaptureHandle`] immediately; capture runs in a background
137    /// task. Frames arrive on `video_tx` and `audio_tx`. Drop the handle or
138    /// call [`CaptureHandle::stop`] to end capture.
139    ///
140    /// `audio_tx` may receive no frames on platforms/targets where system
141    /// audio capture is unavailable (e.g. Linux without PipeWire audio).
142    fn start(
143        config: CaptureConfig,
144        video_tx: mpsc::Sender<RawVideoFrame>,
145        audio_tx: mpsc::Sender<RawAudioFrame>,
146    ) -> Result<CaptureHandle, CaptureError>;
147}

149// ─── Stub capturer (used when no platform impl is available) ─────────────────
150
151/// A capturer that always returns an error.
152///
153/// Used on platforms we don't yet support. This lets the code compile
154/// everywhere; the user sees a clear "Unsupported" error at runtime rather
155/// than a compile failure.
156pub struct StubCapturer;
157
158impl ScreenCapturer for StubCapturer {
159    fn list_targets() -> Result<Vec<CaptureTarget>, CaptureError> {
160        Err(CaptureError::Unsupported)
161    }
162
163    fn start(
164        _config: CaptureConfig,
165        _video_tx: mpsc::Sender<RawVideoFrame>,
166        _audio_tx: mpsc::Sender<RawAudioFrame>,
167    ) -> Result<CaptureHandle, CaptureError> {
168        Err(CaptureError::Unsupported)
169    }
170}

172// ─── scap-backed capturer ────────────────────────────────────────────────────
173
174// `scap` supports macOS (ScreenCaptureKit), Windows (DXGI + WASAPI), and
175// Linux (PipeWire). It is only compiled in on those platforms so we don't
176// pull in heavy native dependencies on unsupported targets.
177#[cfg(any(target_os = "macos", target_os = "windows", target_os = "linux"))]
178mod scap_capturer;

180// ─── PlatformCapturer alias ───────────────────────────────────────────────────
181
182/// The screen capturer implementation for the current platform.
183///
184/// Use this alias everywhere instead of a concrete type so that adding a
185/// new platform only requires a new `impl ScreenCapturer` file, not changes
186/// to callers.
187#[cfg(any(target_os = "macos", target_os = "linux"))]
188pub use scap_capturer::ScapCapturer as PlatformCapturer;
189
190#[cfg(not(any(target_os = "macos", target_os = "linux")))]
191pub use StubCapturer as PlatformCapturer;
