1//! # Tauri voice commands
2//!
3//! These are the functions the TypeScript frontend calls via `invoke(...)`.
4//! Each command maps to one specific action in the voice lifecycle:
5//!
6//! ```text
7//! voice_join  → calls backend /space/{id}/voice/join, stores relay token
8//! voice_leave → calls backend /space/{id}/voice/leave, clears state
9//! voice_list_capture_targets → lists capturable screens/windows
10//! ```
11//!
12//! Commands that require an active session check [`VoiceState`] first and
13//! return [`VoiceError::NotConnected`] if there is no active session.
14//!
15//! ## Error handling
16//!
17//! All commands return `Result<T, VoiceError>`. On the TypeScript side,
18//! `invoke` rejects with a `VoiceError` object `{ kind, message }` that
19//! can be pattern-matched on `kind` for user-facing messages.

36/// The response body from `POST /space/{id}/voice/join`.
37///
38/// Mirrors `VoiceJoinResponse` in the backend. Fields use snake_case because
39/// the backend serialises them that way.
40#[derive(serde::Deserialize)]
41struct JoinResponse {
42    relay_url: String,
43    relay_token: String,
44    profile: JoinProfile,
45    peers: Vec<JoinPeer>,
46    /// iroh node ID of the SFU (hex-encoded public key).
47    ///
48    /// `None` when the backend has not yet been updated to return this field —
49    /// the voice channel degrades gracefully (no SFU connection).
50    #[serde(default)]
51    sfu_node_id: Option<String>,
52}

70/// What we hand back to the TypeScript frontend after a successful join.
71///
72/// The frontend uses this to render the local participant tile and the
73/// initial peer list, and to configure the MediaWorker.
74#[derive(serde::Serialize)]
75#[serde(rename_all = "camelCase")]
76pub struct VoiceJoinResult {
77    /// The joining user's profile (for the local tile).
78    pub profile: JoinProfile,
79    /// Other participants already in the channel.
80    pub peers: Vec<JoinPeer>,
81    /// The iroh relay URL, passed to the MediaWorker so it can configure
82    /// the iroh endpoint. Never hardcoded — always use what the server says.
83    pub relay_url: String,
84    /// The relay token to present to the SFU on connect.
85    pub relay_token: String,
86}

90/// Join a voice channel.
91///
92/// Calls `POST /space/{space_id}/voice/join` on the backend, stores the relay
93/// token in app state, and returns the join result to the frontend.
94///
95/// The frontend is responsible for:
96/// 1. Passing `relay_token` + `relay_url` to the MediaWorker.
97/// 2. Rendering the initial peer list from `peers`.
98/// 3. Starting the heartbeat timer.
99///
100/// # Errors
101///
102/// * [`VoiceError::JoinFailed`] — backend returned a non-2xx response.
103/// * [`VoiceError::Network`] — request failed at the transport level.
104#[tauri::command]
105pub async fn voice_join<R: Runtime>(
106    app: AppHandle<R>,
107    space_id: Uuid,
108    api_base: String,
109    access_token: String,
110    state: State<'_, VoiceState>,
111) -> Result<VoiceJoinResult, VoiceError> {
112    let url = format!("{api_base}/space/{space_id}/voice/join");
113
114    let response = reqwest::Client::new()
115        .post(&url)
116        .bearer_auth(&access_token)
117        .send()
118        .await
119        .map_err(|source| VoiceError::Network {
120            endpoint: "voice/join",
121            source,
122        })?;
123
124    if !response.status().is_success() {
125        return Err(VoiceError::JoinFailed {
126            status: response.status().as_u16(),
127            detail: response.text().await.unwrap_or_default(),
128        });
129    }
130
131    let body: JoinResponse = response
132        .json()
133        .await
134        .map_err(|source| VoiceError::Network {
135            endpoint: "voice/join (parse)",
136            source,
137        })?;
138
139    // Connect to the SFU if the backend provided its node ID.
140    //
141    // When the backend advertises an `sfu_node_id`, the SFU is deployed and a
142    // failure to reach it is a hard error — NOT something to swallow. Previously
143    // a failed connect was logged and the join "succeeded" with no media path,
144    // so the UI showed the user as connected while no audio or video ever
145    // flowed: a silent failure that looked exactly like a working call. We now
146    // surface it so the user (and the logs) see why the call is dead.
147    let sfu = match &body.sfu_node_id {
148        Some(node_id) => {
149            let relay_url: iroh::RelayUrl =
150                body.relay_url.parse().map_err(|_| VoiceError::SfuConnect {
151                    reason: format!("invalid relay URL: {}", body.relay_url),
152                })?;
153            let client = SfuClient::connect(relay_url, node_id, &body.relay_token, app.clone())
154                .await
155                .inspect_err(|e| tracing::error!("SFU connect failed: {e}"))?;
156            tracing::info!(peer_id = client.peer_id, "connected to SFU");
157            Some(Arc::new(client))
158        }
159        None => {
160            // No SFU configured on the backend at all. This is a deployment
161            // state, not a transient failure; log loudly so it is not mistaken
162            // for a working call, but allow the (media-less) join to proceed.
163            tracing::warn!(
164                "backend returned no sfu_node_id — joining WITHOUT a media path; \
165                 no audio or video will flow until the SFU is configured"
166            );
167            None
168        }
169    };
170
171    // Always confirm with the backend — this triggers the voice:join broadcast
172    // to other participants. sfu_peer_id is None when we have no SFU connection
173    // (relay-only mode or SFU not configured); participants still appear, they
174    // just won't have video mapped until an SFU session is established.
175    {
176        #[derive(serde::Serialize)]
177        struct ConfirmRequest {
178            sfu_peer_id: Option<u16>,
179        }
180        let url = format!("{api_base}/space/{space_id}/voice/confirm");
181        let peer_id = sfu.as_ref().map(|c| c.peer_id as u16);
182        tracing::info!(sfu_peer_id = ?peer_id, "confirming voice join with backend");
183        match reqwest::Client::new()
184            .post(&url)
185            .bearer_auth(&access_token)
186            .json(&ConfirmRequest {
187                sfu_peer_id: peer_id,
188            })
189            .send()
190            .await
191        {
192            Ok(resp) if resp.status().is_success() => {
193                tracing::info!("voice confirm succeeded");
194            }
195            Ok(resp) => {
196                let status = resp.status().as_u16();
197                let body = resp.text().await.unwrap_or_default();
198                tracing::warn!(status, body, "voice confirm returned non-2xx");
199            }
200            Err(e) => {
201                tracing::warn!("voice confirm request failed: {e}");
202            }
203        }
204    }
205
206    // Store the session so other commands can check we're connected.
207    *state.session.lock().await = Some(ActiveSession { space_id, sfu });
208
209    tracing::info!(%space_id, relay_url=%body.relay_url, "joined voice channel");
210
211    Ok(VoiceJoinResult {
212        profile: body.profile,
213        peers: body.peers,
214        relay_url: body.relay_url,
215        relay_token: body.relay_token,
216    })
217}

219/// Leave the current voice channel.
220///
221/// Clears session state and notifies the backend. Safe to call even if not
222/// connected (returns `Ok(())` in that case).
223///
224/// # Errors
225///
226/// * [`VoiceError::LeaveFailed`] — backend returned a non-2xx response.
227/// * [`VoiceError::Network`] — request failed at the transport level.
228#[tauri::command]
229pub async fn voice_leave(
230    api_base: String,
231    access_token: String,
232    state: State<'_, VoiceState>,
233) -> Result<(), VoiceError> {
234    let space_id = {
235        let mut guard = state.session.lock().await;
236        match guard.take() {
237            Some(s) => s.space_id,
238            None => return Ok(()), // not connected — nothing to do
239        }
240    };
241
242    let url = format!("{api_base}/space/{space_id}/voice/leave");
243
244    let response = reqwest::Client::new()
245        .post(&url)
246        .bearer_auth(&access_token)
247        .send()
248        .await
249        .map_err(|source| VoiceError::Network {
250            endpoint: "voice/leave",
251            source,
252        })?;
253
254    if !response.status().is_success() {
255        return Err(VoiceError::LeaveFailed {
256            status: response.status().as_u16(),
257        });
258    }
259
260    tracing::info!(%space_id, "left voice channel");
261    Ok(())
262}

264/// List capturable screens and windows on this device.
265///
266/// Returns an empty list on platforms where capture is not yet supported
267/// (rather than an error) so the UI can show "no targets" gracefully.
268///
269/// # Errors
270///
271/// * [`VoiceError::StateNotInitialised`] — Tauri state was not set up.
272#[tauri::command]
273pub async fn voice_list_capture_targets() -> Result<Vec<CaptureTarget>, VoiceError> {
274    PlatformCapturer::list_targets().map_err(|e| {
275        tracing::warn!("list_capture_targets: {e}");
276        // Map capture errors to voice errors. CaptureError::Unsupported returns
277        // an empty list (handled above) — only OS errors reach here.
278        VoiceError::NotConnected {
279            operation: "list capture targets",
280        }
281    })
282}

284/// Send an encoded media frame to the SFU over QUIC.
285///
286/// Called by the TypeScript frontend (via the MediaWorker → VoiceClient →
287/// `invoke`) whenever the local encoder produces a new frame.
288///
289/// * `frame` — raw encoded payload bytes (no header — we add it here)
290/// * `is_keyframe` — if true, the frame is sent reliably (QUIC stream);
291///   otherwise unreliably (QUIC datagram)
292/// * `track` — one of `"cam-video"`, `"cam-audio"`, `"scr-video"`, `"scr-audio"`
293///
294/// # Errors
295///
296/// * [`VoiceError::NotConnected`] — `voice_join` has not been called.
297/// * [`VoiceError::SfuSend`] — the QUIC send failed.
298#[tauri::command]
299pub async fn voice_send_frame(
300    frame: Vec<u8>,
301    is_keyframe: bool,
302    track: String,
303    state: State<'_, VoiceState>,
304) -> Result<(), VoiceError> {
305    let session = state.session.lock().await;
306    let sfu = session
307        .as_ref()
308        .and_then(|s| s.sfu.as_ref())
309        .ok_or(VoiceError::NotConnected {
310            operation: "send frame",
311        })?
312        .clone();
313    // Drop the lock before the async send so other commands aren't blocked.
314    drop(session);
315
316    sfu.send_frame(&frame, &track, is_keyframe).await
317}
