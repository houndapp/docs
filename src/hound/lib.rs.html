1//! # hound.chat Native API
2//! This crate contains all of the definitions for the native API for hound's
3//! desktop & mobile app. If you are adding features to the `ui`, this is the
4//! documentation you should use to find what is available in [`commands`].
5//!
6//! If you are adding new features to the native API and need to know what is
7//! available from the backend, you want to check out the [`backend`] crate.

14/// The entrypoint for the native app.
15#[cfg_attr(mobile, tauri::mobile_entry_point)]
16pub fn run() {
17    // The CLI plugin is desktop-only (`tauri-plugin-cli` has no mobile target,
18    // and command-line args are meaningless on iOS/Android); it is registered
19    // under `#[cfg(desktop)]` in `setup()`.
20    tauri::Builder::default()
21        .plugin(tauri_plugin_http::init())
22        .plugin(tauri_plugin_websocket::init())
23        // Voice state is stored for the lifetime of the app.
24        // VoiceState tracks the active session (relay token, space ID, etc.).
25        .manage(voice::VoiceState::new())
26        .setup(setup)
27        .invoke_handler(tauri::generate_handler![
28            passkey_device_name,
29            voice::commands::voice_join,
30            voice::commands::voice_leave,
31            voice::commands::voice_list_capture_targets,
32            voice::commands::voice_send_frame,
33            spike::spike_render_start,
34            spike::spike_render_stop,
35        ])
36        .run(tauri::generate_context!())
37        .expect("error while running tauri application");
38}

40/// Customises the tauri setup for hound.
41///
42/// On native platforms this will setup logging, plugins, and if in dev mode
43/// spawn a local version of the backend for dev work.
44fn setup<W: tauri::Runtime>(app: &mut tauri::App<W>) -> Result<(), Box<dyn std::error::Error>> {
45    // Logging. `tracing` events from the voice subsystem reach this via the
46    // `tracing` crate's `log` feature (see Cargo.toml). We route to:
47    //   * Stdout  — visible in the terminal / container logs (incl. release, so
48    //               prod failures like a failed SFU connect are no longer silent);
49    //   * Webview — emits a `log://log` event that the frontend prints to the dev
50    //               console (see `ui/src/lib/log.ts`).
51    // Registered unconditionally (not just debug builds) so release builds also
52    // produce diagnostics; the noisy iroh relay-actor reconnect WARNs are dropped
53    // to ERROR since iroh reconnects automatically and they are expected.
54    use log::LevelFilter;
55    app.handle().plugin(
56        tauri_plugin_log::Builder::default()
57            .level(LevelFilter::Info)
58            // The `tracing` → `log` bridge turns every span enter into a `log`
59            // record under the `tracing::span` target. iroh's transport is a
60            // torrent of INFO spans (one `poll_send` per packet, plus `tx`,
61            // `do_holepunching`, `select_path`, …), so unfiltered the dev console
62            // is unusable. We keep our own `hound::*` logs at Info and clamp the
63            // noisy dependency surface:
64            //   * `tracing::span`   — span lifecycle records (pure noise); off.
65            //   * `iroh`            — per-packet/path INFO; warn keeps genuine
66            //                         connectivity warnings (relay lost, probes).
67            //   * relay actor       — expected reconnect churn; error only.
68            // Whether media is actually flowing is shown by the periodic
69            // "voice transport" line emitted from stats_loop (sfu_client.rs).
70            .level_for("tracing::span", LevelFilter::Off)
71            .level_for("iroh", LevelFilter::Warn)
72            .level_for("iroh::socket::transports::relay::actor", LevelFilter::Error)
73            .targets([
74                tauri_plugin_log::Target::new(tauri_plugin_log::TargetKind::Stdout),
75                tauri_plugin_log::Target::new(tauri_plugin_log::TargetKind::Webview),
76            ])
77            .build(),
78    )?;
79    #[cfg(desktop)]
80    app.handle().plugin(tauri_plugin_cli::init())?;
81
82    // let mut backend_cli = backend::cli::Cli::default();
83
84    // #[cfg(desktop)]
85    // if let Ok(matches) = app.cli().matches()
86    //     && let Some(arg) = matches.args.get("host").and_then(|arg| arg.value.as_str())
87    // {
88    //     backend_cli.host = arg.parse().unwrap();
89    // }
90
91    // let (tx, rx) = tokio::sync::oneshot::channel();
92    // tauri::async_runtime::spawn(async move {
93    //     let (address, task) = backend::server::serve(backend_cli).await?;
94
95    //     tx.send(address).unwrap();
96    //     task.await
97    // });
98    // let address = tauri::async_runtime::block_on(rx).unwrap();
99
100    // app.manage(backend::Client::init(address).unwrap());
101    Ok(())
102}

104#[tauri::command]
105fn passkey_device_name() -> String {
106    format!(
107        "{} – {}",
108        whoami::distro().unwrap_or_else(|_| "unknown operating system".into()),
109        whoami::devicename().unwrap_or_else(|_| "unknown device".into()),
110    )
