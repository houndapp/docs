1//! Client-side session state derived purely from [`ServerMessage`]s.
2//!
3//! [`SessionView`] is a deterministic reducer: feed it the server messages as
4//! they arrive and it maintains the current roster of peers and tracks. It holds
5//! no I/O and makes no requests — the UI and media engine read from it, and a
6//! [`ServerMessage::Welcome`] fully resyncs it (used on reconnect), so a dropped
7//! socket never leaves stale state behind.
8
9use std::collections::BTreeMap;
10
11use crate::proto::{DecodeCaps, PeerId, PeerInfo, ServerMessage, TrackId, TrackInfo};
12
13/// The current view of a call: who is present and what tracks exist.
14#[derive(Debug, Default, Clone, PartialEq, Eq)]
15pub struct SessionView {
16    self_id: Option<PeerId>,
17    protocol_version: Option<u16>,
18    peers: BTreeMap<PeerId, PeerInfo>,
19    tracks: BTreeMap<TrackId, TrackInfo>,
20}
21
22impl SessionView {
23    pub fn new() -> Self {
24        Self::default()
25    }
26
27    /// Apply one server message, mutating the view. `Welcome` replaces all state
28    /// (initial sync / reconnect resync); other messages are incremental.
29    /// Messages that carry no roster state (`KeyframeRequested`, `Error`) are
30    /// no-ops here — the caller handles them out of band.
31    pub fn apply(&mut self, msg: &ServerMessage) {
32        match msg {
33            ServerMessage::Welcome {
34                protocol_version,
35                self_id,
36                peers,
37                tracks,
38            } => {
39                self.protocol_version = Some(*protocol_version);
40                self.self_id = Some(*self_id);
41                self.peers = peers.iter().map(|p| (p.id, *p)).collect();
42                self.tracks = tracks.iter().map(|t| (t.id, t.clone())).collect();
43            }
44            ServerMessage::PeerJoined { peer } => {
45                self.peers.insert(peer.id, *peer);
46            }
47            ServerMessage::PeerLeft { peer } => {
48                self.peers.remove(peer);
49                // A departed peer's tracks are gone too; drop them so we don't
50                // leak dead subscriptions.
51                self.tracks.retain(|_, t| &t.peer != peer);
52            }
53            ServerMessage::TrackPublished { track } => {
54                self.tracks.insert(track.id, track.clone());
55            }
56            ServerMessage::TrackUnpublished { track } => {
57                self.tracks.remove(track);
58            }
59            ServerMessage::MuteChanged { track, muted } => {
60                if let Some(t) = self.tracks.get_mut(track) {
61                    t.muted = *muted;
62                }
63            }
64            ServerMessage::KeyframeRequested { .. }
65            | ServerMessage::MediaFeedbackReceived { .. }
66            | ServerMessage::QualityRequested { .. }
67            | ServerMessage::Error { .. }
68            // Carries no roster state; the IO shell (`ControlClient`'s reader
69            // loop) reacts to it by tearing the connection down, not the view.
70            | ServerMessage::SfuMigrate { .. }
71            // No track was added server-side (the rejected `Publish` never
72            // produced a `TrackPublished`), so there is nothing to reconcile here
73            // either; the IO shell just logs it (see `control.rs`).
74            | ServerMessage::PublishRejected { .. } => {}
75        }
76    }
77
78    /// This client's own peer id, once `Welcome` has been received.
79    pub fn self_id(&self) -> Option<PeerId> {
80        self.self_id
81    }
82
83    /// Protocol version the server reported in `Welcome`.
84    pub fn protocol_version(&self) -> Option<u16> {
85        self.protocol_version
86    }
87
88    /// All participants currently in the call (including self).
89    pub fn peers(&self) -> impl Iterator<Item = &PeerInfo> {
90        self.peers.values()
91    }
92
93    /// Look up a single peer's full info (including its authenticated
94    /// [`PeerInfo::user_id`]) — the lookup a caller building a media broadcast
95    /// path for a remote peer needs (see `hound_client::call::subscribe`).
96    pub fn peer(&self, id: PeerId) -> Option<&PeerInfo> {
97        self.peers.get(&id)
98    }
99
100    /// All known tracks (local and remote).
101    pub fn tracks(&self) -> impl Iterator<Item = &TrackInfo> {
102        self.tracks.values()
103    }
104
105    /// Look up a single track.
106    pub fn track(&self, id: TrackId) -> Option<&TrackInfo> {
107        self.tracks.get(&id)
108    }
109
110    /// Tracks produced by `peer`.
111    pub fn tracks_of(&self, peer: PeerId) -> impl Iterator<Item = &TrackInfo> {
112        self.tracks.values().filter(move |t| t.peer == peer)
113    }
114
115    /// Tracks produced by other participants — i.e. what this client can
116    /// subscribe to and render. Empty until `self_id` is known.
117    pub fn remote_tracks(&self) -> impl Iterator<Item = &TrackInfo> {
118        let me = self.self_id;
119        self.tracks
120            .values()
121            .filter(move |t| me.is_some_and(|me| t.peer != me))
122    }
123
124    /// The codecs **every other participant** can decode — the intersection a
125    /// producer must satisfy, since the forward-no-transcode SFU sends one
126    /// encoding to all subscribers (so the chosen codec must play on the weakest
127    /// peer). Self is excluded (you never send video to yourself).
128    ///
129    /// Returns [`DecodeCaps::ALL`] when alone (no remote peer constrains the
130    /// choice, so the producer may pick the best it can *encode*). Feed the result
131    /// — mapped to the engine's caps — into `pick_encode_codec` alongside this
132    /// device's encode capabilities. Recompute on every roster change
133    /// (`PeerJoined`/`PeerLeft`/`Welcome`): a less-capable peer joining shrinks it
134    /// (forcing a downgrade + republish), and their leaving grows it back.
135    pub fn common_remote_decode_caps(&self) -> DecodeCaps {
136        let me = self.self_id;
137        self.peers
138            .iter()
139            .filter(|(id, _)| me != Some(**id))
140            .fold(DecodeCaps::ALL, |acc, (_, p)| acc.intersect(p.decode_caps))
141    }
142}

144#[cfg(test)]
145mod tests {
146    use super::*;
147    use crate::proto::{Codec, MediaKind, Source};
148
149    fn peer(id: u32) -> PeerInfo {
150        peer_with(id, DecodeCaps::ALL)
151    }
152
153    fn peer_with(id: u32, decode_caps: DecodeCaps) -> PeerInfo {
154        PeerInfo {
155            id: PeerId(id),
156            decode_caps,
157            user_id: uuid::Uuid::nil(),
158        }
159    }
160
161    fn track(id: u32, peer: u32, kind: MediaKind, source: Source) -> TrackInfo {
162        TrackInfo {
163            id: TrackId(id),
164            peer: PeerId(peer),
165            kind,
166            source,
167            codec: if matches!(kind, MediaKind::Audio) {
168                Codec::Opus
169            } else {
170                Codec::Av1
171            },
172            muted: false,
173            low: false,
174        }
175    }
176
177    fn welcome_two_peers() -> ServerMessage {
178        ServerMessage::Welcome {
179            protocol_version: crate::proto::PROTOCOL_VERSION,
180            self_id: PeerId(1),
181            peers: vec![peer(1), peer(2)],
182            tracks: vec![
183                track(10, 2, MediaKind::Video, Source::Camera),
184                track(11, 2, MediaKind::Audio, Source::Microphone),
185            ],
186        }
187    }
188
189    #[test]
190    fn welcome_populates_and_sets_self() {
191        let mut s = SessionView::new();
192        s.apply(&welcome_two_peers());
193        assert_eq!(s.self_id(), Some(PeerId(1)));
194        assert_eq!(s.protocol_version(), Some(crate::proto::PROTOCOL_VERSION));
195        assert_eq!(s.peers().count(), 2);
196        assert_eq!(s.tracks().count(), 2);
197    }
198
199    #[test]
200    fn peer_looks_up_a_single_roster_entry_by_id() {
201        let mut s = SessionView::new();
202        s.apply(&ServerMessage::Welcome {
203            protocol_version: crate::proto::PROTOCOL_VERSION,
204            self_id: PeerId(1),
205            peers: vec![peer_with(1, DecodeCaps::ALL), {
206                let mut p = peer_with(2, DecodeCaps::ALL);
207                p.user_id = uuid::Uuid::from_u128(42);
208                p
209            }],
210            tracks: vec![],
211        });
212        assert_eq!(
213            s.peer(PeerId(2)).map(|p| p.user_id),
214            Some(uuid::Uuid::from_u128(42)),
215            "peer() must return the roster entry's authenticated user_id"
216        );
217        assert!(s.peer(PeerId(99)).is_none(), "unknown peer id is None");
218    }

220    #[test]
221    fn remote_tracks_excludes_self() {
222        let mut s = SessionView::new();
223        s.apply(&welcome_two_peers());
224        // I'm peer 1; my own publish should not appear in remote_tracks.
225        s.apply(&ServerMessage::TrackPublished {
226            track: track(20, 1, MediaKind::Audio, Source::Microphone),
227        });
228        assert_eq!(s.tracks().count(), 3);
229        let remote: Vec<_> = s.remote_tracks().map(|t| t.id).collect();
230        assert_eq!(remote, vec![TrackId(10), TrackId(11)]);
231    }

236        s.apply(&welcome_two_peers());

238        assert_eq!(s.peers().count(), 1);

245        s.apply(&welcome_two_peers());

262    #[test]
263    fn welcome_resyncs_clearing_stale_state() {
264        let mut s = SessionView::new();
265        s.apply(&welcome_two_peers());
266        s.apply(&ServerMessage::PeerJoined { peer: peer(3) });
267        assert_eq!(s.peers().count(), 3);
268
269        // Reconnect: a fresh Welcome must replace everything, not merge.
270        s.apply(&ServerMessage::Welcome {
271            protocol_version: crate::proto::PROTOCOL_VERSION,
272            self_id: PeerId(1),
273            peers: vec![peer(1)],
274            tracks: vec![],
275        });
276        assert_eq!(s.peers().count(), 1);
277        assert_eq!(s.tracks().count(), 0);
278    }

280    #[test]
281    fn common_remote_decode_caps_intersects_others_excludes_self() {
282        let mut s = SessionView::new();
283        // I'm peer 1 (all codecs). Alone → no constraint (ALL).
284        s.apply(&ServerMessage::Welcome {
285            protocol_version: crate::proto::PROTOCOL_VERSION,
286            self_id: PeerId(1),
287            peers: vec![peer_with(1, DecodeCaps::ALL)],
288            tracks: vec![],
289        });
290        assert_eq!(
291            s.common_remote_decode_caps(),
292            DecodeCaps::ALL,
293            "alone: my own (all) caps don't constrain me; ALL identity"
294        );
295
296        // An all-codecs peer joins → still ALL.
297        s.apply(&ServerMessage::PeerJoined {
298            peer: peer_with(2, DecodeCaps::ALL),
299        });
300        assert_eq!(s.common_remote_decode_caps(), DecodeCaps::ALL);
301
302        // An H.264-only peer joins → the common set collapses to the floor, which
303        // is exactly what forces a producer to downgrade to H.264.
304        s.apply(&ServerMessage::PeerJoined {
305            peer: peer_with(3, DecodeCaps::default()),
306        });
307        assert_eq!(s.common_remote_decode_caps(), DecodeCaps::default());
308
309        // That peer leaves → the constraint lifts back to ALL (re-upgrade path).
310        s.apply(&ServerMessage::PeerLeft { peer: PeerId(3) });
311        assert_eq!(s.common_remote_decode_caps(), DecodeCaps::ALL);
312    }

314    #[test]
315    fn non_roster_messages_are_noops() {
316        let mut s = SessionView::new();
317        s.apply(&welcome_two_peers());
318        let before = s.clone();
319        s.apply(&ServerMessage::KeyframeRequested { track: TrackId(10) });
320        s.apply(&ServerMessage::MediaFeedbackReceived {
321            track: TrackId(10),
322            from: PeerId(2),
323            received: 30,
324            lost: 1,
325            jitter_us: Some(2_000),
326        });
327        s.apply(&ServerMessage::Error {
328            code: crate::proto::ErrorCode::NotFound,
329            message: "nope".into(),
330        });
331        assert_eq!(s, before);
332    }
