1//! Wire types for the call control plane.
2//!
3//! A **structured, versioned** message protocol carried over WebSocket.
4//! It manages the media *session* — tracks,
5//! codecs, mute, keyframe requests, reconnect/resync — on top of the existing
6//! participant roster (HTTP join + socket.io). It is transport-agnostic: these
7//! types serialize with serde (JSON on the wire) and carry no I/O.
8
9use serde::{Deserialize, Serialize};
10
11/// Bumped on any breaking change to the message set below. [`ClientMessage::Hello`]
12/// and [`ServerMessage::Welcome`] carry it so a mismatch is detected at connect
13/// time rather than as a confusing downstream parse failure.
14pub const PROTOCOL_VERSION: u16 = 1;
15
16/// Stable identifier for a participant within a call (assigned by the server).
17#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, Serialize, Deserialize)]
18#[serde(transparent)]
19pub struct PeerId(pub u32);

46/// Negotiated codec for a track. Mirrors `hound_media::Codec` (keep in sync).
47#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
48#[serde(rename_all = "snake_case")]
49pub enum Codec {
50    Av1,
51    H264,
52    Opus,
53    H265,
54}

56impl Codec {
57    /// A short uppercase label for UI/stats (e.g. the RTC debug panel's codec
58    /// row). Stable across the wire-codec set, including future additions.
59    pub fn label(self) -> &'static str {
60        match self {
61            Codec::Av1 => "AV1",
62            Codec::H264 => "H264",
63            Codec::Opus => "Opus",
64            Codec::H265 => "H265",
65        }
66    }
67}

69/// Which video codecs a peer can **decode**, advertised at join. The sender of a
70/// video track intersects this across every other participant and encodes the
71/// highest-priority codec they can all decode (the forward-no-transcode SFU sends
72/// one encoding to all subscribers, so the sender must satisfy the weakest peer).
73///
74/// H.264 is the universal floor — every client can decode it (HW or SW openh264)
75/// — so [`Default`] is "H.264 only", which is also what an older peer that omits
76/// the field deserializes to (`#[serde(default)]` on the carrying fields). That
77/// makes the absence of the field safe: such a peer is conservatively assumed to
78/// decode only H.264, so no one ever sends it a codec it cannot play.
79#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
80pub struct DecodeCaps {
81    pub av1: bool,
82    pub h265: bool,
83    pub h264: bool,
84}
85
86impl Default for DecodeCaps {
87    fn default() -> Self {
88        // The universal floor: H.264 only. An un-upgraded peer (no field on the
89        // wire) lands here, so a sender never picks a codec it cannot decode.
90        Self {
91            av1: false,
92            h265: false,
93            h264: true,
94        }
95    }
96}
97
98impl DecodeCaps {
99    /// Every video codec — the identity for [`Self::intersect`] folds (and what a
100    /// producer alone in a call may target before any receiver constrains it).
101    pub const ALL: DecodeCaps = DecodeCaps {
102        av1: true,
103        h265: true,
104        h264: true,
105    };
106
107    /// Does this peer decode `codec`? Non-video codecs (Opus) are never part of
108    /// the video ladder, so they report `false`.
109    pub fn supports(self, codec: Codec) -> bool {
110        match codec {
111            Codec::Av1 => self.av1,
112            Codec::H265 => self.h265,
113            Codec::H264 => self.h264,
114            Codec::Opus => false,
115        }
116    }
117
118    /// The codecs **both** peers can decode — used to fold the roster down to the
119    /// set every receiver supports. H.264 stays set as long as both have it (the
120    /// floor never drops out in practice).
121    pub fn intersect(self, other: Self) -> Self {
122        Self {
123            av1: self.av1 && other.av1,
124            h265: self.h265 && other.h265,
125            h264: self.h264 && other.h264,
126        }
127    }
128}

130/// A participant in the call. Profile/display data lives in the existing roster
131/// (HTTP join + socket.io); this is just the session-level identity the media
132/// plane routes by, plus the codecs it can decode (for sender-side selection).
133#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
134pub struct PeerInfo {
135    pub id: PeerId,
136    /// Codecs this peer can decode. Omitted by older peers → [`DecodeCaps::default`]
137    /// (H.264 only), the safe floor.
138    #[serde(default)]
139    pub decode_caps: DecodeCaps,
140}

142/// Full description of a published track, as the server advertises it.
143#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
144pub struct TrackInfo {
145    pub id: TrackId,
146    pub peer: PeerId,
147    pub kind: MediaKind,
148    pub source: Source,
149    pub codec: Codec,
150    /// Producer-signalled mute (the track exists but is currently silent/black).
151    pub muted: bool,
152    /// `true` for the LOW simulcast layer (see [`TrackDecl::low`]). A receiver picks
153    /// the source or low track of a (peer, source) by its demand. Additive-compat
154    /// (omitted on the wire when `false`).
155    #[serde(default, skip_serializing_if = "is_false")]
156    pub low: bool,
157}
158
159/// What the client proposes when publishing; the server assigns the [`TrackId`]
160/// and echoes the full [`TrackInfo`] back via [`ServerMessage::TrackPublished`].
161#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
162pub struct TrackDecl {
163    pub kind: MediaKind,
164    pub source: Source,
165    pub codec: Codec,
166    /// `true` for the LOW simulcast layer of a video source — a separate,
167    /// lower-resolution track a constrained / small-tile receiver subscribes to
168    /// instead of the full source. Additive-compat: omitted on the wire when `false`
169    /// (older clients never see it; the default is the full/source layer).
170    #[serde(default, skip_serializing_if = "is_false")]
171    pub low: bool,
172}
173
174/// `skip_serializing_if` helper: keep additive bool flags off the wire when unset.
175fn is_false(b: &bool) -> bool {
176    !*b
177}

179/// Why the server rejected a request or closed the session.
180#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
181#[serde(rename_all = "snake_case")]
182pub enum ErrorCode {
183    /// Client `protocol_version` is incompatible with the server's.
184    VersionMismatch,
185    /// Referenced a peer or track that does not exist.
186    NotFound,
187    /// Not permitted (e.g. subscribing without a valid session).
188    Unauthorized,
189    /// Server-side failure.
190    Internal,
191}

193/// Client → server. Carries the participant's intent for the media session.
194#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
195#[serde(tag = "type", rename_all = "snake_case")]
196pub enum ClientMessage {
197    /// First message; negotiates the protocol version and advertises which
198    /// codecs this client can decode (so producers pick a codec it can play).
199    /// `decode_caps` is omitted by older clients → [`DecodeCaps::default`].
200    Hello {
201        protocol_version: u16,
202        #[serde(default)]
203        decode_caps: DecodeCaps,
204    },
205    /// Advertise a track this client is now sending.
206    Publish { track: TrackDecl },
207    /// Stop sending a previously published track.
208    Unpublish { track: TrackId },
209    /// Ask to receive a remote track.
210    Subscribe { track: TrackId },
211    /// Stop receiving a remote track.
212    Unsubscribe { track: TrackId },
213    /// Toggle mute on one of this client's own tracks.
214    SetMuted { track: TrackId, muted: bool },
215    /// Relay a keyframe request (PLI) for a subscribed track to its producer.
216    /// Pairs with `hound_media::keyframe`.
217    RequestKeyframe { track: TrackId },
218    /// A receiver's reception report for a subscribed remote `track`, relayed to
219    /// the track's producer so its sender-side BWE can see loss on the downstream
220    /// leg the relay hides (the receiver→producer half of the BWE loop).
221    ///
222    /// Counts are **since this reporter's previous report for the same track**
223    /// (deltas, not cumulative), so the producer can fold them into a per-interval
224    /// loss fraction directly. A reporter sends these on a ~1 s cadence while it
225    /// has something to report.
226    ///
227    /// Additive variant (protocol-compatible): an older server that does not know
228    /// it skips it; an older client never sends it. See [`PROTOCOL_VERSION`].
229    MediaFeedback {
230        track: TrackId,
231        /// Frames the receiver actually decoded/accepted for this track since its
232        /// last report (the denominator of the loss fraction).
233        received: u32,
234        /// Frames the receiver determined missing/skipped since its last report —
235        /// `frame_seq` gaps plus deliberate priority-drops (both signal pressure;
236        /// see the receiver measurement in `src/voice/session`).
237        lost: u32,
238        /// Optional inter-arrival jitter estimate (µs) for the track, if measured.
239        /// Informational: producers fold only the loss fraction.
240        #[serde(default, skip_serializing_if = "Option::is_none")]
241        jitter_us: Option<i64>,
242    },
243    /// A receiver's **standing** quality request for a subscribed remote video
244    /// `track`, relayed to that track's producer to drive demand-driven simulcast.
245    ///
246    /// `cap_bps`: `None` = "I can take the source" (no lower layer needed for me);
247    /// `Some(bps)` = "I can sustain at most `bps`". The producer coalesces every
248    /// receiver's cap into at most one shared low layer (see
249    /// [`hound_media::plan_layers`](../../hound_media/fn.plan_layers.html)) and drops
250    /// the source if no one can take it. Defensive: one standing request per
251    /// (reporter, track), only meaningful below the source (a cap ≥ source is just
252    /// "source"), and it is a *level*, not a stream demand. A receiver re-sends it
253    /// only when its estimate crosses the hysteresis threshold, not every tick.
254    ///
255    /// Additive variant (protocol-compatible) — see [`PROTOCOL_VERSION`].
256    RequestQuality {
257        track: TrackId,
258        cap_bps: Option<u64>,
259    },
260    /// Graceful leave.
261    Bye,
262}

264/// Server → client. Drives the client's [`crate::session::SessionView`].
265#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
266#[serde(tag = "type", rename_all = "snake_case")]
267pub enum ServerMessage {
268    /// Sent right after a successful `Hello`, and again on reconnect: a full
269    /// snapshot for (re)syncing the session from scratch.
270    Welcome {
271        protocol_version: u16,
272        self_id: PeerId,
273        peers: Vec<PeerInfo>,
274        tracks: Vec<TrackInfo>,
275    },
276    PeerJoined {
277        peer: PeerInfo,
278    },
279    PeerLeft {
280        peer: PeerId,
281    },
282    TrackPublished {
283        track: TrackInfo,
284    },
285    TrackUnpublished {
286        track: TrackId,
287    },
288    MuteChanged {
289        track: TrackId,
290        muted: bool,
291    },
292    /// A subscriber asked for a keyframe; delivered to the producing client,
293    /// which forces one (see `hound_media::keyframe::KeyframeThrottle`).
294    KeyframeRequested {
295        track: TrackId,
296    },
297    /// A receiver's reception report, delivered to the **producer** of `track`
298    /// only (the receiver→producer feedback half of the BWE loop). The producer
299    /// folds `received`/`lost` into its sender-side rate controller's
300    /// loss-equivalent. `from` identifies the reporting peer so a producer serving
301    /// multiple receivers can take the worst leg.
302    ///
303    /// Counts are deltas since the reporter's previous report (see
304    /// [`ClientMessage::MediaFeedback`]). Additive variant — an older client
305    /// simply never receives it (it never subscribed to anyone's video in a way
306    /// that produces feedback for it, and unknown server frames are skipped).
307    MediaFeedbackReceived {
308        track: TrackId,
309        from: PeerId,
310        received: u32,
311        lost: u32,
312        #[serde(default, skip_serializing_if = "Option::is_none")]
313        jitter_us: Option<i64>,
314    },
315    /// A receiver's standing quality request, delivered to the **producer** of
316    /// `track` only (never broadcast, never echoed). The producer keeps a
317    /// per-receiver cap map and runs [`hound_media::plan_layers`] to decide which
318    /// layers to publish (source and/or one shared low). `from` identifies the
319    /// requester so a producer serving several receivers can coalesce their caps.
320    /// Additive variant — an older client never receives it.
321    QualityRequested {
322        track: TrackId,
323        from: PeerId,
324        cap_bps: Option<u64>,
325    },
326    Error {
327        code: ErrorCode,
328        message: String,
329    },
330    /// The backend picked a new best SFU node (a rollout finished, or the node
331    /// this call is on started draining) and every live `/control` session is
332    /// being asked to move. `node_id` is the new best node, carried for
333    /// logging/diagnostics only — the client re-resolves via a fresh
334    /// `voice/join` rather than trusting this value directly, so a stale or
335    /// duplicate migrate signal is harmless. The client's reaction is to tear
336    /// down and re-establish exactly as it would for a dropped connection,
337    /// landing it on the new node.
338    ///
339    /// Additive variant — an older client that doesn't recognize `"type":
340    /// "sfu_migrate"` fails to parse just that one frame and skips it (see
341    /// [`ClientMessage`]'s sibling additive variants); it keeps riding the old
342    /// node until the backend forcibly cuts it or the connection drops on its
343    /// own.
344    SfuMigrate {
345        node_id: String,
346    },
347}

499    /// Unknown-variant tolerance: a frame from a *newer* peer carrying a `type`
500    /// this build does not know fails to deserialize (rather than mis-parsing as a
501    /// known variant). The transports (backend `next_client_message`, client reader)

505    fn unknown_variant_is_rejected_not_misparsed() {
506        let future = r#"{"type":"some_future_message","track":9}"#;
507        assert!(

509            "an unknown tagged variant must error (so the transport skips it), not silently map to a known one"

514    /// Additive-compat: an older peer that omits `decode_caps` (on both Hello and
515    /// PeerInfo) deserializes to the H.264-only floor, never a panic or a codec it
516    /// cannot decode.
517    #[test]
518    fn absent_decode_caps_defaults_to_h264_floor() {
519        let hello: ClientMessage = serde_json::from_str(r#"{"type":"hello","protocol_version":1}"#)
520            .expect("older Hello (no decode_caps) must parse");
521        let ClientMessage::Hello { decode_caps, .. } = hello else {
522            panic!("expected Hello");
523        };
524        assert_eq!(decode_caps, DecodeCaps::default());
525        assert!(decode_caps.h264 && !decode_caps.h265 && !decode_caps.av1);
526
527        let peer: PeerInfo = serde_json::from_str(r#"{"id":3}"#)
528            .expect("older PeerInfo (no decode_caps) must parse");
529        assert_eq!(peer.decode_caps, DecodeCaps::default());
530    }

532    #[test]
533    fn decode_caps_supports_and_intersect() {
534        let all = DecodeCaps {
535            av1: true,
536            h265: true,
537            h264: true,
538        };
539        assert!(all.supports(Codec::H265));
540        assert!(
541            !all.supports(Codec::Opus),
542            "Opus is never a video-ladder codec"

545        let floor = DecodeCaps::default();
546        assert_eq!(all.intersect(floor), floor);
