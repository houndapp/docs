1//! Authoritative, sans-IO server logic for one call.
2//!
3//! [`SessionServer`] is the counterpart to the client's
4//! [`crate::session::SessionView`]: it owns the source-of-truth roster, assigns
5//! [`PeerId`]/[`TrackId`]s, validates each [`ClientMessage`], and emits the
6//! [`ServerMessage`]s that result — each tagged with **who should receive it**
7//! ([`Target`]). It performs no I/O: the transport (the backend's socket.io
8//! namespace) feeds it messages and routes the [`Outgoing`]s it returns.
9//!
10//! Keeping this pure means the routing rules (a publish reaches everyone, a
11//! keyframe request reaches only the producer, a leave drops the peer's tracks)
12//! are tested deterministically, independent of the socket layer.
13
14use std::collections::{BTreeMap, BTreeSet};
15
16use uuid::Uuid;
17
18use crate::proto::{
19    ClientMessage, DecodeCaps, ErrorCode, MediaKind, PROTOCOL_VERSION, PeerId, PeerInfo,
20    ServerMessage, TrackId, TrackInfo,
21};
22
23/// Who an outgoing [`ServerMessage`] is addressed to.
24#[derive(Debug, Clone, Copy, PartialEq, Eq)]
25pub enum Target {
26    /// A single peer (e.g. `Welcome` to the joiner, `KeyframeRequested` to the
27    /// producer, an `Error` to the offending client).
28    Peer(PeerId),
29    /// Every peer currently in the call.
30    Broadcast,
31    /// Every peer except one (e.g. announce a joiner to the others).
32    BroadcastExcept(PeerId),
33}
34
35/// A [`ServerMessage`] plus its routing [`Target`], to be sent by the transport.
36#[derive(Debug, Clone, PartialEq, Eq)]
37pub struct Outgoing {
38    pub target: Target,
39    pub message: ServerMessage,
40}
41
42impl Outgoing {
43    fn to(target: Target, message: ServerMessage) -> Self {
44        Self { target, message }
45    }
46}

48/// Authoritative state and logic for a single call.
49#[derive(Debug, Default)]
50pub struct SessionServer {
51    next_peer: u32,
52    next_track: u32,
53    /// Admitted peers and the codecs each can decode (advertised in `Hello`,
54    /// re-broadcast in `PeerInfo` so producers can pick a codec everyone plays).
55    peers: BTreeMap<PeerId, DecodeCaps>,
56    /// The authenticated user each admitted peer belongs to — stamped by the
57    /// transport (from the validated `/control` JWT) and passed into
58    /// [`SessionServer::join`]. Kept alongside `peers` (not merged into it) so
59    /// `Publish`-path decode-caps logic never has to know this field exists.
60    /// Echoed into every [`PeerInfo`] the roster carries; media broadcast paths
61    /// are built from this, not from a realtime-derived identity map.
62    peer_users: BTreeMap<PeerId, Uuid>,
63    tracks: BTreeMap<TrackId, TrackInfo>,
64    /// What each peer is subscribed to receive (for later SFU routing).
65    subscriptions: BTreeMap<PeerId, BTreeSet<TrackId>>,
66}
67
68impl SessionServer {
69    pub fn new() -> Self {
70        Self::default()
71    }
72
73    /// Whether the call has no participants — the transport uses this to drop an
74    /// empty `SessionServer` rather than leak one per call ever started.
75    pub fn is_empty(&self) -> bool {
76        self.peers.is_empty()
77    }
78
79    /// Admit a peer that sent [`ClientMessage::Hello`] (carrying its `decode_caps`),
80    /// authenticated as `user_id` by the transport (the validated `/control` JWT's
81    /// subject — never client-supplied). On a compatible version, assigns a
82    /// [`PeerId`] and returns it with the outgoing messages: a `Welcome` snapshot to
83    /// the joiner (the full roster, each peer's decode caps + authenticated user id
84    /// included, so the joiner can immediately pick a codec everyone plays and build
85    /// correct media broadcast paths) and a `PeerJoined` carrying the joiner's caps +
86    /// identity to everyone else. On a version mismatch, returns `Err` with the
87    /// `Error` to send before closing.
88    pub fn join(
89        &mut self,
90        client_version: u16,
91        decode_caps: DecodeCaps,
92        user_id: Uuid,
93    ) -> Result<(PeerId, Vec<Outgoing>), ServerMessage> {
94        if client_version != PROTOCOL_VERSION {
95            return Err(ServerMessage::Error {
96                code: ErrorCode::VersionMismatch,
97                message: format!("server speaks protocol v{PROTOCOL_VERSION}"),
98            });
99        }
100
101        let id = PeerId(self.next_peer);
102        self.next_peer += 1;
103
104        // Every existing entry in `self.peers` has a matching `self.peer_users`
105        // entry (inserted/removed together — see below and `leave`); `unwrap_or_default`
106        // is a defensive fallback that should never actually trigger.
107        let welcome = ServerMessage::Welcome {
108            protocol_version: PROTOCOL_VERSION,
109            self_id: id,
110            peers: self
111                .peers
112                .iter()
113                .map(|(&pid, &caps)| PeerInfo {
114                    id: pid,
115                    decode_caps: caps,
116                    user_id: self.peer_users.get(&pid).copied().unwrap_or_default(),
117                })
118                .chain(std::iter::once(PeerInfo {
119                    id,
120                    decode_caps,
121                    user_id,
122                }))
123                .collect(),
124            tracks: self.tracks.values().cloned().collect(),
125        };
126
127        self.peers.insert(id, decode_caps);
128        self.peer_users.insert(id, user_id);
129        self.subscriptions.insert(id, BTreeSet::new());
130
131        let out = vec![
132            Outgoing::to(Target::Peer(id), welcome),
133            Outgoing::to(
134                Target::BroadcastExcept(id),
135                ServerMessage::PeerJoined {
136                    peer: PeerInfo {
137                        id,
138                        decode_caps,
139                        user_id,
140                    },
141                },
142            ),
143        ];
144        Ok((id, out))
145    }
146
147    /// Remove a peer (graceful `Bye` or a dropped connection). Drops its tracks
148    /// and subscriptions and announces `PeerLeft` to the rest. Idempotent: a
149    /// peer that isn't present yields no messages.
150    pub fn leave(&mut self, peer: PeerId) -> Vec<Outgoing> {
151        if self.peers.remove(&peer).is_none() {
152            return Vec::new();
153        }
154        self.peer_users.remove(&peer);
155        self.subscriptions.remove(&peer);
156        // Drop the departed peer's tracks (clients infer this from PeerLeft, so
157        // no separate TrackUnpublished is needed) and any subscriptions to them.
158        let gone: Vec<TrackId> = self
159            .tracks
160            .iter()
161            .filter(|(_, t)| t.peer == peer)
162            .map(|(&id, _)| id)
163            .collect();
164        for id in &gone {
165            self.tracks.remove(id);
166        }
167        for subs in self.subscriptions.values_mut() {
168            subs.retain(|t| !gone.contains(t));
169        }
170
171        vec![Outgoing::to(
172            Target::Broadcast,
173            ServerMessage::PeerLeft { peer },
174        )]
175    }
176
177    /// Process a message from an admitted `peer`, returning the messages to send.
178    pub fn handle(&mut self, peer: PeerId, msg: ClientMessage) -> Vec<Outgoing> {
179        match msg {
180            // The handshake is consumed by `join`; a duplicate Hello is a no-op.
181            ClientMessage::Hello { .. } => Vec::new(),
182
183            ClientMessage::Publish { track: decl } => {
184                // Cross-check the declared codec against the roster BEFORE assigning
185                // a TrackId — a sender ignoring negotiation must not hand a receiver
186                // a codec it can't decode. Video only: `DecodeCaps` is the video
187                // ladder, and `DecodeCaps::supports` reports `false` for Opus
188                // unconditionally, so gating audio the same way would reject every
189                // audio publish.
190                if decl.kind == MediaKind::Video {
191                    let common = self.common_decode_caps_excluding(peer);
192                    if !common.supports(decl.codec) {
193                        tracing::warn!(
194                            peer = peer.0,
195                            codec = ?decl.codec,
196                            "publish rejected: codec not decodable by the current roster"
197                        );
198                        return vec![Outgoing::to(
199                            Target::Peer(peer),
200                            ServerMessage::PublishRejected {
201                                source: peer,
202                                kind: decl.kind,
203                                codec: decl.codec,
204                            },
205                        )];
206                    }
207                }
208                let id = TrackId(self.next_track);
209                self.next_track += 1;
210                let info = TrackInfo {
211                    id,
212                    peer,
213                    kind: decl.kind,
214                    source: decl.source,
215                    codec: decl.codec,
216                    muted: false,
217                    low: decl.low,
218                };
219                self.tracks.insert(id, info.clone());
220                // Broadcast to all — including the publisher, which learns the
221                // server-assigned TrackId this way.
222                vec![Outgoing::to(
223                    Target::Broadcast,
224                    ServerMessage::TrackPublished { track: info },
225                )]
226            }
227
228            ClientMessage::Unpublish { track } => match self.tracks.get(&track) {
229                Some(t) if t.peer == peer => {
230                    self.tracks.remove(&track);
231                    for subs in self.subscriptions.values_mut() {
232                        subs.remove(&track);
233                    }
234                    vec![Outgoing::to(
235                        Target::Broadcast,
236                        ServerMessage::TrackUnpublished { track },
237                    )]
238                }
239                _ => vec![self.not_found(peer, "no such track to unpublish")],
240            },
241
242            ClientMessage::Subscribe { track } => {
243                if self.tracks.contains_key(&track) {
244                    self.subscriptions.entry(peer).or_default().insert(track);
245                    Vec::new()
246                } else {
247                    vec![self.not_found(peer, "cannot subscribe to unknown track")]
248                }
249            }
250
251            ClientMessage::Unsubscribe { track } => {
252                if let Some(subs) = self.subscriptions.get_mut(&peer) {
253                    subs.remove(&track);
254                }
255                Vec::new()
256            }
257
258            ClientMessage::SetMuted { track, muted } => match self.tracks.get_mut(&track) {
259                Some(t) if t.peer == peer => {
260                    t.muted = muted;
261                    vec![Outgoing::to(
262                        Target::Broadcast,
263                        ServerMessage::MuteChanged { track, muted },
264                    )]
265                }
266                _ => vec![self.not_found(peer, "cannot mute a track you don't own")],
267            },
268
269            ClientMessage::RequestKeyframe { track } => match self.tracks.get(&track) {
270                Some(t) => {
271                    // Relay the PLI to the producing client only.
272                    vec![Outgoing::to(
273                        Target::Peer(t.peer),
274                        ServerMessage::KeyframeRequested { track },
275                    )]
276                }
277                None => vec![self.not_found(peer, "keyframe requested for unknown track")],
278            },
279
280            ClientMessage::MediaFeedback {
281                track,
282                received,
283                lost,
284                jitter_us,
285            } => match self.tracks.get(&track) {
286                // Ownership rule (mirrors RequestKeyframe): a reporter may only
287                // report on a track that exists, and the report is delivered to the
288                // track's PRODUCER only — never broadcast, never echoed to the
289                // reporter. A peer reporting on its *own* track is a no-op rather
290                // than feeding its BWE its own (meaningless) numbers.
291                Some(t) if t.peer != peer => {
292                    vec![Outgoing::to(
293                        Target::Peer(t.peer),
294                        ServerMessage::MediaFeedbackReceived {
295                            track,
296                            from: peer,
297                            received,
298                            lost,
299                            jitter_us,
300                        },
301                    )]
302                }
303                // Self-report on own track: ignore silently (not an error — a
304                // producer that also subscribes to itself shouldn't be punished).
305                Some(_) => Vec::new(),
306                None => vec![self.not_found(peer, "media feedback for unknown track")],
307            },
308
309            ClientMessage::RequestQuality { track, cap_bps } => match self.tracks.get(&track) {
310                // Same ownership rule as MediaFeedback: relay to the PRODUCER only,
311                // tagged with the requester; a self-request on an own track is a no-op.
312                Some(t) if t.peer != peer => {
313                    vec![Outgoing::to(
314                        Target::Peer(t.peer),
315                        ServerMessage::QualityRequested {
316                            track,
317                            from: peer,
318                            cap_bps,
319                        },
320                    )]
321                }
322                Some(_) => Vec::new(),
323                None => vec![self.not_found(peer, "quality requested for unknown track")],
324            },
325
326            ClientMessage::Bye => self.leave(peer),
327        }
328    }

330    /// Tracks a peer is currently subscribed to (for SFU forwarding decisions).
331    pub fn subscriptions_of(&self, peer: PeerId) -> impl Iterator<Item = TrackId> + '_ {
332        self.subscriptions
333            .get(&peer)
334            .into_iter()
335            .flat_map(|s| s.iter().copied())
336    }
337
338    /// The codecs every OTHER admitted peer can decode — the same fold
339    /// [`crate::session::SessionView::common_remote_decode_caps`] runs client-side
340    /// (identity [`DecodeCaps::ALL`], intersected across every peer but `exclude`),
341    /// so a solo publisher — or a call with no other participants — is never
342    /// constrained: the empty fold returns `ALL`, which every codec satisfies.
343    /// Reimplemented here (not shared) because the server's roster is a flat
344    /// `PeerId -> DecodeCaps` map, not a `SessionView`.
345    fn common_decode_caps_excluding(&self, exclude: PeerId) -> DecodeCaps {
346        self.peers
347            .iter()
348            .filter(|&(&id, _)| id != exclude)
349            .fold(DecodeCaps::ALL, |acc, (_, caps)| acc.intersect(*caps))
350    }
351
352    fn not_found(&self, peer: PeerId, message: &str) -> Outgoing {
353        Outgoing::to(
354            Target::Peer(peer),
355            ServerMessage::Error {
356                code: ErrorCode::NotFound,
357                message: message.to_string(),
358            },
359        )
360    }

434    fn roster_carries_each_peers_decode_caps() {
435        let mut s = SessionServer::new();
436        let user_a = Uuid::new_v4();
437        let user_b = Uuid::new_v4();
438        // First peer is H.264-only (e.g. an un-upgraded / non-Apple client).
439        let (a, _) = s
440            .join(PROTOCOL_VERSION, DecodeCaps::default(), user_a)
441            .unwrap();

443        // (so the joiner knows it can only send A the H.264 floor) and its own.
444        let (_b, out) = s.join(PROTOCOL_VERSION, all_caps(), user_b).unwrap();
445
446        let ServerMessage::Welcome { peers, .. } = &out[0].message else {
447            panic!("expected Welcome");
448        };
449        let a_info = peers.iter().find(|p| p.id == a).unwrap();
450        assert_eq!(
451            a_info.decode_caps,
452            DecodeCaps::default(),
453            "A advertised H.264-only"
454        );
455        assert_eq!(
456            a_info.user_id, user_a,
457            "A's roster entry carries A's authenticated user id"
458        );
459
460        // The PeerJoined announced to A must carry B's full caps (so A's producer
461        // can re-negotiate when B — who can decode more — joins).
462        let ServerMessage::PeerJoined { peer } = &out[1].message else {
463            panic!("expected PeerJoined");
464        };
465        assert_eq!(out[1].target, Target::BroadcastExcept(_b));
466        assert_eq!(peer.decode_caps, all_caps());

468            peer.user_id, user_b,
469            "B's PeerJoined carries B's authenticated user id"
470        );
471    }
