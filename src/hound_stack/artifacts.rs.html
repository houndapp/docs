1//! Artifact directory resolution: an env var (`E2E_OUT_DIR`/`VISUAL_OUT_DIR`
2//! — a durable, CI-collectible dir) if set, else a throwaway temp dir left
3//! in place for local inspection. Same compat contract the two shell
4//! scripts documented.
5
6use std::path::PathBuf;
7
8use eyre::{Context, Result};
9
10pub fn resolve(env_var: &str) -> Result<PathBuf> {
11    let dir = match std::env::var(env_var) {
12        Ok(dir) => PathBuf::from(dir),
13        Err(_) => tempfile::Builder::new()
14            .prefix("hound-stack-")
15            .tempdir()
16            .context("creating a throwaway artifact dir")?
17            // Leaks the dir instead of deleting it on drop — mirrors the
18            // shell scripts' `mktemp -d`, which is also never cleaned up
19            // (`workdir: $WORKDIR (left in place for inspection)`).
20            .keep(),
21    };
22    std::fs::create_dir_all(&dir)
23        .with_context(|| format!("creating artifact dir {}", dir.display()))?;
24    Ok(dir)
25}
