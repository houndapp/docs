1//! SFU (re-)registration: the SFU registers its node ID with the backend at
2//! ITS OWN startup — `compose up -d` only (re)creates containers whose config
3//! changed, so a `hound-sfu` container left running from a previous invocation
4//! may be mid-interval against a freshly `MIGRATE=fresh`'d backend. Restarting
5//! it forces an immediate registration, then we poll the backend's registry
6//! endpoint until a live node appears — the typed contract for "is voice
7//! routable", never a log-line scrape (see MEMORY.md "SFU registration
8//! fragility" / `services/backend/src/api/internal.rs`).
9
10use std::path::Path;
11use std::process::Child;
12use std::time::Duration;
13
14use eyre::Result;
15
16use crate::backend::RELAY_SECRET_DEV;
17use crate::compose::Compose;
18use crate::wait::{Deadline, wait_for_condition_or_exit};
19
20/// Restart `hound-sfu` (compose output → `compose_log`) and wait until the
21/// backend's `GET /internal/sfu/nodes` reports a live, non-draining node.
22pub async fn reregister(
23    compose: &Compose,
24    compose_log: &Path,
25    backend_child: &mut Child,
26    backend_url: &str,
27) -> Result<()> {
28    compose.restart("hound-sfu", compose_log)?;
29
30    let client = reqwest::Client::new();
31    let url = format!("{backend_url}/internal/sfu/nodes");
32    let deadline = Deadline::after(Duration::from_secs(15), Duration::from_secs(1));
33    wait_for_condition_or_exit("SFU registration", backend_child, &deadline, || {
34        let client = client.clone();
35        let url = url.clone();
36        async move {
37            let Ok(resp) = client.get(&url).bearer_auth(RELAY_SECRET_DEV).send().await else {
38                return false;
39            };
40            if !resp.status().is_success() {
41                return false;
42            }
43            match resp.json::<hound_types::SfuNodesResponse>().await {
44                Ok(nodes) => nodes.nodes.iter().any(|n| n.live && !n.draining),
45                Err(_) => false,
46            }
47        }
48    })
49    .await
50}
