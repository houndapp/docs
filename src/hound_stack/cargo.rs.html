1//! Quiet cargo builds, separated from runtime logs. Every binary the harness
2//! runs (backend, houndbot, snapdiff, the nextest test binaries) is built
3//! HERE first — full compiler output captured into the run's `build.log`,
4//! surfaced only on failure — and then executed directly, so each runtime
5//! artifact (`backend.log`, bot logs, test output) contains runtime lines
6//! only, never pages of "Compiling…" chatter.
7
8use std::io::Write;
9use std::path::{Path, PathBuf};
10use std::process::Command;
11
12use eyre::{Context, Result, bail};
13
14use crate::logs;
15
16/// Where `cargo build` puts binaries: `CARGO_TARGET_DIR` if set, else the
17/// workspace-root `target/`.
18pub fn target_debug_dir(root: &Path) -> PathBuf {
19    std::env::var_os("CARGO_TARGET_DIR")
20        .map(PathBuf::from)
21        .unwrap_or_else(|| root.join("target"))
22        .join("debug")
23}
24
25/// Run `cmd` with all output captured into a `build_log` section titled
26/// `title`. Silent on success; on failure prints the log tail inline and
27/// bails with a pointer at the full file.
28pub fn run_captured(mut cmd: Command, build_log: &Path, title: &str) -> Result<()> {
29    let mut f = logs::append_section(build_log, title)?;
30    let out = cmd.output().with_context(|| title.to_owned())?;
31    f.write_all(&out.stdout)?;
32    f.write_all(&out.stderr)?;
33    if !out.status.success() {
34        logs::print_tail(title, build_log, 40);
35        bail!(
36            "{title} exited {} — full output in {}",
37            out.status,
38            build_log.display()
39        );
40    }
41    Ok(())
42}
43
44/// `cargo build -q --message-format=short -p package --bin bin`, output to
45/// `build_log`. Returns the built binary's path for the caller to exec.
46pub fn build_bin(root: &Path, package: &str, bin: &str, build_log: &Path) -> Result<PathBuf> {
47    let mut cmd = Command::new("cargo");
48    cmd.args([
49        "build",
50        "-q",
51        "--message-format=short",
52        "-p",
53        package,
54        "--bin",
55        bin,
56    ])
57    .current_dir(root);
58    run_captured(
59        cmd,
60        build_log,
61        &format!("cargo build -p {package} --bin {bin}"),
62    )?;
63    Ok(target_debug_dir(root).join(bin))
64}
