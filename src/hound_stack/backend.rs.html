1//! The host backend child process: the exact dev env both shell harnesses
2//! ran the backend under — this is the one place that env now lives.
3//!
4//! Takes a PREBUILT binary (see `crate::cargo::build_bin`), never `cargo
5//! run`: the build's compiler chatter belongs in `build.log`, and
6//! `backend.log` must open with the backend's own first runtime line.
7
8use std::fs::{self, File};
9use std::path::{Path, PathBuf};
10use std::process::{Child, Command, Stdio};
11
12use eyre::{Context, Result};
13
14pub struct BackendProcess {
15    pub child: Child,
16    pub log_path: PathBuf,
17}
18
19/// Spawn the prebuilt backend `binary` against the compose stack's fixed dev
20/// ports, redirecting stdout+stderr to `log_path`. `storage_dir` is created
21/// (uploads are unused by these harnesses; the backend's local storage
22/// backend just needs the directory to exist — checked at runtime, not by
23/// clap, see `services/backend/src/api/storage.rs`).
24pub fn spawn(
25    binary: &Path,
26    root: &Path,
27    log_path: &Path,
28    storage_dir: &Path,
29) -> Result<BackendProcess> {
30    fs::create_dir_all(storage_dir).context("creating STORAGE_LOCAL_DIR")?;
31    let log_file = File::create(log_path).context("creating backend log file")?;
32    let stderr_file = log_file
33        .try_clone()
34        .context("cloning backend log file handle")?;
35
36    let child = Command::new(binary)
37        .current_dir(root)
38        .env("PORT", "8080")
39        .env("SECRET_KEY", "secret-key")
40        .env(
41            "DATABASE_URL",
42            "postgres://root:secret@localhost:5432/hound",
43        )
44        .env("ENVIRONMENT", "dev")
45        .env("RP_ID", "localhost")
46        .env("RP_ORIGIN", "http://localhost:5173")
47        .env("MIGRATE", "fresh")
48        .env("SMTP_URL", "smtp://localhost:1025")
49        .env("EMAIL_FROM", "Hound <dev@localhost>")
50        .env("MAGIC_BASE_URL", "http://localhost:3100")
51        // Must match `compose.dev.yaml`'s `hound-sfu.RELAY_SECRET` — see
52        // `services/sfu/src/main.rs`'s `RELAY_SECRET` doc: "must match the
53        // backend's RELAY_HMAC_SECRET".
54        .env("RELAY_HMAC_SECRET", "dev-relay-secret")
55        .env("RELAY_URL", "http://localhost:3340")
56        .env("STORAGE_LOCAL_DIR", storage_dir)
57        .env("RUST_LOG", "info")
58        .stdout(Stdio::from(log_file))
59        .stderr(Stdio::from(stderr_file))
60        .spawn()
61        .with_context(|| format!("spawning backend ({})", binary.display()))?;
62
63    Ok(BackendProcess {
64        child,
65        log_path: log_path.to_owned(),
66    })
67}
