1//! The shared stack lifecycle: compose bring-up, health waits, host backend,
2//! SFU re-registration — and its mirror-image teardown (backend kill, SFU
3//! log capture, `compose down --volumes`). One [`LiveStack`] per invocation
4//! of `e2e`/`visual`/`up`; [`LiveStack::teardown`] is idempotent, so it's
5//! safe to call both explicitly (to observe teardown before the process
6//! exits) and again from `Drop` — a panic, early return, or PARTIAL bring-up
7//! failure (compose up but the backend never came) still tears everything
8//! down, mirroring the shell scripts' `trap cleanup EXIT`.
9//!
10//! Noise contract: one console line per lifecycle step; every subprocess's
11//! full output goes to its own artifact file (`build.log`, `compose.log`,
12//! `backend.log`, `sfu.log`); a failure prints the relevant log's tail
13//! inline. Success prints no log content at all.
14

27/// The stack's fixed filesystem layout, relative to the repo root.
28pub struct StackPaths {
29    pub root: PathBuf,
30    pub backend_dir: PathBuf,
31    pub compose_file: PathBuf,
32    pub override_file: PathBuf,
33}
34
35impl StackPaths {
36    pub fn discover(root: PathBuf) -> Self {
37        let backend_dir = root.join("services/backend");
38        let compose_file = backend_dir.join("compose.dev.yaml");
39        // `compose.dev.yaml`'s `hound-sfu.BACKEND_URL` points at a `backend`
40        // service that only exists in the prod/staging compose topology —
41        // locally the backend runs on the host, so the SFU's self-registration
42        // POST needs the Docker host gateway instead (see the override file's
43        // own doc comment).
44        let override_file = root.join("tools/houndbot/compose.e2e-override.yaml");
45        Self {
46            root,
47            backend_dir,
48            compose_file,
49            override_file,
50        }
51    }
52}

54pub struct LiveStack {
55    compose: Compose,
56    /// `None` until the backend child is spawned — the window where a
57    /// bring-up failure has compose containers to reap but no backend.
58    backend: Option<BackendProcess>,
59    sfu_log_path: PathBuf,
60    compose_log_path: PathBuf,
61    torn_down: bool,
62}
63
64impl LiveStack {
65    /// Bring the full stack up: port preflight, compose (mailpit/postgres/
66    /// iroh-relay/hound-sfu), the host backend (built quietly to
67    /// `build.log`, then exec'd), then a forced SFU restart so it
68    /// (re-)registers against this run's freshly migrated backend.
69    ///
70    /// On a partial failure the already-started pieces are torn down before
71    /// the error returns — the caller never inherits a half-up stack.
72    pub async fn bring_up(
73        paths: &StackPaths,
74        workdir: &Path,
75        backend_url: &str,
76        mailpit_url: &str,
77    ) -> Result<Self> {
78        ports::preflight(STACK_PORTS).context("port preflight")?;
79
80        let compose = Compose::resolve(
81            &paths.backend_dir,
82            &paths.compose_file,
83            &paths.override_file,
84        );
85        let mut stack = Self {
86            compose,
87            backend: None,
88            sfu_log_path: workdir.join("sfu.log"),
89            compose_log_path: workdir.join("compose.log"),
90            torn_down: false,
91        };
92        match stack
93            .bring_up_inner(paths, workdir, backend_url, mailpit_url)
94            .await
95        {
96            Ok(()) => Ok(stack),
97            Err(e) => {
98                stack.teardown();
99                Err(e)
100            }
101        }
102    }
103
104    async fn bring_up_inner(
105        &mut self,
106        paths: &StackPaths,
107        workdir: &Path,
108        backend_url: &str,
109        mailpit_url: &str,
110    ) -> Result<()> {
111        let build_log = workdir.join("build.log");
112        let backend_log = workdir.join("backend.log");
113
114        info!("compose up: mailpit, postgres, iroh-relay, hound-sfu (output → compose.log)");
115        self.compose.up_d_build(&self.compose_log_path)?;
116
117        info!("waiting for postgres to be healthy");
118        let postgres_id = self.compose.ps_q("postgres")?.ok_or_else(|| {
119            eyre::eyre!("postgres container not found after compose up — see docker compose ps")
120        })?;
121        wait_container_healthy(
122            &postgres_id,
123            &Deadline::after(Duration::from_secs(120), Duration::from_secs(2)),
124        )
125        .await
126        .context("postgres never became healthy")?;
127
128        let http = reqwest::Client::new();
129        // Best-effort, matching the shell scripts: mailpit runs as a compose
130        // service (no local child to guard) and a missed wait here just
131        // means the FIRST magic-link poll retries a little longer.
132        info!("waiting for mailpit to be reachable");
133        let _ = wait_http_reachable(
134            &http,
135            &format!("{mailpit_url}/api/v1/messages"),
136            &Deadline::after(Duration::from_secs(30), Duration::from_secs(1)),
137        )
138        .await;
139
140        info!("building backend (output → build.log)");
141        let backend_bin = cargo::build_bin(&paths.root, "backend", "backend", &build_log)?;
142
143        info!("starting backend (output → backend.log)");
144        let storage_dir = workdir.join("storage");
145        let backend = backend::spawn(&backend_bin, &paths.root, &backend_log, &storage_dir)?;
146        self.backend = Some(backend);
147        let backend = self.backend.as_mut().expect("just set");
148
149        info!("waiting for backend to be reachable");
150        if let Err(e) = wait_for_http_or_exit(
151            &http,
152            &format!("{backend_url}/instance"),
153            &mut backend.child,
154            &Deadline::after(Duration::from_secs(120), Duration::from_secs(1)),
155        )
156        .await
157        {
158            logs::print_tail("backend log", &backend_log, 60);
159            return Err(e).context("backend never became reachable");
160        }
161        info!("backend reachable");
162
163        info!("restarting hound-sfu so it (re-)registers against this backend");
164        if let Err(e) = sfu::reregister(
165            &self.compose,
166            &self.compose_log_path,
167            &mut backend.child,
168            &backend_log,
169        )
170        .await
171        {
172            logs::print_tail("backend log", &backend_log, 60);
173            return Err(e).context("SFU never registered with the backend — voice/join would fail");
174        }
175        info!("SFU registered");
176        Ok(())
177    }
178
179    pub fn backend_log_path(&self) -> Option<&Path> {
180        self.backend.as_ref().map(|b| b.log_path.as_path())
181    }
182
183    /// Idempotent: safe to call explicitly (so a caller can observe the
184    /// teardown before choosing its own exit code) and again from `Drop`.
185    /// Prints no log content — a caller that failed decides which tail to
186    /// surface (see `logs::print_tail`).
187    pub fn teardown(&mut self) {
188        if self.torn_down {
189            return;
190        }
191        self.torn_down = true;
192        if let Some(backend) = self.backend.as_mut()
193            && let Ok(None) = backend.child.try_wait()
194        {
195            info!("stopping backend");
196            let _ = backend.child.kill();
197            let _ = backend.child.wait();
198        }
199        // Captured BEFORE `compose down` (which takes the container, and its
200        // log, with it) — an e2e-call/visual CI artifact.
201        info!("capturing hound-sfu log → sfu.log");
202        let sfu_log = self.compose.logs_no_color("hound-sfu");
203        let _ = std::fs::write(&self.sfu_log_path, sfu_log);
204        info!("compose down --volumes (output → compose.log)");
205        self.compose.down_volumes(&self.compose_log_path);
206    }
207}
208
209impl Drop for LiveStack {
210    fn drop(&mut self) {
211        self.teardown();
212    }
