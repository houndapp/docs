1//! `hound-stack e2e`: bring the stack up, run the `hound-testkit` scenario
2//! suite (`cargo nextest run -p hound-testkit --profile e2e`) against it,
3//! collect artifacts, tear down. Mirrors the retired `tools/houndbot/e2e.sh`.
4//! Extra CLI args forward to `cargo nextest run` (e.g. an `-E` filter for a
5//! fast local iteration loop against one scenario file).
6//!
7//! The test binaries are compiled in a separate quiet pass
8//! (`cargo nextest run --no-run`, output → `build.log`) before the real run,
9//! so `nextest.log` holds test results, not compiler chatter.
10
11use std::path::PathBuf;
12use std::process::Command;
13
14use eyre::{Context, Result};
15use tracing::info;
16
17use crate::lifecycle::{LiveStack, StackPaths};
18use crate::tee::run_tee;
19use crate::{artifacts, cargo, logs};
20
21/// Durable-artifact override — the nextest run log + junit report,
22/// build/backend/sfu/compose logs — instead of the default throwaway temp
23/// dir (still left in place for local inspection either way).
24pub const OUT_DIR_ENV: &str = "E2E_OUT_DIR";
25
26const INFRA_FAILURE: i32 = 2;
27
28/// The stack's shared dev relay secret/URL — must match
29/// `compose.dev.yaml`'s `hound-sfu.RELAY_SECRET` and `backend::spawn`'s
30/// `RELAY_HMAC_SECRET`/`RELAY_URL` env (see `src/backend.rs`). Exported to
31/// the nextest run as `HOUND_SFU_RELAY_SECRET`/`HOUND_SFU_RELAY_URL` so the
32/// SFU seamless-rollout scenario's second `hound-sfu` process (a plain host
33/// process, not a compose service — see `crates/test/testkit/src/sfu_process.rs`)
34/// talks to the same relay/backend as the stack's own `sfu1`.
35const RELAY_URL_DEV: &str = "http://localhost:3340";
36
37pub async fn run(root: PathBuf, reuse_stack: bool, extra_nextest_args: Vec<String>) -> Result<i32> {
38    let workdir = artifacts::resolve(OUT_DIR_ENV)?;
39    info!(workdir = %workdir.display(), "artifact dir");
40    let build_log = workdir.join("build.log");
41
42    let backend_url =
43        std::env::var("BACKEND_URL").unwrap_or_else(|_| "http://localhost:8080".into());
44    let mailpit_url =
45        std::env::var("MAILPIT_URL").unwrap_or_else(|_| "http://localhost:8025".into());
46
47    // Build the suite before the stack comes up: a compile failure then
48    // costs nothing to tear down, and the containers don't sit idle through
49    // a long build.
50    info!("building the scenario suite (output → build.log)");
51    let mut build_cmd = Command::new("cargo");
52    build_cmd.current_dir(&root).args([
53        "nextest",
54        "run",
55        "--no-run",
56        "-p",
57        "hound-testkit",
58        "--profile",
59        "e2e",
60    ]);
61    cargo::run_captured(
62        build_cmd,
63        &build_log,
64        "cargo nextest run --no-run (suite build)",
65    )?;
66
67    // Built here (not spawned) — the SFU seamless-rollout scenario spawns
68    // its OWN second `hound-sfu` process from this binary directly (fresh
69    // ports, a short drain deadline), separate from the stack's own `sfu1`
70    // compose service. Building it is orthogonal to compose/backend
71    // bring-up, so this happens unconditionally, `--reuse-stack` included —
72    // the binary path is exported below regardless.
73    info!("building hound-sfu (output → build.log)");
74    let sfu_bin = cargo::build_bin(&root, "hound-sfu", "hound-sfu", &build_log)?;
75
76    let mut stack = if reuse_stack {
77        None
78    } else {
79        let paths = StackPaths::discover(root.clone());
80        match LiveStack::bring_up(&paths, &workdir, &backend_url, &mailpit_url).await {
81            Ok(stack) => Some(stack),
82            Err(e) => {
83                eprintln!("[e2e] stack bring-up failed: {e:#}");
84                return Ok(INFRA_FAILURE);
85            }
86        }
87    };
88
89    info!("running the hound-testkit scenario suite (output → nextest.log)");
90    let mut cmd = Command::new("cargo");
91    cmd.current_dir(&root)
92        .args([
93            "nextest",
94            "run",
95            "-p",
96            "hound-testkit",
97            "--profile",
98            "e2e",
99            // One scenario's failure must not cancel the rest — a full
100            // per-test result set is what makes the CI artifact diagnosable.
101            "--no-fail-fast",
102        ])
103        .args(&extra_nextest_args)
104        .env("BACKEND_URL", &backend_url)
105        .env("MAILPIT_URL", &mailpit_url)
106        .env("HOUND_SFU_BIN", &sfu_bin)
107        .env("HOUND_SFU_RELAY_SECRET", crate::backend::RELAY_SECRET_DEV)
108        .env("HOUND_SFU_RELAY_URL", RELAY_URL_DEV);
109    let nextest_log = workdir.join("nextest.log");
110    let status = run_tee(cmd, &nextest_log).context("running cargo nextest")?;
111
112    let junit_src = root.join("target/nextest/e2e/junit.xml");
113    if junit_src.exists() {
114        let _ = std::fs::copy(&junit_src, workdir.join("junit.xml"));
115    }
116
117    if let Some(stack) = stack.as_mut() {
118        stack.teardown();
119    }
120
121    // Propagate nextest's own exit code (0 pass, its own nonzero on
122    // failures) — the same contract the old shell harness had via
123    // `set -o pipefail`.
124    let code = status.code().unwrap_or(1);
125    if code != 0 {
126        logs::print_tail("backend log", &workdir.join("backend.log"), 100);
127    }
128    info!(code, workdir = %workdir.display(), "e2e suite finished (workdir left in place)");
129    Ok(code)
130}
