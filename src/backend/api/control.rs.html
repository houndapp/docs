1//! The `/control` call control plane.
2//!
3//! Unlike `/community` (a socket.io namespace serving browsers), the control
4//! plane's client is the **Rust app**, so this is a plain authenticated axum
5//! WebSocket — no socket.io. The transport-agnostic [`hound_control::SessionServer`]
6//! holds the authoritative per-call state; this module is only the connection,
7//! routing, and authorization glue around it.
8//!
9//! ## Endpoint & wire shape
10//!
11//! `GET /space/{space_id}/control?token=<jwt>` upgrades to a WebSocket. A "call"
12//! is a **space** (keyed by `space_id`). Every WS text frame is JSON: inbound a
13//! [`hound_control::ClientMessage`], outbound a [`hound_control::ServerMessage`].
14//!
15//! The first inbound frame MUST be `ClientMessage::Hello { protocol_version }`;
16//! the peer is admitted via [`SessionServer::join`]. Subsequent frames are fed to
17//! [`SessionServer::handle`]. A `ClientMessage::Bye`, the stream ending, or a
18//! transport error all run the leave path.
19//!
20//! ## Authorization (trust boundary)
21//!
22//! The control plane is privileged. We authenticate & authorize **before**
23//! accepting the upgrade: the `?token=` query param must be a valid, unrevoked
24//! *access* token, and `claims.sub` must be a member of the space's community
25//! (`is_community_member`). Any failure returns 401/403 and the upgrade never
26//! happens — no `PeerId` is assigned, no `ControlCall` is created.
27//!
28//! ## State / lifecycle (SPOF flag)
29//!
30//! Per-call state lives in [`ControlCall`] in `Inner::control_calls`, created on
31//! the first join for a space and dropped when the last peer leaves
32//! (`SessionServer::is_empty()`). This is **in-memory and per-process** — it does
33//! not survive a backend redeploy and does not coordinate across horizontally
34//! scaled replicas. `/control` is only correct on a single backend replica.
35//!
36//! ## Routing
37//!
38//! Each admitted peer gets an `mpsc::UnboundedSender<ServerMessage>` registered in
39//! `ControlCall::conns[peer]`; a per-connection forward task drains the receiver
40//! to the WS sink. [`route`] resolves each [`Outgoing`]'s [`Target`] to recipient
41//! `PeerId`s against `conns` and pushes the message onto each sender. Because the
42//! transport tracks each connection's own sender, `BroadcastExcept(self)` is a
43//! genuine exclusion (no join-ordering trick needed).
44//!
45//! ## Locking
46//!
47//! The per-call `SessionServer` and `conns` are behind std `Mutex`es. We NEVER
48//! hold either across an `.await`: lock → call the sync `join`/`handle`/`leave`
49//! and collect recipients → drop the lock → then send. `mpsc::UnboundedSender::send`
50//! is sync and non-blocking, so routing never awaits while locked.
51

66/// Per-call control-plane state: the authoritative [`SessionServer`] plus the
67/// live connection senders keyed by [`PeerId`], used to route [`Outgoing`]s.

69pub(crate) struct ControlCall {

96/// Authenticates & authorizes the control-plane connection, then upgrades to a
97/// WebSocket. Auth happens *before* the upgrade so an unauthorized caller gets a
98/// plain 401/403 and never reaches the session machinery.
99async fn control_ws_handler(
100    State(api): State<Api>,
101    Path(space_id): Path<Uuid>,
102    Query(query): Query<ControlQuery>,
103    ws: WebSocketUpgrade,
104) -> Response {
105    // The Rust client may pass either a bare token or a `Bearer `-prefixed one.
106    let token = query
107        .token
108        .strip_prefix("Bearer ")
109        .unwrap_or(&query.token)
110        .to_owned();
111
112    // 1. Validate the JWT (must be a valid, unrevoked *access* token).
113    let claims = match api.auth.jwt.validate_unrevoked_token(&token).await {
114        Ok(c) if c.is_access_token() => c,
115        _ => {
116            tracing::warn!(%space_id, "/control — rejected: invalid access token");
117            return StatusCode::UNAUTHORIZED.into_response();
118        }
119    };
120    let user_id = claims.sub;
121
122    // 2. Authorize: the user must be a member of the space's community.
123    let space = match api.find_space_by_id(space_id).await {
124        Ok(Some(s)) => s,
125        Ok(None) => {
126            tracing::warn!(%space_id, %user_id, "/control — rejected: space not found");
127            return StatusCode::NOT_FOUND.into_response();
128        }
129        Err(err) => {
130            tracing::error!(%err, %space_id, %user_id, "/control — db error finding space");
131            return StatusCode::INTERNAL_SERVER_ERROR.into_response();
132        }
133    };
134    match api.is_community_member(space.community_id, user_id).await {
135        Ok(true) => {}
136        Ok(false) => {
137            tracing::warn!(%space_id, community_id = %space.community_id, %user_id, "/control — rejected: not a community member");
138            return StatusCode::FORBIDDEN.into_response();
139        }
140        Err(err) => {
141            tracing::error!(%err, %space_id, %user_id, "/control — db error checking membership");
142            return StatusCode::INTERNAL_SERVER_ERROR.into_response();
143        }
144    }
145
146    tracing::debug!(%space_id, %user_id, "/control — authorized, upgrading");
147    ws.on_upgrade(move |socket| handle_socket(socket, api, space_id))
148}

150/// Drives one authorized control-plane connection for its full lifetime.
151async fn handle_socket(socket: WebSocket, api: Api, space_id: Uuid) {
152    let (mut sink, mut stream) = socket.split();
153
154    // The first frame must be `Hello`. Anything else (or a closed stream) ends
155    // the connection without ever creating a peer.
156    let hello = match next_client_message(&mut stream).await {
157        Some(msg @ ClientMessage::Hello { .. }) => msg,
158        Some(_) => {
159            tracing::warn!(%space_id, "/control — first frame was not Hello; closing");
160            return;
161        }
162        None => return,
163    };
164    let ClientMessage::Hello {
165        protocol_version,
166        decode_caps,
167    } = hello
168    else {
169        unreachable!("matched Hello above");
170    };
171
172    let call = api.get_or_create_control_call(space_id);
173
174    // Admit the peer with the codecs it advertised it can decode. Lock held only
175    // for the sync `join`.
176    let join_result = {
177        let mut server = call.server.lock().unwrap();
178        server.join(protocol_version, decode_caps)
179    };
180    let (peer, join_outgoings) = match join_result {
181        Ok(ok) => ok,
182        Err(err_msg) => {
183            // Version mismatch (or other rejection): send the Error, then close.
184            let _ = sink.send(to_text(&err_msg)).await;
185            api.drop_control_call_if_empty(space_id);
186            return;
187        }
188    };
189
190    // Register this connection's sender and spawn the forward task that drains
191    // queued ServerMessages to the WS sink.
192    let (tx, mut rx) = mpsc::unbounded_channel::<ServerMessage>();
193    call.conns.lock().unwrap().insert(peer, tx);
194
195    let mut forward = tokio::spawn(async move {
196        while let Some(msg) = rx.recv().await {
197            if sink.send(to_text(&msg)).await.is_err() {
198                break;
199            }
200        }
201    });
202
203    // Route the join outgoings (Welcome → self, PeerJoined → the others). The
204    // sender for `peer` is already registered, so `Welcome` reaches this client.
205    route(&call, &join_outgoings);
206
207    tracing::debug!(%space_id, peer = peer.0, "/control — peer joined");
208
209    // Read loop: feed each inbound ClientMessage to the SessionServer and route
210    // the results. Runs until the client sends Bye, the stream ends/errors, or
211    // the forward task dies (sink closed).
212    loop {
213        tokio::select! {
214            // Forward task exiting means the sink is gone; stop reading.
215            res = &mut forward => {
216                if let Err(err) = res {
217                    tracing::warn!(%err, %space_id, peer = peer.0, "/control — forward task panicked");
218                }
219                break;
220            }
221            incoming = next_client_message(&mut stream) => {
222                let Some(msg) = incoming else { break };
223                let is_leave = matches!(msg, ClientMessage::Bye);
224                let outgoings = {
225                    let mut server = call.server.lock().unwrap();
226                    server.handle(peer, msg)
227                };
228                route(&call, &outgoings);
229                if is_leave {
230                    break;
231                }
232            }
233        }
234    }
235
236    // Leave path (idempotent). Drop this connection's sender first so a final
237    // PeerLeft broadcast is not delivered back to the departing peer, then run
238    // the SessionServer leave and route PeerLeft to the rest.
239    forward.abort();
240    call.conns.lock().unwrap().remove(&peer);
241    let leave_outgoings = {
242        let mut server = call.server.lock().unwrap();
243        server.leave(peer)
244    };
245    route(&call, &leave_outgoings);
246    api.drop_control_call_if_empty(space_id);
247    tracing::debug!(%space_id, peer = peer.0, "/control — peer left");
248}

250/// Reads the next WS frame and deserializes it as a [`ClientMessage`].
251///
252/// Returns `None` when the stream ends, errors, or the peer sends a Close frame.
253/// Malformed JSON or non-text frames (ping/pong/binary) are skipped, not fatal.
254async fn next_client_message(
255    stream: &mut futures::stream::SplitStream<WebSocket>,
256) -> Option<ClientMessage> {
257    loop {
258        match stream.next().await? {
259            Ok(Message::Text(text)) => match serde_json::from_str::<ClientMessage>(&text) {
260                Ok(msg) => return Some(msg),
261                Err(err) => {
262                    tracing::warn!(%err, "/control — ignoring malformed client message");
263                    continue;
264                }
265            },
266            Ok(Message::Close(_)) => return None,
267            // Ping/Pong are handled by axum; binary is unexpected — skip.
268            Ok(_) => continue,
269            Err(err) => {
270                tracing::debug!(%err, "/control — websocket stream error");
271                return None;
272            }
273        }
274    }
275}

277/// Serializes a [`ServerMessage`] to a WS text frame. Serialization of this
278/// closed enum cannot fail; if it ever did we send an empty frame rather than
279/// panic in the transport task.
280fn to_text(msg: &ServerMessage) -> Message {
281    Message::Text(serde_json::to_string(msg).unwrap_or_default().into())
282}

284/// Resolves each [`Outgoing`]'s [`Target`] to recipient senders and pushes the
285/// message onto each. NEVER holds the `conns` lock across an await: collect the
286/// senders, then send (sync, non-blocking) — all within one lock scope is fine
287/// because `send` does not await.
288fn route(call: &ControlCall, outgoings: &[Outgoing]) {
289    if outgoings.is_empty() {
290        return;
291    }
292    let conns = call.conns.lock().unwrap();
293    for out in outgoings {
294        match out.target {
295            Target::Peer(p) => {
296                if let Some(tx) = conns.get(&p) {
297                    let _ = tx.send(out.message.clone());
298                }
299            }
300            Target::Broadcast => {
301                for tx in conns.values() {
302                    let _ = tx.send(out.message.clone());
303                }
304            }
305            Target::BroadcastExcept(except) => {
306                for (p, tx) in conns.iter() {
307                    if *p != except {
308                        let _ = tx.send(out.message.clone());
309                    }
310                }
311            }
312        }
313    }
314}

318impl Api {
319    /// Returns the [`ControlCall`] for a space, creating it if absent.
320    fn get_or_create_control_call(&self, space_id: Uuid) -> Arc<ControlCall> {
321        self.control_calls
322            .write()
323            .unwrap()
324            .entry(space_id)
325            .or_insert_with(|| Arc::new(ControlCall::new()))
326            .clone()
327    }
328
329    /// Drops the [`ControlCall`] for a space if it has no remaining peers, so we
330    /// don't leak one empty entry per call ever started. Idempotent.
331    fn drop_control_call_if_empty(&self, space_id: Uuid) {
332        let mut calls = self.control_calls.write().unwrap();
333        if let Some(call) = calls.get(&space_id)
334            && call.server.lock().unwrap().is_empty()
335        {
336            calls.remove(&space_id);
337        }
338    }
339}

342//
343// Regression guard for the *whole* control stack: a real client WS ↔ the
344// backend's connection/routing glue ↔ the authoritative `SessionServer` ↔
345// another real client. Unlike `crate::api::control::tests` in the sans-IO crate
346// (which test the reducer in isolation), these drive two live `tokio-tungstenite`
347// clients against the router bound to a real TCP port, because a WebSocket
348// upgrade cannot be driven through the in-memory `axum_test` request path.
349//
350// Synchronization rule: never `sleep` to wait for a message. Every assertion
351// either `await`s the specific message it expects (bounded by a timeout) or
352// drains a small, bounded window of messages and asserts the expected one is
353// present. Ordering between two independent peers' streams is not assumed.

373    /// How long any single "expect a frame" read may block before we declare the
374    /// invariant violated. Generous vs. the work involved (a localhost round-trip
375    /// plus a sync state transition is sub-millisecond) so the bound is a real
376    /// liveness assertion, not a perf budget; small enough that a genuinely stuck
377    /// or never-delivered message fails the test quickly rather than hanging CI.

380    /// Window for asserting a message is *absent* (the PLI-targeting negative:
381    /// "B must NOT receive `KeyframeRequested`"). Shorter than [`RECV_TIMEOUT`]
382    /// so the negative tests stay fast, but comfortably longer than localhost
383    /// delivery latency, so a mis-routed message would arrive well within it.
384    const ABSENCE_WINDOW: Duration = Duration::from_millis(400);

487        /// Opens a WS, completes the `Hello`/`Welcome` handshake at the negotiated
488        /// protocol version, and returns the socket plus the assigned `self_id`.
489        async fn connect(&self, token: &str) -> (Ws, PeerId, ServerMessage) {
490            let (mut ws, _resp) = connect_async(self.control_url(token)).await.unwrap();
491            send(
492                &mut ws,
493                &ClientMessage::Hello {
494                    protocol_version: PROTOCOL_VERSION,
495                    decode_caps: DecodeCaps::ALL,
496                },
497            )
498            .await;
499            let welcome = next_server_msg(&mut ws).await;
500            let ServerMessage::Welcome { self_id, .. } = welcome else {
501                panic!("first server frame must be Welcome, got {welcome:?}");
502            };
503            (ws, self_id, welcome)
504        }
505    }

514    /// Reads the next *server message*, skipping ping/pong/non-text frames, and
515    /// fails the test (rather than hanging) if none arrives within [`RECV_TIMEOUT`].
516    async fn next_server_msg(ws: &mut Ws) -> ServerMessage {
517        let fut = async {
518            loop {
519                match ws.next().await {
520                    Some(Ok(WsMessage::Text(text))) => {
521                        return serde_json::from_str::<ServerMessage>(&text)
522                            .unwrap_or_else(|e| panic!("bad server frame {text:?}: {e}"));
523                    }
524                    // Ping/Pong/Binary are not protocol messages — skip.
525                    Some(Ok(_)) => continue,
526                    Some(Err(e)) => panic!("ws stream error: {e}"),
527                    None => panic!("ws closed while awaiting a server message"),
528                }
529            }
530        };
531        tokio::time::timeout(RECV_TIMEOUT, fut)
532            .await
533            .expect("timed out awaiting a server message (liveness violated)")
534    }

536    /// Drains up to `max` server messages, returning the first that satisfies
537    /// `pred`. Used where two events may interleave on one stream so position is
538    /// not guaranteed. Fails if none of the first `max` match.
539    async fn recv_until(
540        ws: &mut Ws,
541        max: usize,
542        mut pred: impl FnMut(&ServerMessage) -> bool,
543    ) -> ServerMessage {
544        let mut seen = Vec::new();
545        for _ in 0..max {
546            let msg = next_server_msg(ws).await;
547            if pred(&msg) {
548                return msg;
549            }
550            seen.push(msg);
551        }
552        panic!("expected a matching server message within {max} frames; saw {seen:?}");
553    }

555    /// Asserts no message matching `pred` arrives within [`ABSENCE_WINDOW`].
556    /// Non-matching messages are tolerated (and ignored).
557    async fn assert_absent(ws: &mut Ws, mut pred: impl FnMut(&ServerMessage) -> bool) {
558        let deadline = tokio::time::Instant::now() + ABSENCE_WINDOW;
559        loop {
560            let remaining = deadline.saturating_duration_since(tokio::time::Instant::now());
561            if remaining.is_zero() {
562                return;
563            }
564            match tokio::time::timeout(remaining, ws.next()).await {
565                // Timed out with nothing matching — the absence held.
566                Err(_) => return,
567                Ok(Some(Ok(WsMessage::Text(text)))) => {
568                    let msg = serde_json::from_str::<ServerMessage>(&text).unwrap();
569                    assert!(!pred(&msg), "expected NO matching message, but got {msg:?}");
570                }
571                Ok(Some(Ok(_))) => continue,
572                Ok(Some(Err(e))) => panic!("ws stream error: {e}"),
573                Ok(None) => return,
574            }
575        }
576    }

587    /// Invariant 1 (join roster) + 2 (track propagation) + 3 (mute propagation)
588    /// + 4 (PLI targeting) + 5 (ownership) + 6 (leave teardown), all in one
589    /// two-client session so each step builds on real prior state.
590    #[tokio::test]
591    async fn two_client_session_invariants() {
592        let h = Harness::start().await;
593
594        // ── Invariant 1: join roster ──────────────────────────────────────────
595        // A joins first: its Welcome lists only itself, with self_id == A.
596        let (mut a, peer_a, welcome_a) = h.connect(&h.token_a).await;
597        {
598            let ServerMessage::Welcome {
599                self_id,
600                peers,
601                tracks,
602                protocol_version,
603            } = &welcome_a
604            else {
605                unreachable!("connect already matched Welcome");
606            };
607            assert_eq!(*protocol_version, PROTOCOL_VERSION);
608            assert_eq!(*self_id, peer_a);
609            let ids: Vec<_> = peers.iter().map(|p| p.id).collect();
610            assert_eq!(ids, vec![peer_a], "first joiner's roster is just itself");
611            assert!(tracks.is_empty(), "no tracks before any publish");
612        }
613
614        // B joins second: its Welcome lists both A and B; A gets PeerJoined { B }.

894    /// assertion is deterministic rather than racing the old socket's teardown.

900        // A's stream is clean. B (second joiner) already saw A in its own Welcome,
901        // so B gets no PeerJoined. We keep A around only as `peer_a`; the live
902        // sockets we drive are `a` (then dropped) and `b` (the witness).
903        let (mut a, peer_a, _) = h.connect(&h.token_a).await;
904        let (mut b, peer_b, _) = h.connect(&h.token_b).await;
905        let _ = recv_until(&mut a, 4, |m| matches!(m, ServerMessage::PeerJoined { .. })).await;

915            matches!(m, ServerMessage::TrackPublished { .. })

954        send(&mut b, &ClientMessage::Bye).await;

1083    /// Polls (bounded by [`RECV_TIMEOUT`]) until the registry holds exactly
1084    /// `expected` calls. Used after a leave, whose cleanup runs asynchronously on
1085    /// the departing connection's read loop — so we converge rather than sleep.
1086    async fn wait_for_call_count(h: &Harness, expected: usize) {
1087        let deadline = tokio::time::Instant::now() + RECV_TIMEOUT;
1088        loop {
1089            let now = call_count(h);
1090            if now == expected {
1091                return;
1092            }
1093            if tokio::time::Instant::now() >= deadline {
1094                panic!("control_calls did not converge to {expected}; still {now}");
1095            }
1096            tokio::time::sleep(Duration::from_millis(10)).await;
1097        }
1098    }
