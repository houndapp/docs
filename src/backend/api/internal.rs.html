1//! Internal service-to-service endpoints, not exposed in the public OpenAPI spec.
2
3use axum::{
4    Json,
5    extract::State,
6    http::{HeaderMap, StatusCode},
7};
8
9use crate::api::{Api, ApiResult};
10
11/// Returns a router for all `/internal` routes.
12pub(crate) fn router() -> axum::Router<Api> {
13    axum::Router::new().route("/sfu/register", axum::routing::post(post_sfu_register))
14}
15
16#[derive(Debug, serde::Deserialize)]
17pub(crate) struct SfuRegisterRequest {
18    pub node_id: String,
19}
20
21/// Called by the SFU on startup to register its iroh node ID with the backend.
22///
23/// Authenticated by `Authorization: Bearer <relay_hmac_secret>`. The node ID
24/// is stored in memory and returned to clients in every `voice/join` response.
25pub(crate) async fn post_sfu_register(
26    State(api): State<Api>,
27    headers: HeaderMap,
28    Json(body): Json<SfuRegisterRequest>,
29) -> ApiResult<StatusCode> {
30    let expected = format!("Bearer {}", api.relay_hmac_secret);
31    let auth = headers
32        .get(axum::http::header::AUTHORIZATION)
33        .and_then(|v| v.to_str().ok())
34        .unwrap_or("");
35    if auth != expected {
36        return Ok(StatusCode::UNAUTHORIZED);
37    }
38
39    tracing::info!(node_id = %body.node_id, "SFU registered node ID");
40    *api.sfu_node_id.write().unwrap() = Some(body.node_id);
41    Ok(StatusCode::OK)
42}
