1//! File upload endpoints.
2//!
3//! Two endpoints are exposed:
4//! - `POST /upload/avatar` — upload a user or community avatar image.
5//! - `POST /upload/attachment` — upload a message attachment.
6//!
7//! Both follow the same two-phase pattern:
8//! 1. Client uploads the file; the server validates, stores it, and returns an `UploadResponse`
9//!    containing the upload ID and public URL.
10//! 2. Client references the upload ID in a subsequent PATCH call (e.g. `PATCH /profile/global`).
11

25pub(crate) fn router() -> utoipa_axum::router::OpenApiRouter<Api> {
26    utoipa_axum::router::OpenApiRouter::with_openapi(Default::default())
27        .routes(utoipa_axum::routes!(upload_avatar))
28        .routes(utoipa_axum::routes!(upload_attachment))
29        // Disable axum's default 2 MiB body limit so `collect_bounded` is the
30        // sole size enforcer and returns a proper 413 instead of axum's 422.
31        .layer(axum::extract::DefaultBodyLimit::disable())
32}

46// ── Magic-byte signatures ────────────────────────────────────────────────────
47
48fn check_magic(mime: &str, data: &[u8]) -> bool {
49    match mime {
50        "image/jpeg" => data.starts_with(&[0xFF, 0xD8, 0xFF]),
51        "image/png" => data.starts_with(&[0x89, 0x50, 0x4E, 0x47]),
52        "image/gif" => data.starts_with(b"GIF8"),
53        "image/webp" => data.starts_with(b"RIFF") && data.get(8..12) == Some(b"WEBP"),
54        // For non-image types we accept the declared content-type header alone.
55        _ => true,
56    }
57}

103pub(crate) async fn upload_avatar(
104    LoggedIn(claims): LoggedIn,
105    Query(params): Query<UploadAvatarParams>,
106    State(api): State<Api>,
107    multipart: Multipart,
108) -> ApiResult<(StatusCode, Json<UploadResponse>)> {
109    let purpose = match params.community_id {
110        Some(cid) => format!("community-avatar/{cid}"),
111        None => format!("avatar/{}", claims.sub),
112    };
113    process_upload(&api, claims.sub, &purpose, AVATAR_MIME_TYPES, multipart).await
114}

140// ── Core upload logic ─────────────────────────────────────────────────────────
141
142async fn process_upload(
143    api: &Api,
144    uploader_id: Uuid,
145    purpose: &str,
146    allowed_mime_types: &[&str],
147    mut multipart: Multipart,
148) -> ApiResult<(StatusCode, Json<UploadResponse>)> {
149    // Find the `file` field in the multipart body.
150    let field: Field<'_> = loop {
151        match multipart.next_field().await {
152            Ok(Some(f)) if f.name() == Some("file") => break f,
153            Ok(Some(_)) => continue, // skip other parts
154            Ok(None) => return Err(ApiError::MultipartError("missing `file` part".into())),
155            Err(e) => return Err(ApiError::MultipartError(e.to_string())),
156        }
157    };
158
159    // Use the declared content-type from the part header as a fast-path check.
160    let declared_mime = field.content_type().map(str::to_string).unwrap_or_default();
161
162    if !allowed_mime_types.contains(&declared_mime.as_str()) {
163        return Err(ApiError::UnsupportedMediaType);
164    }
165
166    // Collect the bytes, enforcing the size limit.
167    let data: Bytes = collect_bounded(field, api.upload_max_bytes).await?;
168
169    // Verify magic bytes match the declared MIME.
170    if !check_magic(&declared_mime, &data) {
171        return Err(ApiError::UnsupportedMediaType);
172    }
173
174    let ext = mime_to_ext(&declared_mime);
175    let upload_id = Uuid::now_v7();
176    let key = format!("{purpose}/{upload_id}.{ext}");
177
178    // Write to storage; on failure we have nothing to clean up yet.
179    api.storage
180        .put_object(&key, &declared_mime, data.clone())
181        .await
182        .map_err(ApiError::Storage)?;
183
184    let size_bytes = data.len() as i64;
185    let created_at = chrono::Utc::now().timestamp();
186
187    // Insert the DB record. On failure, delete the object we just wrote.
188    let model = upload::ActiveModel {
189        id: Set(upload_id),
190        uploader_id: Set(uploader_id),
191        bucket: Set(api.storage.bucket().to_string()),
192        key: Set(key.clone()),
193        content_type: Set(declared_mime.clone()),
194        size_bytes: Set(size_bytes),
195        created_at: Set(created_at),
196        referenced_at: Set(None),
197    }
198    .insert(&api.db)
199    .await;
200
201    let model = match model {
202        Ok(m) => m,
203        Err(db_err) => {
204            // Compensating delete: remove the orphaned S3 object.
205            let _ = api.storage.delete_object(&key).await;
206            return Err(ApiError::Database(db_err));
207        }
208    };
209
210    let url = api.storage.public_url(&model.key);
211    Ok((
212        StatusCode::CREATED,
213        Json(UploadResponse {
214            id: model.id,
215            url,
216            content_type: model.content_type,
217            size_bytes: model.size_bytes,
218        }),
219    ))
220}

222/// Collect bytes from a multipart field, returning `PayloadTooLarge` if the
223/// data exceeds `max_bytes`.
224async fn collect_bounded(mut field: Field<'_>, max_bytes: u64) -> ApiResult<Bytes> {
225    let mut buf: Vec<u8> = Vec::new();
226    loop {
227        match field.chunk().await {
228            Ok(Some(chunk)) => {
229                if buf.len() as u64 + chunk.len() as u64 > max_bytes {
230                    return Err(ApiError::PayloadTooLarge);
231                }
232                buf.extend_from_slice(&chunk);
233            }
234            Ok(None) => break,
235            Err(e) => return Err(ApiError::MultipartError(e.to_string())),
236        }
237    }
238    Ok(Bytes::from(buf))
239}

241fn mime_to_ext(mime: &str) -> &'static str {
242    match mime {
243        "image/jpeg" => "jpg",
244        "image/png" => "png",
245        "image/gif" => "gif",
246        "image/webp" => "webp",
247        "application/pdf" => "pdf",
248        "text/plain" => "txt",
249        _ => "bin",
250    }
251}

255impl Api {
256    /// Looks up an upload by ID, verifies the caller owns it, marks it as
257    /// referenced, and returns the object key.
258    pub(crate) async fn reference_upload(
259        &self,
260        upload_id: Uuid,
261        caller_id: Uuid,
262    ) -> ApiResult<String> {
263        let row = upload::Entity::find_by_id(upload_id)
264            .one(&self.db)
265            .await?
266            .ok_or(ApiError::UploadNotFound(upload_id))?;
267
268        if row.uploader_id != caller_id {
269            return Err(ApiError::Forbidden);
270        }
271
272        let now = chrono::Utc::now().timestamp();
273        upload::ActiveModel {
274            id: Set(upload_id),
275            referenced_at: Set(Some(now)),
276            ..Default::default()
277        }
278        .update(&self.db)
279        .await?;
280
281        Ok(row.key)
282    }
283
284    /// Deletes all upload rows (and their storage objects) that were never
285    /// referenced and are older than `ttl`.
286    #[allow(dead_code)]
287    pub(crate) async fn sweep_orphaned_uploads(&self, ttl: std::time::Duration) {
288        let cutoff = chrono::Utc::now().timestamp() - ttl.as_secs() as i64;
289        let orphans = upload::Entity::find()
290            .filter(upload::Column::ReferencedAt.is_null())
291            .filter(upload::Column::CreatedAt.lte(cutoff))
292            .all(&self.db)
293            .await;
294
295        let orphans = match orphans {
296            Ok(rows) => rows,
297            Err(e) => {
298                tracing::error!(error = %e, "sweep: failed to query orphaned uploads");
299                return;
300            }
301        };
302
303        for row in orphans {
304            if let Err(e) = self.storage.delete_object(&row.key).await {
305                tracing::warn!(key = %row.key, error = %e, "sweep: failed to delete object");
306            }
307            if let Err(e) = upload::Entity::delete_by_id(row.id).exec(&self.db).await {
308                tracing::warn!(id = %row.id, error = %e, "sweep: failed to delete upload row");
309            }
310        }
311    }
312}

324    fn jpeg_bytes() -> Vec<u8> {
325        // Minimal valid JPEG: SOI marker + APP0 JFIF header + EOI
326        let mut v = vec![0xFF, 0xD8, 0xFF, 0xE0];
327        v.extend_from_slice(&[0x00, 0x10]); // APP0 length
328        v.extend_from_slice(b"JFIF\x00");
329        v.extend_from_slice(&[0x01, 0x01, 0x00, 0x00, 0x01, 0x00, 0x01, 0x00, 0x00]);
330        v.extend_from_slice(&[0xFF, 0xD9]); // EOI
331        v
332    }
333
334    fn png_bytes() -> Vec<u8> {
335        let mut v = vec![0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A];
336        v.extend_from_slice(&[0x00; 16]); // padding
337        v
338    }
339
340    fn webp_bytes() -> Vec<u8> {
341        let mut v = b"RIFF".to_vec();
342        v.extend_from_slice(&[0x00; 4]); // file size (dummy)
343        v.extend_from_slice(b"WEBP");
344        v.extend_from_slice(&[0x00; 8]); // padding
345        v
346    }

348    fn multipart_with(name: &str, mime: &str, data: Vec<u8>) -> MultipartForm {
349        MultipartForm::new().add_part(name, Part::bytes(data).mime_type(mime))
350    }
