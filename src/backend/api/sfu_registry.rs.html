1//! The multi-node SFU registry: which SFU nodes are live, and which one is
2//! "best" for a new voice join right now.
3//!
4//! ## Why a registry, not a single slot
5//!
6//! A blue/green SFU rollout runs the old and new pod side by side for a while
7//! (existing calls drain off the old one; new joins should land on the new
8//! one). A single `sfu_node_id: Option<String>` slot (the pre-rollout-support
9//! shape) can only ever name one node, so the last registration to land wins
10//! — during a rollout that's a coin flip between old and new pod, and it
11//! offers no way to say "this node is going away, don't route new joins to
12//! it." The registry keeps every node's last-seen registration plus its
13//! `draining` flag and picks the best live one at read time.
14//!
15//! ## In-process only
16//!
17//! Like [`super::control_calls`](super::Inner::control_calls), this is
18//! in-memory and per-process — it does not survive a backend redeploy and
19//! does not coordinate across horizontally-scaled replicas. That's fine: the
20//! SFU self-heals by re-registering every `REGISTER_INTERVAL`
21//! (`services/sfu/src/main.rs`), so a backend restart clears the registry and
22//! it repopulates within one interval, same as the single-slot predecessor.
23//!
24//! ## Liveness
25//!
26//! The SFU re-registers every 30s. A node is "live" if it registered within
27//! [`LIVENESS_TTL`] (3 missed cycles). Liveness is evaluated at read time
28//! ([`select_best`]) rather than by a background reaper — a stale entry just
29//! stops being selectable; nothing needs to actively evict it, and the map
30//! never grows unbounded because node IDs are stable per-pod-lifetime (a
31//! replaced pod re-registers under a fresh ephemeral identity, but the old
32//! entry ages out and is simply never selected again — see the `SfuNode`
33//! churn note below if that ever needs bounding).
34
35use std::collections::HashMap;
36use std::time::{Duration, Instant};
37
38use crate::api::Api;

40/// A node is live if it registered within this many registration intervals.
41/// The SFU re-registers every 30s; 90s tolerates two missed cycles (a slow
42/// backend, a blip) before presuming the node dead.
43const LIVENESS_TTL: Duration = Duration::from_secs(90);
44
45/// One SFU node's last-known registration state.
46///
47/// Two timestamps, deliberately: `first_registered_at` is the node's stable
48/// age and the SELECTION key; `last_seen_at` is refreshed by every heartbeat
49/// and used ONLY for liveness. Selecting on the heartbeat timestamp would make
50/// two live nodes leapfrog each other as "newest" on alternating 30s
51/// heartbeats — flipping the advertised node and broadcasting a migrate storm
52/// that bounces every in-call client between nodes.
53#[derive(Debug, Clone)]
54pub(crate) struct SfuNode {
55    pub(crate) first_registered_at: Instant,
56    pub(crate) last_seen_at: Instant,
57    pub(crate) draining: bool,
58}
59
60/// Picks the best node to route a new voice join to: the **newest** (by first
61/// registration) live, non-draining node; if every live node is draining, the
62/// newest live node anyway (serving a call on a draining node beats refusing
63/// it outright — the node keeps accepting connections while it drains, see
64/// `services/sfu/src/main.rs`). `None` if no node has been seen within
65/// [`LIVENESS_TTL`].
66///
67/// Pure function of `nodes` and `now` so it's directly unit-testable without
68/// standing up an `Api` or waiting out real wall-clock time.
69pub(crate) fn select_best(nodes: &HashMap<String, SfuNode>, now: Instant) -> Option<String> {
70    let live: Vec<(&String, &SfuNode)> = nodes
71        .iter()
72        .filter(|(_, n)| {
73            now.checked_duration_since(n.last_seen_at)
74                .unwrap_or(Duration::ZERO)
75                <= LIVENESS_TTL
76        })
77        .collect();
78
79    live.iter()
80        .filter(|(_, n)| !n.draining)
81        .max_by_key(|(_, n)| n.first_registered_at)
82        .or_else(|| live.iter().max_by_key(|(_, n)| n.first_registered_at))
83        .map(|(id, _)| (*id).clone())
84}

86/// Seeds the registry at boot from the `SFU_NODE_ID` env var, so a dev/compose
87/// setup that only sets the env (never calls `BACKEND_URL` self-registration)
88/// keeps working exactly as the old single-slot field did. The seeded entry
89/// expires via [`LIVENESS_TTL`] like any other node unless something
90/// re-registers it before then.
91pub(crate) fn seed(env_node_id: Option<String>) -> HashMap<String, SfuNode> {
92    let mut nodes = HashMap::new();
93    if let Some(node_id) = env_node_id {
94        let now = Instant::now();
95        nodes.insert(
96            node_id,
97            SfuNode {
98                first_registered_at: now,
99                last_seen_at: now,
100                draining: false,
101            },
102        );
103    }
104    nodes
105}

107/// What a registration upsert amounted to — drives log level in the handler
108/// (a steady-state heartbeat every 30s must not spam info logs).
109#[derive(Debug, PartialEq, Eq, Clone, Copy)]
110pub(crate) enum Registration {
111    /// First time this node ID has been seen.
112    New,
113    /// The node's `draining` flag changed.
114    DrainChanged,
115    /// Routine heartbeat, nothing changed but `last_seen_at`.
116    Heartbeat,
117}
118
119impl Api {
120    /// Upserts a node's registration (called from `POST /internal/sfu/register`)
121    /// and recomputes the advertised best pick, broadcasting
122    /// [`ServerMessage::SfuMigrate`](hound_control::ServerMessage::SfuMigrate)
123    /// to every live `/control` session if the pick changed. A heartbeat for a
124    /// known node refreshes `last_seen_at` only — `first_registered_at` is
125    /// stable, so selection order between live nodes never churns.
126    pub(crate) fn register_sfu_node(&self, node_id: String, draining: bool) -> Registration {
127        let now = Instant::now();
128        let registration = {
129            let mut nodes = self.sfu_nodes.write().unwrap();
130            match nodes.entry(node_id) {
131                std::collections::hash_map::Entry::Occupied(mut e) => {
132                    let node = e.get_mut();
133                    let drain_changed = node.draining != draining;
134                    node.last_seen_at = now;
135                    node.draining = draining;
136                    if drain_changed {
137                        Registration::DrainChanged
138                    } else {
139                        Registration::Heartbeat
140                    }
141                }
142                std::collections::hash_map::Entry::Vacant(e) => {
143                    e.insert(SfuNode {
144                        first_registered_at: now,
145                        last_seen_at: now,
146                        draining,
147                    });
148                    Registration::New
149                }
150            }
151        };
152        self.recompute_advertised_sfu();
153        registration
154    }
155
156    /// The current best SFU node for a new voice join. Recomputed fresh
157    /// against [`LIVENESS_TTL`] on every call (not the cached
158    /// `advertised_sfu`, which only updates on registration events) — a join
159    /// arriving between registrations still sees an expired node fall out of
160    /// consideration.
161    pub(crate) fn best_sfu_node(&self) -> Option<String> {
162        select_best(&self.sfu_nodes.read().unwrap(), Instant::now())
163    }
164
165    /// Snapshot of the registry for `GET /internal/sfu/nodes` — the observable
166    /// contract for tooling (hound-stack readiness polling, ops debugging)
167    /// instead of anyone scraping backend log lines. Liveness is evaluated
168    /// here with the same TTL selection uses.
169    pub(crate) fn sfu_nodes_status(&self) -> hound_types::SfuNodesResponse {
170        let now = Instant::now();
171        let nodes = self.sfu_nodes.read().unwrap();
172        let mut statuses: Vec<hound_types::SfuNodeStatus> = nodes
173            .iter()
174            .map(|(id, n)| hound_types::SfuNodeStatus {
175                node_id: id.clone(),
176                draining: n.draining,
177                live: now
178                    .checked_duration_since(n.last_seen_at)
179                    .unwrap_or(Duration::ZERO)
180                    <= LIVENESS_TTL,
181            })
182            .collect();
183        let best = select_best(&nodes, now);
184        drop(nodes);
185        // Deterministic order for consumers and tests.
186        statuses.sort_by(|a, b| a.node_id.cmp(&b.node_id));
187        hound_types::SfuNodesResponse {
188            nodes: statuses,
189            best,
190        }
191    }
192
193    /// If the best pick changed since the last registration, updates
194    /// `advertised_sfu` and broadcasts a migrate signal to every live
195    /// `/control` session across all spaces so already-connected clients
196    /// re-resolve `voice_join` onto the new node instead of riding out the old
197    /// one until it's forcibly killed.
198    fn recompute_advertised_sfu(&self) {
199        let best = self.best_sfu_node();
200        let mut advertised = self.advertised_sfu.write().unwrap();
201        if *advertised == best {
202            return;
203        }
204        let old = advertised.clone();
205        *advertised = best.clone();
206        drop(advertised);
207
208        tracing::info!(old_sfu = ?old, new_sfu = ?best, "SFU best-node changed");
209        if let Some(node_id) = best {
210            self.broadcast_sfu_migrate(node_id);
211        }
212    }
213}

215#[cfg(test)]
216impl Api {
217    /// Upserts a node with an explicit `registered_at`, bypassing
218    /// `register_sfu_node`'s `Instant::now()`. Lets TTL-expiry and
219    /// selection-ordering tests in `api::space`'s test module be deterministic
220    /// (backdate a registration) instead of racing the wall clock, and skips
221    /// the migrate-broadcast side effect so ordering setup doesn't itself
222    /// trigger one.
223    pub(crate) fn set_sfu_node_for_test(
224        &self,
225        node_id: &str,
226        registered_at: Instant,
227        draining: bool,
228    ) {
229        self.sfu_nodes.write().unwrap().insert(
230            node_id.to_owned(),
231            SfuNode {
232                first_registered_at: registered_at,
233                last_seen_at: registered_at,
234                draining,
235            },
236        );
237    }
238
239    /// The currently advertised best node, for asserting broadcast behavior
240    /// (`recompute_advertised_sfu` only broadcasts when this changes).
241    pub(crate) fn advertised_sfu_for_test(&self) -> Option<String> {
242        self.advertised_sfu.read().unwrap().clone()
243    }
244}

267        let mut nodes = HashMap::new();

273    #[test]
274    fn draining_flips_selection_to_the_newer_non_draining_node() {
275        let now = Instant::now();
276        let mut nodes = HashMap::new();
277        // "new" registered most recently but is draining — an older,
278        // still-healthy node must win over it.
279        nodes.insert("old", node(now - Duration::from_secs(20), false));
280        nodes.insert("new", node(now - Duration::from_secs(1), true));
281        let nodes: HashMap<String, SfuNode> =
282            nodes.into_iter().map(|(k, v)| (k.to_owned(), v)).collect();
283        assert_eq!(select_best(&nodes, now), Some("old".into()));
284    }

323    /// Regression: two live non-draining nodes heartbeating on interleaved
324    /// cycles must NOT leapfrog each other as "best". Selection keys on
325    /// `first_registered_at` (stable), so the newer NODE wins regardless of
326    /// whose HEARTBEAT arrived last — heartbeats only refresh liveness.
327    #[test]
328    fn interleaved_heartbeats_do_not_leapfrog_selection() {
329        let t0 = Instant::now();
330        let mut nodes = HashMap::new();
331        nodes.insert("old".to_owned(), node(t0, false));
332        nodes.insert("new".to_owned(), node(t0 + Duration::from_secs(10), false));
333
334        // Simulate alternating 30s heartbeats for several cycles: only
335        // last_seen_at moves. The pick must stay "new" at every step.
336        for cycle in 1..=6u64 {
337            let hb = t0 + Duration::from_secs(cycle * 15);
338            let id = if cycle % 2 == 0 { "new" } else { "old" };
339            nodes.get_mut(id).unwrap().last_seen_at = hb;
340            assert_eq!(
341                select_best(&nodes, hb),
342                Some("new".into()),
343                "heartbeat cycle {cycle} flipped the best node"
344            );
345        }
346    }

348    /// Same property at the `Api` level: interleaved heartbeats through the
349    /// real registration path never change `advertised_sfu` after the second
350    /// node's first registration — and an unchanged advertised node is exactly
351    /// the "no SfuMigrate broadcast" condition in `recompute_advertised_sfu`.
352    #[tokio::test]
353    async fn interleaved_heartbeats_do_not_rebroadcast_migrate() {
354        let (api, _server) = crate::api::test().await;
355
356        assert_eq!(
357            api.register_sfu_node("old".into(), false),
358            Registration::New
359        );
360        assert_eq!(api.advertised_sfu_for_test(), Some("old".into()));
361        assert_eq!(
362            api.register_sfu_node("new".into(), false),
363            Registration::New
364        );
365        assert_eq!(api.advertised_sfu_for_test(), Some("new".into()));
366
367        for _ in 0..4 {
368            assert_eq!(
369                api.register_sfu_node("old".into(), false),
370                Registration::Heartbeat
371            );
372            assert_eq!(api.advertised_sfu_for_test(), Some("new".into()));
373            assert_eq!(
374                api.register_sfu_node("new".into(), false),
375                Registration::Heartbeat
376            );
377            assert_eq!(api.advertised_sfu_for_test(), Some("new".into()));
378        }
379
380        // Draining is still a real transition: the flag flips selection back
381        // to the older healthy node and reports DrainChanged.
382        assert_eq!(
383            api.register_sfu_node("new".into(), true),
384            Registration::DrainChanged
385        );
386        assert_eq!(api.advertised_sfu_for_test(), Some("old".into()));
387    }
