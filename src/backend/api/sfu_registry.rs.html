1//! The multi-node SFU registry: which SFU nodes are live, and which one is
2//! "best" for a new voice join right now.
3//!
4//! ## Why a registry, not a single slot
5//!
6//! A blue/green SFU rollout runs the old and new pod side by side for a while
7//! (existing calls drain off the old one; new joins should land on the new
8//! one). A single `sfu_node_id: Option<String>` slot (the pre-rollout-support
9//! shape) can only ever name one node, so the last registration to land wins
10//! — during a rollout that's a coin flip between old and new pod, and it
11//! offers no way to say "this node is going away, don't route new joins to
12//! it." The registry keeps every node's last-seen registration plus its
13//! `draining` flag and picks the best live one at read time.
14//!
15//! ## In-process only
16//!
17//! Like [`super::control_calls`](super::Inner::control_calls), this is
18//! in-memory and per-process — it does not survive a backend redeploy and
19//! does not coordinate across horizontally-scaled replicas. That's fine: the
20//! SFU self-heals by re-registering every `REGISTER_INTERVAL`
21//! (`services/sfu/src/main.rs`), so a backend restart clears the registry and
22//! it repopulates within one interval, same as the single-slot predecessor.
23//!
24//! ## Liveness
25//!
26//! The SFU re-registers every 30s. A node is "live" if it registered within
27//! [`LIVENESS_TTL`] (3 missed cycles). Liveness is evaluated at read time
28//! ([`select_best`]) rather than by a background reaper — a stale entry just
29//! stops being selectable; nothing needs to actively evict it, and the map
30//! never grows unbounded because node IDs are stable per-pod-lifetime (a
31//! replaced pod re-registers under a fresh ephemeral identity, but the old
32//! entry ages out and is simply never selected again — see the `SfuNode`
33//! churn note below if that ever needs bounding).
34
35use std::collections::HashMap;
36use std::time::{Duration, Instant};
37
38use crate::api::Api;

40/// A node is live if it registered within this many registration intervals.
41/// The SFU re-registers every 30s; 90s tolerates two missed cycles (a slow
42/// backend, a blip) before presuming the node dead.
43const LIVENESS_TTL: Duration = Duration::from_secs(90);
44
45/// One SFU node's last-known registration state.
46///
47/// Two timestamps, deliberately: `first_registered_at` is the node's stable
48/// age and the SELECTION key; `last_seen_at` is refreshed by every heartbeat
49/// and used ONLY for liveness. Selecting on the heartbeat timestamp would make
50/// two live nodes leapfrog each other as "newest" on alternating 30s
51/// heartbeats — flipping the advertised node and broadcasting a migrate storm
52/// that bounces every in-call client between nodes.
53#[derive(Debug, Clone)]
54pub(crate) struct SfuNode {
55    pub(crate) first_registered_at: Instant,
56    pub(crate) last_seen_at: Instant,
57    pub(crate) draining: bool,
58    /// Active relay-session count from the node's last heartbeat. `None`
59    /// until a heartbeat carrying it has landed (a pre-upgrade SFU binary
60    /// never sends it — see `SfuRegisterRequest::sessions`).
61    pub(crate) sessions: Option<usize>,
62}
63
64/// Picks the best node to route a new voice join to: the **newest** (by first
65/// registration) live, non-draining node; if every live node is draining, the
66/// newest live node anyway (serving a call on a draining node beats refusing
67/// it outright — the node keeps accepting connections while it drains, see
68/// `services/sfu/src/main.rs`). `None` if no node has been seen within
69/// [`LIVENESS_TTL`].
70///
71/// Pure function of `nodes` and `now` so it's directly unit-testable without
72/// standing up an `Api` or waiting out real wall-clock time.
73pub(crate) fn select_best(nodes: &HashMap<String, SfuNode>, now: Instant) -> Option<String> {
74    let live: Vec<(&String, &SfuNode)> = nodes
75        .iter()
76        .filter(|(_, n)| {
77            now.checked_duration_since(n.last_seen_at)
78                .unwrap_or(Duration::ZERO)
79                <= LIVENESS_TTL
80        })
81        .collect();
82
83    live.iter()
84        .filter(|(_, n)| !n.draining)
85        .max_by_key(|(_, n)| n.first_registered_at)
86        .or_else(|| live.iter().max_by_key(|(_, n)| n.first_registered_at))
87        .map(|(id, _)| (*id).clone())
88}

90/// Seeds the registry at boot from the `SFU_NODE_ID` env var, so a dev/compose
91/// setup that only sets the env (never calls `BACKEND_URL` self-registration)
92/// keeps working exactly as the old single-slot field did. The seeded entry
93/// expires via [`LIVENESS_TTL`] like any other node unless something
94/// re-registers it before then.
95pub(crate) fn seed(env_node_id: Option<String>) -> HashMap<String, SfuNode> {
96    let mut nodes = HashMap::new();
97    if let Some(node_id) = env_node_id {
98        let now = Instant::now();
99        nodes.insert(
100            node_id,
101            SfuNode {
102                first_registered_at: now,
103                last_seen_at: now,
104                draining: false,
105                sessions: None,
106            },
107        );
108    }
109    nodes
110}

112/// What a registration upsert amounted to — drives log level in the handler
113/// (a steady-state heartbeat every 30s must not spam info logs).
114#[derive(Debug, PartialEq, Eq, Clone, Copy)]
115pub(crate) enum Registration {
116    /// First time this node ID has been seen.
117    New,
118    /// The node's `draining` flag changed.
119    DrainChanged,
120    /// Routine heartbeat, nothing changed but `last_seen_at`.
121    Heartbeat,
122}
123
124impl Api {
125    /// Upserts a node's registration (called from `POST /internal/sfu/register`)
126    /// and recomputes the advertised best pick, broadcasting
127    /// [`ServerMessage::SfuMigrate`](hound_control::ServerMessage::SfuMigrate)
128    /// to every live `/control` session if the pick changed. A heartbeat for a
129    /// known node refreshes `last_seen_at` only — `first_registered_at` is
130    /// stable, so selection order between live nodes never churns.
131    ///
132    /// `sessions` is always stored (it rides along on every heartbeat), but a
133    /// change in `sessions` alone is never a [`Registration::DrainChanged`] —
134    /// only `draining` flipping is a loggable state transition (see the
135    /// handler in `api::internal`); a session-count change is exactly the
136    /// steady-state noise `Registration::Heartbeat` exists to keep quiet.
137    pub(crate) fn register_sfu_node(
138        &self,
139        node_id: String,
140        draining: bool,
141        sessions: Option<usize>,
142    ) -> Registration {
143        let now = Instant::now();
144        let registration = {
145            let mut nodes = self.sfu_nodes.write().unwrap();
146            match nodes.entry(node_id) {
147                std::collections::hash_map::Entry::Occupied(mut e) => {
148                    let node = e.get_mut();
149                    let drain_changed = node.draining != draining;
150                    node.last_seen_at = now;
151                    node.draining = draining;
152                    node.sessions = sessions;
153                    if drain_changed {
154                        Registration::DrainChanged
155                    } else {
156                        Registration::Heartbeat
157                    }
158                }
159                std::collections::hash_map::Entry::Vacant(e) => {
160                    e.insert(SfuNode {
161                        first_registered_at: now,
162                        last_seen_at: now,
163                        draining,
164                        sessions,
165                    });
166                    Registration::New
167                }
168            }
169        };
170        self.recompute_advertised_sfu();
171        registration
172    }
173
174    /// The current best SFU node for a new voice join. Recomputed fresh
175    /// against [`LIVENESS_TTL`] on every call (not the cached
176    /// `advertised_sfu`, which only updates on registration events) — a join
177    /// arriving between registrations still sees an expired node fall out of
178    /// consideration.
179    pub(crate) fn best_sfu_node(&self) -> Option<String> {
180        select_best(&self.sfu_nodes.read().unwrap(), Instant::now())
181    }
182
183    /// Snapshot of the registry for `GET /internal/sfu/nodes` — the observable
184    /// contract for tooling (hound-stack readiness polling, ops debugging)
185    /// instead of anyone scraping backend log lines. Liveness is evaluated
186    /// here with the same TTL selection uses.
187    pub(crate) fn sfu_nodes_status(&self) -> hound_types::SfuNodesResponse {
188        let now = Instant::now();
189        let nodes = self.sfu_nodes.read().unwrap();
190        let mut statuses: Vec<hound_types::SfuNodeStatus> = nodes
191            .iter()
192            .map(|(id, n)| hound_types::SfuNodeStatus {
193                node_id: id.clone(),
194                draining: n.draining,
195                live: now
196                    .checked_duration_since(n.last_seen_at)
197                    .unwrap_or(Duration::ZERO)
198                    <= LIVENESS_TTL,
199                sessions: n.sessions,
200            })
201            .collect();
202        let best = select_best(&nodes, now);
203        drop(nodes);
204        // Deterministic order for consumers and tests.
205        statuses.sort_by(|a, b| a.node_id.cmp(&b.node_id));
206        hound_types::SfuNodesResponse {
207            nodes: statuses,
208            best,
209        }
210    }
211
212    /// If the best pick changed since the last registration, updates
213    /// `advertised_sfu` and broadcasts a migrate signal to every live
214    /// `/control` session across all spaces so already-connected clients
215    /// re-resolve `voice_join` onto the new node instead of riding out the old
216    /// one until it's forcibly killed.
217    fn recompute_advertised_sfu(&self) {
218        let best = self.best_sfu_node();
219        let mut advertised = self.advertised_sfu.write().unwrap();
220        if *advertised == best {
221            return;
222        }
223        let old = advertised.clone();
224        *advertised = best.clone();
225        drop(advertised);
226
227        tracing::info!(old_sfu = ?old, new_sfu = ?best, "SFU best-node changed");
228        if let Some(node_id) = best {
229            self.broadcast_sfu_migrate(node_id);
230        }
231    }
232}

234#[cfg(test)]
235impl Api {
236    /// Upserts a node with an explicit `registered_at`, bypassing
237    /// `register_sfu_node`'s `Instant::now()`. Lets TTL-expiry and
238    /// selection-ordering tests in `api::space`'s test module be deterministic
239    /// (backdate a registration) instead of racing the wall clock, and skips
240    /// the migrate-broadcast side effect so ordering setup doesn't itself
241    /// trigger one.
242    pub(crate) fn set_sfu_node_for_test(
243        &self,
244        node_id: &str,
245        registered_at: Instant,
246        draining: bool,
247    ) {
248        self.sfu_nodes.write().unwrap().insert(
249            node_id.to_owned(),
250            SfuNode {
251                first_registered_at: registered_at,
252                last_seen_at: registered_at,
253                draining,
254                sessions: None,
255            },
256        );
257    }
258
259    /// The currently advertised best node, for asserting broadcast behavior
260    /// (`recompute_advertised_sfu` only broadcasts when this changes).
261    pub(crate) fn advertised_sfu_for_test(&self) -> Option<String> {
262        self.advertised_sfu.read().unwrap().clone()
263    }
264}

286    fn picks_the_newest_live_non_draining_node() {
287        let now = Instant::now();
288        let mut nodes = HashMap::new();
289        nodes.insert("old".into(), node(now - Duration::from_secs(20), false));

294    #[test]
295    fn draining_flips_selection_to_the_newer_non_draining_node() {
296        let now = Instant::now();
297        let mut nodes = HashMap::new();
298        // "new" registered most recently but is draining — an older,
299        // still-healthy node must win over it.
300        nodes.insert("old", node(now - Duration::from_secs(20), false));
301        nodes.insert("new", node(now - Duration::from_secs(1), true));
302        let nodes: HashMap<String, SfuNode> =
303            nodes.into_iter().map(|(k, v)| (k.to_owned(), v)).collect();
304        assert_eq!(select_best(&nodes, now), Some("old".into()));
305    }

312        assert_eq!(

344    /// Regression: two live non-draining nodes heartbeating on interleaved
345    /// cycles must NOT leapfrog each other as "best". Selection keys on
346    /// `first_registered_at` (stable), so the newer NODE wins regardless of
347    /// whose HEARTBEAT arrived last — heartbeats only refresh liveness.
348    #[test]
349    fn interleaved_heartbeats_do_not_leapfrog_selection() {
350        let t0 = Instant::now();
351        let mut nodes = HashMap::new();
352        nodes.insert("old".to_owned(), node(t0, false));
353        nodes.insert("new".to_owned(), node(t0 + Duration::from_secs(10), false));
354
355        // Simulate alternating 30s heartbeats for several cycles: only
356        // last_seen_at moves. The pick must stay "new" at every step.
357        for cycle in 1..=6u64 {
358            let hb = t0 + Duration::from_secs(cycle * 15);
359            let id = if cycle % 2 == 0 { "new" } else { "old" };
360            nodes.get_mut(id).unwrap().last_seen_at = hb;
361            assert_eq!(
362                select_best(&nodes, hb),
363                Some("new".into()),
364                "heartbeat cycle {cycle} flipped the best node"
365            );
366        }
367    }

369    /// Same property at the `Api` level: interleaved heartbeats through the
370    /// real registration path never change `advertised_sfu` after the second
371    /// node's first registration — and an unchanged advertised node is exactly
372    /// the "no SfuMigrate broadcast" condition in `recompute_advertised_sfu`.
373    #[tokio::test]
374    async fn interleaved_heartbeats_do_not_rebroadcast_migrate() {
375        let (api, _server) = crate::api::test().await;
376
377        assert_eq!(
378            api.register_sfu_node("old".into(), false, None),
379            Registration::New
380        );
381        assert_eq!(api.advertised_sfu_for_test(), Some("old".into()));
382        assert_eq!(
383            api.register_sfu_node("new".into(), false, None),
384            Registration::New
385        );
386        assert_eq!(api.advertised_sfu_for_test(), Some("new".into()));
387
388        for _ in 0..4 {
389            assert_eq!(
390                api.register_sfu_node("old".into(), false, None),
391                Registration::Heartbeat
392            );
393            assert_eq!(api.advertised_sfu_for_test(), Some("new".into()));
394            assert_eq!(
395                api.register_sfu_node("new".into(), false, None),
396                Registration::Heartbeat
397            );
398            assert_eq!(api.advertised_sfu_for_test(), Some("new".into()));
399        }
400
401        // Draining is still a real transition: the flag flips selection back
402        // to the older healthy node and reports DrainChanged.
403        assert_eq!(
404            api.register_sfu_node("new".into(), true, None),
405            Registration::DrainChanged
406        );
407        assert_eq!(api.advertised_sfu_for_test(), Some("old".into()));
408    }
