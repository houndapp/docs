96/// Marks the instance as provisioned. First-wins-atomic under concurrent
97/// callers: the write is a single conditional `UPDATE ... WHERE provisioned =
98/// false`, so exactly one concurrent caller's rows-affected count is 1 (the
99/// rest see 0 and 409). Returns 409 if already provisioned.
100pub(crate) async fn post_provision(
101    LoggedIn(_claims): LoggedIn,
102    State(api): State<Api>,
103    body: Option<Json<PostProvisionRequest>>,
104) -> ApiResult<Json<InstanceModel>> {
105    // Ensures the singleton row exists; race-safe under concurrent first
106    // callers (see `get_or_create_instance`).
107    api.get_or_create_instance().await?;
108
109    let default_community =
110        body.and_then(|Json(PostProvisionRequest { default_community })| default_community);
111
112    let mut update = Instance::update_many()
113        .col_expr(instance::Column::Provisioned, Expr::value(true))
114        .filter(instance::Column::Id.eq(Api::INSTANCE_ID))
115        .filter(instance::Column::Provisioned.eq(false));
116    if let Some(id) = default_community {
117        update = update.col_expr(instance::Column::DefaultCommunity, Expr::value(id));
118    }
119    let result = update.exec(&api.db).await?;
120    if result.rows_affected == 0 {
121        return Err(ApiError::InstanceAlreadyProvisioned);
122    }
123
124    let updated = api.get_or_create_instance().await?;
125    Ok(Json(InstanceModel::from(updated)))
126}

134#[utoipa::path(
135    put,
136    path = "/user/{user_id}/permissions",
137    tag = TAG,
138    params(
139        ("user_id" = Uuid, Path, description = "The user whose instance permissions to set"),
140    ),
141    request_body(content = PutUserPermissionsRequest),
142    responses(
143        (status = 200, description = "permissions updated"),
144        (status = 403, description = "missing ManageInstance permission"),
145    ),
146    security(("token" = [])),
147)]
148/// Replaces the instance-level permissions for a user. Requires `ManageInstance`.
149pub(crate) async fn put_user_permissions(
150    InstancePerm { .. }: InstancePerm<perm::instance::ManageInstance>,
151    Path(user_id): Path<Uuid>,
152    State(api): State<Api>,
153    Json(req): Json<PutUserPermissionsRequest>,
154) -> ApiResult<()> {
155    let txn = api.db.begin().await?;
156    instance_user_permission::Entity::delete_many()
157        .filter(instance_user_permission::Column::UserId.eq(user_id))
158        .exec(&txn)
159        .await?;
160    for perm in req.permissions {
161        instance_user_permission::ActiveModel {
162            user_id: Set(user_id),
163            permission: Set(perm),
164        }
165        .insert(&txn)
166        .await?;
167    }
168    txn.commit().await?;
169    Ok(())
170}

172impl Api {
173    pub(crate) const INSTANCE_ID: i32 = 1;
174
175    /// Returns the single instance row, inserting a default one if it doesn't
176    /// exist yet. The insert is `ON CONFLICT DO NOTHING` (portable across
177    /// sqlite/postgres) rather than a plain insert: two concurrent first
178    /// callers (e.g. two racing first registrations) both take the "missing"
179    /// branch, and only one insert may land on the fixed `id = 1` row — the
180    /// other must not 500 on the resulting unique-key violation, it must just
181    /// see the winner's row.
182    pub(crate) async fn get_or_create_instance(&self) -> DbResult<instance::Model> {
183        if let Some(instance) = Instance::find_by_id(Self::INSTANCE_ID)
184            .one(&self.db)
185            .await?
186        {
187            return Ok(instance);
188        }
189
190        Instance::insert(instance::ActiveModel {
191            id: Set(Self::INSTANCE_ID),
192            single_community_instance: Set(true),
193            default_community: Set(None),
194            provisioned: Set(false),
195            first_user_granted: Set(false),
196        })
197        .on_conflict(
198            sea_orm::sea_query::OnConflict::column(instance::Column::Id)
199                .do_nothing()
200                .to_owned(),
201        )
202        .exec_without_returning(&self.db)
203        .await?;
204
205        // Either we inserted it, or a concurrent caller beat us to the same
206        // `id = 1` row — either way it now exists.
207        Instance::find_by_id(Self::INSTANCE_ID)
208            .one(&self.db)
209            .await?
210            .ok_or_else(|| {
211                sea_orm::DbErr::RecordNotFound(
212                    "instance row missing immediately after an ON CONFLICT DO NOTHING insert"
213                        .to_owned(),
214                )
215            })
216    }
217
218    /// Atomically claims the one-time "first user" instance-permission grant
219    /// slot. Returns `true` for exactly one caller across any number of
220    /// concurrent registrations — the arbiter is the conditional `UPDATE ...
221    /// WHERE first_user_granted = false`'s rows-affected count, not a
222    /// preceding `SELECT`, so there is no read-then-write window for two
223    /// callers to both observe "not yet granted". Portable across
224    /// sqlite/postgres: relevant only through standard `UPDATE ... WHERE`
225    /// row-affected semantics (a concurrent updater targeting the same row
226    /// either serializes behind the winner, on sqlite, or is re-evaluated
227    /// against the winner's committed row, on postgres — both land on 0 rows
228    /// affected for the loser). Must run in the same transaction as the
229    /// caller's user-row insert so the claim rolls back with it on failure.
230    /// The caller must have already ensured the singleton instance row exists
231    /// (via [`Self::get_or_create_instance`], on its own pooled connection,
232    /// before opening the transaction — every caller does, to read
233    /// `default_community`) — a claim `UPDATE` against a not-yet-existing row
234    /// affects 0 rows, which would wrongly read as "already granted".
235    pub(crate) async fn claim_first_user_grant(
236        &self,
237        conn: &impl ConnectionTrait,
238    ) -> DbResult<bool> {
239        let result = Instance::update_many()
240            .col_expr(instance::Column::FirstUserGranted, Expr::value(true))
241            .filter(instance::Column::Id.eq(Self::INSTANCE_ID))
242            .filter(instance::Column::FirstUserGranted.eq(false))
243            .exec(conn)
244            .await?;
245        Ok(result.rows_affected == 1)
246    }
247
248    /// Clears `default_community` when it points at `community_id`. The FK has no
249    /// cascade, so this must run before deleting the community row. No-op otherwise.
250    pub(crate) async fn clear_default_community(
251        &self,
252        conn: &impl sea_orm::ConnectionTrait,
253        community_id: Uuid,
254    ) -> DbResult<()> {
255        Instance::update_many()
256            .col_expr(
257                instance::Column::DefaultCommunity,
258                sea_orm::sea_query::Expr::value(Option::<Uuid>::None),
259            )
260            .filter(instance::Column::DefaultCommunity.eq(community_id))
261            .exec(conn)
262            .await?;
263        Ok(())
264    }

372    // ── HTTP: post_provision atomicity ──────────────────────────────────────────
373
374    /// Two concurrent `POST /instance/provision` calls (racing on the same
375    /// unprovisioned instance) must settle first-wins-atomic: exactly one 200,
376    /// exactly one 409, and a stable `default_community` that matches whichever
377    /// call actually won (never last-write-wins, never a torn mix of both).
378    #[tokio::test]
379    async fn concurrent_provisions_are_first_wins_atomic() {
380        use crate::api::community::{CommunityModel, PostCommunityRequest};
381
382        let (api, server) = crate::api::test().await;
383
384        let admin = api.insert_test_user().await;
385        api.grant_all_instance_permissions(&api.db, admin.id)
386            .await
387            .unwrap();
388        let admin_token = api.test_access_token(admin.id);
389
390        // Two distinct communities so the settled `default_community` tells us
391        // which of the two concurrent calls actually won.
392        let community_a: CommunityModel = server
393            .post("/community")
394            .authorization_bearer(&admin_token)
395            .json(&PostCommunityRequest { name: "a".into() })
396            .await
397            .json();
398        let community_b: CommunityModel = server
399            .post("/community")
400            .authorization_bearer(&admin_token)
401            .json(&PostCommunityRequest { name: "b".into() })
402            .await
403            .json();
404
405        let server = Arc::new(server);
406        let mut handles = Vec::new();
407        for community_id in [community_a.id, community_b.id] {
408            let server = server.clone();
409            let admin_token = admin_token.clone();
410            handles.push(tokio::spawn(async move {
411                server
412                    .post("/instance/provision")
413                    .authorization_bearer(&admin_token)
414                    .json(&PostProvisionRequest {
415                        default_community: Some(community_id),
416                    })
417                    .await
418                    .status_code()
419                    .as_u16()
420            }));
421        }
422
423        let mut codes = Vec::new();
424        for handle in handles {
425            codes.push(handle.await.unwrap());
426        }
427        codes.sort_unstable();
428        assert_eq!(codes, vec![200, 409], "exactly one winner, one conflict");
429
430        let instance: InstanceModel = server
431            .get("/instance")
432            .authorization_bearer(&admin_token)
433            .await
434            .json();
435        assert!(instance.provisioned);
436        assert!(
437            instance.default_community == Some(community_a.id)
438                || instance.default_community == Some(community_b.id),
439            "settled default_community must be exactly one of the two racers, got {:?}",
440            instance.default_community
441        );
442    }
