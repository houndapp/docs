1//! Login-method management for a signed-in account: list identities, link an
2//! email (the authenticated twin of passkey registration), and unlink.
3//!
4//! Two invariants, both enforced in [`Api::remove_identity`] so no endpoint
5//! can drift:
6//! - never-strand: removal must leave at least one identity on the account.
7//!   Every provider counts unconditionally — a bare email identity (no
8//!   passkeys) used to be excluded, but magic-link (`api/auth/magic.rs`) now
9//!   makes it sign-in-able on its own, so it counts like Steam always did.
10//! - deleting the email identity deletes the account's passkeys with it (they
11//!   are unreachable without an email to start a passkey login ceremony —
12//!   magic-link doesn't need one).
13

49/// Lists the signed-in account's login methods.
50pub(crate) async fn list_identities(
51    LoggedIn(claims): LoggedIn,
52    State(api): State<Api>,
53) -> ApiResult<Json<IdentitiesResponse>> {
54    let identities = api.user_identities(claims.sub).await?;
55    let passkey_count = Passkey::find()
56        .filter(passkey::Column::UserId.eq(claims.sub))
57        .count(&api.db)
58        .await?;
59    Ok(Json(IdentitiesResponse {
60        identities: identities
61            .into_iter()
62            .map(|i| IdentityEntry {
63                provider: i.provider,
64                external_id: i.external_id,
65            })
66            .collect(),
67        passkey_count,
68    }))
69}

84/// Unlinks a login method from the signed-in account.
85pub(crate) async fn unlink_identity(
86    LoggedIn(claims): LoggedIn,
87    State(api): State<Api>,
88    Path(provider): Path<IdentityProvider>,
89) -> ApiResult<StatusCode> {
90    api.remove_identity(claims.sub, provider).await?;

92}

122/// The account behind a link-ceremony request: the session bearer token when
123/// present, else the scoped `ceremony_token` (browser-deferred flow).
124fn link_ceremony_user(
125    api: &Api,
126    claims: &Option<Claims>,
127    ceremony_token: Option<&str>,
128) -> ApiResult<Uuid> {
129    if let Some(claims) = claims {
130        return Ok(claims.sub);
131    }
132    let token = ceremony_token.ok_or(ApiError::Jwt(super::jwt::AuthError::MissingToken))?;
133    Ok(api.auth.jwt.validate_email_link_token(token)?.sub)
134}

143        (status = 401, description = "not signed in and no valid ceremony token"),
144        (status = 409, description = "email already in use, or this account already has one")
145    ),
146    security(("token" = []))

148/// Starts linking an email + passkey to an account. The ceremony is ordinary
149/// passkey registration; no new user or profile is created on finish.
150pub(crate) async fn link_email_start(
151    MaybeLoggedIn(claims): MaybeLoggedIn,
152    State(api): State<Api>,
153    Json(req): Json<LinkEmailStartRequest>,
154) -> ApiResult<Json<StartRegisterResponse>> {
155    let user_id = link_ceremony_user(&api, &claims, req.ceremony_token.as_deref())?;
156    if api.find_user_by_email(&req.email).await?.is_some() {
157        return Err(ApiError::IdentityAlreadyLinked);
158    }
159    if api
160        .user_identities(user_id)
161        .await?
162        .iter()
163        .any(|i| i.provider == IdentityProvider::Email)
164    {
165        return Err(ApiError::IdentityAlreadyLinked);
166    }
167
168    let (options, passkey) =
169        api.auth
170            .rp
171            .start_registration(user_id, &req.email, &req.display_name)?;
172    // Keyed by the linking user id, so it can't collide with fresh
173    // registrations (v7 uuids minted per request).
174    api.auth.inflights.write().await.insert(
175        user_id,
176        InFlightRegistration::new(user_id, req.email, req.display_name, passkey),
177    );
178
179    Ok(Json(StartRegisterResponse { options, user_id }))
180}

182#[utoipa::path(
183    post,
184    path = "/identity/email/link/finish",
185    tag = TAG,
186    request_body = LinkEmailFinishRequest,
187    responses(
188        (status = 204, description = "email linked and passkey registered"),
189        (status = 400, description = "no active link ceremony"),
190        (status = 401, description = "not signed in and no valid ceremony token"),
191        (status = 409, description = "email was claimed while the ceremony ran")
192    ),
193    security(("token" = []))
194)]
195/// Completes the email + passkey link ceremony.
196pub(crate) async fn link_email_finish(
197    MaybeLoggedIn(claims): MaybeLoggedIn,
198    State(api): State<Api>,
199    Json(req): Json<LinkEmailFinishRequest>,
200) -> ApiResult<StatusCode> {
201    let user_id = link_ceremony_user(&api, &claims, req.ceremony_token.as_deref())?;
202    let inflight = api
203        .auth
204        .inflights
205        .write()
206        .await
207        .remove(&user_id)
208        .ok_or(ApiError::RegistrationStateNotFound(user_id))?;
209
210    let passkey = api
211        .auth
212        .rp
213        .finish_registration(&req.credential, &inflight.passkey)?;
214
215    if api.find_user_by_email(&inflight.email).await?.is_some() {
216        return Err(ApiError::IdentityAlreadyLinked);
217    }
218
219    let txn = api.db.begin().await?;
220    api.insert_identity(&txn, user_id, IdentityProvider::Email, inflight.email)
221        .await?;
222    api.insert_passkey(&txn, user_id, inflight.name, passkey)
223        .await?;
224    txn.commit().await?;
225
226    Ok(StatusCode::NO_CONTENT)
227}

229impl Api {
230    pub(crate) async fn user_identities(&self, user_id: Uuid) -> DbResult<Vec<identity::Model>> {
231        identity::Entity::find()
232            .filter(identity::Column::UserId.eq(user_id))
233            .order_by_asc(identity::Column::Provider)
234            .all(&self.db)
235            .await
236    }
237
238    /// Removes a login method, holding both linking invariants (see module
239    /// docs). The sole path for identity removal.
240    pub(crate) async fn remove_identity(
241        &self,
242        user_id: Uuid,
243        provider: IdentityProvider,
244    ) -> ApiResult<()> {
245        let txn = self.db.begin().await?;
246
247        let identities = identity::Entity::find()
248            .filter(identity::Column::UserId.eq(user_id))
249            .all(&txn)
250            .await?;
251        if !identities.iter().any(|i| i.provider == provider) {
252            return Err(ApiError::IdentityNotFound);
253        }
254
255        // Every provider is unconditionally usable to sign in now (magic-link
256        // makes a bare email sign-in-able), so never-strand just needs ≥1
257        // OTHER identity to remain.
258        if identities.iter().filter(|i| i.provider != provider).count() == 0 {
259            return Err(ApiError::LastLoginMethod);
260        }
261
262        identity::Entity::delete_many()
263            .filter(identity::Column::UserId.eq(user_id))
264            .filter(identity::Column::Provider.eq(provider))
265            .exec(&txn)
266            .await?;
267        if provider == IdentityProvider::Email {
268            Passkey::delete_many()
269                .filter(passkey::Column::UserId.eq(user_id))
270                .exec(&txn)
271                .await?;
272        }
273
274        txn.commit().await?;
275        Ok(())
276    }
277}

327        // insert_test_user has a bare email identity (no passkeys) — usable
