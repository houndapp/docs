122    LoggedIn(_claims): LoggedIn,
123    axum::extract::Path(id): axum::extract::Path<Uuid>,
124    State(api): State<Api>,
125) -> ApiResult<Json<CompleteCommunitySpace>> {
126    let Some(space) = api.find_space_by_id(id).await? else {
127        return Err(super::ApiError::CommunitySpaceNotFound(id));
128    };
129
130    let kind = space.kind.into();
131    Ok(Json(CompleteCommunitySpace {
132        id: space.id,
133        community_id: space.community_id,
134        name: space.name,
135        kind,
136        details: match kind {
137            CommunitySpaceKind::Text => Some(CommunitySpaceDetails::Text {
138                messages: api.find_messages_by_space(id, Default::default()).await?,
139            }),
140            _ => None,
141        },
142    }))
143}

260/// Join a voice space: validates membership, issues a LiveKit token, and broadcasts voice:join.
261pub(crate) async fn post_voice_join(
262    Path(space_id): Path<Uuid>,
263    LoggedIn(claims): LoggedIn,
264    State(api): State<Api>,
265) -> ApiResult<Json<VoiceTokenResponse>> {
266    let space = api
267        .find_space_by_id(space_id)
268        .await?
269        .ok_or(ApiError::CommunitySpaceNotFound(space_id))?;
270
271    let is_member = api
272        .is_community_member(space.community_id, claims.sub)
273        .await?;
274    if !is_member {
275        return Err(ApiError::NotInCommunity(claims.sub, space.community_id));
276    }
277
278    let profiles = api
279        .find_merged_profiles(std::iter::once(claims.sub), space.community_id)
280        .await?;
281    let profile = profiles
282        .into_iter()
283        .next()
284        .ok_or_else(|| ApiError::NotInCommunity(claims.sub, space.community_id))?;
285
286    let token = api.generate_livekit_token(space_id, claims.sub, profile.name.clone())?;
287
288    let event = VoiceJoinEvent {
289        user_id: claims.sub,
290        space_id,
291        community_id: space.community_id,
292        name: profile.name.clone(),
293        picture: profile.picture.clone(),
294    };
295    api.add_voice_participant(event.clone());
296    api.bump_eviction_generation(claims.sub); // cancel any pending disconnect eviction
297    api.emit_voice_joined(&event).await;
298
299    Ok(Json(VoiceTokenResponse {
300        token,
301        url: api.livekit_url.clone(),
302        profile,
303    }))
304}

356/// Keepalive heartbeat: resets the server-side eviction timer for the caller.
357/// The client sends this every 10 s; after 30 s of silence the server evicts the user.
358pub(crate) async fn post_voice_heartbeat(
359    Path(_space_id): Path<Uuid>,
360    LoggedIn(claims): LoggedIn,
361    State(api): State<Api>,
362) -> ApiResult<()> {
363    api.bump_eviction_generation(claims.sub);
364    Ok(())
365}

368    pub(crate) fn add_voice_participant(&self, event: VoiceJoinEvent) {
369        let mut map = self.voice_participants.write().unwrap();
370        map.entry(event.space_id)
371            .or_insert_with(std::collections::HashMap::new)
372            .insert(event.user_id, event);
373    }
374
375    pub(crate) fn remove_voice_participant(&self, space_id: Uuid, user_id: Uuid) {
376        let mut map = self.voice_participants.write().unwrap();
377        if let Some(space_map) = map.get_mut(&space_id) {
378            space_map.remove(&user_id);
379        }
380    }

382    /// Increments the eviction generation for a user and returns the new value.
383    /// Invalidates any pending disconnect-eviction task from an earlier socket session.
384    pub(crate) fn bump_eviction_generation(&self, user_id: Uuid) -> u64 {
385        let mut gens = self.voice_eviction_generations.write().unwrap();
386        let eviction_gen = gens.entry(user_id).and_modify(|g| *g += 1).or_insert(1);
387        *eviction_gen
388    }
389
390    pub(crate) fn current_eviction_generation(&self, user_id: Uuid) -> u64 {
391        self.voice_eviction_generations
392            .read()
393            .unwrap()
394            .get(&user_id)
395            .copied()
396            .unwrap_or(0)
397    }

399    /// Evicts a user from all voice channels they are in and broadcasts `voice:leave`.
400    /// Called after the 30 s grace period following a socket disconnect.
401    pub(crate) async fn evict_voice_participant_by_user(&self, user_id: Uuid) {
402        let to_evict: Vec<(Uuid, Uuid)> = {
403            let map = self.voice_participants.read().unwrap();
404            map.iter()
405                .filter_map(|(&space_id, space_map)| {
406                    space_map.get(&user_id).map(|e| (space_id, e.community_id))
407                })
408                .collect()
409        };
410
411        for (space_id, community_id) in to_evict {
412            tracing::info!(%user_id, %space_id, "evicting voice participant after disconnect grace period");
413            self.remove_voice_participant(space_id, user_id);
414            let event = VoiceLeaveEvent {
415                user_id,
416                space_id,
417                community_id,
418            };
419            self.emit_voice_left(&event).await;
420        }
421    }

516    pub(crate) fn generate_livekit_token(
517        &self,
518        space_id: Uuid,
519        user_id: Uuid,
520        display_name: Option<String>,
521    ) -> Result<String, ApiError> {
522        use jsonwebtoken::{Algorithm, EncodingKey, Header, encode};
523
524        #[derive(serde::Serialize)]
525        struct LiveKitClaims {
526            exp: i64,
527            nbf: i64,
528            iss: String,
529            /// Participant identity — must be the user_id so the JS client can
530            /// resolve it against the socket.io participant map.
531            sub: String,
532            /// Optional display name shown in the LiveKit room UI.
533            #[serde(skip_serializing_if = "Option::is_none")]
534            name: Option<String>,
535            video: LiveKitVideoGrants,
536        }
537
538        #[derive(serde::Serialize)]
539        struct LiveKitVideoGrants {
540            #[serde(rename = "roomJoin")]
541            room_join: bool,
542            room: String,
543            #[serde(rename = "canPublish")]
544            can_publish: bool,
545            #[serde(rename = "canSubscribe")]
546            can_subscribe: bool,
547        }
548
549        let now = chrono::Utc::now();
550        let claims = LiveKitClaims {
551            exp: (now + chrono::Duration::hours(6)).timestamp(),
552            nbf: now.timestamp(),
553            iss: self.livekit_api_key.clone(),
554            sub: user_id.to_string(),
555            name: display_name,
556            video: LiveKitVideoGrants {
557                room_join: true,
558                room: space_id.to_string(),
559                can_publish: true,
560                can_subscribe: true,
561            },
562        };
563
564        encode(
565            &Header::new(Algorithm::HS256),
566            &claims,
567            &EncodingKey::from_secret(self.livekit_api_secret.as_bytes()),
568        )
569        .map_err(|e| ApiError::LiveKit(e.to_string()))
570    }
