37/// Registers the `/community` namespace event handlers.
38///
39/// On connect the client must supply a JWT access token in the socket.io
40/// handshake auth object: `{ "token": "<jwt>" }`.  The server validates it,
41/// then auto-joins the socket to a `community:{id}` room for every community
42/// the user belongs to.
43///
44/// The client may then subscribe or unsubscribe to space-level events:
45///
46/// - `space:subscribe`   — payload: UUID string — joins `space:{id}` room (if
47///   the user is a member of the owning community)
48/// - `space:unsubscribe` — payload: UUID string — leaves `space:{id}` room
49pub(crate) fn register_events(api: Api) {
50    api.socket_io.ns("/community", {
51        let api = api.clone();
52        move |socket: SocketRef, TryData(auth): TryData<AuthPayload>| {
53            let api = api.clone();
54            async move {
55                on_connect(socket, api, auth).await;
56            }
57        }
58    });
59}

65#[derive(serde::Deserialize)]
66struct AuthPayload {

80// ── connection handler ───────────────────────────────────────────────────────
81
82async fn on_connect(socket: SocketRef, api: Api, auth: Result<AuthPayload, impl std::fmt::Debug>) {
83    // Validate token from handshake auth payload.
84    let raw_token = match auth {
85        Ok(p) => p.token,
86        Err(_) => {
87            let _ = socket.disconnect();
88            return;
89        }
90    };
91    let token = raw_token
92        .strip_prefix("Bearer ")
93        .unwrap_or(&raw_token)
94        .to_owned();
95
96    let claims = match api.auth.jwt.validate_unrevoked_token(&token).await {
97        Ok(c) if c.is_access_token() => c,
98        _ => {
99            let _ = socket.disconnect();
100            return;
101        }
102    };
103
104    let user_id = claims.sub;
105    socket.extensions.insert(user_id);
106
107    // Register per-socket event handlers immediately after auth so they are
108    // in place before any client events arrive (the CONNECT ack is sent before
109    // this async handler completes, so the client can emit events right away).
110    register_socket_events(&socket, api.clone());
111
112    // Auto-join a room for every community the user is a member of.
113    let ids = match api.find_community_ids_by_user(user_id).await {
114        Ok(ids) => ids,
115        Err(err) => {
116            tracing::error!(%err, "failed to load communities for socket auth");
117            let _ = socket.disconnect();
118            return;
119        }
120    };
121
122    for &id in &ids {
123        socket.join(community_room(id));
124    }
125
126    // Push the current voice state for all joined communities so the client
127    // has a consistent snapshot at the moment the subscription is established,
128    // with no race window between fetching and subscribing.
129    let voice_state: Vec<VoiceJoinEvent> = ids
130        .iter()
131        .flat_map(|&id| api.get_voice_participants_for_community(id))
132        .collect();
133    let _ = socket.emit("voice:state", &voice_state);
134
135    // On disconnect, wait 30 s then evict the user from any voice channel they
136    // were in — unless a newer socket session (or voice rejoin) has since bumped
137    // the generation counter, which cancels this eviction.
138    {
139        let api = api.clone();
140        socket.on_disconnect(async move |_socket: SocketRef, _: DisconnectReason| {
141            let eviction_gen = api.bump_eviction_generation(user_id);
142            let api2 = api.clone();
143            tokio::spawn(async move {
144                tokio::time::sleep(std::time::Duration::from_secs(30)).await;
145                if api2.current_eviction_generation(user_id) == eviction_gen {
146                    api2.evict_voice_participant_by_user(user_id).await;
147                }
148            });
149        });
150    }
151}

153// ── per-socket event registration ───────────────────────────────────────────
154
155fn register_socket_events(socket: &SocketRef, api: Api) {
156    // space:subscribe — join a space room after verifying membership
157    socket.on("space:subscribe", {
158        let api = api.clone();
159        move |socket: SocketRef, Data(space_id): Data<Uuid>| {
160            let api = api.clone();
161            async move {
162                let user_id = match socket.extensions.get::<Uuid>() {
163                    Some(id) => id,
164                    None => return,
165                };
166                on_space_subscribe(socket, api, space_id, user_id).await;
167            }
168        }
169    });
170
171    // space:unsubscribe — leave a space room
172    socket.on(
173        "space:unsubscribe",
174        move |socket: SocketRef, Data(space_id): Data<Uuid>| async move {
175            socket.leave(space_room(space_id));
176        },
177    );
178
179    // voice:screenshare — client toggled screenshare; re-broadcast to community
180    socket.on("voice:screenshare", {
181        let api = api.clone();
182        move |socket: SocketRef, Data(payload): Data<ScreenSharePayload>| {
183            let api = api.clone();
184            async move {
185                let user_id = match socket.extensions.get::<Uuid>() {
186                    Some(id) => id,
187                    None => return,
188                };
189                let space = match api.find_space_by_id(payload.space_id).await {
190                    Ok(Some(s)) => s,
191                    _ => return,
192                };
193                let event = VoiceScreenShareEvent {
194                    user_id,
195                    space_id: payload.space_id,
196                    community_id: space.community_id,
197                    enabled: payload.enabled,
198                };
199                let _ = api
200                    .socket_io
201                    .of("/community")
202                    .unwrap()
203                    .to(community_room(space.community_id))
204                    .emit("voice:screenshare", &event)
205                    .await;
206            }
207        }
208    });
209}

217async fn on_space_subscribe(socket: SocketRef, api: Api, space_id: Uuid, user_id: Uuid) {
218    let space = match api.find_space_by_id(space_id).await {
219        Ok(Some(s)) => s,
220        Ok(None) => {
221            tracing::warn!(%space_id, %user_id, "space:subscribe — space not found");
222            return;
223        }
224        Err(err) => {
225            tracing::error!(%err, %space_id, %user_id, "space:subscribe — db error finding space");
226            return;
227        }
228    };
229    match api.is_community_member(space.community_id, user_id).await {
230        Ok(true) => {
231            tracing::debug!(%space_id, %user_id, "space:subscribe — joined room");
232            socket.join(space_room(space_id));
233        }
234        Ok(false) => {
235            tracing::warn!(%space_id, community_id = %space.community_id, %user_id, "space:subscribe — user is not a community member");
236        }
237        Err(err) => {
238            tracing::error!(%err, %space_id, %user_id, "space:subscribe — db error checking membership");
239        }
240    }
241}

534/// Creates a new community with default general and voice spaces.
535pub(crate) async fn post_community(
536    LoggedIn(claims): LoggedIn,
537    State(api): State<Api>,
538    Json(req): Json<PostCommunityRequest>,
539) -> ApiResult<Json<CommunityModel>> {
540    if req.name.is_empty() {
541        return Err((StatusCode::BAD_REQUEST, ()).into_response().into());
542    }
543
544    let txn = api.db.begin().await?;
545    let community = api.insert_community(&txn).await?;
546    api.join_community(&txn, claims.sub, community.id).await?;
547    let general = api
548        .insert_community_space(
549            &txn,
550            community.id,
551            "general".into(),
552            CommunitySpaceKind::Text,
553        )
554        .await?;
555    let voice = api
556        .insert_community_space(
557            &txn,
558            community.id,
559            "voice".into(),
560            CommunitySpaceKind::Voice,
561        )
562        .await?;
563    let settings = api
564        .insert_community_settings(&txn, community.id, general.id, req.name)
565        .await?;
566    txn.commit().await?;
567
568    Ok(Json(CommunityModel {
569        name: settings.name,
570        default_space: settings.default_space,
571        id: community.id,
572        spaces: Page::single_page(vec![
573            CommunitySpaceSummary::from(general),
574            CommunitySpaceSummary::from(voice),
575        ]),
576    }))
577}

696#[sea_orm(entity = "community::Entity", from_query_result)]
697pub(crate) struct CompleteCommunity {
698    pub id: Uuid,
699    #[sea_orm(nested)]
700    pub settings: community_settings::Model,
701    #[sea_orm(skip)]
702    pub spaces: Page<community_space::Model>,
703}
704
705impl Api {
706    pub(crate) async fn join_community(
707        &self,
708        conn: &impl sea_orm::ConnectionTrait,
709        user_id: Uuid,
710        community_id: Uuid,
711    ) -> DbResult<()> {
712        self.insert_community_member(conn, user_id, community_id)
713            .await?;
714        self.insert_community_profile(conn, user_id, community_id)
715            .await?;
716        Ok(())
717    }
718
719    async fn insert_community(
720        &self,
721        conn: &impl sea_orm::ConnectionTrait,
722    ) -> DbResult<community::Model> {
723        community::ActiveModel {
724            id: Set(uuid::Uuid::now_v7()),
725        }
726        .insert(conn)
727        .await
728    }
729
730    async fn insert_community_settings(
731        &self,
732        conn: &impl sea_orm::ConnectionTrait,
733        community_id: Uuid,
734        space_id: Uuid,
735        name: String,
736    ) -> DbResult<community_settings::Model> {
737        community_settings::ActiveModel {
738            id: Set(uuid::Uuid::now_v7()),
739            name: Set(name),
740            community_id: Set(community_id),
741            default_space: Set(space_id),
742        }
743        .insert(conn)
744        .await
745    }
746
747    async fn find_community(&self, id: Uuid) -> DbResult<Option<CompleteCommunity>> {
748        let Some(mut community): Option<CompleteCommunity> = Community::find_by_id(id)
749            .join(
750                sea_orm::JoinType::InnerJoin,
751                community_settings::Relation::Community.def().rev(),
752            )
753            .into_partial_model()
754            .one(&self.db)
755            .await?
756        else {
757            return Ok(None);
758        };
759
760        community.spaces = CommunitySpace::find()
761            .filter(community_space::Column::CommunityId.eq(community.id))
762            .paginate(&self.db, 50)
763            .to_page(0)
764            .await?;
765
766        Ok(Some(community))
767    }
768
769    async fn find_communities(
770        &self,
771        params: PaginationParameters,
772    ) -> DbResult<Page<CompleteCommunity>> {
773        Community::find()
774            .join(
775                sea_orm::JoinType::InnerJoin,
776                community_settings::Relation::Community.def().rev(),
777            )
778            .into_partial_model()
779            .paginate(&self.db, params.per_page())
780            .to_page(params.page())
781            .await
782    }
