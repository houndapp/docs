15/// Container for all authentication related state.
16#[derive(Clone, Debug)]
17pub struct Auth {
18    /// Responsible for issuing and authentication
19    pub webauthn: Webauthn,
20    // Maps user‑id → registration state (for the duration of the ceremony)
21    pub inflights: Arc<RwLock<HashMap<Uuid, InFlightRegistration>>>,
22    // TODO: Replace auth_states and credentials with database store.
23    pub auth_states: Arc<RwLock<HashMap<Uuid, PasskeyAuthentication>>>,
24    pub credentials: Arc<RwLock<HashMap<Uuid, Passkey>>>,
25    /// The secret key for issuing JSON Web Tokens (JWT)
26    pub secret_key: [u8; 64],
27}

68/// Container for passkey registrations that have started but not finished.
69#[derive(Debug)]
70pub struct InFlightRegistration {
71    passkey: PasskeyRegistration,
72    expires_at: DateTime,
73}
74
75impl InFlightRegistration {
76    const TTL: std::time::Duration = std::time::Duration::from_secs(30 * 60);
77
78    fn new(passkey: PasskeyRegistration) -> Self {
79        Self {
80            passkey,
81            expires_at: chrono::Utc::now() + Self::TTL,
82        }
83    }
84
85    pub fn is_expired(&self, now: DateTime) -> bool {
86        now > self.expires_at
87    }
88}

169            iat: now.timestamp(),
170            exp: (now + std::time::Duration::from_secs(validity_hours * 60 * 60)).timestamp(),
171        }
172    }

186/// Begins the ceremony to register a user to hound.chat
187pub async fn start_register(
188    AxumState(state): AxumState<State>,
189    Json(req): Json<StartRegisterRequest>,
190) -> Result<Json<StartRegisterResponse>, StatusCode> {
191    let user_id = Uuid::new_v4();
192    let (options, reg_state) = state
193        .auth
194        .webauthn
195        .start_passkey_registration(user_id, &req.email, &req.display_name, None)
196        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;
197
198    state
199        .auth
200        .inflights
201        .write()
202        .await
203        .insert(user_id, InFlightRegistration::new(reg_state));
204
205    Ok(Json(StartRegisterResponse { options, user_id }))
206}

220/// Completes the ceremony to register a user to hound.chat
221pub async fn finish_register(
222    AxumState(state): AxumState<State>,
223    Json(req): Json<FinishRegisterRequest>,
224) -> Result<Json<FinishLoginResponse>, StatusCode> {
225    let reg_state = state
226        .auth
227        .inflights
228        .write()
229        .await
230        .remove(&req.user_id)
231        .ok_or(StatusCode::BAD_REQUEST)?;
232
233    let passkey = state
234        .auth
235        .webauthn
236        .finish_passkey_registration(&req.credential, &reg_state.passkey)
237        .map_err(|_| StatusCode::BAD_REQUEST)?;
238    state
239        .auth
240        .credentials
241        .write()
242        .await
243        .insert(req.user_id, passkey);
244
245    let claims = Claims::new(req.user_id, 24);
246    let token = state
247        .create_jwt(&claims)
248        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;
249
250    Ok(Json(FinishLoginResponse {
251        token,
252        user_id: req.user_id,
253        expires_in: claims.exp - claims.iat,
254    }))
255}
