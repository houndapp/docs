11/// Command Line configuration for the backend.
12#[derive(Debug, clap::Parser)]
13#[command(version)]
14#[non_exhaustive]
15pub struct Cli {
16    /// The "Relying Party" ID
17    ///
18    /// WARNING: Changing will break existing passkeys.
19    #[clap(long, env, default_value = "hound.chat")]
20    pub rp_id: String,
21    /// The "Relying Party" origin
22    #[clap(long, env, default_value = "https://hound.chat")]
23    pub rp_origin: url::Url,
24    /// The URL for the database
25    #[clap(long, env, default_value = "sqlite::memory:")]
26    pub database_url: url::Url,
27    /// The host address of the backend API.
28    #[clap(long, env, default_value_t=[0,0,0,0].into())]
29    pub host: std::net::IpAddr,
30    /// The port for the backend API socket.
31    #[clap(long, env, default_value_t = 8000)]
32    pub port: u16,
33    /// Whether the backend will output to stdout/stderr.
34    #[clap(short, long, env, default_value_t = false)]
35    pub quiet: bool,
36    #[clap(
37     long,
38     default_value_t = LogFormats::Auto,
39     value_parser = clap::builder::PossibleValuesParser::new(["auto", "json", "plain", "pretty"])
40     .map(|s| s.parse::<LogFormats>().unwrap()),
41     )]
42    pub log_format: LogFormats,
43
44    /// Commands relating specific domains in hound's backend.
45    #[clap(subcommand)]
46    pub cmd: Option<Commands>,
47}

58#[derive(Clone, Debug, clap::Subcommand)]

63    Refresh,

66impl Cli {
67    pub async fn drive(self) -> eyre::Result<()> {
68        let _guard = self.log_format.init_tracing_subscriber(self.quiet, None);
69        let state = server::State::from_cli(&self).await?;
70        match self.cmd {
71            Some(Commands::Openapi) => println!("{}", state.router().1.to_yaml()?),
72            Some(Commands::Db(DbCommands::Refresh)) => {
73                migration::Migrator::refresh(&state.db).await?
74            }
75            Some(Commands::Db(DbCommands::Fresh)) => migration::Migrator::fresh(&state.db).await?,
76            None => todo!(),
77        }
78
79        Ok(())
80    }
81}

83/// The data format for the log output.
84#[derive(Copy, Clone, PartialEq, Eq, Debug, EnumString, Display, Default)]
85pub enum LogFormats {
86    /// Chooses the format based on whether it is being run in an interactive
87    /// terminal or not. Currently chooses [`Self::Pretty`] if so, and
88    /// [`Self::Json`] otherwise.
89    #[strum(serialize = "auto")]
90    #[default]
91    Auto,
92    #[strum(serialize = "json")]
93    Json,
94    #[strum(serialize = "plain")]
95    Plain,
96    #[strum(serialize = "pretty")]
97    Pretty,
98}
99
100impl LogFormats {
101    /// Creates the tracing subscriber that pulls from the env for filters
102    /// and outputs based on [Self].
103    fn init_tracing_subscriber(
104        self,
105        quiet: bool,
106        file_writer: Option<tracing_appender::non_blocking::NonBlocking>,
107    ) -> tracing::dispatcher::DefaultGuard {
108        use tracing_subscriber::fmt::writer::{BoxMakeWriter, MakeWriterExt};
109
110        let env_filter = tracing_subscriber::EnvFilter::builder()
111            .with_default_directive(tracing_subscriber::filter::LevelFilter::INFO.into())
112            .from_env_lossy();
113
114        let mk_writer: BoxMakeWriter = match file_writer {
115            Some(file_writer) => {
116                if quiet {
117                    BoxMakeWriter::new(file_writer)
118                } else {
119                    BoxMakeWriter::new(std::io::stdout.and(file_writer))
120                }
121            }
122            None => {
123                if quiet {
124                    BoxMakeWriter::new(std::io::sink)
125                } else {
126                    BoxMakeWriter::new(std::io::stdout)
127                }
128            }
129        };
130
131        let subscriber = tracing_subscriber::fmt()
132            .with_file(true)
133            .with_thread_ids(true)
134            .with_env_filter(env_filter)
135            .with_writer(mk_writer);
136
137        tracing::dispatcher::set_default(&match self {
138            LogFormats::Auto => {
139                use std::io::IsTerminal;
140                if !std::io::stdout().is_terminal() {
141                    subscriber.with_ansi(false).json().into()
142                } else {
143                    subscriber.pretty().into()
144                }
145            }
146            LogFormats::Json => subscriber.with_ansi(false).json().into(),
147            LogFormats::Plain => subscriber.into(),
148            LogFormats::Pretty => subscriber.pretty().into(),
149        })
150    }
151}
