253    #[strum(serialize = "up")]
254    Up,

256    #[strum(serialize = "down")]
257    Down,
258    /// Drops all tables in the database and applies all migrations

262    #[strum(serialize = "refresh")]

266impl Cli {
267    pub async fn drive(self) -> eyre::Result<()> {
268        self.log_format.init_tracing_subscriber(self.quiet, None);
269        tracing::debug!(%self.log_format, "cli arguments");
270
271        if let Some(Commands::Openapi) = self.cmd {
272            println!("{}", api::router().get_openapi().to_yaml()?);
273            return Ok(());
274        }
275
276        // DB subcommands only need a database connection, not the full API server.
277        if let Some(Commands::Db(ref db_cmd)) = self.cmd {
278            let db = sea_orm::Database::connect(&self.database_url.to_string()).await?;
279            return match db_cmd {
280                DbCommands::Up => Ok(migration::Migrator::up(&db, None).await?),
281                DbCommands::Down => Ok(migration::Migrator::down(&db, Some(1)).await?),
282                DbCommands::Fresh => Ok(migration::Migrator::fresh(&db).await?),
283                DbCommands::Refresh => Ok(migration::Migrator::refresh(&db).await?),
284            };
285        }
286
287        let (api, server) = api::serve(&self).await?;
288        match self.migrate {
289            None => {}
290            Some(DbCommands::Up) => migration::Migrator::up(&api.db, None).await?,
291            Some(DbCommands::Down) => migration::Migrator::down(&api.db, Some(1)).await?,
292            Some(DbCommands::Fresh) => migration::Migrator::fresh(&api.db).await?,
293            Some(DbCommands::Refresh) => migration::Migrator::refresh(&api.db).await?,
294        }
295        server.await.map_err(From::from)
296    }
297}

299/// The data format for the log output.
300#[derive(Copy, Clone, PartialEq, Eq, Debug, EnumString, Display, Default)]
301pub enum LogFormats {
302    /// Chooses the format based on whether it is being run in an interactive
303    /// terminal or not. Currently chooses [`Self::Pretty`] if so, and
304    /// [`Self::Json`] otherwise.
305    #[strum(serialize = "auto")]
306    #[default]
307    Auto,
308    #[strum(serialize = "json")]
309    Json,
310    #[strum(serialize = "plain")]
311    Plain,
312    #[strum(serialize = "pretty")]
313    Pretty,
314}
315
316impl LogFormats {
317    /// Creates the tracing subscriber that pulls from the env for filters
318    /// and outputs based on [Self].
319    fn init_tracing_subscriber(
320        self,
321        quiet: bool,
322        file_writer: Option<tracing_appender::non_blocking::NonBlocking>,
323    ) {
324        use tracing_subscriber::fmt::writer::{BoxMakeWriter, MakeWriterExt};
325
326        let env_filter = tracing_subscriber::EnvFilter::builder()
327            .with_default_directive(tracing_subscriber::filter::LevelFilter::INFO.into())
328            .from_env_lossy();
329
330        let mk_writer: BoxMakeWriter = match file_writer {
331            Some(file_writer) => {
332                if quiet {
333                    BoxMakeWriter::new(file_writer)
334                } else {
335                    BoxMakeWriter::new(std::io::stdout.and(file_writer))
336                }
337            }
338            None => {
339                if quiet {
340                    BoxMakeWriter::new(std::io::sink)
341                } else {
342                    BoxMakeWriter::new(std::io::stdout)
343                }
344            }
345        };
346
347        tracing_subscriber::fmt()
348            .with_env_filter(env_filter)
349            .with_writer(mk_writer)
350            .pretty()
351            .init()
352    }
