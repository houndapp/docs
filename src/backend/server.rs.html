6/// The primary server backend task. All backend services will be initiated. The
7/// function if successful, returns a tuple of the backend address, and a task
8/// responsible for serving the backend. This task will only return from the
9/// await as a result of a server backend termination (both intentional
10/// and failures).
11pub async fn serve(
12    cli: crate::cli::Cli,
13) -> eyre::Result<(
14    std::net::SocketAddr,
15    impl std::future::Future<Output = eyre::Result<()>>,
16)> {
17    tracing::info!(%cli.host, %cli.port, "binding web server");
18    let listener = tokio::net::TcpListener::bind((cli.host, cli.port))
19        .await
20        .unwrap();
21    let webrtc = webrtc::serve().await?;
22    let (address, trailbase) = trailbase::serve(cli).await?;
23    Ok((address, async move {
24        tokio::select! {
25            result = axum::serve(listener, webrtc) => result.map_err(From::from),
26            result = trailbase => result,
27        }
28    }))
29}
