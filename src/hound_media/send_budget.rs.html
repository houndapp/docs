1//! # Connection-level coordinated send budget
2//!
3//! Sans-IO pieces for the **shared-uplink** problem the MoQ multiplex creates: all
4//! of a client's video tracks (camera + screen share) ride ONE QUIC connection, but
5//! each track runs its own per-track sender bitrate controller that sees only its
6//! receivers' loss/RTT — never the connection's own uplink saturation. Under a
7//! heavy screen share the sum of the per-track targets oversubscribes the wire,
8//! the shared publish channels drop, bufferbloat builds, and the client's OWN
9//! downlink receive starves. This module is the missing coordinator's pure core:
10//!
11//! * [`SendBudget`] — an AIMD estimate of the connection's achievable TOTAL send
12//!   rate, driven by aggregate publish-drop backpressure (the one connection-level
13//!   congestion signal a sender can see without new transport surface). It decreases
14//!   multiplicatively when the connection dropped publish AUs this interval and
15//!   increases additively when it didn't, converging toward the wire's capacity.
16//! * [`allocate_send_budget`] — a max-min-fair split of a video budget across the
17//!   active video tracks by their desired (uncapped) rates, with a per-track floor
18//!   so a greedy screen share can't starve the camera below a usable rate.
19//!
20//! Both are deterministic: time enters only as an explicit `now_us` argument and
21//! there is no I/O, clock read, or hidden state — so the convergence and the
22//! allocation are fully unit-testable. The IO shell (the session coordinator) feeds
23//! the drop signal in and pushes the resulting caps onto each track's adaptation
24//! seam; nothing here touches a socket or a thread.
25
26/// Multiplicative decrease factor applied to the total budget when the connection
27/// dropped publish AUs this interval. 0.85 ≈ a 15% back-off per congested second,
28/// matching the well-known AIMD decrease band: gentle enough that one stray drop
29/// (gated by [`DECREASE_COOLDOWN_US`]) doesn't collapse the estimate, fast enough
30/// that sustained over-capacity converges toward the wire in a handful of seconds.
31const DECREASE_FACTOR: f64 = 0.85;
32
33/// Additive increase per clean (no-drop) control interval. 500 kbps/s is a steady
34/// probe upward that recovers the full budget within a few seconds after the
35/// congestion clears, without overshooting hard enough to immediately re-congest.
36const INCREASE_BPS: u64 = 500_000;
37
38/// Minimum spacing between multiplicative decreases. One isolated drop (a transient
39/// scheduling hiccup, a single late AU) must NOT collapse the budget; only
40/// *sustained* over-capacity — a drop in successive intervals past this cooldown —
41/// walks it down (≈ ×0.85 per second toward capacity). Increases are not cooled
42/// down (they are already gentle and clamped).
43const DECREASE_COOLDOWN_US: i64 = 1_000_000;
44
45/// AIMD estimate of the connection's achievable TOTAL send rate (bits/sec), driven
46/// by aggregate publish-drop backpressure. One per shared connection; the
47/// coordinator calls [`observe`](SendBudget::observe) once per control interval
48/// (~1 Hz) with whether the connection over-produced (dropped publish AUs) this
49/// interval, and feeds the returned total into [`allocate_send_budget`].
50///
51/// Pure and deterministic: `now_us` is an explicit monotonic-microsecond argument;
52/// there is no clock read or hidden state, so convergence is reproducible in tests.
53#[derive(Debug, Clone)]
54pub struct SendBudget {
55    total_bps: u64,
56    min_bps: u64,
57    max_bps: u64,
58    /// Monotonic-µs timestamp of the last multiplicative decrease, for the cooldown
59    /// gate. `None` until the first decrease (so the first drop always applies).
60    last_decrease_us: Option<i64>,
61}
62
63impl SendBudget {
64    /// A fresh budget seeded at `initial_bps`, clamped to `[min_bps, max_bps]`. At
65    /// the session call site the suggested defaults are `initial = max = 12 Mbps`
66    /// (start optimistic — the AIMD only ever clamps DOWN from there toward the wire)
67    /// and `min = 600 kbps` (enough to keep one low video rung alive).
68    pub fn new(initial_bps: u64, min_bps: u64, max_bps: u64) -> Self {
69        // Tolerate a swapped/degenerate min/max by ordering them; an empty range
70        // collapses to the single value rather than panicking.
71        let (lo, hi) = if min_bps <= max_bps {
72            (min_bps, max_bps)
73        } else {
74            (max_bps, min_bps)
75        };
76        Self {
77            total_bps: initial_bps.clamp(lo, hi),
78            min_bps: lo,
79            max_bps: hi,
80            last_decrease_us: None,
81        }
82    }
83
84    /// Advance the AIMD one control interval and return the new total budget.
85    ///
86    /// `over_capacity` is `true` when the connection dropped publish AUs this
87    /// interval (we produced faster than the wire drained). On `true`, decrease
88    /// multiplicatively (×[`DECREASE_FACTOR`]) — but only if at least
89    /// [`DECREASE_COOLDOWN_US`] has elapsed since the last decrease, so one stray
90    /// drop can't collapse the estimate while sustained drops still converge. On
91    /// `false`, increase additively ([`INCREASE_BPS`]). Always clamped to
92    /// `[min_bps, max_bps]`.
93    pub fn observe(&mut self, over_capacity: bool, now_us: i64) -> u64 {
94        if over_capacity {
95            let cooled = match self.last_decrease_us {
96                None => true,
97                Some(prev) => now_us.saturating_sub(prev) >= DECREASE_COOLDOWN_US,
98            };
99            if cooled {
100                // `* 0.85` on a u64 budget: f64 has 52 bits of mantissa, far more
101                // than any sane bitrate needs, so this is exact enough; round to
102                // nearest and re-clamp.
103                let decreased = (self.total_bps as f64 * DECREASE_FACTOR).round() as u64;
104                self.total_bps = decreased.clamp(self.min_bps, self.max_bps);
105                self.last_decrease_us = Some(now_us);
106            }
107        } else {
108            self.total_bps = self
109                .total_bps
110                .saturating_add(INCREASE_BPS)
111                .clamp(self.min_bps, self.max_bps);
112        }
113        self.total_bps
114    }
115
116    /// The current total budget estimate (bits/sec).
117    pub fn total_bps(&self) -> u64 {
118        self.total_bps
119    }
120}

122/// The per-track floor: an active video track is never capped below this, so a
123/// greedy screen share can't starve the camera to a black/frozen tile. 200 kbps is
124/// ≈ the lowest usable video rung (180p15-class); below it, video isn't worth
125/// sending and the track's own adaptation should have paused to AudioOnly.
126pub const MIN_VIDEO_BPS: u64 = 200_000;
127
128/// Split a video budget across the active video tracks, max-min-fair with a floor.
129///
130/// `video_budget_bps` is the connection total minus the audio reserve.
131/// `desired` is each active video track's current uncapped loss/RTT target
132/// (`0` = the track isn't producing yet / hasn't decided). The returned vector is
133/// the per-track cap, in the SAME order as `desired`. Contract:
134///
135/// * If `sum(desired) <= video_budget` OR `sum == 0`: return `video_budget` for
136///   EACH track — a generous ceiling, so no track is artificially limited and its
137///   OWN adaptation governs. (The coordinator only bites when the aggregate
138///   genuinely exceeds the budget.)
139/// * Else: scale each track's cap proportionally to its desire, but never below
140///   [`MIN_VIDEO_BPS`]. If the floors alone can't all fit in `video_budget`, fall
141///   back to an equal split (`video_budget / n`) so the outcome is still total and
142///   deterministic rather than handing out floors that oversubscribe.
143///
144/// Pure: no time, no state. The sum of the returned caps is `≈ video_budget` in the
145/// over-budget case (within rounding), so the coordinator hands out exactly the
146/// estimated wire, no more.
147pub fn allocate_send_budget(video_budget_bps: u64, desired: &[u64]) -> Vec<u64> {
148    let n = desired.len();
149    if n == 0 {
150        return Vec::new();
151    }
152
153    let sum: u128 = desired.iter().map(|&d| d as u128).sum();
154
155    // Headroom (or nobody's producing yet): everyone gets the full ceiling; each
156    // track's own adaptation governs within it.
157    if sum == 0 || sum <= video_budget_bps as u128 {
158        return vec![video_budget_bps; n];
159    }
160
161    // Over budget. If even the per-track floors don't all fit, an equal split is the
162    // only total, fair outcome (proportional-with-floor would oversubscribe).
163    let n_u64 = n as u64;
164    if MIN_VIDEO_BPS.saturating_mul(n_u64) > video_budget_bps {
165        let share = video_budget_bps / n_u64;
166        return vec![share; n];
167    }
168
169    // Proportional split, floored at MIN_VIDEO_BPS. Tracks that the proportional
170    // share would push below the floor are pinned to the floor; the remaining budget
171    // is split proportionally among the rest. One pass suffices because the floors
172    // are guaranteed to fit (checked above) and pinning only ever removes demand.
173    //
174    // First pass: who falls below the floor at a naive proportional share?
175    let budget = video_budget_bps as u128;
176    let mut caps = vec![0u64; n];
177    let mut floored = vec![false; n];
178    let mut remaining_budget = budget;
179    let mut remaining_sum = sum;
180
181    // Pin floors iteratively: pinning a track to the floor reduces the budget and
182    // the desire-sum available to the rest, which can push further tracks below the
183    // floor, so repeat until no new track is floored.
184    loop {
185        let mut newly_floored = false;
186        for i in 0..n {
187            if floored[i] {
188                continue;
189            }
190            // This track's proportional share of the budget still available to the
191            // un-floored tracks.
192            let share = if remaining_sum == 0 {
193                0
194            } else {
195                (desired[i] as u128 * remaining_budget / remaining_sum) as u64
196            };
197            if share < MIN_VIDEO_BPS {
198                caps[i] = MIN_VIDEO_BPS;
199                floored[i] = true;
200                remaining_budget = remaining_budget.saturating_sub(MIN_VIDEO_BPS as u128);
201                remaining_sum = remaining_sum.saturating_sub(desired[i] as u128);
202                newly_floored = true;
203            }
204        }
205        if !newly_floored {
206            break;
207        }
208    }
209
210    // Second pass: the un-floored tracks split the remaining budget proportionally.
211    for i in 0..n {
212        if floored[i] {
213            continue;
214        }
215        let share = if remaining_sum == 0 {
216            remaining_budget as u64
217        } else {
218            (desired[i] as u128 * remaining_budget / remaining_sum) as u64
219        };
220        // A track that survived the floor pass is by construction >= MIN_VIDEO_BPS,
221        // but clamp defensively against rounding.
222        caps[i] = share.max(MIN_VIDEO_BPS);
223    }
224
225    caps
226}

234    /// A sustained over-capacity signal walks the budget down ≈×0.85 per second
235    /// (one decrease per cooldown window), converging toward the wire — but a single
236    /// isolated drop inside the cooldown does NOT collapse it.
237    #[test]
238    fn decrease_is_multiplicative_and_cooldown_gated() {
239        let mut b = SendBudget::new(12_000_000, 600_000, 12_000_000);
240
241        // First over-capacity decreases immediately (no prior decrease).
242        let t0 = 0;
243        let after_first = b.observe(true, t0);
244        assert_eq!(after_first, (12_000_000.0 * 0.85f64).round() as u64);
245
246        // A second drop only 100 ms later is INSIDE the cooldown → no change.
247        let held = b.observe(true, t0 + 100_000);
248        assert_eq!(
249            held, after_first,
250            "a drop inside the cooldown must not decrease"
251        );
252
253        // Past the cooldown, sustained drops decrease again (×0.85 of the held value).
254        let after_second = b.observe(true, t0 + DECREASE_COOLDOWN_US);
255        assert_eq!(after_second, (after_first as f64 * 0.85).round() as u64);
256        assert!(
257            after_second < after_first,
258            "sustained over-capacity converges down"
259        );
260    }
261
262    /// Clean intervals recover the budget additively (+500 kbps each), clamped to
263    /// max — the converse of the decrease, so the budget climbs back after the
264    /// congestion clears.
265    #[test]
266    fn increase_is_additive_and_clamped_to_max() {
267        let mut b = SendBudget::new(1_000_000, 600_000, 2_000_000);
268        assert_eq!(b.observe(false, 0), 1_500_000);
269        assert_eq!(b.observe(false, 1_000_000), 2_000_000); // +500k → hits max
270        assert_eq!(b.observe(false, 2_000_000), 2_000_000, "clamped at max");
271        assert_eq!(b.total_bps(), 2_000_000);
272    }
273
274    /// The budget never falls below `min_bps` no matter how many sustained drops
275    /// land (the video floor stays alive).
276    #[test]
277    fn decrease_clamps_to_min() {
278        let mut b = SendBudget::new(1_000_000, 600_000, 12_000_000);
279        let mut now = 0;
280        for _ in 0..50 {
281            b.observe(true, now);
282            now += DECREASE_COOLDOWN_US;
283        }
284        assert_eq!(b.total_bps(), 600_000, "must not fall below min_bps");
285    }

294        assert_eq!(
295            SendBudget::new(99_000_000, 600_000, 12_000_000).total_bps(),

299        let b = SendBudget::new(5_000_000, 12_000_000, 600_000);

328    /// Over budget → proportional split that sums to ≈ the budget (within rounding),
329    /// preserving the desire ratio.
330    #[test]
331    fn allocate_over_budget_is_proportional() {
332        // cam wants 2 Mbps, screen wants 6 Mbps (1:3), budget only 4 Mbps.
333        let budget = 4_000_000;
334        let caps = allocate_send_budget(budget, &[2_000_000, 6_000_000]);
335        let sum: u64 = caps.iter().sum();
336        assert!(
337            (sum as i64 - budget as i64).abs() <= 2,
338            "proportional caps must sum to ≈ budget, got {sum}"
339        );
340        // The 1:3 ratio is preserved (both well above the floor here).
341        assert!(
342            caps[1] > caps[0],
343            "the greedier track gets the larger share"
344        );
345        let ratio = caps[1] as f64 / caps[0] as f64;
346        assert!((ratio - 3.0).abs() < 0.05, "ratio ≈ 3:1, got {ratio}");
347    }

349    /// Floor protection: a greedy screen share cannot starve the camera below
350    /// `MIN_VIDEO_BPS`. The camera is pinned to the floor and the screen takes the
351    /// rest (still summing to the budget).
352    #[test]
353    fn allocate_floor_protects_the_small_track() {
354        // cam wants 300 kbps, screen wants 20 Mbps; budget 2 Mbps. A pure
355        // proportional split would give cam ~30 kbps — below the floor.
356        let budget = 2_000_000;
357        let caps = allocate_send_budget(budget, &[300_000, 20_000_000]);
358        assert_eq!(
359            caps[0], MIN_VIDEO_BPS,
360            "the small track is pinned to the floor"
361        );
362        assert_eq!(
363            caps[1],
364            budget - MIN_VIDEO_BPS,
365            "the greedy track takes the rest"
366        );
367        let sum: u64 = caps.iter().sum();
368        assert!(
369            (sum as i64 - budget as i64).abs() <= 2,
370            "caps still sum to budget"
371        );
372    }

374    /// When the per-track floors can't all fit, fall back to an equal split — a
375    /// total, fair outcome rather than handing out floors that oversubscribe.
376    #[test]
377    fn allocate_floors_dont_fit_falls_back_to_equal_split() {
378        // 3 tracks, floor 200k each = 600k needed, but budget is only 300k.
379        let budget = 300_000;
380        let caps = allocate_send_budget(budget, &[1_000_000, 1_000_000, 1_000_000]);
381        assert_eq!(caps, vec![100_000, 100_000, 100_000]);
382        let sum: u64 = caps.iter().sum();
383        assert!(sum <= budget, "equal split never oversubscribes");
384    }
