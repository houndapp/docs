1//! # Delay-based bandwidth estimation (sans-IO)
2//!
3//! Loss-based control (the [`crate::congestion::RateController`] AIMD) only reacts
4//! once a queue has already overflowed and packets are being dropped — by then the
5//! call has already stuttered. A bottleneck queue announces itself *earlier*: as it
6//! fills, each packet waits a little longer than the last, so the **one-way delay
7//! trends upward** while loss is still zero. Sensing that trend lets a receiver back
8//! its quality request off before the link actually drops frames.
9//!
10//! This is the delay-based half of Google Congestion Control (GCC), distilled to the
11//! pieces that matter here and kept **sans-IO** (timestamps in, a usage signal +
12//! estimate out — no clock, no sockets, fully unit-tested):
13//!
14//! * [`TrendlineEstimator`] folds each frame's inter-arrival *delay variation*
15//!   `d = (recv_i − recv_{i−1}) − (send_i − send_{i−1})` into a smoothed accumulated
16//!   delay and fits a least-squares **slope** over a sliding window. A positive slope
17//!   means delay is growing — a queue is building.
18//! * an **adaptive threshold** `gamma` that tracks the trend's own scale, so the
19//!   detector neither false-positives on a quiet link nor goes numb on a noisy one,
20//!   plus an overuse-duration gate so a single spike isn't called congestion.
21//! * [`DelayBasedBwe`] runs AIMD over that [`Usage`] signal: multiplicative-decrease
22//!   on `Overuse`, additive-increase on `Normal`, hold on `Underuse` (a queue is
23//!   draining — don't push more in). Its estimate is combined with the loss-based one
24//!   by taking the **minimum** (whichever controller is more worried wins), exactly
25//!   as GCC does.
26//!
27//! Clocks need not be synchronised between sender and receiver: only *differences*
28//! of timestamps are used, so any fixed offset cancels (same property the
29//! [`crate::jitter::JitterEstimator`] relies on).
30
31/// Whether the bottleneck queue is filling, draining, or steady — the overuse
32/// detector's verdict for one observation.
33#[derive(Debug, Clone, Copy, PartialEq, Eq)]
34pub enum Usage {
35    /// Delay is steady: the link is keeping up. Safe to probe for more bandwidth.
36    Normal,
37    /// Delay is trending up: a queue is building. Back off before loss appears.
38    Overuse,
39    /// Delay is trending down: a queue is draining. Hold — don't refill it.
40    Underuse,
41}
42
43/// Fits the trend (slope) of accumulated one-way delay over a sliding window and
44/// classifies it against an adaptive threshold into a [`Usage`].
45///
46/// Pure/sans-IO: feed [`Self::observe`] each frame's `(recv_us, send_us)`; read
47/// [`Self::usage`].
48#[derive(Debug)]
49pub struct TrendlineEstimator {
50    prev_recv_us: Option<i64>,
51    prev_send_us: Option<i64>,
52    /// Running sum of delay variations (µs) — the raw accumulated delay.
53    accumulated_delay_us: f64,
54    /// EWMA-smoothed accumulated delay; the regression is fit over this.
55    smoothed_delay_us: f64,
56    /// Sliding window of `(elapsed_us_since_first_sample, smoothed_delay_us)`.
57    window: std::collections::VecDeque<(f64, f64)>,
58    window_size: usize,
59    first_recv_us: Option<i64>,
60    /// Adaptive overuse threshold (unitless, compared against the modified trend).
61    gamma: f64,
62    /// Consecutive time (µs) the trend has been over threshold (overuse-duration gate).
63    overuse_us: f64,
64    /// Consecutive over-threshold observations (overuse-count gate). Together with
65    /// `overuse_us` this debounces a one- or two-frame jitter spike, which at a 30 fps
66    /// cadence would otherwise satisfy the time gate alone on its very first frame.
67    overuse_count: u32,
68    prev_trend: f64,
69    last_update_us: Option<i64>,
70    /// Total deltas folded (the modified trend scales by `min(this, MIN_DELTAS)` so a
71    /// short window still yields a comparable magnitude — WebRTC's kMinNumDeltas).
72    num_deltas: usize,
73    /// The latest verdict from [`Self::classify`].
74    state: Usage,
75}
76
77impl Default for TrendlineEstimator {
78    fn default() -> Self {
79        Self::new()
80    }
81}
82
83impl TrendlineEstimator {
84    // Tuned after WebRTC's TrendlineEstimator defaults, adapted to per-frame cadence.
85    const SMOOTHING: f64 = 0.9;
86    const TREND_GAIN: f64 = 4.0;
87    const WINDOW: usize = 20;
88    /// The modified trend scales by `min(num_deltas, MIN_DELTAS)`, so the magnitude is
89    /// cadence-independent once enough deltas have accrued (WebRTC's kMinNumDeltas).
90    const MIN_DELTAS: usize = 60;
91    /// Trend must stay over threshold this long before `Overuse` fires (debounce).
92    const OVERUSE_TIME_US: f64 = 10_000.0;
93    /// …AND over threshold for at least this many consecutive observations. WebRTC
94    /// uses `> 1`; video frames are far coarser than its packet-group cadence, so a
95    /// slightly larger count keeps a brief jitter burst from reading as congestion.
96    const OVERUSE_COUNT: u32 = 2;
97    /// Initial/clamped adaptive-threshold bounds.
98    const GAMMA_INIT: f64 = 12.5;
99    const GAMMA_MIN: f64 = 6.0;
100    const GAMMA_MAX: f64 = 600.0;
101    /// How fast the threshold adapts toward the observed trend magnitude.
102    const K_UP: f64 = 0.0087;
103    const K_DOWN: f64 = 0.039;
104
105    pub fn new() -> Self {
106        Self {
107            prev_recv_us: None,
108            prev_send_us: None,
109            accumulated_delay_us: 0.0,
110            smoothed_delay_us: 0.0,
111            window: std::collections::VecDeque::with_capacity(Self::WINDOW),
112            window_size: Self::WINDOW,
113            first_recv_us: None,
114            gamma: Self::GAMMA_INIT,
115            overuse_us: 0.0,
116            overuse_count: 0,
117            prev_trend: 0.0,
118            last_update_us: None,
119            num_deltas: 0,
120            state: Usage::Normal,
121        }
122    }
123
124    /// Fold one frame: `recv_us` local arrival, `send_us` the sender's capture
125    /// timestamp. Updates the trend and the [`Self::usage`] verdict.
126    pub fn observe(&mut self, recv_us: i64, send_us: i64) {
127        if let (Some(pr), Some(ps)) = (self.prev_recv_us, self.prev_send_us) {
128            let d = (recv_us - pr) as f64 - (send_us - ps) as f64;
129            self.num_deltas += 1;
130            self.accumulated_delay_us += d;
131            self.smoothed_delay_us = Self::SMOOTHING * self.smoothed_delay_us
132                + (1.0 - Self::SMOOTHING) * self.accumulated_delay_us;
133            let first = *self.first_recv_us.get_or_insert(recv_us);
134            let t = (recv_us - first) as f64;
135            self.window.push_back((t, self.smoothed_delay_us));
136            while self.window.len() > self.window_size {
137                self.window.pop_front();
138            }
139            self.classify(recv_us);
140        }
141        self.prev_recv_us = Some(recv_us);
142        self.prev_send_us = Some(send_us);
143    }
144
145    /// Least-squares slope of smoothed delay vs. time over the window, scaled into the
146    /// "modified trend" WebRTC compares against the threshold.
147    fn modified_trend(&self) -> f64 {
148        let n = self.window.len();
149        if n < self.window_size {
150            return 0.0; // not enough samples for a stable fit
151        }
152        let nf = n as f64;
153        let (mut sx, mut sy, mut sxx, mut sxy) = (0.0, 0.0, 0.0, 0.0);
154        for &(x, y) in &self.window {
155            sx += x;
156            sy += y;
157            sxx += x * x;
158            sxy += x * y;
159        }
160        let denom = nf * sxx - sx * sx;
161        if denom.abs() < f64::EPSILON {
162            return 0.0;
163        }
164        let slope = (nf * sxy - sx * sy) / denom;
165        // Scale by the delta count (capped) and gain so the threshold magnitude is
166        // cadence-independent — matching WebRTC's `min(num_deltas, kMinNumDeltas)`.
167        let scale = self.num_deltas.min(Self::MIN_DELTAS) as f64;
168        slope * scale * Self::TREND_GAIN
169    }
170
171    /// The detector state machine: compare the modified trend to the adaptive
172    /// threshold (with an overuse-duration gate), then adapt the threshold.
173    fn classify(&mut self, now_us: i64) {
174        let trend = self.modified_trend();
175        let dt_us = self
176            .last_update_us
177            .map(|p| (now_us - p) as f64)
178            .unwrap_or(0.0)
179            .clamp(0.0, 100_000.0);
180        self.last_update_us = Some(now_us);
181
182        if trend > self.gamma {
183            self.overuse_us += dt_us;
184            self.overuse_count += 1;
185            // Sustained (time AND count) AND still rising → overuse. Resetting only
186            // the gates (not the state) means a persistent queue holds `Overuse`
187            // between re-trips, so the rate controller — not this detector — owns how
188            // often to actually back off (see `DelayBasedBwe`).
189            if self.overuse_us > Self::OVERUSE_TIME_US
190                && self.overuse_count > Self::OVERUSE_COUNT
191                && trend >= self.prev_trend
192            {
193                self.overuse_us = 0.0;
194                self.overuse_count = 0;
195                self.state = Usage::Overuse;
196            }
197        } else if trend < -self.gamma {
198            self.overuse_us = 0.0;
199            self.overuse_count = 0;
200            self.state = Usage::Underuse;
201        } else {
202            self.overuse_us = 0.0;
203            self.overuse_count = 0;
204            self.state = Usage::Normal;
205        }
206        self.prev_trend = trend;
207
208        // Adapt the threshold toward the trend magnitude: rise quickly under load
209        // (K_UP when |trend| > gamma) and fall slowly when quiet, clamped.
210        let k = if trend.abs() > self.gamma {
211            Self::K_UP
212        } else {
213            Self::K_DOWN
214        };
215        if dt_us > 0.0 {
216            self.gamma += k * (trend.abs() - self.gamma) * (dt_us / 1000.0);
217            self.gamma = self.gamma.clamp(Self::GAMMA_MIN, Self::GAMMA_MAX);
218        }
219    }
220
221    /// The current congestion verdict.
222    pub fn usage(&self) -> Usage {
223        self.state
224    }
225
226    /// The smoothed accumulated-delay slope scaled as compared to the threshold —
227    /// for stats/debug.
228    pub fn trend(&self) -> f64 {
229        self.modified_trend()
230    }
231}

233/// Delay-based AIMD bandwidth estimate driven by a [`TrendlineEstimator`]'s [`Usage`].
234///
235/// Multiplicative-decrease on `Overuse` (a queue is forming — shed load fast),
236/// additive-increase on `Normal` (probe for more), hold on `Underuse` (a queue is
237/// draining; adding load would refill it). The estimate is meant to be combined with
238/// the loss-based estimate by taking the minimum — whichever controller is more
239/// worried governs.
240#[derive(Debug)]
241pub struct DelayBasedBwe {
242    trend: TrendlineEstimator,
243    estimate_bps: f64,
244    max_bps: f64,
245    min_bps: f64,
246    last_update_us: Option<i64>,
247    /// When the estimate last took an overuse decrease — drives the decrease cooldown.
248    last_decrease_us: Option<i64>,
249}
250
251impl DelayBasedBwe {
252    /// Decrease factor on overuse (shed 15%).
253    const DECREASE: f64 = 0.85;
254    /// Additive increase per second while Normal (bits/sec added per second).
255    const INCREASE_BPS_PER_S: f64 = 0.08; // ×current/sec, i.e. +8%/s additive-ish
256    const MIN_BPS: f64 = 50_000.0;
257    /// Minimum spacing between overuse decreases. A persistent queue holds `Overuse`
258    /// for many frames; without this the estimate would shed 15% PER FRAME and slam
259    /// into `MIN_BPS` in well under a second. One decrease per ~RTT-ish window lets
260    /// the loss-based half and the receiver react before the next cut — standard
261    /// AIMD "decrease then hold".
262    const DECREASE_COOLDOWN_US: i64 = 1_000_000;
263
264    pub fn new(start_bps: u64, max_bps: u64) -> Self {
265        Self {
266            trend: TrendlineEstimator::new(),
267            estimate_bps: start_bps as f64,
268            max_bps: max_bps as f64,
269            min_bps: Self::MIN_BPS,
270            last_update_us: None,
271            last_decrease_us: None,
272        }
273    }
274
275    /// Fold one frame's `(recv_us, send_us)` and step the estimate per the verdict.
276    pub fn observe(&mut self, recv_us: i64, send_us: i64) -> u64 {
277        self.trend.observe(recv_us, send_us);
278        let dt_s = self
279            .last_update_us
280            .map(|p| (recv_us - p) as f64 / 1_000_000.0)
281            .unwrap_or(0.0)
282            .clamp(0.0, 1.0);
283        self.last_update_us = Some(recv_us);
284
285        match self.trend.usage() {
286            Usage::Overuse => {
287                // Decrease at most once per cooldown so a sustained queue can't
288                // ratchet the estimate to the floor frame-by-frame.
289                let cooled = self
290                    .last_decrease_us
291                    .is_none_or(|t| recv_us - t >= Self::DECREASE_COOLDOWN_US);
292                if cooled {
293                    self.estimate_bps *= Self::DECREASE;
294                    self.last_decrease_us = Some(recv_us);
295                }
296            }
297            Usage::Normal => {
298                self.estimate_bps *= 1.0 + Self::INCREASE_BPS_PER_S * dt_s;
299            }
300            Usage::Underuse => { /* hold: let the queue drain */ }
301        }
302        self.estimate_bps = self.estimate_bps.clamp(self.min_bps, self.max_bps);
303        self.estimate_bps as u64
304    }
305
306    /// The current delay-based estimate (bits/sec).
307    pub fn estimate_bps(&self) -> u64 {
308        self.estimate_bps as u64
309    }
310
311    /// The current congestion verdict.
312    pub fn usage(&self) -> Usage {
313        self.trend.usage()
314    }
315
316    /// Move the ceiling when the offered layer set changes.
317    pub fn set_max_bps(&mut self, max_bps: u64) {
318        self.max_bps = max_bps as f64;
319        if self.estimate_bps > self.max_bps {
320            self.estimate_bps = self.max_bps;
321        }
322    }
323}

325#[cfg(test)]
326mod tests {
327    use super::*;

329    const FRAME_US: i64 = 33_333; // ~30 fps cadence
330
331    /// A link with constant propagation delay (arrival cadence == capture cadence)
332    /// never trends up → the detector stays `Normal`.
333    #[test]
334    fn steady_link_is_normal() {
335        let mut t = TrendlineEstimator::new();
336        for i in 0..100i64 {
337            let send = i * FRAME_US;
338            let recv = send + 5_000; // constant 5 ms transit
339            t.observe(recv, send);
340        }
341        assert_eq!(t.usage(), Usage::Normal);
342    }
343
344    /// A bottleneck queue filling — each frame's transit grows a little — drives the
345    /// trend over threshold and the detector into `Overuse` BEFORE any loss.
346    #[test]
347    fn building_queue_trips_overuse() {
348        let mut t = TrendlineEstimator::new();
349        let mut extra = 0i64;
350        let mut saw_overuse = false;
351        for i in 0..200i64 {
352            let send = i * FRAME_US;
353            // Transit grows 1.5 ms each frame: a steadily filling queue.
354            extra += 3_000;
355            let recv = send + 5_000 + extra;
356            t.observe(recv, send);
357            if t.usage() == Usage::Overuse {
358                saw_overuse = true;
359                break;
360            }
361        }
362        assert!(saw_overuse, "a steadily filling queue must trip overuse");
363    }
364
365    /// `DelayBasedBwe` decreases its estimate when the queue builds and climbs back
366    /// on a steady link.
367    #[test]
368    fn delay_bwe_backs_off_then_recovers() {
369        let mut bwe = DelayBasedBwe::new(2_000_000, 4_000_000);
370        let start = bwe.estimate_bps();
371
372        // Phase 1: filling queue → expect a decrease below the start estimate.
373        let mut extra = 0i64;
374        for i in 0..200i64 {
375            let send = i * FRAME_US;
376            extra += 3_000;
377            bwe.observe(send + 5_000 + extra, send);
378        }
379        let backed_off = bwe.estimate_bps();
380        assert!(
381            backed_off < start,
382            "delay-based estimate must drop under a building queue: {backed_off} !< {start}"
383        );
384
385        // Phase 2: steady link for a while → estimate climbs back up.
386        let mut t = 200 * FRAME_US;
387        for _ in 0..400 {
388            bwe.observe(t + 5_000, t);
389            t += FRAME_US;
390        }
391        assert!(
392            bwe.estimate_bps() > backed_off,
393            "a recovered link must let the estimate climb again"
394        );
395    }
396
397    /// The estimate never exceeds the configured ceiling.
398    #[test]
399    fn delay_bwe_respects_ceiling() {
400        let mut bwe = DelayBasedBwe::new(900_000, 1_000_000);
401        let mut t = 0i64;
402        for _ in 0..10_000 {
403            bwe.observe(t + 5_000, t); // steady → always climbing
404            t += FRAME_US;
405        }
406        assert!(bwe.estimate_bps() <= 1_000_000, "must clamp at the ceiling");
407    }

409    /// An ISOLATED jitter spike (one late frame) must NOT read as
410    /// congestion. The overuse-count gate debounces it, so the estimate never backs
411    /// off — it only ever climbed on the surrounding steady frames. (Without the
412    /// count gate, a single spike could satisfy the time threshold on its first
413    /// frame at 30 fps.)
414    #[test]
415    fn isolated_spike_does_not_trip_overuse() {
416        let mut bwe = DelayBasedBwe::new(2_000_000, 4_000_000);
417        let mut t = 0i64;
418        for _ in 0..40 {
419            bwe.observe(t + 5_000, t); // warm the regression window on a clean link
420            t += FRAME_US;
421        }
422        let before = bwe.estimate_bps();
423        bwe.observe(t + 80_000, t); // one ~75 ms transit spike
424        t += FRAME_US;
425        for _ in 0..5 {
426            bwe.observe(t + 5_000, t); // link is steady again
427            t += FRAME_US;
428        }
429        assert!(
430            bwe.estimate_bps() >= before,
431            "an isolated spike must not back the estimate off: {} < {before}",
432            bwe.estimate_bps()
433        );
434    }

436    /// Two arrivals with IDENTICAL timestamps (a zero-duration window: `recv_us`
437    /// equal across frames, e.g. a burst delivered in one poll) must not panic —
438    /// the least-squares fit guards its denominator (`denom.abs() < EPSILON → 0`)
439    /// and the `dt`/`gamma` updates are gated on `dt_us > 0`. The estimate is left
440    /// untouched (no time elapsed → no additive increase) and stays Normal.
441    #[test]
442    fn zero_duration_window_does_not_panic() {
443        let mut bwe = DelayBasedBwe::new(2_000_000, 4_000_000);
444        let start = bwe.estimate_bps();
445        // Many observations all stamped at the SAME recv/send instant.
446        for _ in 0..50 {
447            bwe.observe(1_000, 1_000);
448        }
449        assert_eq!(bwe.usage(), Usage::Normal, "no delay change → Normal");
450        assert_eq!(
451            bwe.estimate_bps(),
452            start,
453            "zero elapsed time contributes no increase and no decrease"
454        );
455    }
456
457    /// A delay trend that sits right around the threshold without sustaining the
458    /// overuse-duration AND overuse-count gates does NOT fire `Overuse` (the
459    /// comparison is strict `trend > gamma`, and a single over-threshold sample is
460    /// debounced). A constant-transit link — every frame the exact same delay — has
461    /// zero trend, which is below `gamma` by definition, so it never trips. Pins the
462    /// at/below-threshold side of the detector deterministically (an exact
463    /// `trend == gamma` float is not constructible).
464    #[test]
465    fn constant_delay_never_trips_overuse() {
466        let mut t = TrendlineEstimator::new();
467        for i in 0..200i64 {
468            let send = i * FRAME_US;
469            // Constant 5 ms transit on every frame: inter-arrival delay variation is
470            // exactly zero, so the modified trend is 0 ≤ gamma → never Overuse.
471            t.observe(send + 5_000, send);
472            assert_ne!(
473                t.usage(),
474                Usage::Overuse,
475                "a zero-trend (at/below threshold) link must never read as overuse"
476            );
477        }
478    }
479
480    /// Under a SUSTAINED queue the estimate backs off, but the decrease is
481    /// rate-limited (≈ one cut per cooldown), so it can't ratchet to the floor
482    /// frame-by-frame: ~5 s of continuous overuse is a handful of cuts, NOT 150
483    /// (an unthrottled per-frame decrease would slam straight to `MIN_BPS`).
484    #[test]
485    fn sustained_overuse_decrease_is_rate_limited() {
486        let mut bwe = DelayBasedBwe::new(2_000_000, 4_000_000);
487        let mut extra = 0i64;
488        let mut t = 0i64;
489        for _ in 0..150 {
490            // ~5 s of a steadily filling queue
491            extra += 5_000;
492            bwe.observe(t + 5_000 + extra, t);
493            t += FRAME_US;
494        }
495        let est = bwe.estimate_bps();
496        assert!(
497            est < 2_000_000,
498            "a real sustained queue must still back off: {est}"
499        );
500        assert!(
501            est > 300_000,
502            "rate-limited decrease must stay well above the {} floor (per-frame would \
503             have slammed there): {est}",
504            DelayBasedBwe::MIN_BPS as u64
505        );
506    }
