1//! Sans-IO frame-rate downsampler.
2//!
3//! The quality ladder picks a target fps per rung, but capture runs at a fixed
4//! (higher) rate. This gate decides, per captured frame, whether to *emit* it so
5//! the encoded stream matches the target — making the ladder's fps real (a
6//! 60→30→15 rung change actually changes the frames sent, and a user fps cap is
7//! honoured) without re-opening the capture device.
8//!
9//! **Downsample only:** a target ≥ the capture rate emits every frame. The count
10//! is self-correcting (an integer µs accumulator carries the remainder), so over
11//! any window it emits ≈ `target_fps` frames even when capture stamps don't align
12//! with the target interval; per-frame spacing jitters by at most one capture
13//! frame. Pure + deterministic (integer math) → trivially unit-testable.
14
15/// A per-stream frame-rate gate. Cheap to hold; reset on a stream restart.
16#[derive(Debug, Default)]
17pub struct FrameRateGate {
18    /// Capture stamp (µs) of the last frame *seen* (`None` until the first).
19    last_us: Option<i64>,
20    /// Accumulated capture time (µs) not yet "spent" on an emitted frame. Bounded
21    /// to one interval so a capture stall never releases a burst on resume.
22    acc_us: i64,
23}
24
25impl FrameRateGate {
26    pub fn new() -> Self {
27        Self::default()
28    }
29
30    /// Should the frame stamped `now_us` be emitted to hit `target_fps`?
31    ///
32    /// `target_fps == 0` means "no limit" → always emit. Stamps are expected
33    /// non-decreasing (a capture clock); a backwards step contributes no time.
34    pub fn should_emit(&mut self, now_us: i64, target_fps: u32) -> bool {
35        if target_fps == 0 {
36            self.last_us = Some(now_us);
37            self.acc_us = 0;
38            return true;
39        }
40        let interval = 1_000_000 / target_fps as i64;
41        match self.last_us {
42            // First frame always emits and anchors the clock.
43            None => {
44                self.last_us = Some(now_us);
45                self.acc_us = 0;
46                true
47            }
48            Some(last) => {
49                self.acc_us += (now_us - last).max(0);
50                self.last_us = Some(now_us);
51                if self.acc_us >= interval {
52                    // Carry the remainder (keeps the long-run count exact) but cap it
53                    // at half an interval: in steady state the leftover is always
54                    // below this, so the count is untouched; after a capture stall it
55                    // bounds the carry so the next frame can't emit back-to-back (no
56                    // burst — emitted frames stay ≥ ~one interval apart).
57                    self.acc_us = (self.acc_us - interval).min(interval / 2);
58                    true
59                } else {
60                    false
61                }
62            }
63        }
64    }
65
66    /// Forget pacing state (e.g. after a stream restart) so the next frame emits.
67    pub fn reset(&mut self) {
68        self.last_us = None;
69        self.acc_us = 0;
70    }
71}

73#[cfg(test)]
74mod tests {
75    use super::*;
76
77    /// Feed `secs` seconds of `capture_fps` frames through a gate targeting
78    /// `target_fps`; return how many were emitted.
79    fn emitted(capture_fps: u32, target_fps: u32, secs: i64) -> usize {
80        let mut g = FrameRateGate::new();
81        let cap_interval = 1_000_000 / capture_fps as i64;
82        let frames = capture_fps as i64 * secs;
83        (0..frames)
84            .filter(|i| g.should_emit(i * cap_interval, target_fps))
85            .count()
86    }
87
88    /// One table drives the downsample check: over a 4 s window the emitted count
89    /// is within one frame of `target_fps × secs` (or `capture_fps × secs` when the
90    /// target meets/exceeds capture — it never upsamples). DRY: a new ratio is a
91    /// new row.
92    #[test]
93    fn downsamples_to_target_count() {
94        // (capture_fps, target_fps) → expected emitted fps over the window.
95        let cases = [
96            (60, 30, 30),
97            (60, 15, 15),
98            (30, 15, 15),
99            (120, 30, 30),
100            (60, 60, 60),  // target == capture → all
101            (60, 120, 60), // target > capture → capped at capture (no upsample)
102            (60, 0, 60),   // 0 == no limit → all
103        ];
104        let secs = 4;
105        for (cap, tgt, want_fps) in cases {
106            let got = emitted(cap, tgt, secs) as i64;
107            let want = want_fps * secs;
108            assert!(
109                (got - want).abs() <= secs, // ≤ ~1 frame/sec of slack
110                "{cap}fps → target {tgt}fps over {secs}s: emitted {got}, want ~{want}"
111            );
112        }
113    }
114
115    /// A target above the capture rate never drops a frame (downsample only).
116    #[test]
117    fn never_upsamples() {
118        let mut g = FrameRateGate::new();
119        let cap = 1_000_000 / 30; // 30 fps capture
120        for i in 0..120 {
121            assert!(
122                g.should_emit(i * cap, 60),
123                "60-target must pass every 30fps frame"
124            );
125        }
126    }
127
128    /// A capture stall does not release a burst: the post-stall frame emits, and no
129    /// two consecutive frames emit back-to-back (emitted frames stay ≥ ~one interval
130    /// apart even right after the gap — the bounded carry, not a flush).
131    #[test]
132    fn stall_does_not_burst() {
133        let mut g = FrameRateGate::new();
134        assert!(g.should_emit(0, 30)); // first
135        // 10 s gap, then a tight run of 60fps capture frames.
136        let cap = 1_000_000 / 60;
137        let base = 10_000_000;
138        let emitted: Vec<bool> = (0..6).map(|i| g.should_emit(base + i * cap, 30)).collect();
139        assert!(emitted[0], "the post-stall frame emits");
140        let burst = emitted.windows(2).any(|w| w[0] && w[1]);
141        assert!(!burst, "no two adjacent emits after a stall: {emitted:?}");
142    }
143
144    /// A backwards timestamp (`now_us < last_us`, e.g. a clock glitch or reordered
145    /// stamp) contributes zero accumulated time (`(now - last).max(0)`), so it does
146    /// NOT push the accumulator over the interval and does NOT emit a spurious frame.
147    /// The clock still advances to the backwards stamp (matching the `Some(last)`
148    /// branch), but emission is gated purely on accumulated forward time.
149    #[test]
150    fn backwards_timestamp_contributes_zero() {
151        let mut g = FrameRateGate::new();
152        assert!(
153            g.should_emit(1_000_000, 30),
154            "first frame anchors and emits"
155        );
156        // A stamp well BEFORE the anchor: no forward time accrues → no emit.
157        assert!(
158            !g.should_emit(500_000, 30),
159            "a backwards stamp must not emit a spurious frame"
160        );
161        // And another backwards step: still no emit (accumulator stays at 0).
162        assert!(
163            !g.should_emit(400_000, 30),
164            "still no forward time → no emit"
165        );
166    }
167
168    /// Switching the target mid-stream takes effect immediately (no reset needed).
169    #[test]
170    fn retarget_takes_effect() {
171        let mut g = FrameRateGate::new();
172        let cap = 1_000_000 / 60;
173        // 1 s at 60fps target → ~60 emitted.
174        let a = (0..60).filter(|i| g.should_emit(i * cap, 60)).count();
175        assert!(a >= 58, "60-target passes ~all 60fps frames: {a}");
176        // Next 1 s at 15fps target → ~15 emitted.
177        let b = (60..120).filter(|i| g.should_emit(i * cap, 15)).count();
178        assert!((b as i64 - 15).abs() <= 2, "15-target downsamples: {b}");
179    }
