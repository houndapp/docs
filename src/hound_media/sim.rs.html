1//! # Sans-IO network impairment simulator
2//!
3//! A deterministic, virtual-clock model of a lossy network link, used to test
4//! the media engine without real sockets. You pump it: `send(now_us, packet)`
5//! to enqueue, `poll(now_us)` to collect whatever has arrived by `now_us`. It
6//! models, all reproducibly from a seed:
7//!
8//! * **loss** — per-packet drop probability;
9//! * **duplication** — per-packet duplicate probability;
10//! * **propagation + jitter** — base one-way delay plus uniform random jitter
11//!   (which naturally produces **reordering**);
12//! * **bandwidth + queue** — a serialization model where sending faster than the
13//!   link capacity builds queuing delay, and a bounded queue drops on overflow
14//!   (congestion loss).
15//!
16//! This is the foundation for the reliability, congestion-control, and
17//! jitter-buffer tests. It is intentionally pure and dependency-free.
18
19use std::cmp::Reverse;
20use std::collections::BinaryHeap;
21
22/// Tunable link impairments. `Default` is a perfect link (no loss/delay/limit).
23#[derive(Debug, Clone)]
24pub struct NetworkConditions {
25    /// Per-packet drop probability, 0.0..=1.0.
26    pub loss: f64,
27    /// Per-packet duplicate probability, 0.0..=1.0.
28    pub duplicate: f64,
29    /// One-way propagation delay in microseconds.
30    pub base_delay_us: u64,
31    /// Extra uniform-random delay in `[0, jitter_us]`, the source of reordering.
32    pub jitter_us: u64,
33    /// Link capacity in bits/second. `None` = unlimited (no serialization delay).
34    pub bandwidth_bps: Option<u64>,
35    /// Max bytes that may sit queued for serialization before packets are
36    /// dropped (congestion loss). Ignored when `bandwidth_bps` is `None`.
37    pub queue_bytes: usize,
38}
39
40impl Default for NetworkConditions {
41    fn default() -> Self {
42        Self {
43            loss: 0.0,
44            duplicate: 0.0,
45            base_delay_us: 0,
46            jitter_us: 0,
47            bandwidth_bps: None,
48            queue_bytes: usize::MAX,
49        }
50    }
51}

53/// A one-directional simulated link. Drive it with a monotonically increasing
54/// `now_us`.
55#[derive(Debug)]
56pub struct SimLink {
57    cond: NetworkConditions,
58    rng: Rng,
59    heap: BinaryHeap<Reverse<Entry>>,
60    /// Tie-breaker so equal delivery times pop in a stable, deterministic order.
61    seqno: u64,
62    /// Time at which the link finishes serializing already-queued packets.
63    link_free_us: u64,
64    /// Bytes currently queued for serialization (for the congestion drop check).
65    queued_bytes: usize,
66    // ── Counters (diagnostics / test assertions) ──
67    pub sent: u64,
68    pub dropped_loss: u64,
69    pub dropped_congestion: u64,
70    pub delivered: u64,
71}
72
73#[derive(Debug)]
74struct Entry {
75    deliver_at_us: u64,
76    seqno: u64,
77    size: usize,
78    packet: Vec<u8>,
79}
80
81impl PartialEq for Entry {
82    fn eq(&self, o: &Self) -> bool {
83        (self.deliver_at_us, self.seqno) == (o.deliver_at_us, o.seqno)
84    }
85}
86impl Eq for Entry {}
87impl PartialOrd for Entry {
88    fn partial_cmp(&self, o: &Self) -> Option<std::cmp::Ordering> {
89        Some(self.cmp(o))
90    }
91}
92impl Ord for Entry {
93    fn cmp(&self, o: &Self) -> std::cmp::Ordering {
94        (self.deliver_at_us, self.seqno).cmp(&(o.deliver_at_us, o.seqno))
95    }
96}
97
98impl SimLink {
99    pub fn new(cond: NetworkConditions, seed: u64) -> Self {
100        Self {
101            cond,
102            rng: Rng::new(seed),
103            heap: BinaryHeap::new(),
104            seqno: 0,
105            link_free_us: 0,
106            queued_bytes: 0,
107            sent: 0,
108            dropped_loss: 0,
109            dropped_congestion: 0,
110            delivered: 0,
111        }
112    }
113
114    /// Offer a packet to the link at time `now_us`. It may be dropped (loss or
115    /// congestion), delayed, and/or duplicated; survivors become deliverable
116    /// later via [`SimLink::poll`].
117    pub fn send(&mut self, now_us: u64, packet: &[u8]) {
118        self.sent += 1;
119
120        if self.cond.loss > 0.0 && self.rng.next_f64() < self.cond.loss {
121            self.dropped_loss += 1;
122            return;
123        }
124
125        let size = packet.len();
126
127        // Serialization (bandwidth) model: the link transmits one packet at a
128        // time; sending faster than capacity makes `link_free_us` run ahead of
129        // `now_us`, i.e. queuing delay. A full queue drops (congestion loss).
130        let serialize_us = match self.cond.bandwidth_bps {
131            Some(bps) if bps > 0 => (size as u64 * 8 * 1_000_000) / bps,
132            _ => 0,
133        };
134        let base_arrive = if serialize_us > 0 {
135            if self.queued_bytes + size > self.cond.queue_bytes {
136                self.dropped_congestion += 1;
137                return;
138            }
139            let start = now_us.max(self.link_free_us);
140            self.link_free_us = start + serialize_us;
141            self.queued_bytes += size;
142            self.link_free_us
143        } else {
144            now_us
145        };
146
147        let arrive = base_arrive + self.cond.base_delay_us;
148        let jitter = self.sample_jitter();
149        self.enqueue(arrive + jitter, size, packet.to_vec());
150
151        if self.cond.duplicate > 0.0 && self.rng.next_f64() < self.cond.duplicate {
152            let dup_jitter = self.sample_jitter();
153            self.enqueue(
154                arrive + dup_jitter,
155                0, // a duplicate doesn't re-consume serialization budget here
156                packet.to_vec(),
157            );
158        }
159    }
160
161    /// A fresh jitter sample in `[0, jitter_us]` (0 when jitter is disabled, which
162    /// also draws no random number so the stream is unperturbed).
163    fn sample_jitter(&mut self) -> u64 {
164        if self.cond.jitter_us > 0 {
165            (self.rng.next_f64() * self.cond.jitter_us as f64) as u64
166        } else {
167            0
168        }
169    }
170
171    fn enqueue(&mut self, deliver_at_us: u64, size: usize, packet: Vec<u8>) {
172        self.heap.push(Reverse(Entry {
173            deliver_at_us,
174            seqno: self.seqno,
175            size,
176            packet,
177        }));
178        self.seqno += 1;
179    }
180
181    /// Collect every packet that has arrived by `now_us`, in delivery-time order
182    /// (which may differ from send order — that is the simulated reordering).
183    pub fn poll(&mut self, now_us: u64) -> Vec<Vec<u8>> {
184        let mut out = Vec::new();
185        while let Some(Reverse(top)) = self.heap.peek() {
186            if top.deliver_at_us > now_us {
187                break;
188            }
189            let Reverse(entry) = self.heap.pop().unwrap();
190            self.queued_bytes = self.queued_bytes.saturating_sub(entry.size);
191            self.delivered += 1;
192            out.push(entry.packet);
193        }
194        out
195    }
196
197    /// Packets still in flight (queued, not yet delivered).
198    pub fn in_flight(&self) -> usize {
199        self.heap.len()
200    }
201}

203/// Tiny deterministic PRNG (SplitMix64) — no external `rand` dependency, fully
204/// reproducible so impairment tests are stable.
205#[derive(Debug)]
206struct Rng(u64);
207
208impl Rng {
209    fn new(seed: u64) -> Self {
210        Self(seed)
211    }
212    fn next_u64(&mut self) -> u64 {
213        self.0 = self.0.wrapping_add(0x9E37_79B9_7F4A_7C15);
214        let mut z = self.0;
215        z = (z ^ (z >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
216        z = (z ^ (z >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
217        z ^ (z >> 31)
218    }
219    /// Uniform in `[0, 1)`.
220    fn next_f64(&mut self) -> f64 {
221        (self.next_u64() >> 11) as f64 / (1u64 << 53) as f64
222    }
223}

297    #[test]
298    fn bandwidth_builds_queue_then_drops_on_overflow() {
299        // 1 Mbps link, 4 KB queue. 1500-byte packets = 12 kbit each = 12 ms to
300        // serialize; bursting many at t=0 must back up and then drop.
301        let cond = NetworkConditions {
302            bandwidth_bps: Some(1_000_000),
303            queue_bytes: 4_000,
304            ..Default::default()
305        };
306        let mut link = SimLink::new(cond, 1);
307        let pkt = vec![0u8; 1500];
308        for _ in 0..20 {
309            link.send(0, &pkt);
310        }
311        assert!(link.dropped_congestion > 0, "overflow should drop some");
312        // Earliest packets are delayed by serialization (not delivered at t=0).
313        assert!(
314            link.poll(0).is_empty(),
315            "first packet still serializing at t=0"
316        );
317        assert!(
318            !link.poll(12_000).is_empty(),
319            "first packet out after ~12ms"
320        );

325        // Stronger than the loss-count check: the entire delivered byte stream

373    #[test]
374    fn poll_is_monotonic_and_drains_in_delivery_order() {
375        // Packets become due as `now` advances; nothing is delivered early, and
376        // the queued count shrinks as they drain.
377        let cond = NetworkConditions {
378            base_delay_us: 10_000,
379            ..Default::default()
380        };
381        let mut link = SimLink::new(cond, 1);
382        for i in 0..5u8 {
383            link.send(i as u64 * 1_000, &[i]);
384        }
385        assert_eq!(link.in_flight(), 5);
386        assert!(link.poll(9_999).is_empty(), "first not due until +10ms");
387        let got = link.poll(14_000); // packets sent at 0..=4ms, +10ms delay
388        assert_eq!(
389            got.iter().map(|p| p[0]).collect::<Vec<_>>(),
390            vec![0, 1, 2, 3, 4]
391        );
392        assert_eq!(link.in_flight(), 0);

429    /// The money test: run real framing through a lossy, reordering link and
430    /// prove the engine never yields a corrupt frame — frames either reassemble
431    /// byte-for-byte or are cleanly dropped.
432    #[test]
433    fn framing_survives_loss_and_reorder_without_corruption() {
434        let cond = NetworkConditions {
435            loss: 0.05,
436            jitter_us: 40_000,
437            duplicate: 0.02,
438            ..Default::default()
439        };
440        let mut link = SimLink::new(cond, 99);
441        let mut pktzr = Packetizer::new(1);
442        let mut reasm = Reassembler::new();
443
444        // Known content per frame_seq so we can verify integrity on delivery.
445        let mut expected: HashMap<(TrackKind, u32), Vec<u8>> = HashMap::new();
446        let mut audio_sent = 0u32;
447        let mut video_sent = 0u32;
448        let mut audio_ok = 0u32;
449        let mut video_ok = 0u32;
450
451        let mut now = 0u64;
452        for f in 0..200u32 {
453            // Alternate: small audio (1 fragment) and large video (multi-fragment).
454            let (track, codec, body): (TrackKind, Codec, Vec<u8>) = if f % 2 == 0 {
455                audio_sent += 1;
456                (
457                    TrackKind::CamAudio,
458                    Codec::Opus,
459                    vec![(f & 0xff) as u8; 200],
460                )
461            } else {
462                video_sent += 1;
463                (
464                    TrackKind::CamVideo,
465                    Codec::Av1,
466                    (0..6000u32).map(|i| (i ^ f) as u8).collect(),
467                )
468            };
469            // Record expected content against the frame_seq the packetizer will use.
470            let pkts = pktzr.packetize(track, codec, f % 8 == 0, now as i64, &body, 1100);
471            let fseq = PacketHeader::parse(&pkts[0]).unwrap().frame_seq;
472            expected.insert((track, fseq), body.clone());
473
474            for p in &pkts {
475                link.send(now, p);
476            }
477            now += 20_000; // 20 ms between frames
478
479            for p in link.poll(now) {
480                let h = PacketHeader::parse(&p).unwrap();
481                if let Some(frame) = reasm.push(&h, &p[HEADER_LEN..]) {
482                    let want = expected
483                        .get(&(frame.track, frame.frame_seq))
484                        .expect("delivered a frame we never sent");
485                    assert_eq!(&frame.data, want, "CORRUPT frame {:?}", frame.frame_seq);
486                    if frame.track == TrackKind::CamAudio {
487                        audio_ok += 1;
488                    } else {
489                        video_ok += 1;
490                    }
491                }
492            }
493        }
494        // Flush the tail.
495        for p in link.poll(now + 1_000_000) {
496            let h = PacketHeader::parse(&p).unwrap();
497            if let Some(frame) = reasm.push(&h, &p[HEADER_LEN..]) {
498                let want = expected.get(&(frame.track, frame.frame_seq)).unwrap();
499                assert_eq!(&frame.data, want);
500                if frame.track == TrackKind::CamAudio {
501                    audio_ok += 1;
502                } else {
503                    video_ok += 1;
504                }
505            }
506        }
507
508        // Most frames get through; integrity already asserted above.
509        assert!(
510            audio_ok as f64 > audio_sent as f64 * 0.85,
511            "audio {audio_ok}/{audio_sent}"
512        );
513        assert!(video_ok > 0 && video_sent > 0);
514        // Single-fragment audio survives loss strictly better than
515        // multi-fragment video (losing any one fragment drops the whole frame).
516        let audio_rate = audio_ok as f64 / audio_sent as f64;
517        let video_rate = video_ok as f64 / video_sent as f64;
518        assert!(
519            audio_rate > video_rate,
520            "audio {audio_rate} should beat video {video_rate}"
521        );
522    }
