1//! # Sans-IO network impairment simulator
2//!
3//! A deterministic, virtual-clock model of a lossy network link, used to test
4//! the media engine without real sockets. You pump it: `send(now_us, packet)`
5//! to enqueue, `poll(now_us)` to collect whatever has arrived by `now_us`. It
6//! models, all reproducibly from a seed:
7//!
8//! * **loss** — per-packet drop probability;
9//! * **duplication** — per-packet duplicate probability;
10//! * **propagation + jitter** — base one-way delay plus uniform random jitter
11//!   (which naturally produces **reordering**);
12//! * **bandwidth + queue** — a serialization model where sending faster than the
13//!   link capacity builds queuing delay, and a bounded queue drops on overflow
14//!   (congestion loss).
15//!
16//! This is the foundation for the reliability tests, the future congestion-control
17//! and jitter-buffer tests, and the record/replay debugger (see
18//! `docs/voice/roadmap.md`). It is intentionally pure and dependency-free.
19
20use std::cmp::Reverse;
21use std::collections::BinaryHeap;
22
23/// Tunable link impairments. `Default` is a perfect link (no loss/delay/limit).
24#[derive(Debug, Clone)]
25pub struct NetworkConditions {
26    /// Per-packet drop probability, 0.0..=1.0.
27    pub loss: f64,
28    /// Per-packet duplicate probability, 0.0..=1.0.
29    pub duplicate: f64,
30    /// One-way propagation delay in microseconds.
31    pub base_delay_us: u64,
32    /// Extra uniform-random delay in `[0, jitter_us]`, the source of reordering.
33    pub jitter_us: u64,
34    /// Link capacity in bits/second. `None` = unlimited (no serialization delay).
35    pub bandwidth_bps: Option<u64>,
36    /// Max bytes that may sit queued for serialization before packets are
37    /// dropped (congestion loss). Ignored when `bandwidth_bps` is `None`.
38    pub queue_bytes: usize,
39}
40
41impl Default for NetworkConditions {
42    fn default() -> Self {
43        Self {
44            loss: 0.0,
45            duplicate: 0.0,
46            base_delay_us: 0,
47            jitter_us: 0,
48            bandwidth_bps: None,
49            queue_bytes: usize::MAX,
50        }
51    }
52}
53
54/// A one-directional simulated link. Drive it with a monotonically increasing
55/// `now_us`.
56#[derive(Debug)]
57pub struct SimLink {
58    cond: NetworkConditions,
59    rng: Rng,
60    heap: BinaryHeap<Reverse<Entry>>,
61    /// Tie-breaker so equal delivery times pop in a stable, deterministic order.
62    seqno: u64,
63    /// Time at which the link finishes serializing already-queued packets.
64    link_free_us: u64,
65    /// Bytes currently queued for serialization (for the congestion drop check).
66    queued_bytes: usize,
67    // ── Counters (diagnostics / test assertions) ──
68    pub sent: u64,
69    pub dropped_loss: u64,
70    pub dropped_congestion: u64,
71    pub delivered: u64,
72}
73
74#[derive(Debug)]
75struct Entry {
76    deliver_at_us: u64,
77    seqno: u64,
78    size: usize,
79    packet: Vec<u8>,
80}
81
82impl PartialEq for Entry {
83    fn eq(&self, o: &Self) -> bool {
84        (self.deliver_at_us, self.seqno) == (o.deliver_at_us, o.seqno)
85    }
86}
87impl Eq for Entry {}
88impl PartialOrd for Entry {
89    fn partial_cmp(&self, o: &Self) -> Option<std::cmp::Ordering> {
90        Some(self.cmp(o))
91    }
92}
93impl Ord for Entry {
94    fn cmp(&self, o: &Self) -> std::cmp::Ordering {
95        (self.deliver_at_us, self.seqno).cmp(&(o.deliver_at_us, o.seqno))
96    }
97}
98
99impl SimLink {
100    pub fn new(cond: NetworkConditions, seed: u64) -> Self {
101        Self {
102            cond,
103            rng: Rng::new(seed),
104            heap: BinaryHeap::new(),
105            seqno: 0,
106            link_free_us: 0,
107            queued_bytes: 0,
108            sent: 0,
109            dropped_loss: 0,
110            dropped_congestion: 0,
111            delivered: 0,
112        }
113    }
114
115    /// Offer a packet to the link at time `now_us`. It may be dropped (loss or
116    /// congestion), delayed, and/or duplicated; survivors become deliverable
117    /// later via [`SimLink::poll`].
118    pub fn send(&mut self, now_us: u64, packet: &[u8]) {
119        self.sent += 1;
120
121        if self.cond.loss > 0.0 && self.rng.next_f64() < self.cond.loss {
122            self.dropped_loss += 1;
123            return;
124        }
125
126        let size = packet.len();
127
128        // Serialization (bandwidth) model: the link transmits one packet at a
129        // time; sending faster than capacity makes `link_free_us` run ahead of
130        // `now_us`, i.e. queuing delay. A full queue drops (congestion loss).
131        let serialize_us = match self.cond.bandwidth_bps {
132            Some(bps) if bps > 0 => (size as u64 * 8 * 1_000_000) / bps,
133            _ => 0,
134        };
135        let base_arrive = if serialize_us > 0 {
136            if self.queued_bytes + size > self.cond.queue_bytes {
137                self.dropped_congestion += 1;
138                return;
139            }
140            let start = now_us.max(self.link_free_us);
141            self.link_free_us = start + serialize_us;
142            self.queued_bytes += size;
143            self.link_free_us
144        } else {
145            now_us
146        };
147
148        let jitter = if self.cond.jitter_us > 0 {
149            (self.rng.next_f64() * self.cond.jitter_us as f64) as u64
150        } else {
151            0
152        };
153        let deliver_at_us = base_arrive + self.cond.base_delay_us + jitter;
154
155        self.enqueue(deliver_at_us, size, packet.to_vec());
156
157        if self.cond.duplicate > 0.0 && self.rng.next_f64() < self.cond.duplicate {
158            let dup_jitter = if self.cond.jitter_us > 0 {
159                (self.rng.next_f64() * self.cond.jitter_us as f64) as u64
160            } else {
161                0
162            };
163            self.enqueue(
164                base_arrive + self.cond.base_delay_us + dup_jitter,
165                0, // a duplicate doesn't re-consume serialization budget here
166                packet.to_vec(),
167            );
168        }
169    }
170
171    fn enqueue(&mut self, deliver_at_us: u64, size: usize, packet: Vec<u8>) {
172        self.heap.push(Reverse(Entry {
173            deliver_at_us,
174            seqno: self.seqno,
175            size,
176            packet,
177        }));
178        self.seqno += 1;
179    }
180
181    /// Collect every packet that has arrived by `now_us`, in delivery-time order
182    /// (which may differ from send order — that is the simulated reordering).
183    pub fn poll(&mut self, now_us: u64) -> Vec<Vec<u8>> {
184        let mut out = Vec::new();
185        while let Some(Reverse(top)) = self.heap.peek() {
186            if top.deliver_at_us > now_us {
187                break;
188            }
189            let Reverse(entry) = self.heap.pop().unwrap();
190            self.queued_bytes = self.queued_bytes.saturating_sub(entry.size);
191            self.delivered += 1;
192            out.push(entry.packet);
193        }
194        out
195    }
196
197    /// Packets still in flight (queued, not yet delivered).
198    pub fn in_flight(&self) -> usize {
199        self.heap.len()
200    }
201}

203/// Tiny deterministic PRNG (SplitMix64) — no external `rand` dependency, fully
204/// reproducible so impairment tests are stable.
205#[derive(Debug)]
206struct Rng(u64);
207
208impl Rng {
209    fn new(seed: u64) -> Self {
210        Self(seed)
211    }
212    fn next_u64(&mut self) -> u64 {
213        self.0 = self.0.wrapping_add(0x9E37_79B9_7F4A_7C15);
214        let mut z = self.0;
215        z = (z ^ (z >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
216        z = (z ^ (z >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
217        z ^ (z >> 31)
218    }
219    /// Uniform in `[0, 1)`.
220    fn next_f64(&mut self) -> f64 {
221        (self.next_u64() >> 11) as f64 / (1u64 << 53) as f64
222    }
223}

232    #[test]
233    fn perfect_link_delivers_all_in_order() {
234        let mut link = SimLink::new(NetworkConditions::default(), 1);
235        for i in 0..10u8 {
236            link.send(0, &[i]);
237        }
238        let got = link.poll(0);
239        assert_eq!(got.len(), 10);
240        assert_eq!(
241            got.iter().map(|p| p[0]).collect::<Vec<_>>(),
242            (0..10).collect::<Vec<_>>()
243        );
244        assert_eq!(link.delivered, 10);
245    }

247    #[test]
248    fn base_delay_holds_until_due() {
249        let cond = NetworkConditions {
250            base_delay_us: 50_000,
251            ..Default::default()
252        };
253        let mut link = SimLink::new(cond, 1);
254        link.send(0, b"x");
255        assert!(link.poll(49_999).is_empty(), "not due yet");
256        assert_eq!(link.poll(50_000).len(), 1, "due at exactly base_delay");
257    }

259    #[test]
260    fn loss_rate_is_approximately_honoured_and_deterministic() {
261        let cond = NetworkConditions {
262            loss: 0.3,
263            ..Default::default()
264        };
265        let mut a = SimLink::new(cond.clone(), 42);
266        for _ in 0..10_000 {
267            a.send(0, b"p");
268        }
269        let frac = a.dropped_loss as f64 / 10_000.0;
270        assert!((0.27..0.33).contains(&frac), "loss frac {frac} not ~0.30");
271
272        // Same seed → identical outcome (deterministic).
273        let mut b = SimLink::new(cond, 42);
274        for _ in 0..10_000 {
275            b.send(0, b"p");
276        }
277        assert_eq!(a.dropped_loss, b.dropped_loss);
278    }

280    #[test]
281    fn jitter_causes_reordering() {
282        let cond = NetworkConditions {
283            jitter_us: 100_000,
284            ..Default::default()
285        };
286        let mut link = SimLink::new(cond, 7);
287        for i in 0..50u8 {
288            link.send(0, &[i]);
289        }
290        // Deliver everything well after max jitter.
291        let got = link.poll(1_000_000);
292        assert_eq!(got.len(), 50);
293        let order: Vec<u8> = got.iter().map(|p| p[0]).collect();
294        assert_ne!(order, (0..50).collect::<Vec<_>>(), "jitter should reorder");
295    }

297    #[test]
298    fn bandwidth_builds_queue_then_drops_on_overflow() {
299        // 1 Mbps link, 4 KB queue. 1500-byte packets = 12 kbit each = 12 ms to
300        // serialize; bursting many at t=0 must back up and then drop.
301        let cond = NetworkConditions {
302            bandwidth_bps: Some(1_000_000),
303            queue_bytes: 4_000,
304            ..Default::default()
305        };
306        let mut link = SimLink::new(cond, 1);
307        let pkt = vec![0u8; 1500];
308        for _ in 0..20 {
309            link.send(0, &pkt);
310        }
311        assert!(link.dropped_congestion > 0, "overflow should drop some");
312        // Earliest packets are delayed by serialization (not delivered at t=0).
313        assert!(
314            link.poll(0).is_empty(),
315            "first packet still serializing at t=0"
316        );
317        assert!(
318            !link.poll(12_000).is_empty(),
319            "first packet out after ~12ms"
320        );
321    }

323    /// The money test: run real framing through a lossy, reordering link and
324    /// prove the engine never yields a corrupt frame — frames either reassemble
325    /// byte-for-byte or are cleanly dropped.
326    #[test]
327    fn framing_survives_loss_and_reorder_without_corruption() {
328        let cond = NetworkConditions {
329            loss: 0.05,
330            jitter_us: 40_000,
331            duplicate: 0.02,
332            ..Default::default()
333        };
334        let mut link = SimLink::new(cond, 99);
335        let mut pktzr = Packetizer::new(1);
336        let mut reasm = Reassembler::new();
337
338        // Known content per frame_seq so we can verify integrity on delivery.
339        let mut expected: HashMap<(TrackKind, u32), Vec<u8>> = HashMap::new();
340        let mut audio_sent = 0u32;
341        let mut video_sent = 0u32;
342        let mut audio_ok = 0u32;
343        let mut video_ok = 0u32;
344
345        let mut now = 0u64;
346        for f in 0..200u32 {
347            // Alternate: small audio (1 fragment) and large video (multi-fragment).
348            let (track, codec, body): (TrackKind, Codec, Vec<u8>) = if f % 2 == 0 {
349                audio_sent += 1;
350                (
351                    TrackKind::CamAudio,
352                    Codec::Opus,
353                    vec![(f & 0xff) as u8; 200],
354                )
355            } else {
356                video_sent += 1;
357                (
358                    TrackKind::CamVideo,
359                    Codec::Av1,
360                    (0..6000u32).map(|i| (i ^ f) as u8).collect(),
361                )
362            };
363            // Record expected content against the frame_seq the packetizer will use.
364            let pkts = pktzr.packetize(track, codec, f % 8 == 0, now as i64, &body, 1100);
365            let fseq = PacketHeader::parse(&pkts[0]).unwrap().frame_seq;
366            expected.insert((track, fseq), body.clone());
367
368            for p in &pkts {
369                link.send(now, p);
370            }
371            now += 20_000; // 20 ms between frames
372
373            for p in link.poll(now) {
374                let h = PacketHeader::parse(&p).unwrap();
375                if let Some(frame) = reasm.push(&h, &p[HEADER_LEN..]) {
376                    let want = expected
377                        .get(&(frame.track, frame.frame_seq))
378                        .expect("delivered a frame we never sent");
379                    assert_eq!(&frame.data, want, "CORRUPT frame {:?}", frame.frame_seq);
380                    if frame.track == TrackKind::CamAudio {
381                        audio_ok += 1;
382                    } else {
383                        video_ok += 1;
384                    }
385                }
386            }
387        }
388        // Flush the tail.
389        for p in link.poll(now + 1_000_000) {
390            let h = PacketHeader::parse(&p).unwrap();
391            if let Some(frame) = reasm.push(&h, &p[HEADER_LEN..]) {
392                let want = expected.get(&(frame.track, frame.frame_seq)).unwrap();
393                assert_eq!(&frame.data, want);
394                if frame.track == TrackKind::CamAudio {
395                    audio_ok += 1;
396                } else {
397                    video_ok += 1;
398                }
399            }
400        }
401
402        // Most frames get through; integrity already asserted above.
403        assert!(
404            audio_ok as f64 > audio_sent as f64 * 0.85,
405            "audio {audio_ok}/{audio_sent}"
406        );
407        assert!(video_ok > 0 && video_sent > 0);
408        // Single-fragment audio survives loss strictly better than
409        // multi-fragment video (losing any one fragment drops the whole frame).
410        let audio_rate = audio_ok as f64 / audio_sent as f64;
411        let video_rate = video_ok as f64 / video_sent as f64;
412        assert!(
413            audio_rate > video_rate,
414            "audio {audio_rate} should beat video {video_rate}"
415        );
416    }
