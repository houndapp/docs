1//! # Congestion control + adaptive quality
2//!
3//! Sans-IO pieces for graceful degradation under congestion:
4//!
5//! * [`RateController`] — a loss-based AIMD bandwidth estimator (the
6//!   well-known WebRTC loss controller): increase slowly when loss is low, hold
7//!   in the middle band, and back off fast when loss is high. Produces a target
8//!   send bitrate.
9//! * [`QualityLadder`] — maps an available bitrate to a discrete video [`Rung`]
10//!   with hysteresis (**drop fast, climb slow**) and a manual ceiling, falling
11//!   back to [`Quality::AudioOnly`] when even the lowest rung won't fit.
12//!
13//! Principle: **audio is sacred, video is the release valve.** This governs
14//! video only; audio runs at a fixed low bitrate underneath and is never
15//! starved. Both types are pure and clock-driven for deterministic tests.
16
17/// Tunables for [`RateController`]. Defaults mirror the classic WebRTC loss
18/// controller.
19#[derive(Debug, Clone)]
20pub struct RateConfig {
21    pub min_bps: u64,
22    pub max_bps: u64,
23    pub start_bps: u64,
24    /// Below this loss fraction, increase. Default 2%.
25    pub increase_loss: f64,
26    /// Above this loss fraction, decrease. Default 10%.
27    pub decrease_loss: f64,
28    /// Multiplicative increase per update step. Default 1.08.
29    pub increase_factor: f64,
30    /// Minimum spacing between *increase* steps ("up slow"). Decreases are not
31    /// rate-limited ("down fast").
32    pub increase_interval_us: i64,
33}
34
35impl Default for RateConfig {
36    fn default() -> Self {
37        Self {
38            min_bps: 100_000,   // 100 kbps floor (≈ lowest video rung)
39            max_bps: 4_000_000, // 4 Mbps ceiling
40            start_bps: 600_000,
41            increase_loss: 0.02,
42            decrease_loss: 0.10,
43            increase_factor: 1.08,
44            increase_interval_us: 1_000_000,
45        }
46    }
47}

49/// Loss-based AIMD send-rate estimator.
50#[derive(Debug)]
51pub struct RateController {
52    cfg: RateConfig,
53    target_bps: u64,
54    last_increase_us: i64,
55}
56
57impl RateController {
58    pub fn new(cfg: RateConfig) -> Self {
59        let target_bps = cfg.start_bps.clamp(cfg.min_bps, cfg.max_bps);
60        Self {
61            cfg,
62            target_bps,
63            last_increase_us: i64::MIN / 2,
64        }
65    }
66
67    /// Feed an observed loss fraction (0.0..=1.0) for the last interval; returns
68    /// the updated target bitrate.
69    ///
70    /// * loss > `decrease_loss`: back off immediately to `target * (1 - 0.5·loss)`.
71    /// * loss < `increase_loss`: grow by `increase_factor`, at most once per
72    ///   `increase_interval_us`.
73    /// * in between: hold.
74    pub fn on_feedback(&mut self, loss: f64, now_us: i64) -> u64 {
75        let loss = loss.clamp(0.0, 1.0);
76        if loss > self.cfg.decrease_loss {
77            let factor = 1.0 - 0.5 * loss;
78            self.target_bps = (self.target_bps as f64 * factor) as u64;
79            // Suppress an immediate re-increase after backing off.
80            self.last_increase_us = now_us;
81        } else if loss < self.cfg.increase_loss
82            && now_us - self.last_increase_us >= self.cfg.increase_interval_us
83        {
84            self.target_bps = (self.target_bps as f64 * self.cfg.increase_factor) as u64;
85            self.last_increase_us = now_us;
86        }
87        self.target_bps = self.target_bps.clamp(self.cfg.min_bps, self.cfg.max_bps);
88        self.target_bps
89    }
90
91    pub fn target_bps(&self) -> u64 {
92        self.target_bps
93    }
94
95    /// Move the bitrate ceiling at runtime — e.g. the quality ladder was
96    /// reconfigured to a richer (or capped) top rung, so the estimate is allowed to
97    /// climb higher (or must drop). The current estimate is re-clamped into the new
98    /// `[min, max]` so it never sits above the ceiling.
99    pub fn set_max_bps(&mut self, max_bps: u64) {
100        self.cfg.max_bps = max_bps.max(self.cfg.min_bps);
101        self.target_bps = self.target_bps.clamp(self.cfg.min_bps, self.cfg.max_bps);
102    }
103}

105/// Receiver-side downlink bitrate estimator for **simulcast layer selection**.
106///
107/// [`RateController`] sizes the *uplink* on the sender from receiver-reported loss;
108/// this is its mirror on the *receive* side. A receiver subscribed to one simulcast
109/// layer sees its own downlink loss (from the sequence tracker) and runs the same
110/// loss-based AIMD to estimate the bitrate its link can sustain. A [`LayerSelector`]
111/// then maps that estimate to a layer: a clean link climbs (probing the next layer
112/// up), a lossy one backs off, so each receiver independently converges on the
113/// richest layer its downlink carries — instead of every receiver being pinned to
114/// the worst leg by the single send-side ladder.
115///
116/// Pure/sans-IO: time is an argument, signals are explicit, the output is the
117/// estimate. Wraps a [`RateController`] so the AIMD behaviour is identical to (and
118/// shares the tests of) the send path.
119#[derive(Debug)]
120pub struct DownlinkEstimator {
121    rate: RateController,
122    /// Delay-based half (GCC): senses a filling bottleneck queue from the one-way
123    /// delay trend *before* loss appears. The estimate is the MIN of this and the
124    /// loss-based `rate` — whichever controller is more worried governs.
125    delay: crate::bwe_delay::DelayBasedBwe,
126}
127
128impl DownlinkEstimator {
129    /// `start_bps` seeds the estimate (typically the currently-subscribed layer's
130    /// bitrate); `max_bps` caps it at the top layer's bitrate (× the selector's climb
131    /// margin) so it never probes past the richest layer the sender offers.
132    pub fn new(start_bps: u64, max_bps: u64) -> Self {
133        Self {
134            rate: RateController::new(RateConfig {
135                start_bps,
136                max_bps,
137                ..Default::default()
138            }),
139            // Seed the delay-based half at the CEILING: with no arrival samples it
140            // has no evidence of congestion and must not constrain the loss-based
141            // estimate (min-combine). It only drops below the ceiling once the delay
142            // trend actually senses a filling queue.
143            delay: crate::bwe_delay::DelayBasedBwe::new(max_bps, max_bps),
144        }
145    }
146
147    /// Fold one reception-report interval into the LOSS-based estimate. `received`/
148    /// `lost` are the frame deltas the sequence tracker reports (the same pair sent
149    /// upstream as `MediaFeedback`); `now_us` is the receive clock. An interval with
150    /// no frames at all carries no signal and leaves the estimate untouched (no
151    /// phantom climb). Returns the combined estimate (bits/sec).
152    pub fn on_report(&mut self, received: u32, lost: u32, now_us: i64) -> u64 {
153        let total = received as u64 + lost as u64;
154        if total == 0 {
155            return self.estimate_bps();
156        }
157        let loss = lost as f64 / total as f64;
158        self.rate.on_feedback(loss, now_us);
159        self.estimate_bps()
160    }
161
162    /// Fold one frame's arrival into the DELAY-based estimate. `recv_us` is the local
163    /// arrival time, `send_us` the sender's capture timestamp (only their differences
164    /// are used, so the clocks need not be synchronised). Returns the combined
165    /// estimate. Drives the early back-off that loss-based AIMD can't see.
166    pub fn on_arrival(&mut self, recv_us: i64, send_us: i64) -> u64 {
167        self.delay.observe(recv_us, send_us);
168        self.estimate_bps()
169    }
170
171    /// The current downlink estimate (bits/sec): the lower of the loss-based and
172    /// delay-based controllers (GCC's min-combine).
173    pub fn estimate_bps(&self) -> u64 {
174        self.rate.target_bps().min(self.delay.estimate_bps())
175    }
176
177    /// The current delay-based congestion verdict (for stats/debug).
178    pub fn delay_usage(&self) -> crate::bwe_delay::Usage {
179        self.delay.usage()
180    }
181
182    /// Move the ceiling when the offered layer set changes (mirrors
183    /// [`RateController::set_max_bps`]).
184    pub fn set_max_bps(&mut self, max_bps: u64) {
185        self.rate.set_max_bps(max_bps);
186        self.delay.set_max_bps(max_bps);
187    }
188}

190/// Picks which published simulcast layer a receiver should subscribe to, given its
191/// [`DownlinkEstimator`] output. The layers are the sender's published bitrates,
192/// ascending. Same anti-thrash contract as [`QualityLadder`]: **drop fast** (step
193/// down as far as needed the moment the estimate falls below the current layer) but
194/// **climb slow** (one step at a time, and only with `climb_margin` headroom over
195/// the next layer's cost) so a flapping link can't thrash subscriptions.
196#[derive(Debug)]
197pub struct LayerSelector {
198    /// Published layer bitrates, ascending. Index = layer id.
199    layer_bps: Vec<u64>,
200    current: usize,
201    climb_margin: f64,
202}
203
204impl LayerSelector {
205    /// `layer_bps` is the published layers' bitrates; it is sorted ascending here, so
206    /// the returned index is into the sorted order (lowest layer = 0). Starts at the
207    /// lowest layer (safe bootstrap; climbs as the estimate proves headroom).
208    pub fn new(mut layer_bps: Vec<u64>) -> Self {
209        assert!(!layer_bps.is_empty(), "at least one layer");
210        layer_bps.sort_unstable();
211        Self {
212            layer_bps,
213            current: 0,
214            climb_margin: 1.15,
215        }
216    }
217
218    /// The layer to subscribe to for `estimate_bps`. Drops as many steps as needed to
219    /// fit; climbs at most one step, and only with headroom.
220    pub fn update(&mut self, estimate_bps: u64) -> usize {
221        while self.current > 0 && self.layer_bps[self.current] > estimate_bps {
222            self.current -= 1;
223        }
224        if self.current + 1 < self.layer_bps.len() {
225            let next = self.layer_bps[self.current + 1];
226            if estimate_bps as f64 >= next as f64 * self.climb_margin {
227                self.current += 1;
228            }
229        }
230        self.current
231    }
232
233    /// The layer chosen by the last [`Self::update`].
234    pub fn current(&self) -> usize {
235        self.current
236    }
237}

239/// One receiver's standing quality request for a video track, as the sender sees it.
240///
241/// Demand-driven simulcast: the sender publishes only the **source** until a receiver
242/// proves it cannot keep up and asks for less. The request is deliberately
243/// constrained — a receiver asks for AT MOST one tier, and only *below* the source
244/// (it can never ask for more than the publisher captures; a cap at/above source is
245/// just "I can take source").
246#[derive(Debug, Clone, Copy, PartialEq, Eq)]
247pub enum QualityRequest {
248    /// This receiver handles the source/top quality — no lower layer needed for it.
249    Source,
250    /// This receiver can sustain at most this many bits/sec; it wants a lower layer.
251    Capped(u64),
252}
253
254/// What a sender should publish for a video track given ALL its receivers' requests
255/// ([`plan_layers`]). At most two streams: the source and one shared low layer.
256#[derive(Debug, Clone, Copy, PartialEq, Eq)]
257pub struct LayerPlan {
258    /// Encode + publish the source/top layer (at least one receiver can take it).
259    pub send_source: bool,
260    /// `Some(bps)` → also (or *only*, if `!send_source`) publish ONE shared low layer
261    /// sized for the most-constrained requester (so every requester can decode it).
262    /// `None` → no low layer demanded.
263    pub low_layer_bps: Option<u64>,
264}
265
266impl LayerPlan {
267    /// The zero-cost default: encode only the source, no low layer. Byte-identical
268    /// to a build without simulcast — the plan a sender acts on whenever no receiver
269    /// has asked for less, and the safe plan the planner publishes while actuation is
270    /// disabled (see [`SimulcastPlanner`]).
271    pub const SOURCE_ONLY: LayerPlan = LayerPlan {
272        send_source: true,
273        low_layer_bps: None,
274    };
275}

277/// Aggregate every receiver's [`QualityRequest`] into the sender's [`LayerPlan`].
278///
279/// The whole demand-driven, coalesced, source-droppable policy in one pure function:
280/// * a receiver wanting `Source` (or a cap ≥ source — it can only ask for *lower*)
281///   keeps the source alive;
282/// * receivers wanting less than source collapse to ONE shared low layer at the
283///   **minimum** of their caps, so the weakest can still decode it (bounded to two
284///   total streams — no per-receiver explosion);
285/// * if NO receiver can take the source, `send_source` is `false` and only the low
286///   layer is published — the optimisation that stops encoding a stream nobody wants.
287///
288/// With no receivers the source stays on (idle default; nothing to optimise yet).
289/// Pure/sans-IO → unit-tested below.
290pub fn plan_layers(requests: &[QualityRequest], source_bps: u64) -> LayerPlan {
291    let mut send_source = requests.is_empty();
292    let mut min_cap: Option<u64> = None;
293    for req in requests {
294        match *req {
295            // "Only lower than what's being sent": a cap at/above source means this
296            // receiver can take the source itself.
297            QualityRequest::Source => send_source = true,
298            QualityRequest::Capped(bps) if bps >= source_bps => send_source = true,
299            QualityRequest::Capped(bps) => {
300                min_cap = Some(min_cap.map_or(bps, |m| m.min(bps)));
301            }
302        }
303    }
304    LayerPlan {
305        send_source,
306        low_layer_bps: min_cap,
307    }
308}

310/// Producer-side aggregator: tracks each receiver's standing [`QualityRequest`] for
311/// one published video track and yields the current [`LayerPlan`] via [`plan_layers`].
312///
313/// The sender feeds it every `QualityRequested` (peer → cap) and every peer-leave,
314/// and acts on the returned plan only when it *changes* — so a steady call costs
315/// nothing and a flapping receiver (already damped by its own send-side hysteresis)
316/// can at most toggle one layer. Peers are keyed by a plain `u32` (the session passes
317/// `PeerId.0`) so this core stays free of the control-plane types. Pure/sans-IO.
318#[derive(Debug)]
319pub struct SimulcastPlanner {
320    source_bps: u64,
321    /// Per-receiver standing request. A peer absent here has not asked for anything
322    /// → it is assumed to take the source (the default before any request arrives).
323    caps: std::collections::BTreeMap<u32, QualityRequest>,
324    /// The plan the requests imply — what the sender *would* publish if every
325    /// receiver could consume the layer it asked for. Tracked even while actuation
326    /// is off so the demand pipeline stays fully observable (logged on every change).
327    desired: LayerPlan,
328    /// Whether the sender may actually act on a non-default `desired` (drop the
329    /// source / spin up a low layer). This is only SAFE once a constrained receiver
330    /// can switch its subscription from the source track to the low track — until
331    /// that exists, dropping the source would starve receivers still subscribed to
332    /// it, and any low layer is consumed by nobody. While `false`, [`Self::plan`]
333    /// clamps to [`LayerPlan::SOURCE_ONLY`], so the encode path is byte-identical to
334    /// a build without simulcast regardless of what receivers request.
335    actuate: bool,
336}
337
338impl SimulcastPlanner {
339    /// Start with the given source bitrate and no receiver requests (plan = source
340    /// only — the zero-cost common case). Actuation is OFF: see [`Self::set_actuation`].
341    pub fn new(source_bps: u64) -> Self {
342        Self {
343            source_bps,
344            caps: std::collections::BTreeMap::new(),
345            desired: plan_layers(&[], source_bps),
346            actuate: false,
347        }
348    }
349
350    /// Enable or disable acting on the demand-driven plan. Pass `true` only once
351    /// constrained receivers actually move their subscription to the low layer (so
352    /// dropping the source can't starve anyone). Returns `true` if the EFFECTIVE
353    /// plan ([`Self::plan`]) changed as a result — the caller must then re-publish it.
354    pub fn set_actuation(&mut self, actuate: bool) -> bool {
355        if self.actuate == actuate {
356            return false;
357        }
358        let before = self.plan();
359        self.actuate = actuate;
360        before != self.plan()
361    }
362
363    fn recompute(&mut self) -> bool {
364        let reqs: Vec<QualityRequest> = self.caps.values().copied().collect();
365        let next = plan_layers(&reqs, self.source_bps);
366        let changed = next != self.desired;
367        self.desired = next;
368        changed
369    }
370
371    /// Record a receiver's standing request; returns `true` if the resulting DESIRED
372    /// plan changed (the sender logs the new demand). Whether that translates into a
373    /// reconfiguration depends on actuation — see [`Self::plan`] vs [`Self::desired`].
374    pub fn set_request(&mut self, peer: u32, req: QualityRequest) -> bool {
375        if self.caps.get(&peer) == Some(&req) {
376            return false;
377        }
378        self.caps.insert(peer, req);
379        self.recompute()
380    }
381
382    /// Drop a receiver (it left or unsubscribed); returns `true` if the plan changed.
383    pub fn remove_peer(&mut self, peer: u32) -> bool {
384        if self.caps.remove(&peer).is_none() {
385            return false;
386        }
387        self.recompute()
388    }
389
390    /// Reconcile against the live roster: drop caps for any peer not in `keep` (a
391    /// receiver that left without retracting its request). Returns `true` if the plan
392    /// changed. `keep` membership is tested via the closure so the caller needn't
393    /// allocate a set.
394    pub fn retain_peers(&mut self, keep: impl Fn(u32) -> bool) -> bool {
395        let before = self.caps.len();
396        self.caps.retain(|&peer, _| keep(peer));
397        if self.caps.len() == before {
398            return false;
399        }
400        self.recompute()
401    }
402
403    /// Update the source bitrate (the source rung re-negotiated); returns `true` if
404    /// the plan changed (a cap that was below the old source may now be ≥ the new one).
405    pub fn set_source_bps(&mut self, source_bps: u64) -> bool {
406        if self.source_bps == source_bps {
407            return false;
408        }
409        self.source_bps = source_bps;
410        self.recompute()
411    }
412
413    /// The peer ids currently requesting a CAPPED (below-source) layer — the receivers
414    /// the shared low layer serves. The sender's feedback partitioner excludes these
415    /// from the SOURCE adaptation so a low-layer receiver can't drag the source down
416    /// (see the simulcast notes). Independent of actuation; the caller gates on its own
417    /// flag.
418    pub fn capped_peers(&self) -> std::collections::BTreeSet<u32> {
419        self.caps
420            .iter()
421            .filter_map(|(&p, req)| match req {
422                QualityRequest::Capped(bps) if *bps < self.source_bps => Some(p),
423                _ => None,
424            })
425            .collect()
426    }
427
428    /// The EFFECTIVE plan the sender must act on: the demand-driven [`Self::desired`]
429    /// once actuation is enabled, otherwise the safe [`LayerPlan::SOURCE_ONLY`] (so a
430    /// constrained receiver can never trigger a source drop while it is still
431    /// subscribed to the source track).
432    pub fn plan(&self) -> LayerPlan {
433        if self.actuate {
434            self.desired
435        } else {
436            LayerPlan::SOURCE_ONLY
437        }
438    }
439
440    /// The plan the current requests imply, regardless of actuation — for logging and
441    /// observing the demand pipeline before the receive-side switch exists.
442    pub fn desired(&self) -> LayerPlan {
443        self.desired
444    }
445}

447/// The receiver's brain for demand-driven simulcast: folds its downlink reports into
448/// a [`DownlinkEstimator`] and decides the [`QualityRequest`] to send the producer,
449/// with hysteresis so it doesn't flap requests at the boundary.
450///
451/// Policy: request [`QualityRequest::Source`] only when the estimate reaches the
452/// source bitrate (a genuinely clean link — the estimator is capped at source, so
453/// this is "I'm keeping up with the full stream"); drop to `Capped(estimate)` once
454/// the estimate falls a clear band below source; hold in between. [`Self::on_report`]
455/// returns `Some` only when the decision *changes*, so the receiver sends a control
456/// message on a level crossing, not every tick.
457#[derive(Debug)]
458pub struct QualityRequestDecider {
459    est: DownlinkEstimator,
460    /// Request the low layer only once the estimate falls BELOW this. Set WELL under
461    /// source (source/4), because asking for the low layer is a heavy hammer: the
462    /// shared low layer is a small fixed-class stream, so a receiver should only want
463    /// it when its link genuinely cannot carry a sender-adapted source. A
464    /// moderately-constrained link (say 60-80% of source) must KEEP taking source and
465    /// let the sender's own loss-based adaptation right-size that one stream — a
466    /// threshold near source turns a minor dip into a resolution cliff.
467    drop_below: u64,
468    /// Restore the Source request once the estimate climbs back ABOVE this (source/2).
469    /// The wide `[drop_below, restore_above]` band is the anti-flap hysteresis: a
470    /// narrow band lets ordinary estimate wander flap Source↔Capped.
471    restore_above: u64,
472    current: QualityRequest,
473}
474
475impl QualityRequestDecider {
476    /// Seed assuming we can take the source (the optimistic default — a fresh
477    /// subscriber asks for nothing until its link proves it can't keep up).
478    ///
479    /// NOTE (design): the receiver can only report ITS capacity; whether a separate
480    /// low layer actually *helps* is a GLOBAL decision (do the receivers diverge?) the
481    /// producer makes from the aggregate. This decider is deliberately conservative so
482    /// it caps only genuinely-poor links — see the crate-level simulcast notes.
483    pub fn new(source_bps: u64) -> Self {
484        Self {
485            // The estimator can't usefully probe above what the source sends.
486            est: DownlinkEstimator::new(source_bps, source_bps),
487            drop_below: source_bps / 4,
488            restore_above: source_bps / 2,
489            current: QualityRequest::Source,
490        }
491    }
492
493    /// Fold one downlink reception report; returns `Some(req)` only when the request
494    /// the receiver should send *changes* (a level crossing). `received`/`lost` are
495    /// the sequence-tracker deltas; `now_us` the receive clock.
496    pub fn on_report(&mut self, received: u32, lost: u32, now_us: i64) -> Option<QualityRequest> {
497        let estimate = self.est.on_report(received, lost, now_us);
498        let next = match self.current {
499            // Currently taking source: only down-shift once clearly below source.
500            QualityRequest::Source if estimate < self.drop_below => {
501                QualityRequest::Capped(estimate)
502            }
503            QualityRequest::Source => return None, // holding source
504            // Currently capped: restore source once the estimate clears the (wide)
505            // hysteresis band — well before full source, so a recovered-but-moderate
506            // link rejoins source and lets the sender adapt rather than staying pinned
507            // to the low layer.
508            QualityRequest::Capped(_) if estimate >= self.restore_above => QualityRequest::Source,
509            // Still capped (and below source): track the estimate so the producer's
510            // shared-low coalescing follows our true capacity, but only re-emit on a
511            // meaningful (≥ ~10%) move to avoid chatter.
512            QualityRequest::Capped(prev) => {
513                let moved = prev.abs_diff(estimate) * 10 > prev;
514                if moved {
515                    QualityRequest::Capped(estimate)
516                } else {
517                    return None;
518                }
519            }
520        };
521        if next == self.current {
522            return None;
523        }
524        self.current = next;
525        Some(next)
526    }
527
528    /// Fold one frame's arrival into the delay-based half of the estimate (see
529    /// [`DownlinkEstimator::on_arrival`]). Call per received frame; the loss-based
530    /// [`Self::on_report`] still drives the level-crossing decision, but a building
531    /// queue sensed here lowers the estimate it reads, so the receiver backs off
532    /// *before* the link starts dropping frames.
533    pub fn on_arrival(&mut self, recv_us: i64, send_us: i64) {
534        self.est.on_arrival(recv_us, send_us);
535    }
536
537    /// The current standing request.
538    pub fn current(&self) -> QualityRequest {
539        self.current
540    }
541
542    /// The current delay-based congestion verdict (for stats/debug).
543    pub fn delay_usage(&self) -> crate::bwe_delay::Usage {
544        self.est.delay_usage()
545    }
546}

565/// Maps available bitrate to a [`Rung`] with hysteresis and a manual ceiling.
566#[derive(Debug)]
567pub struct QualityLadder {
568    /// Rungs sorted ascending by bitrate.
569    rungs: Vec<Rung>,
570    current: usize,
571    /// Whether the last [`Self::update`] decided [`Quality::AudioOnly`] (the
572    /// available bitrate fell below the lowest rung). Kept so [`Self::current`] is
573    /// a faithful projection of the last decision rather than silently reporting
574    /// `Video(rungs[0])` while video is actually dropped.
575    audio_only: bool,
576    /// Manual maximum rung index (inclusive); `None` = no cap.
577    ceiling: Option<usize>,
578    /// Headroom required to climb a rung (e.g. 1.15 = need 15% above its cost).
579    climb_margin: f64,
580}

620    /// 1080p60, i.e. `screen_for(ScreenQuality::default())`: 720p15 → 1080p15 →
621    /// 1080p30 → 1080p60 (climb resolution before fps; on the way down, drop fps
622    /// before resolution). Kept as a named constructor for callers that don't
623    /// expose the per-stream control; [`Self::screen_for`] is the configurable one.
624    pub fn screen() -> Self {
625        Self::screen_for(ScreenQuality::default())
626    }

628    /// Build a screen-share ladder for a per-stream [`ScreenQuality`] preference
629    /// (degrade priority + resolution/fps caps). See [`screen_rungs`] for the rung
630    /// construction; this just wraps the result. Pair with [`Self::reconfigure`]
631    /// to change the preference on a live stream without losing the bitrate
632    /// position.
633    pub fn screen_for(quality: ScreenQuality) -> Self {
634        Self::new(screen_rungs(quality))
635    }
636
637    /// Swap in a new rung set (e.g. from a mid-stream [`ScreenQuality`] change)
638    /// while preserving the *bitrate position*: the next [`Self::update`] re-picks
639    /// from the live available bitrate, so the only state to carry is "are we
640    /// paused (audio-only)". `current` is clamped into the new ladder; a stale
641    /// ceiling index is dropped (the new ladder's caps are baked into its rungs).
642    pub fn reconfigure(&mut self, rungs: Vec<Rung>) {
643        assert!(!rungs.is_empty(), "ladder needs at least one rung");
644        self.current = self.current.min(rungs.len() - 1);
645        self.ceiling = None;
646        self.rungs = rungs;
647    }
648
649    /// The BWE estimate at which the *top* rung becomes selectable: the top rung's
650    /// bitrate plus the [`Self::update`] climb-margin headroom. The rate controller
651    /// seeds its ceiling from this so it can actually climb to the richest rung on a
652    /// fast link. Without it a flat BWE cap below the top rung × margin would strand
653    /// the high rungs entirely — e.g. a 1440p120 top needs ~20 Mbps, so a 4 Mbps cap
654    /// would top out at 1080p15 and never offer the quality a gaming LAN can carry.
655    pub fn bwe_ceiling_bps(&self) -> u64 {
656        let top = self.rungs.last().map(|r| r.bitrate_bps).unwrap_or(0);
657        (top as f64 * self.climb_margin).ceil() as u64
658    }
659
660    /// `rungs` must be non-empty and sorted ascending by bitrate.
661    pub fn new(rungs: Vec<Rung>) -> Self {
662        assert!(!rungs.is_empty(), "ladder needs at least one rung");
663        Self {
664            rungs,
665            current: 0,
666            audio_only: false,
667            ceiling: None,
668            climb_margin: 1.15,
669        }
670    }
671
672    /// Set the manual ceiling by rung index (clamped). `None` removes the cap.
673    pub fn set_ceiling(&mut self, idx: Option<usize>) {
674        self.ceiling = idx.map(|i| i.min(self.rungs.len() - 1));
675    }
676
677    /// Pick the quality for the given available bitrate. Drops as many rungs as
678    /// needed immediately, but climbs at most one rung per call and only with
679    /// `climb_margin` headroom (and never above the ceiling).
680    pub fn update(&mut self, available_bps: u64) -> Quality {
681        let max_idx = self
682            .ceiling
683            .unwrap_or(self.rungs.len() - 1)
684            .min(self.rungs.len() - 1);
685
686        if available_bps < self.rungs[0].bitrate_bps {
687            self.current = 0;
688            self.audio_only = true;
689            return Quality::AudioOnly;
690        }
691        self.audio_only = false;
692        if self.current > max_idx {
693            self.current = max_idx;
694        }
695        // Drop fast: step down until the current rung fits.
696        while self.current > 0 && self.rungs[self.current].bitrate_bps > available_bps {
697            self.current -= 1;
698        }
699        // Climb slow: a single step, only with headroom and under the ceiling.
700        if self.current < max_idx {
701            let next = self.rungs[self.current + 1];
702            if available_bps as f64 >= next.bitrate_bps as f64 * self.climb_margin {
703                self.current += 1;
704            }
705        }
706        Quality::Video(self.rungs[self.current])
707    }
708
709    /// The quality from the last [`Self::update`] (a faithful projection: reports
710    /// [`Quality::AudioOnly`] if that was the last decision). Before any update,
711    /// reports the lowest video rung (the starting state).
712    pub fn current(&self) -> Quality {
713        if self.audio_only {
714            Quality::AudioOnly
715        } else {
716            Quality::Video(self.rungs[self.current])
717        }
718    }
719}

723/// Which dimension a screen-share ladder sacrifices **first** as bandwidth drops
724/// — a per-stream user choice, switchable on a live stream (not asked at stream
725/// start). Gameplay wants motion; an IDE/slides wants legibility.
726#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
727pub enum DegradePriority {
728    /// Hold frame rate, drop **resolution** first — smooth motion over sharpness
729    /// (gameplay, video). On the way up: climb fps at low res, then climb res.
730    Motion,
731    /// Hold resolution, drop **frame rate** first — readable detail over smooth
732    /// motion (text, IDEs, slides). On the way up: climb res, then climb fps.
733    #[default]
734    Legibility,
735}

737/// The user's maximum-resolution cap for a stream (the "Source / 1440p / 1080p /
738/// 720p / 480p" control). `Source` imposes no cap beyond the capture's own
739/// geometry — the session still clamps it to the encoder's native target.
740#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
741pub enum MaxResolution {
742    #[default]
743    Source,
744    P1440,
745    P1080,
746    P720,
747    P480,

750impl MaxResolution {

756            MaxResolution::P1440 => 1440,

764/// One stream's video-quality preference, changeable mid-stream. Drives
765/// [`QualityLadder::screen_for`] / [`QualityLadder::reconfigure`]. The default
766/// is legibility-first, capped at 1080p60.
767#[derive(Debug, Clone, Copy, PartialEq, Eq)]
768pub struct ScreenQuality {
769    pub degrade: DegradePriority,
770    pub max_resolution: MaxResolution,
771    /// Frame-rate cap; the UI offers 30 / 60 / 120.
772    pub max_fps: u32,
773}

775impl Default for ScreenQuality {
776    fn default() -> Self {

779            max_resolution: MaxResolution::P1080,
780            max_fps: 60,

785/// Candidate ladder heights (16:9 luma height), ascending.
786const LADDER_HEIGHTS: [u32; 4] = [480, 720, 1080, 1440];
787/// Candidate ladder frame rates, ascending.
788const LADDER_FPS: [u32; 4] = [15, 30, 60, 120];
789
790/// Make `v` even (encoders want even luma dimensions).
791const fn even(v: u32) -> u32 {
792    v & !1
793}
794
795/// The 16:9 luma width for a height (even). Descriptive: the encoder fits the
796/// *source* aspect, so this is the reference geometry the rung reports.
797const fn width_16x9(height: u32) -> u32 {
798    even(height * 16 / 9)
799}
800
801/// Target bitrate (bits/s) for one screen rung at `height`×`fps`. A single
802/// model — `pixels × fps^0.72 × k`, rounded to 100 kbps — instead of a hand-tuned
803/// table per (res, fps), staying monotonic in both axes (the ladder ordering relies
804/// on it).
805///
806/// **Calibrated for SCREEN content, not camera video.** Screen content (text, UI,
807/// large static regions, sharp intra-friendly edges) compresses several times better
808/// than motion video; a camera-video rate table (1080p30 → ~8 Mbps) oversizes a
809/// screenshare and saturates the shared connection, starving everything else. `k` is
810/// set so 1080p30 ≈ 2.4 Mbps — in line with production-RTC screenshare rates over
811/// codecs less efficient than H.265, and ample for crisp text. Rungs: 720p15≈0.6M,
812/// 1080p15≈1.5M, 1080p30≈2.4M, 1080p60≈4.0M. The BWE ceiling tracks the top rung,
813/// so a fast link reaches it.
814fn screen_bitrate_bps(height: u32, fps: u32) -> u64 {
815    // 0.10 solves 1080p30 → ~2.4 Mbps (screen-content target).
816    const K: f64 = 0.10;
817    let pixels = (width_16x9(height) as f64) * (height as f64);
818    let raw = pixels * (fps as f64).powf(0.72) * K;
819    let rounded = (raw / 100_000.0).round() * 100_000.0;
820    (rounded as u64).max(100_000)
821}

823/// Build the screen-share rungs for a [`ScreenQuality`], **sans-IO and pure**.
824///
825/// Candidate heights/fps are filtered to the caps, then assembled into one
826/// ascending-bitrate ladder whose *order of climb* encodes the degrade priority:
827/// * [`DegradePriority::Legibility`] climbs resolution at the floor fps, then fps
828///   at the cap resolution — so descending drops fps before resolution. Floor
829///   resolution is 720p (below that text is unreadable) unless the cap is lower.
830/// * [`DegradePriority::Motion`] climbs fps at the floor resolution, then
831///   resolution at the cap fps — so descending drops resolution before fps. Floor
832///   fps is 30 (below that it isn't "motion") unless the cap is lower; floor
833///   resolution is 480p.
834///
835/// The result is sorted ascending by bitrate (the ladder invariant) and
836/// deduplicated, so callers get a clean monotonic ladder for any cap combination.
837pub fn screen_rungs(quality: ScreenQuality) -> Vec<Rung> {
838    let cap_h = quality.max_resolution.height_cap();
839    let (floor_h, floor_fps) = match quality.degrade {
840        DegradePriority::Legibility => (720, 15),
841        DegradePriority::Motion => (480, 30),
842    };
843
844    // Heights ≤ cap and ≥ the priority's floor — but always keep the cap itself,
845    // so a cap below the floor (e.g. legibility @ 480p) still yields one height.
846    let heights: Vec<u32> = LADDER_HEIGHTS
847        .iter()
848        .copied()
849        .filter(|h| *h <= cap_h && (*h >= floor_h || *h >= cap_h))
850        .collect();
851    let heights = if heights.is_empty() {
852        vec![LADDER_HEIGHTS[0].min(cap_h)]
853    } else {
854        heights
855    };
856    // Frame rates ≤ cap and ≥ the priority's floor — always keep at least one.
857    let fpses: Vec<u32> = LADDER_FPS
858        .iter()
859        .copied()
860        .filter(|f| *f <= quality.max_fps && *f >= floor_fps)
861        .collect();
862    let fpses = if fpses.is_empty() {
863        vec![floor_fps.min(quality.max_fps)]
864    } else {
865        fpses
866    };
867
868    let max_h = *heights.last().unwrap();
869    let min_h = *heights.first().unwrap();
870    let max_f = *fpses.last().unwrap();
871    let min_f = *fpses.first().unwrap();
872
873    // (height, fps) levels in climb order per priority.
874    let mut levels: Vec<(u32, u32)> = Vec::new();
875    match quality.degrade {
876        DegradePriority::Legibility => {
877            for &h in &heights {
878                levels.push((h, min_f));
879            }
880            for &f in fpses.iter().skip(1) {
881                levels.push((max_h, f));
882            }
883        }
884        DegradePriority::Motion => {
885            for &f in &fpses {
886                levels.push((min_h, f));
887            }
888            for &h in heights.iter().skip(1) {
889                levels.push((h, max_f));
890            }
891        }
892    }
893
894    let mut rungs: Vec<Rung> = levels
895        .into_iter()
896        .map(|(h, f)| Rung {
897            bitrate_bps: screen_bitrate_bps(h, f),
898            width: width_16x9(h),
899            height: h,
900            fps: f,
901        })
902        .collect();
903    rungs.sort_by_key(|r| r.bitrate_bps);
904    // Bitrate is a pure function of (height, fps), so equal bitrate ⇒ the same
905    // bandwidth target; dedup on it to keep the ladder STRICTLY ascending — the
906    // `QualityLadder` climb/drop hysteresis compares bitrates and `new` documents
907    // an ascending ladder.
908    rungs.dedup_by_key(|r| r.bitrate_bps);
909    rungs
910}

1273    /// Closed-loop: model loss as the fraction by which the offered target
1274    /// exceeds the link capacity, feed it back, and assert the controller
1275    /// converges near capacity, then backs off when capacity is cut and recovers
1276    /// when it returns.
1277    #[test]
1278    fn converges_to_capacity_and_reacts_to_changes() {
1279        let mut c = RateController::new(RateConfig {
1280            min_bps: 100_000,
1281            max_bps: 4_000_000,
1282            start_bps: 300_000,
1283            ..Default::default()
1284        });
1285        let mut ladder = QualityLadder::standard();
1286
1287        let observe = |target: u64, capacity: u64| -> f64 {
1288            if target <= capacity {
1289                0.0
1290            } else {
1291                (target - capacity) as f64 / target as f64
1292            }
1293        };
1294
1295        // Capacity ~1.2 Mbps; run for a while.
1296        let mut now = 0i64;

1300            c.on_feedback(loss, now);

1302            now += 1_000_000;
1303        }

1306            t > 900_000 && t < 1_400_000,
1307            "did not converge near cap: {t}"
1308        );

1310        assert_eq!(
1311            ladder.current(),
1312            Quality::Video(Rung {
1313                bitrate_bps: 800_000,
1314                width: 960,
1315                height: 540,
1316                fps: 30
1317            })
1318        );
1319
1320        // Capacity collapses to 250 kbps → fast back-off.
1321        let cap = 250_000;
1322        for _ in 0..10 {
1323            let loss = observe(c.target_bps(), cap);

1762        assert!(p.set_request(1, QualityRequest::Capped(400_000)));
1763        assert!(p.set_request(2, QualityRequest::Capped(300_000)));
1764        assert_eq!(
1765            p.desired(),
1766            LayerPlan {
1767                send_source: false,
1768                low_layer_bps: Some(300_000)
1769            }
1770        );
1771        assert_eq!(p.plan(), LayerPlan::SOURCE_ONLY);
1772
1773        // Enabling actuation flips the effective plan to the desired one (and reports
1774        // the change so the caller re-publishes the seam).
1775        assert!(p.set_actuation(true));
1776        assert_eq!(p.plan(), p.desired());
1777        // Idempotent.
1778        assert!(!p.set_actuation(true));
1779        // Disabling clamps back to source-only and reports the change.
1780        assert!(p.set_actuation(false));
1781        assert_eq!(p.plan(), LayerPlan::SOURCE_ONLY);
1782    }

1784    /// The receiver decider holds `Source` on a clean link, drops to `Capped` only
1785    /// after sustained loss, recovers to `Source` when the link clears, and emits a
1786    /// request only on a change (level-triggered).
1787    #[test]
1788    fn quality_request_decider_hysteresis() {
1789        let mut d = QualityRequestDecider::new(SRC);
1790        assert_eq!(d.current(), QualityRequest::Source);
1791
1792        // Clean link: no change emitted (we keep taking source).
1793        let mut now = 0i64;
1794        for _ in 0..10 {
1795            now += 1_000_000;
1796            assert_eq!(d.on_report(30, 0, now), None, "clean link holds source");
1797        }
1798
1799        // Sustained heavy loss → eventually drops to a Capped request (once).
1800        let mut dropped_to = None;
1801        for _ in 0..10 {
1802            now += 1_000_000;
1803            if let Some(QualityRequest::Capped(bps)) = d.on_report(20, 12, now) {
1804                dropped_to = Some(bps);
1805                break;
1806            }
1807        }
1808        let cap = dropped_to.expect("heavy loss must produce a Capped request");
1809        assert!(cap < SRC, "the cap is below source: {cap}");
1810        assert!(matches!(d.current(), QualityRequest::Capped(_)));
1811
1812        // Link clears: the estimate climbs back; eventually restores Source (once).
1813        let mut restored = false;
1814        for _ in 0..120 {
1815            now += 1_000_000;
1816            if d.on_report(30, 0, now) == Some(QualityRequest::Source) {
1817                restored = true;
1818                break;
1819            }
1820        }
1821        assert!(restored, "a recovered link must restore the Source request");
1822        assert_eq!(d.current(), QualityRequest::Source);
1823    }
