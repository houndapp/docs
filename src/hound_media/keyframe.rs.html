1//! # Keyframe requests (PLI) and keyframe-generation throttling
2//!
3//! Sans-IO building blocks for the "ask for a fresh keyframe" recovery path —
4//! the complement to NACK ([`crate::recovery`]). When loss can't be repaired by
5//! retransmission (the NACK generator gave up, a frame failed to reassemble, or
6//! a receiver joined mid-stream), the decoder is stuck until the next
7//! **keyframe**. The receiver asks the producer for one with a Picture Loss
8//! Indication (PLI); the SFU relays it; the producer decides whether to force a
9//! keyframe.
10//!
11//! * [`KeyframeRequester`] (receiver) — tracks whether we are waiting for a
12//!   decodable keyframe and rate-limits PLIs, so one in-flight request (RTT +
13//!   encode time) does not trigger a storm before the keyframe can arrive.
14//! * [`KeyframeThrottle`] (producer) — coalesces a burst of PLIs (e.g. every
15//!   receiver after one shared loss) into a single keyframe and enforces a
16//!   minimum spacing between forced keyframes so the bitrate does not spike.
17//!
18//! Both are pure and clock-driven by the caller, so they are deterministic and
19//! testable. One [`KeyframeRequester`] exists per `(sender, track)` we receive;
20//! one [`KeyframeThrottle`] per track we produce.
21
22/// Tunables for [`KeyframeRequester`].
23#[derive(Debug, Clone)]
24pub struct KeyframeConfig {
25    /// Minimum spacing between PLIs for the same stream while still waiting —
26    /// roughly one RTT plus encode time, so we don't re-ask before a keyframe
27    /// sent in response could plausibly have arrived.
28    pub pli_interval_us: i64,
29}
30
31impl Default for KeyframeConfig {
32    fn default() -> Self {
33        Self {
34            pli_interval_us: 1_000_000, // 1 s
35        }
36    }
37}
38
39/// Receiver-side: decides when to send a PLI (keyframe request) for one stream.
40#[derive(Debug)]
41pub struct KeyframeRequester {
42    cfg: KeyframeConfig,
43    waiting: bool,
44    last_request_us: Option<i64>,
45}
46
47impl KeyframeRequester {
48    pub fn new(cfg: KeyframeConfig) -> Self {
49        Self {
50            cfg,
51            waiting: false,
52            last_request_us: None,
53        }
54    }
55
56    /// Mark the stream undecodable until a keyframe — e.g. NACK gave up on a
57    /// missing packet, a frame failed to reassemble, or we just subscribed.
58    /// Idempotent: calling it again while already waiting does not reset the
59    /// PLI cadence (so repeated loss reports don't cause a storm).
60    pub fn request_keyframe(&mut self) {
61        self.waiting = true;
62    }
63
64    /// A decodable keyframe arrived — clears the waiting state and the cadence.
65    pub fn on_keyframe(&mut self) {
66        self.waiting = false;
67        self.last_request_us = None;
68    }
69
70    /// Whether we currently need a keyframe.
71    pub fn needs_keyframe(&self) -> bool {
72        self.waiting
73    }
74
75    /// Returns `true` at most once per `pli_interval_us` while waiting, meaning
76    /// "emit a PLI now". The caller attaches the `(sender, track)` context and
77    /// forwards it (via the SFU) to the producer.
78    pub fn poll_pli(&mut self, now_us: i64) -> bool {
79        if !self.waiting {
80            return false;
81        }
82        let due = match self.last_request_us {
83            None => true,
84            Some(last) => now_us - last >= self.cfg.pli_interval_us,
85        };
86        if due {
87            self.last_request_us = Some(now_us);
88        }
89        due
90    }
91}

93/// Tunables for [`KeyframeThrottle`].
94#[derive(Debug, Clone)]
95pub struct KeyframeThrottleConfig {
96    /// Minimum spacing between forced keyframes. A burst of PLIs within this
97    /// window collapses to a single keyframe; later requests wait. Protects the
98    /// bitrate (keyframes are large) when many receivers ask at once.
99    pub min_interval_us: i64,
100}
101
102impl Default for KeyframeThrottleConfig {
103    fn default() -> Self {
104        Self {
105            min_interval_us: 1_000_000, // 1 s
106        }
107    }
108}
109
110/// Producer-side: coalesces incoming PLIs and decides when to actually force a
111/// keyframe, respecting a minimum interval.
112#[derive(Debug)]
113pub struct KeyframeThrottle {
114    cfg: KeyframeThrottleConfig,
115    pending: bool,
116    last_keyframe_us: Option<i64>,
117}
118
119impl KeyframeThrottle {
120    pub fn new(cfg: KeyframeThrottleConfig) -> Self {
121        Self {
122            cfg,
123            pending: false,
124            last_keyframe_us: None,
125        }
126    }
127
128    /// A PLI arrived (from any receiver, relayed by the SFU). Coalesced — many
129    /// PLIs before the next [`Self::should_generate`] still mean one keyframe.
130    pub fn on_pli(&mut self) {
131        self.pending = true;
132    }
133
134    /// Returns `true` exactly once when a forced keyframe is due: there is a
135    /// pending request and at least `min_interval_us` has elapsed since the last
136    /// keyframe. On `true` it records the keyframe time and clears `pending`, so
137    /// a burst of PLIs yields a single keyframe.
138    pub fn should_generate(&mut self, now_us: i64) -> bool {
139        if !self.pending {
140            return false;
141        }
142        let allowed = match self.last_keyframe_us {
143            None => true,
144            Some(last) => now_us - last >= self.cfg.min_interval_us,
145        };
146        if allowed {
147            self.last_keyframe_us = Some(now_us);
148            self.pending = false;
149        }
150        allowed
151    }
152
153    /// Call whenever a keyframe is emitted for any reason — a forced one, or the
154    /// encoder's own periodic keyframe. Resets the interval and clears `pending`:
155    /// a keyframe that just went out satisfies any outstanding request, so we
156    /// don't immediately force another.
157    pub fn on_keyframe_emitted(&mut self, now_us: i64) {
158        self.last_keyframe_us = Some(now_us);
159        self.pending = false;
160    }
161}

168    fn requester_silent_until_loss() {
169        let mut r = KeyframeRequester::new(KeyframeConfig::default());

174    #[test]
175    fn requester_rate_limits_then_reasks() {
176        let cfg = KeyframeConfig {
177            pli_interval_us: 1_000_000,
178        };
179        let mut r = KeyframeRequester::new(cfg);
180        r.request_keyframe();
181        assert!(r.needs_keyframe());
182
183        // First poll emits immediately.
184        assert!(r.poll_pli(0));
185        // Within the interval: suppressed.
186        assert!(!r.poll_pli(500_000));
187        assert!(!r.poll_pli(999_999));
188        // At/after the interval: re-ask (keyframe still hasn't arrived).
189        assert!(r.poll_pli(1_000_000));
190        assert!(!r.poll_pli(1_500_000));
191    }

331    #[test]
332    fn throttle_coalesces_plis_arriving_during_the_interval() {
333        // PLIs that arrive while we're inside the min-interval all collapse into
334        // the single keyframe generated when the interval elapses.
335        let mut t = KeyframeThrottle::new(KeyframeThrottleConfig {
336            min_interval_us: 1_000_000,
337        });
338        t.on_pli();
339        assert!(t.should_generate(0)); // first keyframe at t=0
340        // A storm of PLIs over the next second.
341        for ts in [100_000, 300_000, 700_000, 999_000] {
342            t.on_pli();
343            assert!(!t.should_generate(ts), "held within the interval");
344        }
345        // Exactly one keyframe emerges at the interval, then quiet.
346        assert!(t.should_generate(1_000_000));
347        assert!(
348            !t.should_generate(1_000_001),
349            "no second keyframe for the storm"
350        );
351    }

353    /// End-to-end: a receiver detects unrecoverable loss, PLIs are relayed and
354    /// coalesced at the producer, one keyframe is generated, and the receiver
355    /// stops asking once it arrives.
356    #[test]
357    fn end_to_end_pli_to_keyframe() {
358        let mut rx_a = KeyframeRequester::new(KeyframeConfig::default());
359        let mut rx_b = KeyframeRequester::new(KeyframeConfig::default());
360        let mut producer = KeyframeThrottle::new(KeyframeThrottleConfig::default());
361
362        // Shared loss at t=0: both receivers need a keyframe.
363        rx_a.request_keyframe();
364        rx_b.request_keyframe();
365
366        // They emit PLIs; the SFU relays both to the producer.
367        let now = 0;
368        if rx_a.poll_pli(now) {
369            producer.on_pli();
370        }
371        if rx_b.poll_pli(now) {
372            producer.on_pli();
373        }
374
375        // Producer forces exactly one keyframe for the burst.
376        assert!(producer.should_generate(now));
377        assert!(!producer.should_generate(now));
378
379        // The keyframe arrives at both receivers; they stop asking.
380        rx_a.on_keyframe();
381        rx_b.on_keyframe();
382        assert!(!rx_a.needs_keyframe());
383        assert!(!rx_b.needs_keyframe());
384        assert!(!rx_a.poll_pli(5_000_000));
385        assert!(!rx_b.poll_pli(5_000_000));
386    }
