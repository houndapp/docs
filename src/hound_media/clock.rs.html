1//! # Capture clock
2//!
3//! One monotonic timestamp source per session, shared by **both** the audio and
4//! video capture paths. Every captured frame is stamped from the same
5//! [`CaptureClock`], so audio and video timestamps live on one timeline and the
6//! receiver can align them (A/V sync).
7//!
8//! The shared epoch is load-bearing: stamping each stream from its own capture
9//! API puts audio and video on unrelated epochs, and timestamps from unrelated
10//! epochs can never be compared or synced.
11//!
12//! ## Design
13//!
14//! The pure, testable core is [`CaptureClock::stamp`], which converts a
15//! monotonic [`Instant`] reading into microseconds since the session epoch —
16//! the value carried in [`crate::packet::PacketHeader::timestamp_us`]. Tests
17//! pass synthetic instants (`epoch + Duration`); production passes
18//! `Instant::now()` (or uses the [`CaptureClock::now`] convenience).
19//!
20//! The clock is `Copy`, so both capture tasks hold the *same* epoch and produce
21//! comparable stamps — the type makes "one clock" the easy path.
22
23use std::time::{Duration, Instant};
24
25/// A monotonic, microsecond capture clock anchored at a session epoch.
26#[derive(Debug, Clone, Copy)]
27pub struct CaptureClock {
28    epoch: Instant,
29}
30
31impl CaptureClock {
32    /// Start a clock whose epoch is now. Create one per call session and share
33    /// it (it is `Copy`) with every capture path.
34    pub fn start() -> Self {
35        Self {
36            epoch: Instant::now(),
37        }
38    }
39
40    /// Construct a clock with an explicit epoch (used in tests).
41    pub fn from_epoch(epoch: Instant) -> Self {
42        Self { epoch }
43    }
44
45    /// The session epoch.
46    pub fn epoch(&self) -> Instant {
47        self.epoch
48    }
49
50    /// Microseconds from the epoch to `at`. Saturates at 0 for instants before
51    /// the epoch (impossible with a monotonic source, but defensive). Saturates
52    /// at `i64::MAX` rather than overflowing (≈292,000 years — never reached).
53    pub fn stamp(&self, at: Instant) -> i64 {
54        i64::try_from(at.saturating_duration_since(self.epoch).as_micros()).unwrap_or(i64::MAX)
55    }
56
57    /// Stamp the current instant. Convenience over `stamp(Instant::now())`; this
58    /// is the only method that reads the real clock.
59    pub fn now(&self) -> i64 {
60        self.stamp(Instant::now())
61    }
62
63    /// Convert a microsecond stamp produced by this clock back into an
64    /// [`Instant`] on the local timeline (e.g. for a playout deadline).
65    pub fn instant_of(&self, timestamp_us: i64) -> Instant {
66        self.epoch + Duration::from_micros(timestamp_us.max(0) as u64)
67    }
68}

70#[cfg(test)]
71mod tests {
72    use super::*;
73
74    #[test]
75    fn epoch_stamps_to_zero() {
76        let epoch = Instant::now();
77        let c = CaptureClock::from_epoch(epoch);
78        assert_eq!(c.stamp(epoch), 0);
79    }
80
81    #[test]
82    fn monotonic_advances_in_microseconds() {
83        let epoch = Instant::now();
84        let c = CaptureClock::from_epoch(epoch);
85        assert_eq!(c.stamp(epoch + Duration::from_millis(20)), 20_000);
86        assert_eq!(c.stamp(epoch + Duration::from_secs(1)), 1_000_000);
87        assert!(
88            c.stamp(epoch + Duration::from_millis(40)) > c.stamp(epoch + Duration::from_millis(20))
89        );
90    }
91
92    #[test]
93    fn audio_and_video_share_one_timeline() {
94        // The whole point: a clock cloned to two capture paths yields comparable
95        // stamps, so a video frame and an audio frame captured 10 ms apart are
96        // exactly 10 ms apart in their timestamps.
97        let epoch = Instant::now();
98        let video_clock = CaptureClock::from_epoch(epoch);
99        let audio_clock = video_clock; // Copy — same epoch
100
101        let t = epoch + Duration::from_millis(100);
102        let video_ts = video_clock.stamp(t);
103        let audio_ts = audio_clock.stamp(t + Duration::from_millis(10));
104        assert_eq!(audio_ts - video_ts, 10_000);
105    }
106
107    #[test]
108    fn before_epoch_saturates_to_zero() {
109        let epoch = Instant::now();
110        let c = CaptureClock::from_epoch(epoch + Duration::from_secs(5));
111        assert_eq!(c.stamp(epoch), 0); // `at` is before the epoch
112    }
113
114    #[test]
115    fn instant_of_round_trips() {
116        let epoch = Instant::now();
117        let c = CaptureClock::from_epoch(epoch);
118        let at = epoch + Duration::from_millis(250);
119        let ts = c.stamp(at);
120        assert_eq!(c.instant_of(ts), at);
121    }
122
123    #[test]
124    fn very_large_offset_saturates_at_i64_max() {
125        // An absurd offset (past i64::MAX microseconds) saturates instead of
126        // wrapping/panicking. ~292,000 years — never reached in practice.
127        let epoch = Instant::now();
128        let c = CaptureClock::from_epoch(epoch);
129        // u128 micros for ~300_000 years exceeds i64::MAX; from_secs keeps it
130        // representable as a Duration while overflowing i64 micros.
131        let huge = epoch + Duration::from_secs(400_000 * 365 * 24 * 3600);
132        assert_eq!(c.stamp(huge), i64::MAX, "must saturate, not overflow");
133    }
134
135    #[test]
136    fn instant_of_clamps_negative_timestamps_to_epoch() {
137        // A negative stamp (unsynced clock) maps to the epoch, not a panic from
138        // subtracting a Duration below the epoch.
139        let epoch = Instant::now();
140        let c = CaptureClock::from_epoch(epoch);
141        assert_eq!(c.instant_of(-1_000_000), epoch);
142        assert_eq!(c.instant_of(i64::MIN), epoch);
143    }
144
145    #[test]
146    fn stamp_and_instant_of_are_inverse_for_nonnegative() {
147        let epoch = Instant::now();
148        let c = CaptureClock::from_epoch(epoch);
149        for ms in [0u64, 1, 250, 1_000, 60_000] {
150            let at = epoch + Duration::from_millis(ms);
151            assert_eq!(c.instant_of(c.stamp(at)), at);
152        }
153    }
154
155    #[test]
156    fn now_is_nonnegative_and_increases() {
157        let c = CaptureClock::start();
158        let a = c.now();
159        let b = c.now();
160        assert!(a >= 0 && b >= a);
161    }
