1//! # Latency histogram (sans-IO)
2//!
3//! A fixed-bucket histogram for per-stage media latency, the measurement
4//! primitive the pipeline instruments each stage with (decode time, jitter-buffer
5//! dwell, capture→encode, publish-queue, the clock-synced network leg).
6//! Percentiles, not means, because the **tail** is what
7//! causes stutter: a p95 dwell spike matters even when the mean looks fine.
8//!
9//! Deterministic and pure — `record(us)` slots a sample into a log-spaced bucket,
10//! `percentile(q)` interpolates within the bucket holding the q-th sample, and
11//! `drain_percentiles` reads + clears for a windowed (e.g. 1 Hz) emit. No clock, no
12//! atomics, no I/O: the IO shell wraps it (a `Mutex` at frame cadence is free) and
13//! feeds it durations. Bucket bounds span 0.25 ms … 2 s — fine resolution where
14//! real-time stages live (1–100 ms), coarse in the tail where only "it's bad" matters.
15
16/// Upper bounds (µs) of each bucket, ascending. A sample lands in the first bucket
17/// whose bound is ≥ it; anything larger saturates the top bucket. DENSE through the
18/// 10–150 ms band where the real-time stages and frame-interval cushions live
19/// (16/33/67 ms boundaries among them), so a value clustered at a frame interval
20/// reads back tightly rather than smearing across a wide bucket on interpolation;
21/// coarse in the tail where only "it's bad" matters.
22const BOUNDS_US: [u64; 25] = [
23    250, 500, 1_000, 2_000, 4_000, 6_000, 8_000, 12_000, 16_000, 20_000, 24_000, 28_000, 33_000,
24    40_000, 50_000, 67_000, 85_000, 100_000, 125_000, 150_000, 250_000, 500_000, 1_000_000,
25    2_000_000, 4_000_000,
26];
27
28/// A fixed-bucket latency histogram over microsecond samples. One per
29/// instrumented stage per stream; the stats layer drains it on the 1 Hz emit tick.
30#[derive(Debug, Clone)]
31pub struct LatencyHistogram {
32    /// One count per `BOUNDS_US` bucket (same length).
33    buckets: [u64; BOUNDS_US.len()],
34    count: u64,
35}
36
37impl Default for LatencyHistogram {
38    fn default() -> Self {
39        Self::new()
40    }
41}
42
43impl LatencyHistogram {
44    pub const fn new() -> Self {
45        Self {
46            buckets: [0; BOUNDS_US.len()],
47            count: 0,
48        }
49    }
50
51    /// Record one latency sample. Samples above the top bound saturate the top
52    /// bucket (so a pathological stall reads as "≥ 4 s", never lost).
53    pub fn record(&mut self, us: u64) {
54        let idx = BOUNDS_US
55            .iter()
56            .position(|&b| us <= b)
57            .unwrap_or(BOUNDS_US.len() - 1);
58        self.buckets[idx] += 1;
59        self.count += 1;
60    }
61
62    /// Number of samples recorded since the last [`clear`](Self::clear).
63    pub fn count(&self) -> u64 {
64        self.count
65    }
66
67    /// The `q`-quantile (`q` in `[0, 1]`) in microseconds, linearly interpolated
68    /// within the bucket that holds the q-th sample. `0` when empty. Returns the
69    /// top bound for the saturating bucket (no extrapolation past it).
70    pub fn percentile(&self, q: f64) -> u64 {
71        if self.count == 0 {
72            return 0;
73        }
74        let target = (q.clamp(0.0, 1.0) * self.count as f64).ceil().max(1.0);
75        let mut cum = 0u64;
76        for (i, &count) in self.buckets.iter().enumerate() {
77            if count == 0 {
78                continue;
79            }
80            if (cum + count) as f64 >= target {
81                let lo = if i == 0 { 0 } else { BOUNDS_US[i - 1] };
82                let hi = BOUNDS_US[i];
83                // Fraction into this bucket the target sample sits at.
84                let into = (target - cum as f64) / count as f64;
85                return lo + ((hi - lo) as f64 * into) as u64;
86            }
87            cum += count;
88        }
89        BOUNDS_US[BOUNDS_US.len() - 1]
90    }
91
92    /// Convenience: the median (p50).
93    pub fn p50(&self) -> u64 {
94        self.percentile(0.50)
95    }
96    /// Convenience: p95 (the tail that drives perceived stutter).
97    pub fn p95(&self) -> u64 {
98        self.percentile(0.95)
99    }
100
101    /// Clear all samples (window reset).
102    pub fn clear(&mut self) {
103        self.buckets = [0; BOUNDS_US.len()];
104        self.count = 0;
105    }
106
107    /// Read `(p50, p95)` in microseconds and clear — the windowed emit step. `(0, 0)`
108    /// when no samples landed this window.
109    pub fn drain_percentiles(&mut self) -> (u64, u64) {
110        let out = (self.p50(), self.p95());
111        self.clear();
112        out
113    }
114}

116#[cfg(test)]
117mod tests {
118    use super::*;
119
120    #[test]
121    fn empty_reads_zero() {
122        let h = LatencyHistogram::new();
123        assert_eq!(h.count(), 0);
124        assert_eq!(h.p50(), 0);
125        assert_eq!(h.p95(), 0);
126    }
127
128    /// A tight cluster around 10 ms reads back a p50 in that bucket band.
129    #[test]
130    fn percentiles_track_a_tight_cluster() {
131        let mut h = LatencyHistogram::new();
132        for _ in 0..1000 {
133            h.record(10_000); // 10 ms
134        }
135        // 10 ms lands in the (8 ms, 12 ms] bucket; interpolation keeps p50 inside it.
136        let p50 = h.p50();
137        assert!(
138            (8_000..=12_000).contains(&p50),
139            "p50 {p50} should be in the 8–12 ms bucket"
140        );
141        assert_eq!(h.count(), 1000);
142    }
143
144    /// p95 sits in the tail: a 10% body of 200 ms spikes pulls p95 into the high
145    /// band while p50 stays at the healthy ~5 ms body. This is the whole point — a
146    /// mean would smear the two together; the percentile keeps them apart. (With
147    /// exactly 5% in the tail the 95th sample is still in the body, so the tail must
148    /// be >5% for p95 to land in it — that is correct percentile behaviour.)
149    #[test]
150    fn p95_captures_the_tail_the_mean_would_hide() {
151        let mut h = LatencyHistogram::new();
152        for _ in 0..90 {
153            h.record(5_000); // 5 ms
154        }
155        for _ in 0..10 {
156            h.record(200_000); // 200 ms spikes (10% tail)
157        }
158        assert!(h.p50() <= 6_000, "p50 stays at the healthy ~5 ms body");
159        assert!(
160            h.p95() >= 100_000,
161            "p95 {} must surface the 200 ms tail",
162            h.p95()
163        );
164    }
165
166    /// Samples above the top bound saturate the top bucket rather than being lost.
167    #[test]
168    fn oversized_samples_saturate_the_top_bucket() {
169        let mut h = LatencyHistogram::new();
170        h.record(60_000_000); // 60 s — far past the top bound
171        assert_eq!(h.count(), 1);
172        assert_eq!(h.p95(), *BOUNDS_US.last().unwrap());
173    }
174
175    /// `drain_percentiles` returns the window's percentiles and resets to empty.
176    #[test]
177    fn drain_reads_then_clears() {
178        let mut h = LatencyHistogram::new();
179        for _ in 0..10 {
180            h.record(3_000);
181        }
182        let (p50, _p95) = h.drain_percentiles();
183        assert!(p50 > 0);
184        assert_eq!(h.count(), 0, "drain clears the window");
185        assert_eq!(h.drain_percentiles(), (0, 0), "empty after drain");
186    }
187}
