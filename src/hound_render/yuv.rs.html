1//! Pure I420 → RGBA8888 conversion with nearest-neighbor scaling.
2//!
3//! The non-macOS render seam (Windows/Linux) hands decoded I420 to a CPU
4//! pixel-buffer texture: there is no GPU shader (the macOS path renders via wgpu
5//! into an IOSurface), so the convert + fit-to-texture happens here on the render
6//! thread and writes straight into the plugin's RGBA buffer.
7//!
8//! Pure + deterministic, so the pixel math is unit-tested on ANY host (including
9//! macOS builds, which never *use* it) — the part most likely to be wrong when
10//! a platform can't be exercised locally. Scaling is a plain nearest-neighbor
11//! stretch to the fixed texture size; Flutter's `FittedBox` applies the tile fit,
12//! so aspect handling (and any SIMD convert) stays out of this seam.
13
14/// Convert a tightly-packed I420 source (`sw`×`sh`: `Y(sw*sh) ‖ U ‖ V`, even
15/// extents, stride == width) to RGBA8888 (`dw`×`dh`, row-major, 4 bytes/px) written
16/// into `out`, nearest-neighbor scaled. BT.601 limited-range YUV→RGB (the engine's
17/// encode/decode matrix). No-ops (leaves `out` unchanged) on a malformed source or an
18/// undersized `out`, so a single bad frame never panics the render thread.
19#[cfg_attr(target_os = "macos", allow(dead_code))] // used by the non-macOS render seam + tests
20pub(crate) fn i420_to_rgba_scaled(
21    src: &[u8],
22    sw: usize,
23    sh: usize,
24    out: &mut [u8],
25    dw: usize,
26    dh: usize,
27) {
28    if sw == 0 || sh == 0 || dw == 0 || dh == 0 {
29        return;
30    }
31    if !sw.is_multiple_of(2) || !sh.is_multiple_of(2) {
32        return; // 4:2:0 needs even extents
33    }
34    if out.len() < dw * dh * 4 {
35        return;
36    }
37    let (cw, ch) = (sw / 2, sh / 2);
38    let y_size = sw * sh;
39    let u_off = y_size;
40    let v_off = y_size + cw * ch;
41    if src.len() < v_off + cw * ch {
42        return; // short I420 buffer
43    }
44
45    for dy in 0..dh {
46        // Nearest source row + its chroma row.
47        let sy = (dy * sh / dh).min(sh - 1);
48        let cy = (sy / 2).min(ch - 1);
49        let y_row = sy * sw;
50        let u_row = u_off + cy * cw;
51        let v_row = v_off + cy * cw;
52        let out_row = dy * dw * 4;
53        for dx in 0..dw {
54            let sx = (dx * sw / dw).min(sw - 1);
55            let cx = (sx / 2).min(cw - 1);
56            // BT.601 limited-range integer transform (×256 fixed-point).
57            let c = src[y_row + sx] as i32 - 16;
58            let d = src[u_row + cx] as i32 - 128;
59            let e = src[v_row + cx] as i32 - 128;
60            let r = ((298 * c + 409 * e + 128) >> 8).clamp(0, 255);
61            let g = ((298 * c - 100 * d - 208 * e + 128) >> 8).clamp(0, 255);
62            let b = ((298 * c + 516 * d + 128) >> 8).clamp(0, 255);
63            let o = out_row + dx * 4;
64            out[o] = r as u8;
65            out[o + 1] = g as u8;
66            out[o + 2] = b as u8;
67            out[o + 3] = 255;
68        }
69    }
70}

72#[cfg(test)]
73mod tests {
74    use super::*;

76    /// Build a solid-color I420 frame (`y`,`u`,`v` constant across all planes).
77    fn solid_i420(w: usize, h: usize, y: u8, u: u8, v: u8) -> Vec<u8> {
78        let (cw, ch) = (w / 2, h / 2);
79        let mut buf = vec![y; w * h];
80        buf.extend(std::iter::repeat_n(u, cw * ch));
81        buf.extend(std::iter::repeat_n(v, cw * ch));
82        buf
83    }
84
85    /// A mid-gray frame (Y=126, U=V=128 → no chroma) converts to ~mid-gray RGBA with
86    /// a fully-opaque alpha, identically across all output pixels (uniform source).
87    #[test]
88    fn gray_frame_is_uniform_gray_rgba() {
89        let src = solid_i420(4, 4, 126, 128, 128);
90        let (dw, dh) = (8, 8); // upscale 2×
91        let mut out = vec![0u8; dw * dh * 4];
92        i420_to_rgba_scaled(&src, 4, 4, &mut out, dw, dh);
93        // Y=126 → 298*(126-16)>>8 = 128 on every channel; A=255.
94        for px in out.chunks_exact(4) {
95            assert_eq!(px[0], px[1], "gray ⇒ R==G");
96            assert_eq!(px[1], px[2], "gray ⇒ G==B");
97            assert!((127..=129).contains(&px[0]), "≈mid gray, got {}", px[0]);
98            assert_eq!(px[3], 255, "opaque alpha");
99        }
100    }
101
102    /// A saturated-red source (Y/U/V for full red) yields a red-dominant pixel — the
103    /// channel order is RGBA (R first), the byte order the Flutter pixel-buffer
104    /// textures expect.
105    #[test]
106    fn red_source_maps_to_red_dominant_rgba() {
107        // BT.601 limited red ≈ Y=82, U=90, V=240.
108        let src = solid_i420(2, 2, 82, 90, 240);
109        let mut out = vec![0u8; 2 * 2 * 4];
110        i420_to_rgba_scaled(&src, 2, 2, &mut out, 2, 2);
111        let px = &out[..4];
112        assert!(px[0] > 200, "R high, got {}", px[0]);
113        assert!(px[1] < 80, "G low, got {}", px[1]);
114        assert!(px[2] < 80, "B low, got {}", px[2]);
115        assert_eq!(px[3], 255);
116    }
117
118    /// Malformed inputs never panic and leave `out` untouched (a skippable frame).
119    #[test]
120    fn malformed_inputs_are_noops() {
121        let mut out = vec![7u8; 4 * 4 * 4];
122        i420_to_rgba_scaled(&[], 0, 0, &mut out, 4, 4); // empty
123        i420_to_rgba_scaled(&[1, 2, 3], 2, 2, &mut out, 4, 4); // short src
124        let small = solid_i420(2, 2, 100, 128, 128);
125        i420_to_rgba_scaled(&small, 2, 2, &mut out[..8], 4, 4); // undersized out
126        assert!(out.iter().all(|&b| b == 7), "untouched on bad input");
127    }
