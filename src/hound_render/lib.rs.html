1//! # hound-render — decoded video frames → Flutter-registered native textures
2//!
3//! The display side of the video pipeline: the call orchestration decodes remote
4//! video (and the capture path previews local video) and pushes each frame here
5//! by `stream_key` (`"self"`, `"cam:{peer}"`, `"screen:{peer}"`); the Flutter
6//! side registers a native texture per tile and binds it to the same key via
7//! [`attach`]. This crate connects the two with one platform-uniform API — the
8//! caller (the FRB layer) never carries a `#[cfg(target_os = …)]`.
9//!
10//! ## Render model
11//! * **Mailboxes** — each stream key owns a drop-oldest one-frame mailbox.
12//!   [`push_frame`]/[`push_surface`] overwrite it and never block the producer;
13//!   a slow renderer skips stale frames. Mailboxes live independently of the
14//!   attached texture, so a producer can push before attach and across
15//!   re-attach.
16//! * **Shared render loop (macOS)** — ONE render thread draws every stream's
17//!   latest frame into its target IOSurface via wgpu(Metal), batching all draws
18//!   into one command submit per wake, and marks each drawn Flutter
19//!   `texture_id` "dirty". The Swift shared `CVDisplayLink` drains the dirty
20//!   set through the C ABI export [`hound_video_take_dirty`] and re-pulls
21//!   exactly the tiles that got a new frame — repaint is frame-arrival-driven,
22//!   not vsync-driven.
23//! * **D3D11 path (Windows)** — NV12 frames (the engine's zero-copy decode/
24//!   capture textures, or uploaded CPU I420) are blitted into the plugin's
25//!   shared BGRA texture by the fixed-function video processor, synchronously
26//!   in the push; the plugin's ticker drains the same dirty-set C ABI as macOS.
27//! * **CPU path (Linux)** — a per-stream render thread converts the latest
28//!   I420 into the plugin's RGBA pixel buffer (`crate::yuv`) at ~30 fps; the
29//!   plugin pings its texture on its own timer.
30//!
31//! ## Platform layout
32//! The concrete renderers live under `platform/` and are selected at compile
33//! time ([`platform`]); macOS-only primitives that callers may legitimately
34//! need (the IOSurface ⇄ wgpu interop) are exposed as an explicit escape hatch
35//! under [`os::macos`], never as cfg'd root exports.
36
37mod platform;
38// Pure I420→RGBA converter for the CPU render path. Always compiled (it has no
39// platform deps), so its pixel math is unit-tested on every host — including
40// macOS builds, which never use it (macOS renders via wgpu instead).
41pub(crate) mod yuv;
42
43/// Platform-only extension points, in the `std::os` shape: each submodule
44/// exists only on its platform, and nothing under it is re-exported at the
45/// crate root — a caller that reaches in here is knowingly platform-specific.
46pub mod os {
47    /// macOS-only render primitives.
48    #[cfg(target_os = "macos")]
49    pub mod macos {
50        /// The IOSurface-backed `CVPixelBuffer` ⇄ `wgpu::Texture` interop
51        /// primitive (render targets and NV12 plane sampling).
52        pub use crate::platform::apple::iosurface::{MetalPixelBuffer, PlaneFormat};
53        /// The dirty-texture-id drain for the Swift shared `CVDisplayLink`
54        /// (also exported as a C ABI symbol; see its docs).
55        pub use crate::platform::apple::render::hound_video_take_dirty;
56    }
57}

59/// Bind `stream_key` to the native buffer the Flutter plugin registered as
60/// `texture_id`, and start rendering pushed frames into it. Idempotent: an
61/// attach under an existing key replaces the previous binding.
62///
63/// `native_buffer_ptr` is platform-defined: an IOSurface-backed
64/// `CVPixelBufferRef` on macOS (geometry is read off the buffer; `width`/
65/// `height` are advisory), a DXGI **shared handle** to the plugin's BGRA
66/// texture on Windows, and a raw `width`×`height`×4 RGBA byte buffer on Linux
67/// (neither can be introspected, so the size is load-bearing). It must
68/// stay valid until [`detach`] returns for this key. Returns an error string
69/// (FRB maps it to a Dart exception) on a null pointer or setup failure.
70pub fn attach(
71    stream_key: String,
72    native_buffer_ptr: usize,
73    texture_id: i64,
74    width: u32,
75    height: u32,
76) -> Result<(), String> {
77    platform::imp::attach(stream_key, native_buffer_ptr, texture_id, width, height)
78}
79
80/// Unbind `stream_key` (no-op if absent). When this returns, no render work —
81/// CPU thread or in-flight GPU command — still touches the native buffer, so
82/// the caller may free/destroy the texture immediately. The stream's mailbox
83/// survives (a producer may still be pushing; a re-attach may be coming).
84pub fn detach(stream_key: &str) {
85    platform::imp::detach(stream_key)
86}
87
88/// Push the latest decoded I420 frame for `stream_key` (drop-oldest). `i420`
89/// is tightly packed `w*h + 2*(w/2 * h/2)` bytes, stride == width, even
90/// extents. Cheap and non-blocking on every platform — safe to call from a
91/// decode/capture thread; never from a real-time audio thread.
92pub fn push_frame(stream_key: &str, w: u32, h: u32, i420: Vec<u8>) {
93    platform::imp::push_frame(stream_key, w, h, i420)
94}
95
96/// Push a decoded frame that is still GPU-resident, identified by a
97/// platform-defined surface pointer — on macOS an IOSurface-backed NV12
98/// `CVPixelBufferRef` (a retain is taken synchronously), on Windows a borrowed
99/// NV12 `ID3D11Texture2D` (consumed synchronously: the blit into the Flutter
100/// target happens inside this call). Valid for the duration of the call.
101/// Returns `true` if the platform renderer took the surface; `false` where
102/// there is no surface renderer, the tile is not attached yet (Windows), or
103/// zero-copy is disabled via `HOUND_ZEROCOPY_VIDEO=0` — the caller then falls
104/// back to [`push_frame`] with materialized I420.
105pub fn push_surface(stream_key: &str, native_surface_ptr: usize, width: u32, height: u32) -> bool {
106    platform::imp::push_surface(stream_key, native_surface_ptr, width, height)
107}
