1//! The orchestration facade: ties the REST data plane, the session lifecycle, and
2//! the realtime (`/community`) plane into one observable, UI-free app core.
3//!
4//! [`HoundClient`] is what the FRB layer (and any future native UI) drives. It owns
5//! the pieces and an [`AppState`] of `tokio::sync::watch` channels — one per
6//! observable snapshot, latest-wins, which is exactly what a UI wants (it never
7//! needs the history, only the current view). REST loads *seed* the snapshots; a
8//! single background task consumes the typed [`RealtimeEvent`] stream, folds each
9//! event through the pure reducers ([`CommunityTree`]/[`RosterView`]/[`MessageList`]),
10//! and republishes the mutated snapshot onto the matching `watch` sender.
11//!
12//! Commands are async and return [`Result<_, HoundError>`]; observable getters
13//! return [`watch::Receiver`]s the binding subscribes. No FFI and no UI types live
14//! here — the crate stays standalone-buildable.
15

33/// The observable app state: one `watch` channel per snapshot the UI renders. Each
34/// holds the latest value (a new subscriber sees it immediately) and notifies on
35/// change. Per-space message lists are created lazily on [`HoundClient::open_space`].
36struct AppState {
37    auth: watch::Sender<AuthState>,
38    community_tree: watch::Sender<CommunityTree>,
39    roster: watch::Sender<RosterView>,
40    /// space id → that space's message-list channel (lazily created on open).
41    messages: Mutex<HashMap<Uuid, watch::Sender<MessageList>>>,
42    /// Call-state observables, driven by the live [`CallSession`]:
43    /// the roster/track view (from the control plane's `SessionView`), the
44    /// connection state, and the mic + per-peer speaking levels.
45    call_view: watch::Sender<CallView>,
46    connection: watch::Sender<ConnectionState>,
47    levels: watch::Sender<LevelsView>,
48    /// 1 Hz RTC debug-stats snapshot (the data behind the debug panel + per-tile
49    /// overlay). Driven by the call's `spawn_emitter`; default/empty between calls.
50    stats: watch::Sender<crate::stats::StatsSnapshot>,
51    /// Live `peer_id → user_id` map for the current call. Seeded by the call at join
52    /// (from the REST voice roster + self) and kept current by the realtime voice
53    /// events here; the call's control reader joins it into each `CallView` so tiles
54    /// show real identities. Cleared on call join/leave so it never leaks across calls.
55    call_identities: PeerIdentities,
56}
57
58impl AppState {
59    fn new() -> Self {
60        Self {
61            auth: watch::channel(AuthState::SignedOut).0,
62            community_tree: watch::channel(CommunityTree::new()).0,
63            roster: watch::channel(RosterView::new()).0,
64            messages: Mutex::new(HashMap::new()),
65            call_view: watch::channel(CallView::default()).0,
66            connection: watch::channel(ConnectionState::default()).0,
67            levels: watch::channel(LevelsView::default()).0,
68            stats: watch::channel(crate::stats::StatsSnapshot::default()).0,
69            call_identities: PeerIdentities::default(),
70        }
71    }
72}

74/// The headless application core. Cheap to share via the returned `Arc<Self>`; every
75/// command takes `&self`. Holds the REST client, the session, the observable state,
76/// and the live realtime connection (replaced on each (re)connect).
77pub struct HoundClient {
78    rest: Arc<RestClient>,
79    /// The base URL (kept so the call orchestration can derive the control WS URL).
80    base_url: String,
81    session: Session,
82    state: AppState,
83    /// The live realtime connection, if connected. Swapped on connect; dropped (and
84    /// its background task ended) on disconnect/logout.
85    realtime: Mutex<Option<RealtimeClient>>,
86    /// The live voice/video call, if joined. At most one at a time.
87    call: Mutex<Option<CallSession>>,
88    /// The decode→render sink the call hands decoded video to (keyed by peer). The
89    /// FRB layer installs one whose body calls `video_render::push_frame(...)`;
90    /// until then a [`NullVideoSink`] keeps the call running (audio + state) headless.
91    video_sink: Mutex<Arc<dyn VideoSink>>,
92}

185    /// Join the voice/video call for `space_id`. Establishes the control WS, dials
186    /// the MoQ relay, publishes mic (+ camera if `camera_enabled`), and starts the
187    /// subscribe loop. Drives the `watch_call`/`watch_connection`/`watch_levels`
188    /// snapshots. Leaving any prior call first (one call at a time).
189    pub async fn call_join(&self, space_id: Uuid, camera_enabled: bool) -> Result<()> {
190        // Leave any prior call so we never hold two.
191        if let Some(prev) = self.call.lock().await.take() {
192            prev.leave().await;
193        }
194        // Fresh call: drop any stale identities (the call re-seeds from its join roster).
195        self.state.call_identities.store(Default::default());
196        let sink = self.video_sink.lock().await.clone();
197        // Drive join on the dedicated call runtime (not FRB's per-core pool) so the
198        // connection setup's internally-spawned tasks — notably moq-net's session
199        // driver — land there alongside every task `join` spawns. See
200        // [`crate::call::call_runtime`].
201        let join = CallSession::join(
202            self.rest.clone(),
203            self.base_url.clone(),
204            space_id,
205            sink,
206            camera_enabled,
207            self.state.call_view.clone(),
208            self.state.connection.clone(),
209            self.state.levels.clone(),
210            self.state.stats.clone(),
211            self.state.call_identities.clone(),
212        );
213        let session = crate::call::call_runtime()
214            .spawn(join)
215            .await
216            .map_err(|e| HoundError::Internal(format!("call join task failed: {e}")))??;
217        *self.call.lock().await = Some(session);
218        Ok(())
219    }

427    /// Sign out: tear down realtime, clear the session + tokens, publish signed-out,
428    /// and reset the observable snapshots (so a re-login starts from a clean view).
429    pub async fn logout(self: &Arc<Self>) -> Result<()> {
430        if let Some(rt) = self.realtime.lock().await.take() {
431            let _ = rt.disconnect().await;
432        }
433        self.session.sign_out().await?;
434        self.publish_auth(AuthState::SignedOut);
435        let _ = self.state.community_tree.send(CommunityTree::new());
436        let _ = self.state.roster.send(RosterView::new());
437        self.state.messages.lock().await.clear();
438        Ok(())
439    }

455    /// Load the signed-in user's communities and their spaces, seeding the
456    /// community-tree snapshot. Idempotent (re-run on reconnect to re-seed).
457    pub async fn load_communities(&self) -> Result<()> {
458        let page = self.rest.list_communities(0, 100).await?;
459        let mut tree = self.state.community_tree.borrow().clone();
460        tree.apply_communities(page.items.clone());
461        for community in &page.items {
462            let spaces = self.rest.list_spaces(community.id, 0, 100).await?;
463            tree.set_spaces(community.id, spaces.items);
464        }
465        let _ = self.state.community_tree.send(tree);
466        Ok(())
467    }

469    /// Open a space: load its message history (seeding the per-space `MessageList`
470    /// watch and merging the authors into the roster), then subscribe to its
471    /// realtime room so subsequent `message:*` events flow into the same channel.
472    pub async fn open_space(&self, space_id: Uuid) -> Result<()> {
473        let page = self.rest.list_messages(space_id, 0, 50).await?;
474
475        let sender = self.message_sender(space_id).await;
476        sender.send_modify(|list| list.apply_page(page.messages.items));
477
478        if !page.members.is_empty() {
479            self.state
480                .roster
481                .send_modify(|roster| roster.apply_members(page.members));
482        }
483
484        if let Some(rt) = self.realtime.lock().await.as_ref() {
485            rt.subscribe_space(space_id).await?;
486        }
487        Ok(())
488    }

512    /// Connect the realtime plane with the current access token and spawn the
513    /// reducer task. Replaces any existing connection. A signed-out client errors.
514    async fn connect(self: &Arc<Self>) -> Result<()> {
515        let token = self
516            .rest
517            .token()
518            .ok_or_else(|| HoundError::Auth("cannot connect realtime: no session token".into()))?;
519        let base = self.base_url();
520        let (client, rx) = RealtimeClient::connect(&base, &token).await?;
521        *self.realtime.lock().await = Some(client);
522        self.spawn_reducer_task(rx);
523        Ok(())
524    }

531    /// Spawn the task that drains the realtime event stream and applies the pure
532    /// reducers to the observable snapshots. Holds a weak ref to `self` so a dropped
533    /// client lets the task exit (the channel also closes when the connection drops).
534    fn spawn_reducer_task(self: &Arc<Self>, rx: mpsc::UnboundedReceiver<RealtimeEvent>) {
535        let weak = Arc::downgrade(self);
536        tokio::spawn(async move {
537            let mut rx = rx;
538            while let Some(event) = rx.recv().await {
539                let Some(this) = weak.upgrade() else { break };
540                this.apply_event(event).await;
541            }
542        });
543    }

545    /// Fold one realtime event into every snapshot it touches, republishing only the
546    /// channels that actually changed (so subscribers don't wake spuriously). This is
547    /// the single place realtime → reducer → watch happens; it is what the unit test
548    /// drives directly with a fake event source.
549    async fn apply_event(&self, event: RealtimeEvent) {
550        // Roster: voice events carry display identities.
551        match &event {
552            RealtimeEvent::VoiceJoined(_) | RealtimeEvent::VoiceState(_) => {
553                self.state.roster.send_modify(|r| r.apply_event(&event));
554            }
555            _ => {}
556        }
557
558        // Call identities: keep the live peer_id → user_id map current so an in-call
559        // tile resolves to a real name even for a peer who joins/leaves mid-call.
560        // (Merge semantics: never drops self; a stale identity is harmless because
561        // `build_participants` only emits entries for peers in the control roster.)
562        let identities_changed = match &event {
563            RealtimeEvent::VoiceJoined(e) => match e.sfu_peer_id {
564                Some(pid) => {
565                    self.state.call_identities.rcu(|m| {
566                        let mut m = HashMap::clone(m);
567                        m.insert(pid as u32, e.user_id);
568                        m
569                    });
570                    true
571                }
572                None => false,
573            },
574            RealtimeEvent::VoiceState(list) => {
575                self.state.call_identities.rcu(|m| {
576                    let mut m = HashMap::clone(m);
577                    for e in list {
578                        if let Some(pid) = e.sfu_peer_id {
579                            m.insert(pid as u32, e.user_id);
580                        }
581                    }
582                    m
583                });
584                true
585            }
586            RealtimeEvent::VoiceLeft(e) => {
587                self.state.call_identities.rcu(|m| {
588                    let mut m = HashMap::clone(m);
589                    m.retain(|_, uid| *uid != e.user_id);
590                    m
591                });
592                true
593            }
594            _ => false,
595        };
596        if identities_changed {
597            let map = self.state.call_identities.load();
598            self.state.call_view.send_modify(|cv| {
599                cv.participants = build_participants(&cv.peers, &map);
600            });
601        }
602
603        // Community tree: space create/remove.
604        match &event {
605            RealtimeEvent::SpaceCreated(_) | RealtimeEvent::SpaceRemoved { .. } => {
606                self.state
607                    .community_tree
608                    .send_if_modified(|t| t.apply_event(&event));
609            }
610            _ => {}
611        }
612
613        // Messages: route to the matching space's channel (if a UI ever opened it).
614        if let RealtimeEvent::MessageCreated(m) | RealtimeEvent::MessageUpdated(m) = &event {
615            let space_id = m.space_id;
616            let sender = {
617                let map = self.state.messages.lock().await;
618                map.get(&space_id).cloned()
619            };
620            if let Some(sender) = sender {
621                sender.send_if_modified(|list| list.apply_event(space_id, &event));
622            }
623        }
624    }
625
626    fn publish_auth(&self, state: AuthState) {
627        let _ = self.state.auth.send(state);
628    }

637    fn uid(n: u8) -> Uuid {
638        let mut b = [0u8; 16];
639        b[15] = n;
640        Uuid::from_bytes(b)
641    }
642
643    fn msg_id(ms: u64, n: u8) -> Uuid {
644        let mut b = [0u8; 16];
645        b[0] = (ms >> 40) as u8;
646        b[1] = (ms >> 32) as u8;
647        b[2] = (ms >> 24) as u8;
648        b[3] = (ms >> 16) as u8;
649        b[4] = (ms >> 8) as u8;
650        b[5] = ms as u8;
651        b[15] = n;
652        Uuid::from_bytes(b)
653    }
