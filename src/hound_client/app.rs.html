1//! The orchestration facade: ties the REST data plane, the session lifecycle, and
2//! the realtime (`/community`) plane into one observable, UI-free app core.
3//!
4//! [`HoundClient`] is what the FRB layer (and any future native UI) drives. It owns
5//! the pieces and an [`AppState`] of `tokio::sync::watch` channels — one per
6//! observable snapshot, latest-wins, which is exactly what a UI wants (it never
7//! needs the history, only the current view). REST loads *seed* the snapshots; a
8//! single background task consumes the typed [`RealtimeEvent`] stream, folds each
9//! event through the pure reducers ([`CommunityTree`]/[`RosterView`]/[`MessageList`]),
10//! and republishes the mutated snapshot onto the matching `watch` sender.
11//!
12//! Commands are async and return [`Result<_, HoundError>`]; observable getters
13//! return [`watch::Receiver`]s the binding subscribes. No FFI and no UI types live
14//! here — the crate stays standalone-buildable.
15

33/// The observable app state: one `watch` channel per snapshot the UI renders. Each
34/// holds the latest value (a new subscriber sees it immediately) and notifies on
35/// change. Per-space message lists are created lazily on [`HoundClient::open_space`].
36struct AppState {
37    auth: watch::Sender<AuthState>,
38    community_tree: watch::Sender<CommunityTree>,
39    roster: watch::Sender<RosterView>,
40    /// space id → that space's message-list channel (lazily created on open).
41    messages: Mutex<HashMap<Uuid, watch::Sender<MessageList>>>,
42    /// Call-state observables, driven by the live [`CallSession`]:
43    /// the roster/track view (from the control plane's `SessionView`), the
44    /// connection state, and the mic + per-peer speaking levels.
45    call_view: watch::Sender<CallView>,
46    connection: watch::Sender<ConnectionState>,
47    levels: watch::Sender<LevelsView>,
48    /// 1 Hz RTC debug-stats snapshot (the data behind the debug panel + per-tile
49    /// overlay). Driven by the call's `spawn_emitter`; default/empty between calls.
50    stats: watch::Sender<crate::stats::StatsSnapshot>,
51    /// Live `peer_id → user_id` map for the current call. Seeded by the call at join
52    /// (from the REST voice roster + self) and kept current by the realtime voice
53    /// events here; the call's control reader joins it into each `CallView` so tiles
54    /// show real identities. Cleared on call join/leave so it never leaks across calls.
55    call_identities: PeerIdentities,
56}
57
58impl AppState {
59    fn new() -> Self {
60        Self {
61            auth: watch::channel(AuthState::SignedOut).0,
62            community_tree: watch::channel(CommunityTree::new()).0,
63            roster: watch::channel(RosterView::new()).0,
64            messages: Mutex::new(HashMap::new()),
65            call_view: watch::channel(CallView::default()).0,
66            connection: watch::channel(ConnectionState::default()).0,
67            levels: watch::channel(LevelsView::default()).0,
68            stats: watch::channel(crate::stats::StatsSnapshot::default()).0,
69            call_identities: PeerIdentities::default(),
70        }
71    }
72}

74/// The headless application core. Cheap to share via the returned `Arc<Self>`; every
75/// command takes `&self`. Holds the REST client, the session, the observable state,
76/// and the live realtime connection (replaced on each (re)connect).
77pub struct HoundClient {
78    rest: Arc<RestClient>,
79    /// The base URL (kept so the call orchestration can derive the control WS URL).
80    base_url: String,
81    session: Session,
82    state: AppState,
83    /// The live realtime connection, if connected. Swapped on connect; dropped (and
84    /// its background task ended) on disconnect/logout.
85    realtime: Mutex<Option<RealtimeClient>>,
86    /// The live voice/video call, if joined. At most one at a time.
87    call: Mutex<Option<CallSession>>,
88    /// The Steamworks connection when launched through Steam (`None` otherwise,
89    /// including all non-Steam builds). Init success *is* the launch detection.
90    steam: Option<hound_steam::SteamHandle>,
91    /// A pending Steam `link_token`: proof of Steam-account ownership awaiting
92    /// the user's create-or-link decision (or a settings link). Set when a
93    /// Steam sign-in meets an unknown Steam ID; consumed by
94    /// [`steam_register`](Self::steam_register) /
95    /// [`link_steam_pending`](Self::link_steam_pending), auto-consumed by a
96    /// successful sign-in, cleared by [`steam_dismiss`](Self::steam_dismiss).
97    pending_steam_link: Mutex<Option<String>>,
98    /// A pending browser-deferred login (the bound loopback listener), if the UI
99    /// has begun one. At most one at a time; replaced by a re-begin, consumed by
100    /// wait, dropped by cancel.
101    browser_login: Mutex<Option<hound_browser_auth::BrowserLogin>>,
102    /// Wakes a blocked [`browser_login_wait`](Self::browser_login_wait) when the
103    /// UI cancels (the wait has already consumed the listener by then).
104    browser_login_cancelled: tokio::sync::Notify,
105    /// The decode→render sink the call hands decoded video to (keyed by peer). The
106    /// FRB layer installs one whose body calls `video_render::push_frame(...)`;
107    /// until then a [`NullVideoSink`] keeps the call running (audio + state) headless.
108    video_sink: Mutex<Arc<dyn VideoSink>>,
109}

112    /// Build a client bound to `base_url`, persisting tokens through `store`. Does
113    /// not touch the network — call [`restore`](Self::restore) or a login command.
114    /// `steam_app_id` opts into Steamworks detection (Steam-store builds only);
115    /// init failure just means "not launched through Steam".
116    pub fn new(

118        store: Arc<dyn SessionStore>,
119        steam_app_id: Option<u32>,

121        let rest = Arc::new(RestClient::new(base_url)?);
122        let session = Session::new(rest.clone(), store);
123        Ok(Arc::new(Self {
124            rest,
125            base_url: base_url.to_owned(),
126            session,
127            state: AppState::new(),
128            realtime: Mutex::new(None),
129            call: Mutex::new(None),
130            steam: steam_app_id.and_then(hound_steam::SteamHandle::init),
131            pending_steam_link: Mutex::new(None),
132            browser_login: Mutex::new(None),
133            browser_login_cancelled: tokio::sync::Notify::new(),
134            video_sink: Mutex::new(Arc::new(NullVideoSink)),
135        }))
136    }

212    /// Join the voice/video call for `space_id`. Establishes the control WS, dials
213    /// the MoQ relay, publishes mic (+ camera if `camera_enabled`), and starts the
214    /// subscribe loop. Drives the `watch_call`/`watch_connection`/`watch_levels`
215    /// snapshots. Leaving any prior call first (one call at a time).
216    pub async fn call_join(&self, space_id: Uuid, camera_enabled: bool) -> Result<()> {
217        // Leave any prior call so we never hold two.
218        if let Some(prev) = self.call.lock().await.take() {
219            prev.leave().await;
220        }
221        // Fresh call: drop any stale identities (the call re-seeds from its join roster).
222        self.state.call_identities.store(Default::default());
223        let sink = self.video_sink.lock().await.clone();
224        // Drive join on the dedicated call runtime (not FRB's per-core pool) so the
225        // connection setup's internally-spawned tasks — notably moq-net's session
226        // driver — land there alongside every task `join` spawns. See
227        // [`crate::call::call_runtime`].
228        let join = CallSession::join(
229            self.rest.clone(),
230            self.base_url.clone(),
231            space_id,
232            sink,
233            camera_enabled,
234            self.state.call_view.clone(),
235            self.state.connection.clone(),
236            self.state.levels.clone(),
237            self.state.stats.clone(),
238            self.state.call_identities.clone(),
239        );
240        let session = crate::call::call_runtime()
241            .spawn(join)
242            .await
243            .map_err(|e| HoundError::Internal(format!("call join task failed: {e}")))??;
244        *self.call.lock().await = Some(session);
245        Ok(())
246    }

248    /// Leave the current call (best-effort `voice/leave`, then tear down). No-op if
249    /// not in a call.
250    pub async fn call_leave(&self) -> Result<()> {
251        if let Some(session) = self.call.lock().await.take() {
252            session.leave().await;
253        }
254        // Identities are per-call; clear so the next CallView snapshot starts clean.
255        self.state.call_identities.store(Default::default());
256        Ok(())
257    }

466    /// Sign in via the local Steam client (in-Steam launch). A linked Steam ID
467    /// signs straight in; an unknown one parks a link token and returns
468    /// [`AuthFlowOutcome::SteamUnknown`] for the create-or-link screen.
469    pub async fn steam_login(self: &Arc<Self>) -> Result<AuthFlowOutcome> {
470        let steam = self
471            .steam
472            .as_ref()
473            .ok_or_else(|| HoundError::InvalidState("not launched through steam".into()))?;
474        let ticket = steam
475            .auth_ticket_for_web_api()
476            .await
477            .map_err(|e| HoundError::Auth(e.to_string()))?;
478        match self.rest.steam_session(&ticket).await? {
479            crate::rest::SteamLogin::SignedIn(login) => Ok(AuthFlowOutcome::SignedIn(
480                self.adopt_and_connect(login).await?,
481            )),
482            crate::rest::SteamLogin::Unknown {
483                link_token,
484                persona,
485            } => {
486                *self.pending_steam_link.lock().await = Some(link_token);
487                Ok(AuthFlowOutcome::SteamUnknown { persona })
488            }
489        }
490    }

492    /// "Continue as <persona>": create a new account for the pending Steam
493    /// link token, with `name` as the profile display name.
494    pub async fn steam_register(self: &Arc<Self>, name: &str) -> Result<AuthState> {
495        let token = self.pending_steam_link.lock().await.take().ok_or_else(|| {

499        self.adopt_and_connect(login).await
500    }

538    /// Unlink a login method (`"email"` / `"steam"`). The backend refuses to

615    /// Await the browser callback for the flow begun by
616    /// [`browser_login_begin`](Self::browser_login_begin) /
617    /// [`browser_steam_begin`](Self::browser_steam_begin). A token callback is
618    /// adopted as a session (auth published, realtime up); a Steam link-token
619    /// callback parks the token and returns [`AuthFlowOutcome::SteamUnknown`] for
620    /// the create-or-link screen (or the settings link). Errors if no flow is
621    /// pending, on timeout, or when
622    /// [`browser_login_cancel`](Self::browser_login_cancel) interrupts the wait.
623    pub async fn browser_login_wait(self: &Arc<Self>) -> Result<AuthFlowOutcome> {
624        /// Generous next to houndctl's 120 s: the UI shows a cancel button, so a
625        /// slow ceremony (finding the phone, cross-device passkey) shouldn't
626        /// time the flow out from under the user.
627        const BROWSER_LOGIN_TIMEOUT: std::time::Duration = std::time::Duration::from_secs(300);
628
629        let login = self
630            .browser_login
631            .lock()
632            .await
633            .take()
634            .ok_or_else(|| HoundError::InvalidState("no browser login in progress".into()))?;
635
636        let callback = tokio::select! {
637            r = login.wait_callback(BROWSER_LOGIN_TIMEOUT) => {
638                r.map_err(|e| HoundError::Auth(e.to_string()))?
639            }
640            _ = self.browser_login_cancelled.notified() => {
641                return Err(HoundError::Auth("browser login cancelled".into()));
642            }
643        };
644
645        match callback {
646            hound_browser_auth::CallbackResult::Tokens(tokens) => {
647                let state = self
648                    .session
649                    .adopt_tokens(&tokens.access_token, &tokens.refresh_token)?;
650                self.publish_auth(state.clone());
651                self.connect().await?;
652                self.load_communities().await?;
653                self.autolink_pending_steam().await;
654                Ok(AuthFlowOutcome::SignedIn(state))
655            }
656            hound_browser_auth::CallbackResult::SteamLink {
657                link_token,
658                persona,
659            } => {
660                *self.pending_steam_link.lock().await = Some(link_token);
661                Ok(AuthFlowOutcome::SteamUnknown {
662                    persona: persona.unwrap_or_default(),
663                })
664            }
665            hound_browser_auth::CallbackResult::Linked => Ok(AuthFlowOutcome::EmailLinked),
666        }
667    }

669    /// Cancel a pending browser login: drop an un-waited listener and wake a
670    /// blocked [`browser_login_wait`](Self::browser_login_wait). Idempotent.
671    pub async fn browser_login_cancel(&self) {

676    /// Sign out: tear down realtime, clear the session + tokens, publish signed-out,
677    /// and reset the observable snapshots (so a re-login starts from a clean view).
678    pub async fn logout(self: &Arc<Self>) -> Result<()> {
679        if let Some(rt) = self.realtime.lock().await.take() {
680            let _ = rt.disconnect().await;
681        }
682        self.session.sign_out().await?;
683        self.publish_auth(AuthState::SignedOut);
684        let _ = self.state.community_tree.send(CommunityTree::new());
685        let _ = self.state.roster.send(RosterView::new());
686        self.state.messages.lock().await.clear();
687        Ok(())
688    }

775    /// Connect the realtime plane with the current access token and spawn the
776    /// reducer task. Replaces any existing connection. A signed-out client errors.
777    async fn connect(self: &Arc<Self>) -> Result<()> {
778        let token = self
779            .rest
780            .token()
781            .ok_or_else(|| HoundError::Auth("cannot connect realtime: no session token".into()))?;
782        let base = self.base_url();
783        let (client, rx) = RealtimeClient::connect(&base, &token).await?;
784        *self.realtime.lock().await = Some(client);
785        self.spawn_reducer_task(rx);
786        Ok(())
787    }

794    /// Spawn the task that drains the realtime event stream and applies the pure
795    /// reducers to the observable snapshots. Holds a weak ref to `self` so a dropped
796    /// client lets the task exit (the channel also closes when the connection drops).
797    fn spawn_reducer_task(self: &Arc<Self>, rx: mpsc::UnboundedReceiver<RealtimeEvent>) {
798        let weak = Arc::downgrade(self);
799        tokio::spawn(async move {
800            let mut rx = rx;
801            while let Some(event) = rx.recv().await {
802                let Some(this) = weak.upgrade() else { break };
803                this.apply_event(event).await;
804            }
805        });
806    }

808    /// Fold one realtime event into every snapshot it touches, republishing only the
809    /// channels that actually changed (so subscribers don't wake spuriously). This is
810    /// the single place realtime → reducer → watch happens; it is what the unit test
811    /// drives directly with a fake event source.
812    async fn apply_event(&self, event: RealtimeEvent) {
813        // Roster: voice events carry display identities.
814        match &event {
815            RealtimeEvent::VoiceJoined(_) | RealtimeEvent::VoiceState(_) => {
816                self.state.roster.send_modify(|r| r.apply_event(&event));
817            }
818            _ => {}
819        }
820
821        // Call identities: keep the live peer_id → user_id map current so an in-call
822        // tile resolves to a real name even for a peer who joins/leaves mid-call.
823        // (Merge semantics: never drops self; a stale identity is harmless because
824        // `build_participants` only emits entries for peers in the control roster.)
825        let identities_changed = match &event {
826            RealtimeEvent::VoiceJoined(e) => match e.sfu_peer_id {
827                Some(pid) => {
828                    self.state.call_identities.rcu(|m| {
829                        let mut m = HashMap::clone(m);
830                        m.insert(pid as u32, e.user_id);
831                        m
832                    });
833                    true
834                }
835                None => false,
836            },
837            RealtimeEvent::VoiceState(list) => {
838                self.state.call_identities.rcu(|m| {
839                    let mut m = HashMap::clone(m);
840                    for e in list {
841                        if let Some(pid) = e.sfu_peer_id {
842                            m.insert(pid as u32, e.user_id);
843                        }
844                    }
845                    m
846                });
847                true
848            }
849            RealtimeEvent::VoiceLeft(e) => {
850                self.state.call_identities.rcu(|m| {
851                    let mut m = HashMap::clone(m);
852                    m.retain(|_, uid| *uid != e.user_id);
853                    m
854                });
855                true
856            }
857            _ => false,
858        };
859        if identities_changed {
860            let map = self.state.call_identities.load();
861            self.state.call_view.send_modify(|cv| {
862                cv.participants = build_participants(&cv.peers, &map);
863            });
864        }
865
866        // Community tree: space create/remove.
867        match &event {
868            RealtimeEvent::SpaceCreated(_) | RealtimeEvent::SpaceRemoved { .. } => {
869                self.state
870                    .community_tree
871                    .send_if_modified(|t| t.apply_event(&event));
872            }
873            _ => {}
874        }
875
876        // Messages: route to the matching space's channel (if a UI ever opened it).
877        if let RealtimeEvent::MessageCreated(m) | RealtimeEvent::MessageUpdated(m) = &event {
878            let space_id = m.space_id;
879            let sender = {
880                let map = self.state.messages.lock().await;
881                map.get(&space_id).cloned()
882            };
883            if let Some(sender) = sender {
884                sender.send_if_modified(|list| list.apply_event(space_id, &event));
885            }
886        }
887    }
