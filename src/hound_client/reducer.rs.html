1//! Pure reducers: fold REST loads + realtime events into the typed view-models the UI
2//! streams. Each reducer is a plain struct with no IO and no async, so it is trivially
3//! unit-testable and deterministic — the caller drives it from the [`crate::realtime`]
4//! event stream and REST results, then snapshots it for the FRB layer.
5//!
6//! Three reducers, matching the three things the UI renders:
7//! - [`RosterView`] — display identities keyed by user id, merged from every source
8//!   (member list, message authors, voice peers). Last writer with real fields wins.
9//! - [`MessageList`] — an ordered, de-duplicated per-space message list that applies
10//!   create/edit events.
11//! - [`CommunityTree`] — communities and their spaces, applying space create/remove.
12

20// ─── RosterView ───────────────────────────────────────────────────────────────────
21
22/// The display roster: a [`Member`] per user id, the UI joins messages + call peers
23/// against. Merged from many sources; an entry with a real name/picture is never
24/// overwritten by one with empty fields (so a bare member-list `user_id` doesn't
25/// clobber a name learned from a message author).
26#[derive(Clone, Debug, Default, PartialEq)]
27pub struct RosterView {
28    members: HashMap<Uuid, Member>,
29}
30
31impl RosterView {
32    pub fn new() -> Self {
33        Self::default()
34    }
35
36    /// Look up a member's display identity.
37    pub fn get(&self, user_id: Uuid) -> Option<&Member> {
38        self.members.get(&user_id)
39    }
40
41    /// Iterate every known member (unordered). The FRB layer projects this into a
42    /// flat snapshot; the UI sorts/joins as it needs.
43    pub fn iter(&self) -> impl Iterator<Item = &Member> {
44        self.members.values()
45    }
46
47    /// Number of known members.
48    pub fn len(&self) -> usize {
49        self.members.len()
50    }
51
52    pub fn is_empty(&self) -> bool {
53        self.members.is_empty()
54    }
55
56    /// Seed/merge members from a REST load (member list or message-author profiles).
57    pub fn apply_members(&mut self, members: impl IntoIterator<Item = Member>) {
58        for m in members {
59            self.upsert(m);
60        }
61    }
62
63    /// Fold a realtime event into the roster (voice events carry display fields).
64    pub fn apply_event(&mut self, event: &RealtimeEvent) {
65        match event {
66            RealtimeEvent::VoiceJoined(v) => self.upsert(member_from_voice(v)),
67            RealtimeEvent::VoiceState(list) => {
68                for v in list {
69                    self.upsert(member_from_voice(v));
70                }
71            }
72            _ => {}
73        }
74    }
75
76    /// Insert or merge one member. A new entry with a real name/picture upgrades an
77    /// existing bare entry; it never downgrades a populated one to empty.
78    fn upsert(&mut self, incoming: Member) {
79        match self.members.get_mut(&incoming.user_id) {
80            Some(existing) => {
81                if incoming.name.is_some() {
82                    existing.name = incoming.name;
83                }
84                if !incoming.picture.is_empty() {
85                    existing.picture = incoming.picture;
86                }
87            }
88            None => {
89                self.members.insert(incoming.user_id, incoming);
90            }
91        }
92    }
93}
94
95fn member_from_voice(v: &VoiceJoinEvent) -> Member {
96    Member {
97        user_id: v.user_id,
98        name: v.name.clone(),
99        picture: v.picture.clone(),
100    }
101}

103// ─── MessageList ──────────────────────────────────────────────────────────────────
104
105/// An ordered per-space message list. Ordered by `sent_at_ms` then id (the v7-uuid
106/// tiebreak keeps same-ms messages stable). De-duplicates by id so a create event for
107/// a message already loaded via REST is idempotent; an edit updates contents in place.
108#[derive(Clone, Debug, Default, PartialEq)]
109pub struct MessageList {
110    /// Sorted, de-duplicated messages (oldest first).
111    messages: Vec<Message>,
112    /// id → index into `messages` for O(1) edit/dedupe lookup.
113    index: HashMap<Uuid, usize>,
114}
115
116impl MessageList {
117    pub fn new() -> Self {
118        Self::default()
119    }
120
121    /// The messages, oldest first.
122    pub fn messages(&self) -> &[Message] {
123        &self.messages
124    }
125
126    pub fn len(&self) -> usize {
127        self.messages.len()
128    }
129
130    pub fn is_empty(&self) -> bool {
131        self.messages.is_empty()
132    }
133
134    /// Merge a page of messages from a REST load.
135    pub fn apply_page(&mut self, page: impl IntoIterator<Item = Message>) {
136        for m in page {
137            self.upsert(m);
138        }
139    }
140
141    /// Apply a realtime event. Returns `true` if it mutated this list (so the caller
142    /// knows whether to re-snapshot). Ignores events for other spaces / non-message
143    /// events. `space_id` is the space this list represents.
144    pub fn apply_event(&mut self, space_id: Uuid, event: &RealtimeEvent) -> bool {
145        match event {
146            RealtimeEvent::MessageCreated(m) | RealtimeEvent::MessageUpdated(m)
147                if m.space_id == space_id =>
148            {
149                self.upsert(message_from_event(m));
150                true
151            }
152            _ => false,
153        }
154    }
155
156    /// Insert a new message in sorted order, or replace an existing one with the same
157    /// id (an edit). Rebuilds the index on insert (lists are small + bounded per view).
158    fn upsert(&mut self, msg: Message) {
159        if let Some(&i) = self.index.get(&msg.id) {
160            self.messages[i] = msg;
161            return;
162        }
163        let pos = self
164            .messages
165            .partition_point(|m| (m.sent_at_ms, m.id) < (msg.sent_at_ms, msg.id));
166        self.messages.insert(pos, msg);
167        self.reindex_from(pos);
168    }
169
170    fn reindex_from(&mut self, from: usize) {
171        for (i, m) in self.messages.iter().enumerate().skip(from) {
172            self.index.insert(m.id, i);
173        }
174    }
175}
176
177fn message_from_event(m: &MessageEvent) -> Message {
178    Message {
179        id: m.id,
180        author: m.user_id,
181        contents: m.contents.clone(),
182        sent_at_ms: uuid_v7_millis(m.id),
183        edited_at_ms: m.edited_at,
184        attachments: m
185            .attachments
186            .iter()
187            .map(|a| crate::model::Attachment {
188                id: a.id,
189                url: a.url.clone(),
190                content_type: a.content_type.clone(),
191                size_bytes: a.size_bytes.max(0) as u64,
192            })
193            .collect(),
194    }
195}

199/// The community → spaces tree the sidebar renders. Holds communities and, per
200/// community, its spaces; applies `space:create`/`space:remove` realtime events.
201#[derive(Clone, Debug, Default, PartialEq)]
202pub struct CommunityTree {
203    communities: HashMap<Uuid, Community>,
204    /// community id → its spaces (insertion order preserved via a Vec).
205    spaces: HashMap<Uuid, Vec<Space>>,
206}
207
208impl CommunityTree {
209    pub fn new() -> Self {
210        Self::default()
211    }
212
213    pub fn community(&self, id: Uuid) -> Option<&Community> {
214        self.communities.get(&id)
215    }
216
217    /// Iterate every known community (unordered). The FRB layer projects this — with
218    /// each community's [`spaces`](Self::spaces) — into the sidebar snapshot.
219    pub fn communities(&self) -> impl Iterator<Item = &Community> {
220        self.communities.values()
221    }
222
223    /// Spaces of a community, in insertion order (empty slice if unknown community).
224    pub fn spaces(&self, community_id: Uuid) -> &[Space] {
225        self.spaces.get(&community_id).map_or(&[], Vec::as_slice)
226    }
227
228    pub fn community_count(&self) -> usize {
229        self.communities.len()
230    }
231
232    /// Merge communities from a REST load.
233    pub fn apply_communities(&mut self, communities: impl IntoIterator<Item = Community>) {
234        for c in communities {
235            self.communities.insert(c.id, c);
236        }
237    }
238
239    /// Replace a community's space list from a REST load.
240    pub fn set_spaces(&mut self, community_id: Uuid, spaces: Vec<Space>) {
241        self.spaces.insert(community_id, spaces);
242    }
243
244    /// Apply a realtime event; returns `true` if it mutated the tree.
245    pub fn apply_event(&mut self, event: &RealtimeEvent) -> bool {
246        match event {
247            RealtimeEvent::SpaceCreated(s) => {
248                self.upsert_space(space_from_event(s));
249                true
250            }
251            RealtimeEvent::SpaceRemoved { space_id } => self.remove_space(*space_id),
252            _ => false,
253        }
254    }
255
256    fn upsert_space(&mut self, space: Space) {
257        let list = self.spaces.entry(space.community_id).or_default();
258        if let Some(existing) = list.iter_mut().find(|s| s.id == space.id) {
259            *existing = space;
260        } else {
261            list.push(space);
262        }
263    }
264
265    fn remove_space(&mut self, space_id: Uuid) -> bool {
266        let mut removed = false;
267        for list in self.spaces.values_mut() {
268            let before = list.len();
269            list.retain(|s| s.id != space_id);
270            removed |= list.len() != before;
271        }
272        removed
273    }
274}
275
276fn space_from_event(s: &SpaceEvent) -> Space {
277    Space {
278        id: s.id,
279        community_id: s.community_id,
280        name: s.name.clone(),
281        kind: match s.kind.as_str() {
282            "Voice" => SpaceKind::Voice,
283            _ => SpaceKind::Text,
284        },
285        // The create event carries no privacy flag; default public (matches the REST
286        // summary mapping). A subsequent detail/patch load corrects it if needed.
287        is_private: false,
288    }
289}

291/// Same UUIDv7 → unix-ms extraction the REST layer uses, duplicated here so the
292/// reducer stays self-contained (it must derive send time from realtime ids too).
293fn uuid_v7_millis(id: Uuid) -> i64 {
294    let b = id.as_bytes();
295    (((b[0] as u64) << 40)
296        | ((b[1] as u64) << 32)
297        | ((b[2] as u64) << 24)
298        | ((b[3] as u64) << 16)
299        | ((b[4] as u64) << 8)
300        | (b[5] as u64)) as i64
301}

308    fn uid(n: u8) -> Uuid {
309        let mut b = [0u8; 16];
310        b[15] = n;
311        Uuid::from_bytes(b)
312    }
313
314    /// A v7-style uuid whose timestamp bytes encode `ms`, with `n` as a tiebreak tail.
315    fn msg_id(ms: u64, n: u8) -> Uuid {
316        let mut b = [0u8; 16];
317        b[0] = (ms >> 40) as u8;
318        b[1] = (ms >> 32) as u8;
319        b[2] = (ms >> 24) as u8;
320        b[3] = (ms >> 16) as u8;
321        b[4] = (ms >> 8) as u8;
322        b[5] = ms as u8;
323        b[15] = n;
324        Uuid::from_bytes(b)
325    }

384    #[test]
385    fn messages_sorted_and_deduped() {
386        let mut list = MessageList::new();
387        let space = uid(9);
388        let a = msg_id(100, 1);
389        let b = msg_id(200, 1);
390        let c = msg_id(150, 1);
391        // Insert out of order.
392        assert!(list.apply_event(space, &ev_msg(b, space, "b", None)));
393        assert!(list.apply_event(space, &ev_msg(a, space, "a", None)));
394        assert!(list.apply_event(space, &ev_msg(c, space, "c", None)));
395        let order: Vec<_> = list
396            .messages()
397            .iter()
398            .map(|m| m.contents.as_str())
399            .collect();
400        assert_eq!(order, ["a", "c", "b"]);
401
402        // Re-applying the same create is idempotent (deduped by id).
403        assert!(list.apply_event(space, &ev_msg(a, space, "a", None)));
404        assert_eq!(list.len(), 3);
405    }

444    #[test]
445    fn tree_applies_space_create_and_remove() {
446        let mut tree = CommunityTree::new();
447        let cid = uid(8);
448        tree.apply_communities([Community {
449            id: cid,
450            name: "Hounds".into(),
451            default_space: uid(20),
452        }]);
453        tree.set_spaces(
454            cid,
455            vec![Space {
456                id: uid(20),
457                community_id: cid,
458                name: "general".into(),
459                kind: SpaceKind::Text,
460                is_private: false,
461            }],
462        );
463
464        let created = RealtimeEvent::SpaceCreated(SpaceEvent {
465            id: uid(21),
466            community_id: cid,
467            name: "voice".into(),
468            kind: "Voice".into(),
469        });
470        assert!(tree.apply_event(&created));
471        assert_eq!(tree.spaces(cid).len(), 2);
472        assert_eq!(tree.spaces(cid)[1].kind, SpaceKind::Voice);
473
474        // Idempotent create (same id) replaces, doesn't duplicate.
475        assert!(tree.apply_event(&created));
476        assert_eq!(tree.spaces(cid).len(), 2);
477
478        let removed = RealtimeEvent::SpaceRemoved { space_id: uid(20) };
479        assert!(tree.apply_event(&removed));
480        assert_eq!(tree.spaces(cid).len(), 1);
481        assert_eq!(tree.spaces(cid)[0].id, uid(21));
482
483        // Removing an unknown space is a no-op.
484        assert!(!tree.apply_event(&RealtimeEvent::SpaceRemoved { space_id: uid(99) }));
485    }
