1//! Demand-driven simulcast, producer side: [`LayerControl`] folds every
2//! receiver's standing quality demand ([`QualityReport`] via
3//! [`spawn_quality_listener`]) into a [`hound_media::SimulcastPlanner`], driving
4//! the shared LOW-layer supervisor; [`spawn_low_video_publish`] drains the
5//! capture thread's low encode onto its own moq track, announcing it on the
6//! FIRST frame only (dark until actually producing). Invariant: the source
7//! encode is never dropped — the low layer is purely additive, so any low-layer
8//! fault costs a thumbnail, never the full-quality source. That additive-only
9//! design is why the low layer runs unconditionally: a receiver only switches to
10//! the low track for a small/hidden tile it explicitly requested.
11
12use super::*;

14/// The per-source producer seam between the quality listener (records each receiver's
15/// demand into a [`SimulcastPlanner`]) and the low-track publish supervisor (spawns /
16/// tears the low encoder). Mirrors [`AdaptShared`] as a shared producer control block;
17/// the planner is touched ~1 Hz (per report / roster change), never per frame, so the
18/// `Mutex` is free.
19pub(super) struct LayerControl {
20    planner: std::sync::Mutex<hound_media::SimulcastPlanner>,
21    /// Drives the low-publish supervisor: `Some(bps)` = run a shared low layer at this
22    /// bitrate; `None` = tear it down (no receiver wants a below-source layer).
23    pub(super) low_active: watch::Sender<Option<u64>>,
24    /// Lock-free snapshot of the planner's capped-peer set, refreshed on every demand
25    /// change (`set_request`/`retain_peers`) — read on the feedback listener's hot
26    /// path so a per-report fold never contends the `Mutex` the ~1Hz demand listener
27    /// holds. See `Self::capped_peers`.
28    capped: ArcSwap<std::collections::BTreeSet<u32>>,
29}
30
31impl LayerControl {
32    pub(super) fn new(source_bps: u64) -> Self {
33        let planner = hound_media::SimulcastPlanner::new(source_bps);
34        Self {
35            planner: std::sync::Mutex::new(planner),
36            low_active: watch::channel(None).0,
37            capped: ArcSwap::from_pointee(std::collections::BTreeSet::new()),
38        }
39    }
40
41    fn lock(&self) -> std::sync::MutexGuard<'_, hound_media::SimulcastPlanner> {
42        self.planner.lock().unwrap_or_else(|e| e.into_inner())
43    }
44
45    /// Push the planner's effective low-layer target to the supervisor (no-op if same).
46    fn sync_low(&self, plan: hound_media::LayerPlan) {
47        self.low_active.send_if_modified(|cur| {
48            if *cur == plan.low_layer_bps {
49                false
50            } else {
51                tracing::info!(
52                    low_bps = ?plan.low_layer_bps,
53                    send_source = plan.send_source,
54                    "simulcast[send]: low layer active-state changed"
55                );
56                *cur = plan.low_layer_bps;
57                true
58            }
59        });
60    }
61
62    /// Record a receiver's standing request and re-derive the low-layer target.
63    pub(super) fn set_request(&self, peer: u32, req: hound_media::QualityRequest) {
64        let (changed, plan, capped) = {
65            let mut p = self.lock();
66            let changed = p.set_request(peer, req);
67            (changed, p.plan(), p.capped_peers())
68        };
69        self.capped.store(std::sync::Arc::new(capped));
70        if changed {
71            tracing::info!(?plan, "simulcast: receiver demand changed");
72        }
73        self.sync_low(plan);
74    }
75
76    /// Drop departed receivers from the cap map (roster shrank).
77    fn retain_peers(&self, keep: impl Fn(u32) -> bool) {
78        let (plan, capped) = {
79            let mut p = self.lock();
80            p.retain_peers(keep);
81            (p.plan(), p.capped_peers())
82        };
83        self.capped.store(std::sync::Arc::new(capped));
84        self.sync_low(plan);
85    }
86
87    /// Peers currently requesting a capped (below-source) layer — a lock-free snapshot
88    /// read (no `Mutex` contention). Excluded from the SOURCE track's BWE feedback fold
89    /// (see `spawn_feedback_listener` in adapt.rs) so a low-layer receiver — including one
90    /// still transiently subscribed to the source track before its low track's first
91    /// frame — can't drag the source adaptation down.
92    pub(super) fn capped_peers(&self) -> std::sync::Arc<std::collections::BTreeSet<u32>> {
93        self.capped.load_full()
94    }
95}

97/// Spawn the producer side of demand-driven simulcast: watch `quality` and, when a
98/// request names OUR video track of `source`, record it into `layers` (driving the
99/// low-layer supervisor). A slow tick drops requesters who left the call. "Our track"
100/// is resolved exactly like the feedback/keyframe listeners.
101pub(super) fn spawn_quality_listener(
102    mut quality: broadcast::Receiver<QualityReport>,
103    view_rx: watch::Receiver<SessionView>,
104    self_id: PeerId,
105    source: Source,
106    layers: Arc<LayerControl>,
107) -> JoinHandle<()> {
108    call_runtime().spawn(async move {
109        let mut prune = tokio::time::interval(Duration::from_secs(2));
110        prune.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
111        loop {
112            tokio::select! {
113                r = quality.recv() => match r {
114                    Ok(report) => {
115                        let is_ours = {
116                            let view = view_rx.borrow();
117                            view.track(report.track).is_some_and(|t| {
118                                t.peer == self_id
119                                    && t.kind == MediaKind::Video
120                                    && t.source == source
121                            })
122                        };
123                        if is_ours {
124                            let req = match report.cap_bps {
125                                Some(bps) => hound_media::QualityRequest::Capped(bps),
126                                None => hound_media::QualityRequest::Source,
127                            };
128                            layers.set_request(report.from.0, req);
129                        }
130                    }
131                    Err(broadcast::error::RecvError::Lagged(_)) => {}
132                    Err(broadcast::error::RecvError::Closed) => break,
133                },
134                _ = prune.tick() => {
135                    let present: std::collections::HashSet<u32> =
136                        view_rx.borrow().peers().map(|p| p.id.0).collect();
137                    layers.retain_peers(|peer| present.contains(&peer));
138                }
139            }
140        }
141    })
142}

144/// Publish a LOW simulcast layer: drain the capture thread's downscaled encode and
145/// forward it on its own moq track. The control-plane `Publish` announce is sent on
146/// the FIRST frame only, so the low track stays invisible to receivers until it's
147/// actually producing (dark while no one demands it). No own BWE tick — the low
148/// layer's bitrate is the planner's cap, applied by the capture thread's `low_adapt`.
149///
150/// NB keyframe-on-(re)subscribe for the low layer is routed by the keyframe listener
151/// (a PLI for the low track must arm the low encoder's IDR, not the source's) — wired
152/// when the layer is taken live; while dark this never runs.
153pub(super) fn spawn_low_video_publish(
154    relay: Arc<MoqRelaySession>,
155    path: String,
156    track_name: &'static str,
157    source: Source,
158    codec: CtlCodec,
159    mut low_rx: tokio::sync::mpsc::Receiver<WireFrame>,
160    control_outbound: tokio::sync::mpsc::UnboundedSender<ClientMessage>,
161) -> JoinHandle<()> {
162    call_runtime().spawn(async move {
163        let mut publisher = match relay.publish_track(&path, track_name, VIDEO_PRIORITY) {
164            Ok(p) => p,
165            Err(e) => {
166                tracing::error!(%e, track = track_name, "low: failed to publish track");
167                return;
168            }
169        };
170        let mut announced = false;
171        while let Some(frame) = low_rx.recv().await {
172            if !announced {
173                let _ = control_outbound.send(ClientMessage::Publish {
174                    track: TrackDecl {
175                        kind: MediaKind::Video,
176                        source,
177                        codec,
178                        low: true,
179                    },
180                });
181                announced = true;
182                tracing::info!(
183                    track = track_name,
184                    "simulcast[send]: low layer now publishing"
185                );
186            }
187            if let Err(e) = publish_one(&mut publisher, &frame) {
188                tracing::warn!(%e, track = track_name, "low: publish frame failed");
189            }
190        }
191    })
