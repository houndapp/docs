1//! Sender-side BWE/congestion adaptation. The camera +
2//! screen video the client PUBLISHES adapts its bitrate/resolution to network
3//! conditions and serves keyframes on demand. The brain —
4//! [`hound_media::VideoAdaptation`] — is shared; this is the local IO glue:
5//!
6//! * [`MediaFeedback`] — the in-process form of `ServerMessage::MediaFeedbackReceived`
7//!   fanned out of the control reader to the per-track feedback listeners.
8//! * [`AdaptShared`] — the atomics seam between the async 1 s adaptation TICK (owns
9//!   the `VideoAdaptation` core, reads `conn.rtt()`) and the `!Send` capture/encode
10//!   thread (owns the encoder, applies the decision per frame). One writer per field.
11//! * [`apply_adaptation_tick`] — the pure glue: drain backpressure + a pending PLI +
12//!   the worst receiver-reported loss, advance the core, write the decision back.
13//! * [`spawn_keyframe_listener`] / [`spawn_feedback_listener`] — the producer side of
14//!   the PLI relay + the receiver→producer BWE loop, each FILTERED to OUR track of a
15//!   given `Source` (Camera vs Screen) via the `SessionView`.
16//!
17//! With good RTT + no loss the ladder sits at the top rung, so a healthy network
18//! is byte-identical to the pre-adaptation publish path.
19

22/// Open an encoder at `target`, falling back DOWN the ladder
23/// ([`hound_engine::types::ENCODE_PRIORITY`]: AV1 → H.265 → H.264) to the first codec
24/// that opens. Safe to descend without re-checking decode caps: the negotiated
25/// `target` means every peer decodes it, and real decode hardware is monotonic
26/// (AV1-capable ⇒ H.265-capable ⇒ H.264), so every lower rung is decodable too.
27/// H.264 is the always-buildable floor, so this returns `None` only if even H.264
28/// won't open.
29pub(super) fn open_encoder_laddered(
30    target: VideoCodec,
31    dims: Dimensions,
32    fps: Fps,
33    bitrate: Bitrate,
34    cadence: KeyframeCadence,
35) -> Option<hound_engine::api::VideoEncoder> {
36    use hound_engine::types::ENCODE_PRIORITY;
37    let start = ENCODE_PRIORITY
38        .iter()
39        .position(|&c| c == target)
40        .unwrap_or(0);
41    for &c in &ENCODE_PRIORITY[start..] {
42        if let Ok(e) = Engine::new().video_encoder(c, dims, fps, bitrate, cadence) {
43            return Some(e);
44        }
45    }
46    None
47}

49/// Owns a [`JoinHandle`] and aborts the task on drop. Ties an adaptation listener's
50/// lifetime to its publish task: when the publish task returns (or is aborted on
51/// `leave`/reconnect teardown), the listener is aborted too.
52pub(super) struct AbortOnDrop(pub(super) JoinHandle<()>);
53
54impl Drop for AbortOnDrop {
55    fn drop(&mut self) {
56        self.0.abort();
57    }
58}

60/// One receiver reception report relayed to us (the producer) for one of our
61/// published tracks — the in-process form of [`ServerMessage::MediaFeedbackReceived`].
62/// The per-track feedback listener folds the loss fraction into its sender-side BWE
63/// (serve-the-worst-leg across reporters). Counts are deltas since the reporter's
64/// previous report.
65#[derive(Debug, Clone, Copy)]
66pub(super) struct MediaFeedback {
67    /// Our published track the report concerns.
68    pub(super) track: TrackId,
69    /// The reporting peer (so a producer serving many receivers can attribute it).
70    /// Unused today — carried to mirror the wire shape and keep the broadcast type
71    /// stable; per-layer feedback partitioning is the intended consumer.
72    #[allow(dead_code)]
73    pub(super) from: PeerId,
74    /// Frames the receiver decoded since its last report (loss-fraction denominator).
75    pub(super) received: u32,
76    /// Frames the receiver determined missing/skipped since its last report.
77    pub(super) lost: u32,
78    /// Receiver-reported jitter (µs), if present. Dropped here — the sender folds
79    /// only the loss fraction today (kept on the struct to mirror the wire shape).
80    #[allow(dead_code)]
81    pub(super) jitter_us: Option<i64>,
82}

84/// A receiver's standing quality demand for one of OUR published video tracks — the
85/// in-process form of [`ServerMessage::QualityRequested`]. The per-track quality
86/// listener feeds it into a [`hound_media::SimulcastPlanner`] (per-receiver cap map);
87/// the resulting [`LayerPlan`](hound_media::LayerPlan) decides whether to spin up the

89#[derive(Debug, Clone, Copy)]

93    /// The requesting peer (the planner coalesces every receiver's cap).
94    pub(super) from: PeerId,
95    /// `None` = "I can take source"; `Some(bps)` = "I want at most this many bits/sec".
96    pub(super) cap_bps: Option<u64>,

99/// The shared seam between the async **adaptation tick** (owns the sans-IO
100/// [`VideoAdaptation`] core + observes RTT) and the `!Send` **capture/encode thread**
101/// (owns the encoder, applies the decision per frame). Plain atomics — no lock
102/// crosses the thread/async boundary and each field has exactly one writer.
103///
104/// Data flow:
105/// * capture thread → tick: `produced` / `dropped` (per-tick backpressure).
106/// * tick → capture thread: `target_bps` (encoder retarget), `target_fps` (fps gate),
107///   `target_w/target_h` (downscale to the rung), `paused` (AudioOnly ⇒ skip
108///   encode/publish), `force_keyframe` (a throttle-approved IDR).
109/// * keyframe listener → tick: `pli_pending`; feedback listener → tick:
110///   `remote_loss_bps`/`remote_reported`.
111#[derive(Default)]
112pub(super) struct AdaptShared {
113    /// Encoded access units handed to the publisher since the last tick.
114    produced: AtomicU32,
115    /// Of those, how many were dropped at the bounded channel (publisher behind).
116    dropped: AtomicU32,
117    /// The encoder's current target bitrate (bits/sec); the capture thread retargets
118    /// only when it changes. `0` until the first decision.
119    pub(super) target_bps: AtomicU64,
120    /// The chosen rung's frame rate (fps); the capture thread downsamples to it.
121    /// `0` = no limit (no decision yet / AudioOnly — the publish path is skipped).
122    pub(super) target_fps: AtomicU32,
123    /// The chosen rung's encode geometry (width/height); the capture thread downscales
124    /// the frame to this when smaller than capture. `0` = no decision yet / AudioOnly.
125    target_w: AtomicU32,
126    target_h: AtomicU32,
127    /// Whether video is paused (the ladder chose AudioOnly). The capture thread keeps
128    /// capturing (self-preview, A/V-sync clock) but skips encode/publish.
129    pub(super) paused: AtomicBool,
130    /// A PLI arrived for our track (set by the keyframe listener); the tick drains it
131    /// into the keyframe throttle to coalesce a storm.
132    pli_pending: AtomicBool,
133    /// Latched (never cleared) once ANY keyframe request named our track — proof a
134    /// receiver wants this video. The publish loop's live gate accepts it as a
135    /// fallback for the moq subscriber signal, which can be lost when a broadcast
136    /// is replaced on the same path (a share restart): every subscribe sends
137    /// `RequestKeyframe`, so demand always reaches us on the control plane even
138    /// when `used()` never fires.
139    pub(super) demand_seen: AtomicBool,
140    /// Worst receiver-reported loss fraction this interval, in basis points
141    /// (loss × 10_000), max-reduced across reporters. Read+cleared by the tick.
142    remote_loss_bps: AtomicU32,
143    /// Whether any receiver reported this interval — lets the tick pass `None` (no
144    /// remote signal) vs `Some(0.0)` (a receiver explicitly reported a clean leg).
145    remote_reported: AtomicBool,
146    /// A throttle-approved keyframe is due — the capture thread swap-clears this and
147    /// forces the encoder's next frame to an IDR.
148    pub(super) force_keyframe: AtomicBool,
149    /// The codec the encoder should be producing, as a [`codec_rank`] byte. Seeded
150    /// at publish from the roster pick and re-driven by the mid-call re-negotiator
151    /// ([`spawn_codec_renegotiator`]); the encode loop rebuilds its encoder when this
152    /// no longer matches what it built. Defaults to `0` = H.264 (the floor).
153    pub(super) target_codec: AtomicU8,
154}

156impl AdaptShared {
157    /// Record one receiver's reception report, max-reducing into this interval's worst
158    /// loss fraction (serve-the-worst-leg). `received`/`lost` are deltas since the
159    /// reporter's previous report. An empty report (`received + lost == 0`) is ignored.
160    fn record_remote_feedback(&self, received: u32, lost: u32) {
161        let total = received as u64 + lost as u64;
162        if total == 0 {
163            return;
164        }
165        // Quantize to basis points (loss × 10_000), rounding to nearest.
166        let bps = ((lost as u64 * 10_000 + total / 2) / total).min(10_000) as u32;
167        self.remote_reported.store(true, Ordering::Relaxed);
168        let mut cur = self.remote_loss_bps.load(Ordering::Relaxed);
169        while bps > cur {
170            match self.remote_loss_bps.compare_exchange_weak(
171                cur,
172                bps,
173                Ordering::Relaxed,
174                Ordering::Relaxed,
175            ) {
176                Ok(_) => break,
177                Err(observed) => cur = observed,
178            }
179        }
180    }
181
182    /// Drain this interval's worst receiver-reported loss fraction (and clear it).
183    /// `None` when no receiver reported — so the tick does not pin the loss-equivalent
184    /// to 0 and mask the local signals on a feedback-free interval.
185    fn take_remote_loss(&self) -> Option<f64> {
186        if self.remote_reported.swap(false, Ordering::Relaxed) {
187            Some(self.remote_loss_bps.swap(0, Ordering::Relaxed) as f64 / 10_000.0)
188        } else {
189            self.remote_loss_bps.store(0, Ordering::Relaxed);
190            None
191        }
192    }
193}

195/// Read the connection RTT (the one bit of IO) and advance the adaptation one tick.
196/// Splitting the RTT read out keeps [`apply_adaptation_tick`] pure + unit-testable.
197pub(super) fn run_adaptation_tick(
198    adaptation: &mut VideoAdaptation,
199    shared: &AdaptShared,
200    conn: &iroh::endpoint::Connection,
201    now_us: i64,
202    stats: Option<(&CallStats, bool)>,
203) {
204    use iroh::endpoint::PathId;
205    let rtt_us = conn
206        .rtt(PathId::ZERO)
207        .map(|d| d.as_micros().min(i64::MAX as u128) as i64);
208    apply_adaptation_tick(adaptation, shared, rtt_us, now_us, stats);
209}

211/// The pure glue between [`AdaptShared`] (the atomics the capture thread reads/writes)
212/// and the sans-IO [`VideoAdaptation`] core, with RTT passed in. Drains the
213/// backpressure counters + a pending PLI + the worst receiver-reported loss, advances
214/// the core, and writes the decision back (target bitrate/fps/geometry, paused,
215/// forced keyframe). No IO here — unit-testable by driving the atomics directly.
216fn apply_adaptation_tick(
217    adaptation: &mut VideoAdaptation,
218    shared: &AdaptShared,
219    rtt_us: Option<i64>,
220    now_us: i64,
221    // RTC-debug stats sink: `(registry, is_screen)`. `None` in unit tests. RTT is
222    // transport-wide; the decision (target/geometry/rung/loss) is per-track.
223    stats: Option<(&CallStats, bool)>,
224) {
225    let produced = shared.produced.swap(0, Ordering::Relaxed);
226    let dropped = shared.dropped.swap(0, Ordering::Relaxed);
227
228    // Drain a pending PLI into the throttle (coalesces a storm into one IDR).
229    if shared.pli_pending.swap(false, Ordering::Relaxed) {
230        adaptation.on_pli();
231    }
232
233    let remote_loss = shared.take_remote_loss();
234
235    let decision = adaptation.on_tick(
236        &AdaptSignals {
237            produced,
238            dropped,
239            rtt_us,
240            remote_loss,
241        },
242        now_us,
243    );
244
245    shared
246        .target_bps
247        .store(decision.target_bps, Ordering::Relaxed);
248    // The rung's fps + geometry drive the capture thread; AudioOnly carries no rung
249    // (the publish path is skipped while paused anyway).
250    let (target_fps, target_w, target_h) = match decision.quality {
251        Quality::Video(rung) => (rung.fps, rung.width, rung.height),
252        Quality::AudioOnly => (0, 0, 0),
253    };
254    shared.target_fps.store(target_fps, Ordering::Relaxed);
255    shared.target_w.store(target_w, Ordering::Relaxed);
256    shared.target_h.store(target_h, Ordering::Relaxed);
257    let was_paused = shared
258        .paused
259        .swap(decision.is_audio_only(), Ordering::Relaxed);
260
261    // A throttle-approved keyframe? Signal the encode loop via the existing seam.
262    if adaptation.poll_keyframe(now_us) {
263        shared.force_keyframe.store(true, Ordering::Relaxed);
264    }
265
266    // Record the RTC-debug stats: transport RTT + this track's outbound decision.
267    if let Some((stats, is_screen)) = stats {
268        if let Some(rtt) = rtt_us {
269            stats.set_rtt_us(rtt);
270        }
271        let loss_bps = (remote_loss.unwrap_or(0.0) * 10_000.0) as u32;
272        let rung_label = match decision.quality {
273            Quality::Video(rung) => format!("{}x{}@{}", rung.width, rung.height, rung.fps),
274            Quality::AudioOnly => "audio_only".to_string(),
275        };
276        if is_screen {
277            stats.set_screen_decision(
278                decision.target_bps,
279                target_w,
280                target_h,
281                &rung_label,
282                loss_bps,
283            );
284        } else {
285            stats.set_video_decision(
286                decision.target_bps,
287                target_w,
288                target_h,
289                &rung_label,
290                loss_bps,
291            );
292        }
293    }
294
295    if decision.quality_changed {
296        match decision.quality {
297            Quality::Video(rung) => tracing::info!(
298                target_bps = decision.target_bps,
299                width = rung.width,
300                height = rung.height,
301                fps = rung.fps,
302                rtt_us,
303                produced,
304                dropped,
305                "call: adapt → video rung"
306            ),
307            Quality::AudioOnly => tracing::info!(
308                target_bps = decision.target_bps,
309                rtt_us,
310                "call: adapt → AudioOnly (video paused)"
311            ),
312        }
313    } else if was_paused && !decision.is_audio_only() {
314        tracing::info!("call: adapt → video resumed");
315    }
316}

318/// Spawn the producer side of the PLI relay: watch the `keyframe_tx` broadcast and,
319/// when the requested `TrackId` is OUR video track of `source` (Camera vs Screen),
320/// mark a PLI pending in [`AdaptShared`]. The adaptation tick drains it through the
321/// throttle, coalescing a storm into a single IDR. "Our track" is resolved from the
322/// `SessionView` (a video track whose `peer == self` AND `source == source`).
323pub(super) fn spawn_keyframe_listener(
324    mut requests: broadcast::Receiver<TrackId>,
325    view_rx: watch::Receiver<SessionView>,
326    self_id: PeerId,
327    source: Source,
328    adapt: Arc<AdaptShared>,
329    // A PLI for OUR LOW track sets this (the capture loop forces a low IDR), instead
330    // of the source's `pli_pending` — so a receiver subscribing to the low layer gets
331    // a fresh keyframe without waiting for the periodic one. `None` for tracks with no
332    // low layer (e.g. camera today).
333    low_keyframe: Option<Arc<AtomicBool>>,
334) -> JoinHandle<()> {
335    call_runtime().spawn(async move {
336        loop {
337            match requests.recv().await {
338                Ok(track) => {
339                    let (is_ours, is_low) = {
340                        let view = view_rx.borrow();
341                        match view.track(track) {
342                            Some(t)
343                                if t.peer == self_id
344                                    && t.kind == MediaKind::Video
345                                    && t.source == source =>
346                            {
347                                (true, t.low)
348                            }
349                            _ => (false, false),
350                        }
351                    };
352                    if is_ours {
353                        adapt.demand_seen.store(true, Ordering::Relaxed);
354                        match (is_low, &low_keyframe) {
355                            (true, Some(lk)) => lk.store(true, Ordering::Relaxed),
356                            _ => adapt.pli_pending.store(true, Ordering::Relaxed),
357                        }
358                    }
359                }
360                // Lagged: a coalesced keyframe is exactly the debounce we want —
361                // mark pending and continue.
362                Err(broadcast::error::RecvError::Lagged(_)) => {
363                    adapt.pli_pending.store(true, Ordering::Relaxed);
364                }
365                // Closed = the broadcast sender dropped (session teardown).
366                Err(broadcast::error::RecvError::Closed) => break,
367            }
368        }
369    })
370}

372/// Spawn the producer side of the receiver→producer BWE loop: watch the `feedback_tx`
373/// broadcast and, when a report names OUR video track of `source`, fold its loss into
374/// [`AdaptShared`] (max-reduced across reporters this interval — serve the worst leg).
375/// The adaptation tick drains it into the sender-side rate controller. Reports for
376/// other tracks (or a stale id) are ignored. "Our track" is resolved exactly like the
377/// keyframe listener.
378pub(super) fn spawn_feedback_listener(
379    mut feedback: broadcast::Receiver<MediaFeedback>,
380    view_rx: watch::Receiver<SessionView>,
381    self_id: PeerId,
382    source: Source,
383    adapt: Arc<AdaptShared>,
384) -> JoinHandle<()> {
385    call_runtime().spawn(async move {
386        loop {
387            match feedback.recv().await {
388                Ok(report) => {
389                    let is_ours = {
390                        let view = view_rx.borrow();
391                        view.track(report.track).is_some_and(|t| {
392                            // SOURCE track only: a low-layer receiver reports on the
393                            // `_l0` track (also Video/`source`), and folding its loss
394                            // here would let a constrained low viewer drag the source's
395                            // BWE down for the focused viewers. The low layer runs at the
396                            // planner's cap, not loss-based adaptation.
397                            t.peer == self_id
398                                && t.kind == MediaKind::Video
399                                && t.source == source
400                                && !t.low
401                        })
402                    };
403                    if is_ours {
404                        adapt.record_remote_feedback(report.received, report.lost);
405                    }
406                }
407                // Lagged: the next interval's report supersedes the dropped ones
408                // (per-interval deltas) — keep going.
409                Err(broadcast::error::RecvError::Lagged(_)) => {}
410                Err(broadcast::error::RecvError::Closed) => break,
411            }
412        }
413    })
414}

416/// The shared per-frame encode→adapt→publish core for both video capture loops.
417/// Retargets the
418/// encoder to the BWE's latest bitrate (only on a real move), honours a
419/// throttle-approved keyframe (swap-clears so it fires once), downsamples to the
420/// rung's fps, downscales to the rung geometry when smaller than capture, encodes,
421/// and forwards each access unit via `forward`. `forward` returns `false` on
422/// backpressure (its bounded channel was full); the caller's drop bumps `dropped`,
423/// which the adaptation tick reads as the congestion signal. A forced keyframe is
424/// re-armed when it is dropped, so the next captured frame re-emits it.
425#[allow(clippy::too_many_arguments)]
426pub(super) fn encode_and_publish_adaptive(
427    frame: &hound_engine::types::RawFrame,
428    encoder: &mut hound_engine::api::VideoEncoder,
429    adapt: &AdaptShared,
430    applied_bps: &mut u32,
431    fps_gate: &mut hound_media::FrameRateGate,
432    capture_dims: Dimensions,
433    mut forward: impl FnMut(WireFrame) -> bool,
434    frame_seq: &mut u32,
435    sender: u16,
436    track: TrackKind,
437    stats: &CallStats,
438) {
439    use hound_engine::backend::VideoEncode;
440
441    // Retarget if the BWE moved the target since we last applied it. The rung target
442    // is an H.264-equivalent quality ceiling; scale it down for a more efficient codec
443    // so the SAME quality ships at fewer bits (the receive-side packet CPU, which
444    // dominates a screenshare, scales with bits — this is where the codec win lands).
445    let target = adapt.target_bps.load(Ordering::Relaxed);
446    let scaled = (target as f64 * codec_bitrate_factor(encoder.codec())) as u64;
447    let target32 = scaled.min(u32::MAX as u64) as u32;
448    if let Some(rate) = Bitrate::try_from_bps(target32)
449        && target32 != *applied_bps
450    {
451        if let Err(e) = encoder.set_bitrate(rate) {
452            tracing::warn!("call: set_bitrate({target32}) failed: {e}");
453        } else {
454            *applied_bps = target32;
455        }
456    }
457
458    // A forced keyframe (new subscriber / PLI) must always emit — gating it would
459    // starve a fresh decoder of its IDR. Otherwise downsample to the rung's fps. A
460    // frame dropped here is a deliberate downsample, NOT backpressure (don't bump
461    // `dropped`).
462    let force_keyframe = adapt.force_keyframe.swap(false, Ordering::Relaxed);
463    let target_fps = adapt.target_fps.load(Ordering::Relaxed);
464    if !force_keyframe && !fps_gate.should_emit(frame.timestamp_us.micros(), target_fps) {
465        return;
466    }
467    if force_keyframe {
468        encoder.request_keyframe();
469    }
470
471    // Resolution-rung downscaling is DISABLED. The engine's `VideoEncoder` is created
472    // at a fixed geometry (`capture_dims`) and rejects any differently-sized frame
473    // ("requested format/resolution not supported: frame WxH != encoder W'xH'"), so
474    // feeding a rung-downscaled frame would drop ~every frame (produced≈0): outgoing
475    // video silently dies while the failed encodes spam the log and burn CPU.
476    //
477    // The captured frame is already at `capture_dims` (the engine scales native →
478    // capture), so encoding it as-is always matches the encoder. We still adapt the
479    // two primary levers — bitrate (above) and fps (the gate) — which carry the bulk
480    // of the congestion response. Honoring the rung's *resolution* needs the encoder
481    // rebuilt whenever the geometry changes; that's the follow-up that re-enables the
482    // bottom of the ladder. `capture_dims` is retained for that work.
483    let _ = capture_dims;
484    let encode_frame = std::borrow::Cow::Borrowed(frame);
485
486    let wire_codec = encoder.codec().to_wire();
487    let is_screen = matches!(track, TrackKind::ScrVideo);
488    // Time the HW-encode call for the RTC-debug encode-latency stat.
489    let encode_start = std::time::Instant::now();
490    let encoded = encoder.encode(&encode_frame);
491    let encode_us = encode_start.elapsed().as_micros().min(i64::MAX as u128) as i64;
492    match encoded {
493        Ok(chunks) => {
494            for chunk in chunks {
495                adapt.produced.fetch_add(1, Ordering::Relaxed);
496                let is_keyframe = chunk.keyframe;
497                // RTC-debug: one encoded AU (bytes drive send_kbps; keyframe count).
498                let bytes = chunk.data.len();
499                if is_screen {
500                    stats.screen_encoded(bytes, encode_us, is_keyframe);
501                } else {
502                    stats.video_encoded(bytes, encode_us, is_keyframe);
503                }
504                let wire = WireFrame {
505                    sender,
506                    track,
507                    codec: wire_codec,
508                    keyframe: is_keyframe,
509                    frame_seq: *frame_seq,
510                    timestamp_us: chunk.timestamp_us.micros(),
511                    data: chunk.data,
512                };
513                *frame_seq = frame_seq.wrapping_add(1);
514                if !forward(wire) {
515                    adapt.dropped.fetch_add(1, Ordering::Relaxed);
516                    if is_screen {
517                        stats.screen_publish_drop();
518                    } else {
519                        stats.video_publish_drop();
520                    }
521                    // A dropped keyframe must re-arm the forced-IDR flag so the next
522                    // captured frame re-emits it.
523                    if is_keyframe {
524                        adapt.force_keyframe.store(true, Ordering::Relaxed);
525                    }
526                }
527            }
528        }
529        Err(e) => tracing::warn!("call: video encode failed: {e}"),
530    }
531}

533/// Drive one video publish loop: the 1 s adaptation tick + frame forwarding, with
534/// the **live gate** deciding when forwarding starts. The tick runs from the first
535/// iteration — before any subscriber — so PLIs are serviced and the BWE keeps
536/// state even while dark; only *forwarding* waits for proof of a receiver.
537///
538/// Live = the moq subscriber signal ([`MoqTrackPublisher::subscribed`]) OR
539/// control-plane demand (`adapt.demand_seen`, latched by the keyframe listener).
540/// The fallback matters when a broadcast is replaced on the same path (a share
541/// restart): moq-net keeps the newcomer as a *backup* until the old broadcast
542/// closes, so `used()` can arrive late or not at all — while the receiver's
543/// subscribe always sends `RequestKeyframe`. Going live forces a fresh IDR, so
544/// the first delivered group always opens decodable; frames before the gate are
545/// dropped (a dropped keyframe re-arms the forced IDR).
546///
547/// `before_tick` runs ahead of each adaptation tick (the screen path reloads its
548/// live quality preference there). Returns when the encode channel closes or the
549/// track dies.
550#[allow(clippy::too_many_arguments)]
551pub(super) async fn drive_video_publish(
552    publisher: &mut MoqTrackPublisher,
553    enc_rx: &mut tokio::sync::mpsc::Receiver<WireFrame>,
554    adapt: &AdaptShared,
555    adaptation: &mut VideoAdaptation,
556    rtt_conn: &iroh::endpoint::Connection,
557    stats: &CallStats,
558    is_screen: bool,
559    track_label: &'static str,
560    mut before_tick: impl FnMut(&mut VideoAdaptation),
561) {
562    let adapt_clock = CaptureClock::start();
563    let mut tick = tokio::time::interval(ADAPT_TICK_INTERVAL);
564    tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
565
566    let subscribed = publisher.subscribed();
567    tokio::pin!(subscribed);
568    let mut live = false;
569    // Live but not yet opened on a keyframe (the go-live forced IDR in flight).
570    let mut primed = false;
571    let go_live = |live: &mut bool| {
572        *live = true;
573        adapt.force_keyframe.store(true, Ordering::Relaxed);
574        tracing::info!(
575            track = track_label,
576            "call: video publish live — forcing opening IDR"
577        );
578    };
579
580    loop {
581        tokio::select! {
582            biased;
583            r = &mut subscribed, if !live => {
584                if r.is_err() {
585                    return; // track closed before anyone attached
586                }
587                go_live(&mut live);
588            }
589            _ = tick.tick() => {
590                before_tick(adaptation);
591                run_adaptation_tick(adaptation, adapt, rtt_conn, adapt_clock.now(), Some((stats, is_screen)));
592                if !live && adapt.demand_seen.load(Ordering::Relaxed) {
593                    go_live(&mut live);
594                }
595            }
596            frame = enc_rx.recv() => {
597                let Some(frame) = frame else { return };
598                if !live {
599                    if adapt.demand_seen.load(Ordering::Relaxed) {
600                        go_live(&mut live);
601                    } else {
602                        // Dark: no receiver yet. Don't ship (or queue) stale frames;
603                        // a dropped keyframe re-arms so the first live frame is an IDR.
604                        if frame.keyframe {
605                            adapt.force_keyframe.store(true, Ordering::Relaxed);
606                        }
607                        continue;
608                    }
609                }
610                // The publish stream must OPEN on a keyframe (a delta has no group
611                // to ride); skip pre-IDR deltas still in flight from before the
612                // go-live forced keyframe.
613                if !primed {
614                    if !frame.keyframe {
615                        continue;
616                    }
617                    primed = true;
618                }
619                // Tell the throttle a keyframe went out so it measures the next
620                // forced-IDR interval from here.
621                if frame.keyframe {
622                    adaptation.on_keyframe_emitted(adapt_clock.now());
623                }
624                if let Err(e) = publish_one(publisher, &frame) {
625                    tracing::warn!(%e, track = track_label, "call: publish video frame failed");
626                }
627            }
628        }
629    }
630}

632/// Watch the roster and re-negotiate the encode codec as peers join/leave, writing
633/// the decision into `adapt.target_codec` for the encode loop to act on. `encode_caps`
634/// is probed once by the caller (the probe isn't free) and reused here. Lives for the
635/// publish — ends when `view_rx` closes (session torn down).
636pub(super) fn spawn_codec_renegotiator(
637    mut view_rx: watch::Receiver<SessionView>,
638    encode_caps: hound_engine::types::VideoCodecCaps,
639    adapt: Arc<AdaptShared>,
640    initial: VideoCodec,
641    track: &'static str,
642) -> JoinHandle<()> {
643    call_runtime().spawn(async move {
644        let mut switch = CodecSwitch::new(initial);
645        loop {
646            let want = negotiate_codec(encode_caps, &view_rx.borrow());
647            if let Some(new) = switch.observe(want, Instant::now(), CODEC_UPGRADE_DEBOUNCE) {
648                adapt.target_codec.store(codec_rank(new), Ordering::Relaxed);
649                tracing::info!(track, codec = ?new, "call: re-negotiated encode codec (roster change)");
650            }
651            // Wake on the next roster change, or when a debounced upgrade is due.
652            let wait = switch
653                .pending_deadline(CODEC_UPGRADE_DEBOUNCE)
654                .map(|d| d.saturating_duration_since(Instant::now()));
655            tokio::select! {
656                r = view_rx.changed() => { if r.is_err() { break; } }
657                _ = async {
658                    match wait {
659                        Some(d) => tokio::time::sleep(d).await,
660                        None => std::future::pending::<()>().await,
661                    }
662                } => {}
663            }
664        }
665    })
666}

670    //! The sender-side adaptation wiring: the `AdaptShared` ↔

678    /// A healthy link (good RTT, no drops, no remote loss) sits at the top video rung:
679    /// the tick decides Video, never pauses, and writes a non-zero target the encode
680    /// loop applies — so a healthy network is byte-identical to no adaptation.
681    #[test]
682    fn healthy_link_stays_on_video_and_writes_a_target() {
683        let mut adaptation = VideoAdaptation::standard(CAM_BITRATE.bps() as u64);
684        let shared = AdaptShared::default();
685        shared.produced.store(30, Ordering::Relaxed);
686        let mut now = 0i64;
687        for _ in 0..5 {
688            apply_adaptation_tick(&mut adaptation, &shared, Some(40_000), now, None);
689            now += 1_000_000;
690            shared.produced.store(30, Ordering::Relaxed);
691        }
692        assert!(
693            !shared.paused.load(Ordering::Relaxed),
694            "healthy link must keep video"
695        );
696        assert!(
697            shared.target_bps.load(Ordering::Relaxed) > 0,
698            "tick must publish a target bitrate"
699        );
700        assert!(
701            shared.target_w.load(Ordering::Relaxed) > 0
702                && shared.target_h.load(Ordering::Relaxed) > 0,
703            "tick must publish a rung geometry"
704        );
705        // Backpressure counters are drained each tick (single-reader race-free swap).
706        assert_eq!(
707            shared.produced.swap(0, Ordering::Relaxed),
708            30,
709            "last store survives until next tick"
710        );

759    /// `record_remote_feedback` quantizes to basis points and max-reduces across
760    /// reporters (serve the worst leg); `take_remote_loss` drains + clears it, and
761    /// returns `None` (not `Some(0.0)`) when nobody reported.
762    #[test]
763    fn remote_feedback_max_reduces_and_drains() {
764        let shared = AdaptShared::default();
765        assert_eq!(
766            shared.take_remote_loss(),
767            None,
768            "no report ⇒ None (don't mask local signals)"
769        );
770
771        // 1 lost of 100 total = 1% = 100 bps; 10 of 100 = 10% = 1000 bps → keep worst.
772        shared.record_remote_feedback(99, 1);
773        shared.record_remote_feedback(90, 10);
774        let loss = shared.take_remote_loss().expect("a report was folded");
775        assert!(
776            (loss - 0.10).abs() < 1e-6,
777            "worst leg (10%) kept, got {loss}"
778        );
779        // Drained: the next interval with no report is None again.
780        assert_eq!(shared.take_remote_loss(), None);
781
782        // An empty report carries no information and is ignored.
783        shared.record_remote_feedback(0, 0);
784        assert_eq!(shared.take_remote_loss(), None, "empty report ignored");
785    }
