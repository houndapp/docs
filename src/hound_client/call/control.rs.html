1//! The control-plane IO shell: [`ControlClient`] drives the `/control` WebSocket
2//! (applies each `ServerMessage` to the `SessionView` reducer, publishes
3//! `CallView` snapshots, exposes the outbound intent queue, and fans the
4//! keyframe/feedback/quality signals out to the per-track listeners), plus
5//! [`dial_relay`] — the iroh endpoint + QUIC connection the MoQ session rides.
6//! Invariant: the returned iroh `Endpoint` must outlive the connection — the SFU
7//! is relay-only reachable, so dropping the endpoint wedges every stream after
8//! the handshake.

14/// The Rust side of the `/control` WebSocket: connects, drives a [`SessionView`],
15/// publishes a [`CallView`] snapshot on every change, and exposes an outbound
16/// intent channel.
17pub(super) struct ControlClient {
18    pub(super) outbound: mpsc::UnboundedSender<ClientMessage>,
19    pub(super) view_rx: watch::Receiver<SessionView>,
20    /// Fan-out of incoming `KeyframeRequested` PLIs (the producer side of the PLI
21    /// relay). The reader forwards every requested `TrackId` here; the per-track
22    /// keyframe listener filters to OUR cam/screen track and arms a forced IDR. A
23    /// `broadcast` so cam + screen listeners each observe it independently.
24    pub(super) keyframe_tx: broadcast::Sender<TrackId>,
25    /// Fan-out of incoming `MediaFeedbackReceived` reception reports (the producer
26    /// side of the receiver→producer BWE loop). The reader forwards every report
27    /// here; the per-track feedback listener folds the loss into OUR track's
28    /// adaptation. A `broadcast` for the same multi-consumer reason.
29    pub(super) feedback_tx: broadcast::Sender<MediaFeedback>,
30    /// Fan-out of incoming `QualityRequested` standing demands (the producer side of
31    /// demand-driven simulcast). The reader forwards every request here; the per-track
32    /// quality listener filters to OUR cam/screen track and drives the layer planner.
33    pub(super) quality_tx: broadcast::Sender<QualityReport>,
34    /// Resolves when the reader loop exits (server closed / read error / `view_tx`
35    /// dropped). The supervisor's [`LiveSession::monitor`] selects on this; the
36    /// reader never pushes a connection-state transition itself — the supervisor
37    /// owns those.
38    pub(super) closed: oneshot::Receiver<()>,
39}
40
41impl ControlClient {
42    pub(super) async fn connect(
43        base_url: &str,
44        space_id: Uuid,
45        token: &str,
46        channels: CallChannels,
47        identities: PeerIdentities,
48    ) -> Result<Self> {
49        use futures::{SinkExt, StreamExt};
50        use tokio_tungstenite::tungstenite::Message;
51
52        let url = control_ws_url(base_url, space_id, token)?;
53        let (ws, _resp) = tokio_tungstenite::connect_async(&url)
54            .await
55            .map_err(|e| HoundError::Realtime(format!("control WS connect failed: {e}")))?;
56        let (mut sink, mut stream) = ws.split();
57
58        let (tx, mut rx) = mpsc::unbounded_channel::<ClientMessage>();
59        // First frame MUST be Hello. Advertise our decode caps so producers only
60        // ever send us a codec we can play.
61        let decode_caps = local_decode_caps();
62        tx.send(ClientMessage::Hello {
63            protocol_version: PROTOCOL_VERSION,
64            decode_caps,
65        })
66        .expect("receiver just created");
67
68        let (view_tx, view_rx) = watch::channel(SessionView::new());
69
70        // Writer task: drains outbound, with a periodic Ping keepalive (a quiet
71        // call sends nothing for minutes and NAT/proxy idle timeouts kill the WS).
72        const KEEPALIVE: std::time::Duration = std::time::Duration::from_secs(20);
73        let writer = call_runtime().spawn(async move {
74            loop {
75                let msg = match tokio::time::timeout(KEEPALIVE, rx.recv()).await {
76                    Ok(Some(m)) => m,
77                    Ok(None) => break, // tx dropped — teardown
78                    Err(_) => {
79                        if sink.send(Message::Ping(Vec::new().into())).await.is_err() {
80                            return;
81                        }
82                        continue;
83                    }
84                };
85                let Ok(json) = serde_json::to_string(&msg) else {
86                    continue;
87                };
88                if sink.send(Message::Text(json.into())).await.is_err() {
89                    break;
90                }
91            }
92            let _ = sink.close().await;
93        });
94
95        // The control-closed signal: the reader fires this when its loop exits, and
96        // the supervisor's `LiveSession::monitor` selects on it. The reader does NOT
97        // push a connection-state transition itself — the supervisor owns those (a WS
98        // drop must heal, not surface `Failed`, except on a fatal re-establish error).
99        let (closed_tx, closed_rx) = oneshot::channel::<()>();
100
101        // Producer-side BWE seams: fan `KeyframeRequested` PLIs and
102        // `MediaFeedbackReceived` reception reports out of the reader to the per-track
103        // adaptation listeners (each `.subscribe()`s a fresh receiver per establish).
104        // A small bounded buffer is plenty — a lag just coalesces into one keyframe /
105        // is superseded by the next per-interval report. Returned so the publish loops
106        // subscribe; the reader holds clones to forward onto.
107        let (keyframe_tx, _keyframe_rx) = broadcast::channel::<TrackId>(16);
108        let (feedback_tx, _feedback_rx) = broadcast::channel::<MediaFeedback>(32);
109        let (quality_tx, _quality_rx) = broadcast::channel::<QualityReport>(32);
110        let keyframe_tx_reader = keyframe_tx.clone();
111        let feedback_tx_reader = feedback_tx.clone();
112        let quality_tx_reader = quality_tx.clone();
113
114        // Reader loop: apply each ServerMessage to the view, publish CallView, forward
115        // roster updates, AND fan keyframe/feedback signals to the adaptation
116        // listeners. On close/error, signal `closed` and exit.
117        let view_out = channels.view.clone();
118        call_runtime().spawn(async move {
119            let _writer = writer;
120            let mut view = SessionView::new();
121            while let Some(frame) = stream.next().await {
122                match frame {
123                    Ok(Message::Text(text)) => {
124                        let Ok(msg) = serde_json::from_str::<ServerMessage>(&text) else {
125                            continue;
126                        };
127                        // Producer-side BWE/keyframe signals: fan to the per-track
128                        // adaptation listeners (which filter to OUR track). `send` errs
129                        // only with no live receivers (no video publishing) — ignore.
130                        match &msg {
131                            ServerMessage::KeyframeRequested { track } => {
132                                let _ = keyframe_tx_reader.send(*track);
133                            }
134                            ServerMessage::MediaFeedbackReceived {
135                                track,
136                                from,
137                                received,
138                                lost,
139                                jitter_us,
140                            } => {
141                                let _ = feedback_tx_reader.send(MediaFeedback {
142                                    track: *track,
143                                    from: *from,
144                                    received: *received,
145                                    lost: *lost,
146                                    jitter_us: *jitter_us,
147                                });
148                            }
149                            ServerMessage::QualityRequested {
150                                track,
151                                from,
152                                cap_bps,
153                            } => {
154                                let _ = quality_tx_reader.send(QualityReport {
155                                    track: *track,
156                                    from: *from,
157                                    cap_bps: *cap_bps,
158                                });
159                            }
160                            _ => {}
161                        }
162                        view.apply(&msg);
163                        let _ = view_tx.send_replace(view.clone());
164                        // Join the control-plane peers against the latest identity
165                        // snapshot so the published view carries display identities.
166                        // `send_replace`: a headless caller may not have subscribed
167                        // `watch_call` yet (plain `send` drops the publish with zero
168                        // receivers and the last roster change would be lost forever).
169                        view_out.send_replace(CallView::from_view_with(&view, &identities.load()));
170                    }
171                    // Distinguish the two ways the control WS ends, so a recurring
172                    // drop is diagnosable: a server-sent Close (with its reason) vs a
173                    // read/transport error (network blip, reset). Either way the
174                    // supervisor heals; this is purely for visibility.
175                    Ok(Message::Close(frame)) => {
176                        tracing::warn!(?frame, "call: control WS closed by server");
177                        break;
178                    }
179                    Err(e) => {
180                        tracing::warn!(error = %e, "call: control WS read error");
181                        break;
182                    }
183                    Ok(_) => {}
184                }
185            }
186            // The control plane dropped — wake the supervisor's monitor so it heals
187            // the whole session (a fresh PeerId; the roster fully resyncs).
188            let _ = closed_tx.send(());
189        });
190
191        Ok(Self {
192            outbound: tx,
193            view_rx,
194            keyframe_tx,
195            feedback_tx,
196            quality_tx,
197            closed: closed_rx,
198        })
199    }
200}

202/// This device's video decode capabilities, advertised at `Hello`. The engine
203/// reports HW/SW support; H.264 is always present (the universal floor).
204fn local_decode_caps() -> hound_control::DecodeCaps {
205    let caps = Engine::new().decode_caps();
206    hound_control::DecodeCaps {
207        av1: caps.av1,
208        h265: caps.h265,
209        h264: caps.h264,
210    }
211}

213/// Build the `ws(s)://.../space/{id}/control?token=...` URL from the HTTP base.
214fn control_ws_url(base_url: &str, space_id: Uuid, token: &str) -> Result<String> {
215    let base = base_url.trim_end_matches('/');
216    let ws_base = if let Some(rest) = base.strip_prefix("https://") {
217        format!("wss://{rest}")
218    } else if let Some(rest) = base.strip_prefix("http://") {
219        format!("ws://{rest}")
220    } else if base.starts_with("ws://") || base.starts_with("wss://") {
221        base.to_owned()
222    } else {
223        return Err(HoundError::Media(format!(
224            "api_base has no http(s) scheme: {base_url}"
225        )));
226    };
227    Ok(format!(
228        "{ws_base}/space/{space_id}/control?token={}",
229        url_token(token)
230    ))
231}

233/// Minimal percent-encoding for the query-string token (JWTs are URL-safe already).
234fn url_token(s: &str) -> String {
235    let mut out = String::with_capacity(s.len());
236    for b in s.bytes() {
237        match b {
238            b'A'..=b'Z' | b'a'..=b'z' | b'0'..=b'9' | b'-' | b'_' | b'.' | b'~' => {
239                out.push(b as char);
240            }
241            _ => out.push_str(&format!("%{b:02X}")),
242        }
243    }
244    out
245}

247/// Wait until the control reducer has a `self_id` (the server's `Welcome`).
248pub(super) async fn wait_for_self_id(view_rx: &mut watch::Receiver<SessionView>) -> Result<PeerId> {
249    loop {
250        if let Some(id) = view_rx.borrow().self_id() {
251            return Ok(id);
252        }
253        view_rx
254            .changed()
255            .await
256            .map_err(|_| HoundError::Media("control closed before Welcome".into()))?;
257    }
258}

262/// Dial the MoQ relay (its node id + relay URL) on a fresh iroh endpoint, returning
263/// the QUIC connection the MoQ session rides. The endpoint is per-call with an
264/// ephemeral key (no persisted device identity yet) — fine for the call's lifetime;
265/// sharing it across calls is a later optimization.
266/// Returns BOTH the endpoint and the connection: the caller MUST keep the
267/// [`Endpoint`] alive for the connection's lifetime. The SFU is reachable only via
268/// the relay (it has no client-routable direct address), so all packets — including
269/// every stream after the handshake — ride the endpoint's magicsock/relay actor.
270/// Dropping the endpoint stops that actor, so the QUIC handshake still completes but
271/// the very next stream silently never transmits (the relay's `accept_bi` then times
272/// out). Holding only the `Connection` is not enough.
273pub(super) async fn dial_relay(
274    relay_url: &str,
275    sfu_node_id: &str,
276) -> Result<(iroh::endpoint::Endpoint, iroh::endpoint::Connection)> {
277    use iroh::endpoint::presets;
278    use iroh::endpoint::{QuicTransportConfig, VarInt};
279    use iroh::{Endpoint, EndpointAddr, EndpointId, RelayMap, RelayMode, RelayUrl};
280
281    let relay_url: RelayUrl = relay_url
282        .parse()
283        .map_err(|e| HoundError::Media(format!("bad relay url: {e}")))?;
284    let relay_id: EndpointId = sfu_node_id
285        .parse()
286        .map_err(|_| HoundError::Media(format!("unparseable sfu node id: {sfu_node_id:?}")))?;
287
288    // Flow-control windows sized for the heaviest video track (a ~2.5 Mbps 1080p
289    // screenshare, or a fast link's 4 Mbps 1080p60 top rung):
290    // a too-small receive window forces the sender to stop-and-wait for window
291    // updates, and every such update is another inbound packet → another wakeup
292    // on a receive path that has no UDP GRO on macOS to coalesce them. Hold well
293    // over a high-RTT BDP per stream, and the sum across the call's concurrent
294    // moq streams (mic + cam + screen + screen-audio) at the connection level.
295    // MTU is left at iroh's conservative default — PMTUD raises it, and a high
296    // initial MTU risks black-holing on constrained paths. `max_concurrent_uni`
297    // gives moq's per-group streams headroom so a screenshare burst never stalls
298    // on the stream-count limit.
299    let transport = QuicTransportConfig::builder()
300        .stream_receive_window(VarInt::from_u32(4 * 1024 * 1024))
301        .receive_window(VarInt::from_u32(32 * 1024 * 1024))
302        .max_concurrent_uni_streams(VarInt::from_u32(1024))
303        .build();
304
305    let endpoint = Endpoint::builder(presets::N0)
306        .relay_mode(RelayMode::Custom(RelayMap::from(relay_url.clone())))
307        .transport_config(transport)
308        .bind()
309        .await
310        .map_err(|e| HoundError::Media(format!("bind iroh endpoint: {e}")))?;
311
312    let addr = EndpointAddr::new(relay_id).with_relay_url(relay_url);
313    let conn = endpoint
314        .connect(addr, hound_transport::ALPN)
315        .await
316        .map_err(|e| HoundError::Media(format!("dial MoQ relay: {e}")))?;
317    Ok((endpoint, conn))
318}
