1//! Inbound audio: pull → reorder by timestamp → Opus decode (bounded PLC on
2//! genuine gaps) → per-peer jitter buffer on the mix bus, plus level metering.
3//! The encoded-frame reorder hold undoes the cross-stream reordering a
4//! screenshare induces on the shared connection BEFORE the stateful decoder;
5//! release is purely count-based — the moq group read is not cancel-safe
6//! mid-object, so no recv timeout is ever raced against it.
7
8use super::*;
9
10/// Frames held in the encoded-audio reorder buffer to undo network reordering before
11/// the (stateful) Opus decode. `recv_group` delivers frames in ARRIVAL order; on the
12/// shared connection a screenshare reorders the tiny per-frame audio streams, so a
13/// frame can arrive after a later one. Decoding in arrival order forces dropping the
14/// late frame and concealing it with PLC — reordered becomes lost, and on a busy
15/// connection that is continuous distortion. Holding a few frames and releasing the oldest by
16/// timestamp restores a clean in-order stream into the decoder; genuine loss still
17/// surfaces as a real gap. 3 frames (~60 ms) absorbs the multi-frame reordering the
18/// screenshare induces; the added latency is recovered by the PCM jitter buffer
19/// shrinking once arrival is smooth. NOT done via a recv timeout — the moq group read
20/// is not cancel-safe mid-object — so release is purely count-based.
21const AUDIO_REORDER_HOLD: usize = 3;
22
23/// Insert `frame` into `buf` kept sorted ascending by timestamp, dropping an exact
24/// duplicate-timestamp frame (a retransmit of something already buffered). Returns
25/// false if dropped. Pure + unit-tested — the reorder correctness hinges on this.
26fn reorder_insert(buf: &mut Vec<WireFrame>, frame: WireFrame) -> bool {
27    match buf.binary_search_by_key(&frame.timestamp_us, |f| f.timestamp_us) {
28        Ok(_) => false,
29        Err(i) => {
30            buf.insert(i, frame);
31            true
32        }
33    }
34}

36/// Audio decode pipeline: pull frames, **reorder by timestamp**, Opus-decode (PLC on
37/// genuine gaps via the decoder's `None`), push into the per-peer jitter buffer on the
38/// mix bus, and meter levels.
39pub(super) async fn audio_pipeline(
40    mut subscriber: MoqTrackSubscriber,
41    peer: PeerId,
42    source: Source,
43    playout_bus: Arc<PlayoutBus>,
44    levels: watch::Sender<LevelsView>,
45    clock: CaptureClock,
46    stop: Arc<AtomicBool>,
47    peer_stats: Arc<crate::stats::PeerStats>,
48    gain: Arc<std::sync::atomic::AtomicU32>,
49    delivered: Arc<AtomicBool>,
50) {
51    use hound_engine::api::level_from_rms;
52    use hound_engine::types::OpusPacket;
53
54    let Ok(mut decoder) = OpusDecoder::new() else {
55        return;
56    };
57    // The mix bus mixes all slots regardless of label; the label only attributes
58    // starve-stats, so name screen app-audio honestly.
59    let is_screen = source == Source::Screen;
60    let akind = if is_screen {
61        InboundAudioKind::Screen
62    } else {
63        InboundAudioKind::Mic
64    };
65    // Tag the producer with the peer id so a recorded underflow glitch names the peer.
66    let producer =
67        playout_bus.register_tagged(if is_screen { "screen_audio" } else { "mic" }, peer.0);
68    let mut decoded: u64 = 0;
69    // Loss/reorder detection from the sender's capture timestamps. `prev_ts` advances
70    // only on in-order frames; `frame_us` (the Opus frame duration) is learned from the
71    // first decoded frame so the gap math doesn't assume 20 ms. The engine decodes at
72    // 48 kHz mono, so duration = samples · 1e6 / 48000.
73    const ENGINE_RATE_HZ: i64 = 48_000;
74    let mut prev_ts: Option<i64> = None;
75    let mut frame_us: i64 = 20_000; // sane default until the first frame refines it
76    let mut learned_frame_us = false;
77    // Cadence diagnostics — a retained debug tool. Distinguishes the THREE causes of an
78    // active-stream gap that the smoothed jitter number hides: brief network loss
79    // (small ts gaps → PLC), sender DTX / capture discontinuity (large ts gaps → NOT
80    // loss, must not be PLC-fabricated), and sender↔receiver clock skew (ts-elapsed vs
81    // local-elapsed divergence → buffer slowly drains). Logged once per window.
82    let mut diag = CadenceDiag::default();
83    // Encoded-frame reorder buffer (sorted by ts). We pull from the network into it and
84    // release the oldest only once HOLD frames are buffered, so network reordering is
85    // undone BEFORE the stateful decode (see AUDIO_REORDER_HOLD).
86    let mut reorder: Vec<WireFrame> = Vec::with_capacity(AUDIO_REORDER_HOLD + 2);
87    let mut ended = false;
88    while !stop.load(Ordering::Relaxed) {
89        // Acquire the next frame to PROCESS, in TIMESTAMP order. Refill from the network
90        // until we hold more than HOLD frames (then the oldest is safe to release), or
91        // the track ends (drain whatever remains, oldest-first). `next_frame` is never
92        // cancelled — only fully-received frames enter the buffer.
93        let frame = loop {
94            if reorder.len() > AUDIO_REORDER_HOLD {
95                break Some(reorder.remove(0));
96            }
97            if ended {
98                // Track ended/errored: drain remaining buffered frames oldest-first.
99                break (!reorder.is_empty()).then(|| reorder.remove(0));
100            }
101            let now = clock.now();
102            match subscriber.next_frame(now).await {
103                Ok(Some(f)) => {
104                    // RTC-debug: one inbound Opus frame received (counted on arrival).
105                    peer_stats.audio_received(akind, f.data.len());
106                    reorder_insert(&mut reorder, f);
107                }
108                Ok(None) | Err(_) => ended = true,
109            }
110        };
111        let Some(frame) = frame else { break };
112        let ts = frame.timestamp_us;
113        // Local time at processing (after any reorder hold) — for the cadence skew calc.
114        let now = clock.now();
115
116        // Frames now arrive in timestamp order, so a non-advancing timestamp is a true
117        // duplicate/straggler the reorder buffer couldn't place (rare), not routine
118        // network reordering — skip it to keep the stateful decoder coherent.
119        if let Some(pt) = prev_ts {
120            let gap = ts - pt;
121            if gap <= 0 {
122                peer_stats.record_audio_loss(akind, 0, true);
123                continue;
124            }
125            diag.note_gap(gap);
126            let missing = frames_missing(gap, frame_us);
127            // PLC conceals BRIEF loss only. A larger gap is the sender being silent
128            // (Opus DTX sends no frames during silence) or a capture discontinuity —
129            // NOT packet loss. Fabricating one extrapolated PLC frame per 20 ms of
130            // gap would mislabel silence as loss (a silent peer produces more
131            // concealment than real audio), waste CPU, and — worst — splice garbage
132            // PLC into an ACTIVE stream where it leapfrogs/drops the real frame
133            // behind it. Cap it: conceal up to MAX_PLC_FRAMES, then resync silently.
134            if missing > 0 && missing <= MAX_PLC_FRAMES {
135                for i in 1..=missing {
136                    let plc_ts = pt + frame_us * i;
137                    if let Ok(plc) = decoder.decode(None, TimestampUs::from_micros(plc_ts)) {
138                        let _ = producer.push(clock.now(), plc_ts, &plc.samples);
139                    }
140                }
141                peer_stats.record_audio_loss(akind, missing as u64, false);
142                diag.plc += missing as u64;
143            } else if missing > MAX_PLC_FRAMES {
144                // DTX / discontinuity: play silence across the gap (correct — there was
145                // no audio). Resync to the new frame (prev_ts advances after decode).
146                diag.big_gaps += 1;
147            }
148        }
149
150        let packet = OpusPacket {
151            data: frame.data,
152            timestamp_us: TimestampUs::from_micros(ts),
153        };
154        let Ok(mut pcm) = decoder.decode(Some(&packet), TimestampUs::from_micros(ts)) else {
155            continue;
156        };
157        delivered.store(true, Ordering::Relaxed);
158        prev_ts = Some(ts);
159        if !learned_frame_us && !pcm.samples.is_empty() {
160            frame_us = (pcm.samples.len() as i64 * 1_000_000 / ENGINE_RATE_HZ).max(1);
161            learned_frame_us = true;
162        }
163        // Per-stream playback gain (the tile volume/mute control), applied before the
164        // level meter and the mix push so both reflect what is actually played (a
165        // muted stream's meter reads 0 and its speaking ring stays off). Unity skips
166        // the pass entirely (the common case costs one atomic load).
167        let g = f32::from_bits(gain.load(Ordering::Relaxed));
168        if g != 1.0 {
169            apply_gain(&mut pcm.samples, g);
170        }
171        // Both mic + screen audio record their decoded level for the panel; only the
172        // mic drives the per-peer level meter / speaking ring (both key on peer id).
173        let level = level_from_rms(pcm.rms());
174        peer_stats.set_audio_level(akind, level);
175        if !is_screen {
176            levels.send_modify(|l| {
177                l.peers.insert(peer.0, level);
178            });
179        }
180        // The mix bus pulls on its own RT clock; we only FILL the jitter buffer.
181        // Surface the post-push jitter-buffer health to the panel/report: depth
182        // (cur vs recommended), late/overflow drops, and this slot's underflow total.
183        // A depth that climbs while silent = buffered-but-never-released (a playout
184        // stall); recommended ≫ target = the buffer is too shallow for the link.
185        let status = producer.push(clock.now(), ts, &pcm.samples);
186        peer_stats.set_audio_playout_depth(akind, status.buffered_frames);
187        peer_stats.set_audio_jitter(akind, status.jitter_us.max(0) as u32);
188        peer_stats.set_audio_jitterbuf(
189            akind,
190            status.target_delay_us.max(0) as u32,
191            status.recommended_delay_us.max(0) as u32,
192            status.late_drops,
193            status.overflow_drops,
194            status.underflow_total,
195        );
196        decoded += 1;
197        // Feed the cadence diagnostic and emit a window summary periodically. This is
198        // the tool for attributing an active-stream gap: high `plc` = brief loss;
199        // high `big_gaps`/`max_gap_ms` = sender DTX or a capture stall (a sender-side
200        // problem, not network); large |skew_ppm| = clock drift slowly draining the
201        // buffer (needs resampling/retiming, not a deeper buffer).
202        if let Some((real, plc, big_gaps, max_gap_ms, skew_ppm)) = diag.observe(ts, now) {
203            tracing::info!(
204                ?source,
205                peer = peer.0,
206                real,
207                plc,
208                big_gaps,
209                max_gap_ms,
210                skew_ppm,
211                jitterbuf_ms = status.target_delay_us / 1000,
212                underflow_total = status.underflow_total,
213                "audio: cadence window"
214            );
215        }
216        if decoded == 1 || decoded.is_multiple_of(500) {
217            tracing::info!(
218                ?source,
219                peer = peer.0,
220                decoded,
221                "audio: decoded+mixed frames"
222            );
223        }
224    }
225    if !is_screen {
226        levels.send_modify(|l| {
227            l.peers.remove(&peer.0);
228        });
229    }
230    tracing::info!(?source, peer = peer.0, decoded, "audio: pipeline ended");
231}

233#[cfg(test)]
234mod tests {
235    use super::*;
236
237    // ─── Encoded-audio reorder buffer ───────────────────────────────────────────
238
239    fn afr(ts: i64) -> WireFrame {
240        WireFrame {
241            sender: 0,
242            track: TrackKind::CamAudio,
243            codec: WireCodec::Opus,
244            keyframe: true,
245            frame_seq: 0,
246            timestamp_us: ts,
247            data: vec![ts as u8],
248        }
249    }
250
251    #[test]
252    fn reorder_insert_sorts_by_timestamp() {
253        // Frames arriving out of order are placed in ascending-timestamp order, so the
254        // stateful decoder downstream sees a clean monotonic stream.
255        let mut buf = Vec::new();
256        assert!(reorder_insert(&mut buf, afr(40_000)));
257        assert!(reorder_insert(&mut buf, afr(20_000))); // arrives late
258        assert!(reorder_insert(&mut buf, afr(60_000)));
259        assert!(reorder_insert(&mut buf, afr(0))); // arrives very late
260        let order: Vec<i64> = buf.iter().map(|f| f.timestamp_us).collect();
261        assert_eq!(order, vec![0, 20_000, 40_000, 60_000], "sorted by ts");
262    }
263
264    #[test]
265    fn reorder_insert_drops_exact_duplicate_ts() {
266        let mut buf = Vec::new();
267        assert!(reorder_insert(&mut buf, afr(20_000)));
268        assert!(
269            !reorder_insert(&mut buf, afr(20_000)),
270            "duplicate ts dropped"
271        );
272        assert_eq!(buf.len(), 1);
273    }
274
275    #[test]
276    fn reorder_release_yields_monotonic_stream() {
277        // Simulate the HOLD release policy over a reordered arrival sequence and assert
278        // the released (oldest-first) timestamps are monotonic — no frame is ever
279        // released out of order, which is what keeps the decoder coherent.
280        let arrivals = [0i64, 40_000, 20_000, 80_000, 60_000, 100_000];
281        let mut buf: Vec<WireFrame> = Vec::new();
282        let mut released = Vec::new();
283        for &ts in &arrivals {
284            reorder_insert(&mut buf, afr(ts));
285            while buf.len() > AUDIO_REORDER_HOLD {
286                released.push(buf.remove(0).timestamp_us);
287            }
288        }
289        while !buf.is_empty() {
290            released.push(buf.remove(0).timestamp_us);
291        }
292        assert!(
293            released.windows(2).all(|w| w[0] < w[1]),
294            "released stream must be strictly increasing: {released:?}"
295        );
296        assert_eq!(released.len(), arrivals.len(), "every frame released once");
297    }
