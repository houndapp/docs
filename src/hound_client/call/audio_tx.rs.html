1//! Outbound audio: the mic capture loop + Opus publish, and the screen APP-AUDIO
2//! leg (its `!Send` capture thread + publish task). Encode + publish run on the
3//! RT audio runtime ("audio is sacred": a screenshare encode can't stall the
4//! mic), and audio frames publish via the ordered-substream path (count-rolled
5//! groups) so successive Opus frames ride one in-order QUIC stream — never one
6//! stream per frame.

12/// Spawn the mic publish task. A blocking capture thread pulls PCM from the engine
13/// `Microphone` (hot-swappable via `device_rx`); the async half Opus-encodes and
14/// publishes each frame on the relay session's `mic` track. Muting gates the
15/// encoder (no frames sent) and zeroes the reported level.
16#[allow(clippy::too_many_arguments)]
17pub(super) fn spawn_mic_publish(
18    relay: Arc<MoqRelaySession>,
19    space_id: Uuid,
20    self_id: PeerId,
21    clock: CaptureClock,
22    device_rx: watch::Receiver<DeviceChoice<MicId>>,
23    mute: Arc<MuteFlags>,
24    levels: watch::Sender<LevelsView>,
25    stop: Arc<AtomicBool>,
26    stats: Arc<CallStats>,
27    #[cfg(feature = "synthetic")] capture_source: CaptureSource,
28) -> JoinHandle<()> {
29    let (pcm_tx, mut pcm_rx) = tokio::sync::mpsc::channel::<hound_engine::types::PcmFrame>(8);
30
31    // Blocking capture thread (the cpal Microphone is !Send — own it on one thread).
32    let capture_stop = stop.clone();
33    let stats_capture = stats.clone();
34    std::thread::Builder::new()
35        .name("call-mic-capture".into())
36        .spawn(move || {
37            mic_capture_loop(
38                capture_stop,
39                clock,
40                device_rx,
41                pcm_tx,
42                mute,
43                levels,
44                stats_capture,
45                #[cfg(feature = "synthetic")]
46                capture_source,
47            )
48        })
49        .ok();
50
51    let path = broadcast_path(space_id, self_id, MIC_TRACK);
52    let sender = self_id.0 as u16;
53    // Outbound voice is sacred too: encode + publish on the RT audio runtime so a
54    // screenshare encode can't stall the mic and make us stutter for everyone else.
55    audio_runtime().spawn(async move {
56        let mut encoder = match OpusEncoder::new(Bitrate::kbps(24), OpusProfile::Voice) {
57            Ok(e) => e,
58            Err(e) => {
59                tracing::error!(%e, "mic: failed to build Opus encoder");
60                return;
61            }
62        };
63        let mut frame_seq = 0u32;
64        let mut publisher = match relay.publish_track(&path, MIC_TRACK, AUDIO_PRIORITY) {
65            Ok(p) => p,
66            Err(e) => {
67                tracing::error!(%e, "mic: failed to publish track");
68                return;
69            }
70        };
71        if publisher.wait_subscribed().await.is_err() {
72            return;
73        }
74        while let Some(pcm) = pcm_rx.recv().await {
75            let Ok(packet) = encoder.encode(&pcm) else {
76                continue;
77            };
78            let bytes = packet.data.len();
79            let frame = WireFrame {
80                sender,
81                track: TrackKind::CamAudio,
82                codec: WireCodec::Opus,
83                // Grouping is count-driven for audio (ordered substream), not keyframe-
84                // driven — every Opus frame is independently decodable, so the bit is moot.
85                keyframe: false,
86                frame_seq,
87                timestamp_us: packet.timestamp_us.micros(),
88                data: packet.data,
89            };
90            frame_seq = frame_seq.wrapping_add(1);
91            if let Err(e) = publish_audio_one(&mut publisher, &frame) {
92                tracing::warn!(%e, "mic: publish frame failed");
93            } else {
94                // RTC-debug: one Opus mic frame published (drives outbound send_kbps).
95                stats.audio_sent(bytes);
96            }
97        }
98    })
99}

101/// Publish one AUDIO frame via the ordered-substream path (count-rolled groups), so
102/// successive Opus frames ride one in-order QUIC stream instead of one stream per frame
103/// (the source of the cross-stream reordering). Use for mic + screen-audio.
104fn publish_audio_one(
105    publisher: &mut MoqTrackPublisher,
106    frame: &WireFrame,
107) -> std::result::Result<(), String> {
108    publisher
109        .publish_audio_frame(frame)
110        .map_err(|e| e.to_string())
111}

113/// Open the mic per `capture_source`: real hardware, or (behind `synthetic`)
114/// the engine's mic-free test tone. The one seam every (re)open site in
115/// [`mic_capture_loop`] routes through, so a headless build never needs its
116/// own copy of the branch.
117fn open_mic(
118    choice: &DeviceChoice<MicId>,
119    clock: CaptureClock,
120    #[cfg(feature = "synthetic")] capture_source: CaptureSource,
121) -> Option<hound_engine::api::Microphone> {
122    #[cfg(feature = "synthetic")]
123    if capture_source != CaptureSource::Default {
124        return Engine::new().open_synthetic_microphone(clock).ok();
125    }
126    Engine::new().open_microphone(choice, clock).ok()
127}

129/// The mic capture loop: pull PCM, gate on mute, meter the level, forward to the
130/// encoder. Hot-swaps the device when `device_rx` changes; re-opens on error.
131#[allow(clippy::too_many_arguments)]
132fn mic_capture_loop(
133    stop: Arc<AtomicBool>,
134    clock: CaptureClock,
135    mut device_rx: watch::Receiver<DeviceChoice<MicId>>,
136    pcm_tx: tokio::sync::mpsc::Sender<hound_engine::types::PcmFrame>,
137    mute: Arc<MuteFlags>,
138    levels: watch::Sender<LevelsView>,
139    stats: Arc<CallStats>,
140    #[cfg(feature = "synthetic")] capture_source: CaptureSource,
141) {
142    use hound_engine::api::level_from_rms;
143
144    let mut choice = device_rx.borrow_and_update().clone();
145    let mut mic = open_mic(
146        &choice,
147        clock,
148        #[cfg(feature = "synthetic")]
149        capture_source,
150    );
151    let mut warmup = mic.as_ref().map_or(MicRouteWarmup { due: None }, |m| {
152        MicRouteWarmup::arm(m.device_rate(), Instant::now())
153    });
154    let mut silence = MicSilenceWatch::default();
155
156    while !stop.load(Ordering::Relaxed) {
157        if device_rx.has_changed().unwrap_or(false) {
158            choice = device_rx.borrow_and_update().clone();
159            drop(mic.take());
160            mic = open_mic(
161                &choice,
162                clock,
163                #[cfg(feature = "synthetic")]
164                capture_source,
165            );
166            if let Some(m) = mic.as_ref() {
167                warmup = MicRouteWarmup::arm(m.device_rate(), Instant::now());
168            }
169        }
170        // BT/HFP route warmup: one reopen after the SCO link has had time to
171        // settle, mirroring the playout side's warmup in the mix bus.
172        if warmup.fire(Instant::now()) {
173            tracing::info!("voice: warmup reopen of non-48k (BT/HFP) mic — silent-route fix");
174            drop(mic.take());
175            mic = open_mic(
176                &choice,
177                clock,
178                #[cfg(feature = "synthetic")]
179                capture_source,
180            );
181            // Deliberately not re-armed: one warmup per open cycle.
182        }
183        let Some(cap) = mic.as_mut() else {
184            std::thread::sleep(std::time::Duration::from_millis(200));
185            mic = open_mic(
186                &choice,
187                clock,
188                #[cfg(feature = "synthetic")]
189                capture_source,
190            );
191            if let Some(m) = mic.as_ref() {
192                warmup = MicRouteWarmup::arm(m.device_rate(), Instant::now());
193            }
194            continue;
195        };
196        let pulled = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| cap.next_frame()));
197        match pulled {
198            Ok(Ok(frame)) => {
199                // RTC-debug: one PCM frame captured from the mic device.
200                stats.audio_captured();
201                if silence.observe(&frame.samples) {
202                    tracing::warn!(
203                        device_rate = cap.device_rate(),
204                        "voice: mic delivering PURE SILENCE (~3 s of bit-exact zeros) — \
205                         dead capture route (BT/SCO not live, permission, or a silent \
206                         virtual device)"
207                    );
208                }
209                if mute.mic.load(Ordering::Relaxed) {
210                    levels.send_modify(|l| l.mic = 0.0);
211                    stats.set_self_level(0.0);
212                    continue;
213                }
214                let level = level_from_rms(frame.rms());
215                levels.send_modify(|l| l.mic = level);
216                stats.set_self_level(level);
217                let _ = pcm_tx.try_send(frame);
218            }
219            _ => {
220                drop(mic.take());
221                std::thread::sleep(std::time::Duration::from_millis(200));
222            }
223        }
224    }
225}

227/// The `!Send` screen APP-AUDIO capture thread: block on the engine `ScreenAudio`
228/// (the mic-thread analogue — no APM, no levels, no hot-swap) and forward each 20 ms
229/// PCM frame to the publish task. A `next_frame` error/panic is the leg's clean end
230/// (its parent screen stream stopped); drop the leg on this owning thread and exit.
231pub(super) fn screen_audio_capture_loop(
232    stop: Arc<AtomicBool>,
233    mut audio: hound_engine::api::ScreenAudio,
234    pcm_tx: tokio::sync::mpsc::Sender<hound_engine::types::PcmFrame>,
235) {
236    let mut captured: u64 = 0;
237    while !stop.load(Ordering::Relaxed) {
238        let pulled = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| audio.next_frame()));
239        match pulled {
240            Ok(Ok(pcm)) => {
241                captured += 1;
242                if captured == 1 || captured.is_multiple_of(500) {
243                    tracing::info!(captured, "screen: captured app-audio frames");
244                }
245                let _ = pcm_tx.try_send(pcm); // drop on backpressure
246            }
247            Ok(Err(e)) => {
248                tracing::warn!(%e, captured, "screen: app-audio capture ended (error)");
249                break;
250            }
251            Err(_panic) => {
252                tracing::warn!(captured, "screen: app-audio capture panicked");
253                break;
254            }
255        }
256    }
257    tracing::info!(captured, "screen: app-audio capture loop stopped");
258    drop(audio);
259}

261/// The async screen AUDIO publish task: the mic publish path MINUS APM / levels /
262/// hot-swap (app audio is not a mic). Opus-encodes each forwarded PCM frame and
263/// publishes it on the `screen_audio` track; the receiver plays it on the shared
264/// mix bus exactly like a mic.
265pub(super) fn spawn_screen_audio_publish(
266    relay: Arc<MoqRelaySession>,
267    path: String,
268    sender: u16,
269    mut pcm_rx: tokio::sync::mpsc::Receiver<hound_engine::types::PcmFrame>,
270    stats: Arc<CallStats>,
271    mute: Arc<MuteFlags>,
272) -> JoinHandle<()> {
273    // App-audio is audio: encode + publish on the RT audio runtime, off the video pool.
274    audio_runtime().spawn(async move {
275        let mut encoder = match OpusEncoder::new(SCREEN_AUDIO_BITRATE, OpusProfile::Music) {
276            Ok(e) => e,
277            Err(e) => {
278                tracing::error!(%e, "screen: failed to build app-audio Opus encoder");
279                return;
280            }
281        };
282        let mut publisher = match relay.publish_track(&path, SCREEN_AUDIO_TRACK, AUDIO_PRIORITY) {
283            Ok(p) => p,
284            Err(e) => {
285                tracing::error!(%e, "screen: failed to publish audio track");
286                return;
287            }
288        };
289        if publisher.wait_subscribed().await.is_err() {
290            return;
291        }
292        tracing::info!("screen: app-audio track subscribed — publishing");
293        let mut frame_seq = 0u32;
294        let mut sent: u64 = 0;
295        while let Some(pcm) = pcm_rx.recv().await {
296            // The broadcaster's "share audio" toggle: drop BEFORE encode, so a muted
297            // share costs zero CPU and zero wire data. Receivers read the gap as a
298            // quiet source (their starve metric is DTX-aware), and un-muting resumes
299            // instantly on this same track.
300            if mute.screen_audio.load(Ordering::Relaxed) {
301                continue;
302            }
303            let Ok(packet) = encoder.encode(&pcm) else {
304                continue;
305            };
306            let bytes = packet.data.len();
307            let frame = WireFrame {
308                sender,
309                track: TrackKind::ScrAudio,
310                codec: WireCodec::Opus,
311                // Count-driven grouping (ordered substream); keyframe bit is moot for audio.
312                keyframe: false,
313                frame_seq,
314                timestamp_us: packet.timestamp_us.micros(),
315                data: packet.data,
316            };
317            frame_seq = frame_seq.wrapping_add(1);
318            if let Err(e) = publish_audio_one(&mut publisher, &frame) {
319                tracing::warn!(%e, "screen: publish audio frame failed");
320            } else {
321                // RTC-debug: one screen app-audio frame published.
322                stats.screen_audio_sent(bytes);
323                sent += 1;
324                if sent == 1 || sent.is_multiple_of(500) {
325                    tracing::info!(sent, "screen: published app-audio frames");
326                }
327            }
328        }
329        tracing::info!(sent, "screen: app-audio publish ended");
330    })
