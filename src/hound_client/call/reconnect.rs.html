1//! Reconnect policy + the retry loop: the pure exponential [`Backoff`] with
2//! bounded jitter, the fatal-vs-retryable [`classify`] verdict over
3//! [`HoundError`], and [`reconnect_loop`] — which re-runs `establish_live` until
4//! success, a fatal error, or cancel. Invariant: only an auth/permission outcome
5//! is fatal; everything transient retries FOREVER (each attempt re-mints the
6//! relay token via `voice/join`, so a stale token heals on retry).

10// ─── Reconnect policy: backoff + failure classification (pure, headless) ────────
11//
12// A live call must heal a dropped control WS or MoQ/iroh
13// connection WITHOUT the user re-joining; these two pieces are the pure policy the
14// supervisor's IO shell drives. Recovery is a FULL re-establish (fresh `PeerId`) —
15// see the module docs.
16
17/// First reconnect delay. Short enough that a blip (a relay restart, a brief network
18/// stall) heals almost immediately, long enough not to busy-loop a restarting backend.
19const BACKOFF_INITIAL: Duration = Duration::from_millis(500);
20/// Per-attempt growth factor: 0.5s → 1 → 2 → 4 → 8 → (cap) 15s. Classic doubling.
21const BACKOFF_FACTOR: u32 = 2;
22/// Ceiling on the base delay; past this every attempt waits ~15s (± jitter).
23const BACKOFF_MAX: Duration = Duration::from_secs(15);
24/// Jitter band, ±20% of the base delay — de-correlates a fleet that all dropped at
25/// once (a relay bounce) so they don't reconnect in a thundering herd.
26const BACKOFF_JITTER_FRAC: f64 = 0.20;
27
28/// A pure exponential-backoff sequencer with bounded jitter. Sans-IO: it computes
29/// delays but never sleeps, so its whole policy is unit-testable. `jitter` is injected
30/// (a closure returning a fraction in `[-1.0, 1.0]`) so tests can pin it.
31#[derive(Debug, Clone)]
32pub(super) struct Backoff {
33    base: Duration,
34    attempt: u32,
35}
36
37impl Default for Backoff {
38    fn default() -> Self {
39        Self {
40            base: BACKOFF_INITIAL,
41            attempt: 0,
42        }
43    }
44}
45
46impl Backoff {
47    pub(super) fn new() -> Self {
48        Self::default()
49    }
50
51    /// Reset to the initial delay after a successful reconnect (next outage starts
52    /// the ladder over at 0.5s); the attempt counter resets too.
53    pub(super) fn reset(&mut self) {
54        self.base = BACKOFF_INITIAL;
55        self.attempt = 0;
56    }
57
58    /// Delays produced since the last reset — the 1-based count of the upcoming attempt.
59    fn attempt(&self) -> u32 {
60        self.attempt
61    }
62
63    /// Produce the next delay and advance the ladder. The base doubles (saturating at
64    /// the cap) per call; `jitter()` supplies a fraction in `[-1.0, 1.0]` scaled by
65    /// [`BACKOFF_JITTER_FRAC`], clamped non-negative.
66    fn next_delay(&mut self, mut jitter: impl FnMut() -> f64) -> Duration {
67        let base = self.base;
68        self.attempt = self.attempt.saturating_add(1);
69        self.base = base.saturating_mul(BACKOFF_FACTOR).min(BACKOFF_MAX);
70
71        let frac = jitter().clamp(-1.0, 1.0);
72        let base_secs = base.as_secs_f64();
73        let delay_secs = (base_secs + frac * BACKOFF_JITTER_FRAC * base_secs).max(0.0);
74        Duration::from_secs_f64(delay_secs)
75    }
76}

78/// A lightweight jitter source for production: a fraction in `[-1.0, 1.0]` derived
79/// from the wall-clock nanosecond field hashed through a SplitMix64 step. No RNG
80/// dependency needed — the goal is only to de-correlate a fleet of reconnecting
81/// clients, not cryptographic randomness. (Tests pin `next_delay`'s jitter directly,
82/// so this stays out of the deterministic path.)
83fn rng_jitter() -> f64 {
84    let nanos = std::time::SystemTime::now()
85        .duration_since(std::time::UNIX_EPOCH)
86        .map(|d| d.subsec_nanos() as u64)
87        .unwrap_or(0);
88    // One SplitMix64 round to smear the low-entropy subsec nanos across the word.
89    let mut z = nanos.wrapping_add(0x9E37_79B9_7F4A_7C15);
90    z = (z ^ (z >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
91    z = (z ^ (z >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
92    z ^= z >> 31;
93    // Map the top 53 bits to [0,1), then to [-1, 1).
94    let unit = (z >> 11) as f64 / (1u64 << 53) as f64;
95    unit * 2.0 - 1.0
96}

98/// The verdict on a failure: keep reconnecting, or give up and surface `Failed`.
99#[derive(Debug, Clone, Copy, PartialEq, Eq)]
100enum Disposition {
101    /// Transient — re-mint via `voice/join` and re-establish after a backoff. Retried
102    /// FOREVER (the user can always `leave`).
103    Retryable,
104    /// Fatal — retrying cannot fix it: the user lost access (`Http 403`) or the JWT
105    /// was rejected (`Http 401` / [`HoundError::Auth`]) and there is no refresh path,
106    /// so a retry re-presents the same dead token forever. The ONLY non-recoverable
107    /// class.
108    Fatal,
109}
110
111/// Classify a join/re-establish failure. Only an auth/permission outcome is fatal:
112/// `Http {401|403}` or [`HoundError::Auth`]. Everything else — network, realtime
113/// (control WS), media (relay dial/handshake), internal blips, an invalid-state or
114/// not-found/decode hiccup mid-redeploy — is transient and retried forever (a relay
115/// token is re-minted fresh each attempt, so a stale token heals on retry).
116fn classify(err: &HoundError) -> Disposition {
117    match err {
118        HoundError::Http { status, .. } if *status == 401 || *status == 403 => Disposition::Fatal,
119        HoundError::Auth(_) => Disposition::Fatal,
120        HoundError::Http { .. }
121        | HoundError::Network(_)
122        | HoundError::Realtime(_)
123        | HoundError::Media(_)
124        | HoundError::Internal(_)
125        | HoundError::InvalidState(_)
126        | HoundError::NotFound(_)
127        | HoundError::Decode(_) => Disposition::Retryable,
128    }
129}

131/// The result of the inner reconnect loop.
132pub(super) enum ReconnectOutcome {
133    /// A fresh live session is up. Boxed so the tiny `Cancelled`/`Fatal` variants
134    /// don't pay `LiveSession`'s size on the stack.
135    Reconnected(Box<LiveSession>),
136    /// `leave` was requested while we were retrying.
137    Cancelled,
138    /// A fatal (auth/permission) error — give up.
139    Fatal(HoundError),
140}
141
142/// Retry [`establish_live`] with backoff until it succeeds, hits a fatal error, or is
143/// cancelled. Publishes `Reconnecting { attempt }` per attempt. The backoff sleep is
144/// inside a `select!` against cancel, so a `leave` mid-backoff wakes immediately.
145pub(super) async fn reconnect_loop(
146    params: &ConnectionParams,
147    cancel: &Arc<Cancel>,
148    backoff: &mut Backoff,
149    first_error: HoundError,
150) -> ReconnectOutcome {
151    let mut last_error = first_error;
152    loop {
153        if cancel.is_cancelled() {
154            return ReconnectOutcome::Cancelled;
155        }
156
157        let delay = backoff.next_delay(rng_jitter);
158        // `send_replace`: see `CallSession::join` — must not depend on a live receiver.
159        params
160            .channels
161            .connection
162            .send_replace(ConnectionState::Reconnecting {
163                attempt: backoff.attempt(),
164            });
165        tracing::info!(
166            attempt = backoff.attempt(),
167            delay_ms = delay.as_millis() as u64,
168            %last_error,
169            "call: reconnect backoff"
170        );
171
172        // Cancellable sleep: a `leave` wakes us immediately rather than waiting out a
173        // (possibly 15s) delay.
174        tokio::select! {
175            biased;
176            _ = cancel.notify.notified() => return ReconnectOutcome::Cancelled,
177            _ = tokio::time::sleep(delay) => {}
178        }
179        if cancel.is_cancelled() {
180            return ReconnectOutcome::Cancelled;
181        }
182
183        match establish_live(params).await {
184            Ok(live) => return ReconnectOutcome::Reconnected(Box::new(live)),
185            Err(err) => match classify(&err) {
186                Disposition::Fatal => return ReconnectOutcome::Fatal(err),
187                Disposition::Retryable => {
188                    tracing::warn!(%err, "call: reconnect attempt failed (retrying)");
189                    last_error = err;
190                }
191            },
192        }
193    }
194}

196#[cfg(test)]
197mod backoff_tests {
198    use super::*;
199
200    /// With jitter pinned to zero the ladder is the exact documented doubling
201    /// sequence, saturating at the cap and then holding there forever.
202    #[test]
203    fn zero_jitter_is_exact_doubling_to_cap() {
204        let mut b = Backoff::new();
205        let zero = || 0.0;
206        assert_eq!(b.next_delay(zero), Duration::from_millis(500));
207        assert_eq!(b.next_delay(zero), Duration::from_secs(1));
208        assert_eq!(b.next_delay(zero), Duration::from_secs(2));
209        assert_eq!(b.next_delay(zero), Duration::from_secs(4));
210        assert_eq!(b.next_delay(zero), Duration::from_secs(8));
211        // 16s would exceed the 15s cap → clamped, then held.
212        assert_eq!(b.next_delay(zero), Duration::from_secs(15));
213        assert_eq!(b.next_delay(zero), Duration::from_secs(15));
214    }
215
216    /// The attempt counter advances per delay and resets with the ladder.
217    #[test]
218    fn attempt_counter_advances_then_resets() {
219        let mut b = Backoff::new();
220        assert_eq!(b.attempt(), 0, "no delays produced yet");
221        b.next_delay(|| 0.0);
222        assert_eq!(b.attempt(), 1);
223        b.next_delay(|| 0.0);
224        b.next_delay(|| 0.0);
225        assert_eq!(b.attempt(), 3);
226        b.reset();
227        assert_eq!(b.attempt(), 0, "reset clears the attempt count");
228        assert_eq!(b.next_delay(|| 0.0), BACKOFF_INITIAL);
229    }
230
231    /// Jitter stays within the ±20% band at both edges and never goes negative.
232    #[test]
233    fn jitter_stays_within_band_and_non_negative() {
234        let mut hi = Backoff::new();
235        assert_eq!(hi.next_delay(|| 1.0), Duration::from_millis(600));
236        let mut lo = Backoff::new();
237        assert_eq!(lo.next_delay(|| -1.0), Duration::from_millis(400));
238
239        // At the cap, the band is 15s ± 3s = [12s, 18s].
240        let mut at_cap = Backoff::new();
241        for _ in 0..6 {
242            at_cap.next_delay(|| 0.0);
243        }
244        assert_eq!(at_cap.clone().next_delay(|| 1.0), Duration::from_secs(18));
245        assert_eq!(at_cap.next_delay(|| -1.0), Duration::from_secs(12));
246    }
247
248    /// A jitter source out of range is clamped, so a buggy RNG can't blow the band.
249    #[test]
250    fn out_of_range_jitter_is_clamped() {
251        let mut b = Backoff::new();
252        assert_eq!(b.next_delay(|| 5.0), Duration::from_millis(600));
253    }
254
255    /// The production jitter source stays inside `[-1, 1]` so the band holds.
256    #[test]
257    fn rng_jitter_is_in_unit_range() {
258        for _ in 0..1000 {
259            let j = rng_jitter();
260            assert!((-1.0..=1.0).contains(&j), "rng jitter {j} out of [-1,1]");
261        }
262    }
263}
