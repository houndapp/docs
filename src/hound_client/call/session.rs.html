1//! The live call: [`CallSession`] (join/leave/commands + the cross-reconnect
2//! command seams), [`establish_live`] (one full control+MoQ session), and the
3//! auto-reconnect supervisor — which owns the live session, the backoff, and the
4//! screen-share state (so a share is republished against each fresh session).
5//! Invariant: recovery is a FULL re-establish (fresh `PeerId`) — the broken
6//! session's tasks/threads are fully torn down before a rebuild, and only an
7//! auth/permission failure (or an explicit leave) ends the call.

22/// The cross-reconnect outbound-control seam. `control_outbound` changes on every
23/// (re-)establish (it is the new control WS's writer queue), so commands must reach
24/// the CURRENT session: the supervisor swaps the live sender in here on each
25/// establish, and `set_muted` (etc.) loads it to send. An `ArcSwap` so the command
26/// path reads a cheap snapshot while the supervisor swaps lock-free.
27type SharedControlOutbound = Arc<ArcSwap<mpsc::UnboundedSender<ClientMessage>>>;

29/// A leave/drop cancel primitive: an edge-triggered [`Notify`] (wakes a task parked
30/// in a `select!`, e.g. the reconnect backoff sleep) PLUS a durable [`AtomicBool`]
31/// flag (observed by a task that checks after a cancel that fired before it parked).
32#[derive(Default)]
33pub(super) struct Cancel {
34    pub(super) notify: Notify,
35    cancelled: AtomicBool,
36}
37
38impl Cancel {
39    /// Request cancellation: set the durable flag, then wake every parked waiter.
40    fn cancel(&self) {
41        self.cancelled.store(true, Ordering::SeqCst);
42        self.notify.notify_waiters();
43    }
44
45    pub(super) fn is_cancelled(&self) -> bool {
46        self.cancelled.load(Ordering::SeqCst)
47    }
48}

50/// A screen-share command routed to the supervisor (which OWNS the running share so
51/// it can republish on reconnect). `Start` carries a `reply` oneshot so the async
52/// `screen_share_start` can await the open result (a permission/unsupported error
53/// must surface to the caller, not be swallowed). `Ended`/`Resized`/`Retry` are the
54/// capture thread's self-reported exits (via [`ScreenExitNotify`]): `Ended` is a clean
55/// user/OS stop (tear down + unpublish + drop the desire), while `Resized`/`Retry`
56/// RESTART the same share in place against the current session (the share must NOT end
57/// on a resize or a transient glitch).
58enum ScreenCommand {
59    Start {
60        source: hound_engine::types::ScreenSource,
61        reply: oneshot::Sender<Result<()>>,
62    },
63    Stop,
64    /// The capture ended on its own — a window close / OS "stop sharing" / source gone.
65    Ended,
66    /// The shared window resized past the debounce — restart at the new geometry.
67    Resized,
68    /// A transient capture fault/panic — restart the share in place.
69    Retry,
70}

72/// One live voice/video call. The supervisor task owns the live session(s) + every
73/// capture thread; this struct holds the cross-reconnect command seams + the
74/// supervisor handle. Dropping it (or [`CallSession::leave`]) cancels + tears down.
75///
76/// Construct with [`CallSession::join`]. The `HoundClient` holds at most one at a
77/// time (a [`Mutex<Option<CallSession>>`], see [`crate::app`]).
78pub struct CallSession {
79    space_id: Uuid,
80    /// REST client (clone) for the leave broadcast.
81    rest: Arc<RestClient>,
82    /// The watch senders (kept so `leave` can reset them; the live reader reads
83    /// `channels.view` for `own_track`).
84    channels: CallChannels,
85    /// Outbound control intent into the CURRENT session (swapped by the supervisor
86    /// on each establish): queue `SetMuted`/`Unpublish` here.
87    control_outbound: SharedControlOutbound,
88    /// Live device selections (hot-swap), shared across reconnects: the capture
89    /// loops re-subscribe to these on each establish so selection survives.
90    devices: DeviceControl,
91    /// Per-track mute flags the capture loops read (shared across reconnects).
92    mute: Arc<MuteFlags>,
93    /// Per-remote-audio-stream playback gain (the tile volume/mute control).
94    volumes: Arc<StreamVolumes>,
95    /// Whether the camera should be publishing (shared across reconnects).
96    camera_enabled: Arc<AtomicBool>,
97    /// Pause inbound video (the subscribe loop drops video subs, keeps audio).
98    /// Driven by app foreground/visibility; survives reconnects.
99    video_paused: watch::Sender<bool>,
100    /// Per-remote-video tile demand (size/visibility → quality request). The subscribe
101    /// loop reads it to pick source vs low layer; driven by the UI per tile.
102    tile_quality: watch::Sender<HashMap<(PeerId, Source), hound_media::QualityRequest>>,
103    /// The sharer's live screen-quality preference (shared across reconnects); a
104    /// running screen adaptation tick reads it and reconfigures its ladder live.
105    screen_quality: SharedScreenQuality,
106    /// Leave/drop cancel for the supervisor (wakes a backoff sleep promptly).
107    cancel: Arc<Cancel>,
108    /// Screen commands to the supervisor (it owns the share + republishes it).
109    screen_tx: mpsc::UnboundedSender<ScreenCommand>,
110    /// The auto-reconnect supervisor task; awaited on `leave`, aborted on drop.
111    /// `Option` so `leave` can take it (await by value) without fighting `Drop`.
112    supervisor: Option<JoinHandle<()>>,
113}

115/// The handles for one running screen share: the capture-thread stop flag and the
116/// async publish tasks (video + optional app-audio). Stopping it latches the flag
117/// (the `!Send` capture thread drops its engine handles on its own thread) and
118/// aborts the tasks. Owned by the supervisor; replaced on a new share, cleared on
119/// stop/leave.
120pub(super) struct ScreenShareHandle {
121    pub(super) capture_stop: Arc<AtomicBool>,
122    pub(super) tasks: Vec<JoinHandle<()>>,
123}
124
125impl ScreenShareHandle {
126    /// Stop the share and WAIT until its async tasks are gone. The publish tasks
127    /// own the moq broadcast announces for the screen paths, and a restart
128    /// re-announces the SAME paths — moq-net parks a same-path announce as a
129    /// backup until the old broadcast closes, so the old tasks must be fully
130    /// dropped (announces withdrawn) before a replacement is spawned or
131    /// receivers can wedge on the zombie broadcast. The `!Send` capture thread
132    /// stops via the latch on its own time (it owns no announces).
133    async fn stop(self) {
134        self.capture_stop.store(true, Ordering::Relaxed);
135        for task in &self.tasks {
136            task.abort();
137        }
138        for task in self.tasks {
139            let _ = task.await;
140        }
141    }
142}

184/// The per-establish layers: one full control+MoQ session built by
185/// [`establish_live`]. The supervisor monitors it ([`LiveSession::monitor`]) and, on
186/// any layer drop, tears it fully down ([`LiveSession::teardown`]) and rebuilds it.
187pub(super) struct LiveSession {
188    /// The relay session (multiplexes every track over one connection). Held to keep
189    /// our announces alive for the establish's lifetime.
190    relay: Arc<MoqRelaySession>,
191    /// The iroh endpoint backing [`Self::conn`]. Held for the call's lifetime: the
192    /// SFU is relay-only reachable, so dropping the endpoint stops the magicsock/relay
193    /// actor and wedges the connection right after the handshake. Closed gracefully
194    /// by [`Self::teardown`] (a bare drop aborts the socket ungracefully).
195    endpoint: iroh::endpoint::Endpoint,
196    /// The iroh connection under the relay session; `closed()` is one monitor leg,
197    /// and it is closed gracefully on teardown.
198    conn: iroh::endpoint::Connection,
199    /// Outbound control intent for THIS session (the live control WS writer queue).
200    pub(super) control_outbound: mpsc::UnboundedSender<ClientMessage>,
201    /// Our assigned peer id (from this session's `Welcome`).
202    pub(super) self_id: PeerId,
203    /// This client's own authenticated user id (`voice/join`'s `profile.user_id`),
204    /// used to build SELF broadcast paths (`broadcast_path`) under the SFU's
205    /// enforced peer-namespace convention.
206    pub(super) self_user_id: Uuid,
207    /// The shared session receive/capture clock for this establish.
208    clock: CaptureClock,
209    /// The control view receiver — the keyframe/feedback listeners resolve "our
210    /// track" from it. Cloned per-listener; held so a screen-share republish on this
211    /// session can spawn its own listeners.
212    view_rx: watch::Receiver<SessionView>,
213    /// Producer-side PLI broadcast for THIS session — each publish task's keyframe
214    /// listener `.subscribe()`s a fresh receiver.
215    keyframe_tx: broadcast::Sender<TrackId>,
216    /// Producer-side BWE-feedback broadcast for THIS session — each publish task's
217    /// feedback listener `.subscribe()`s a fresh receiver.
218    feedback_tx: broadcast::Sender<MediaFeedback>,
219    /// Producer-side quality-demand broadcast for THIS session — each publish task's
220    /// quality listener `.subscribe()`s a fresh receiver (demand-driven simulcast).
221    quality_tx: broadcast::Sender<QualityReport>,
222    /// The shared iroh connection clone the per-track adaptation tick samples RTT off.
223    rtt_conn: iroh::endpoint::Connection,
224    /// Fires (resolves) when the control-WS reader loop exits — the other monitor leg.
225    control_closed: oneshot::Receiver<()>,
226    /// All spawned async tasks (publish/subscribe/heartbeat), aborted on teardown.
227    tasks: Vec<JoinHandle<()>>,
228    /// Capture-thread stop flag (the threads flag-stop; not `JoinHandle`s here).
229    capture_stop: Arc<AtomicBool>,
230}
231
232impl LiveSession {
233    /// Block until a layer of this session drops, returning a typed error naming the
234    /// layer. Selects over (a) the control-WS-closed signal and (b) the iroh
235    /// connection's `closed()`. Detection is per-layer (so the log names the layer);
236    /// recovery is whole-session (the supervisor rebuilds everything).
237    async fn monitor(&mut self) -> HoundError {
238        tokio::select! {
239            biased;
240            _ = &mut self.control_closed => {
241                HoundError::Realtime("control WebSocket dropped".into())
242            }
243            _ = self.conn.closed() => {
244                HoundError::Network("MoQ relay connection closed".into())
245            }
246        }
247    }
248
249    /// Fully stop this session: latch the capture stop flag, abort every task, and
250    /// gracefully close the QUIC connection. Idempotent (abort/store/close are all
251    /// safe to repeat). Awaited before a rebuild so no capture thread or publish task
252    /// survives under the old (now-stale) `PeerId`.
253    async fn teardown(mut self) {
254        self.capture_stop.store(true, Ordering::Relaxed);
255        for task in self.tasks.drain(..) {
256            task.abort();
257        }
258        self.conn.close(0u32.into(), b"bye");
259        // Graceful endpoint shutdown (flushes the close frame); bounded so a wedged
260        // socket can never stall a reconnect or leave.
261        let _ =
262            tokio::time::timeout(std::time::Duration::from_secs(1), self.endpoint.close()).await;
263    }
264}

285impl CallSession {
286    /// Join the voice channel for `space_id`: run the first `voice/join` +
287    /// [`establish_live`], then spawn the auto-reconnect [`spawn_supervisor`]. Drives
288    /// the three observable snapshots through `channels`. The `sink` receives decoded
289    /// video frames. Device selection / mute / camera-enabled survive a reconnect
290    /// (their shared cells live on [`ConnectionParams`]).
291    ///
292    /// On the FIRST establish failing the call is NOT created (the error is returned
293    /// and `ConnectionState::Failed` is published). After that, a layer drop heals via
294    /// the supervisor without the user re-joining.
295    #[allow(clippy::too_many_arguments)]
296    pub async fn join(
297        rest: Arc<RestClient>,
298        base_url: String,
299        space_id: Uuid,
300        sink: Arc<dyn VideoSink>,
301        camera_enabled: bool,
302        #[cfg(feature = "synthetic")] capture_source: CaptureSource,
303        view: watch::Sender<CallView>,
304        connection: watch::Sender<ConnectionState>,
305        levels: watch::Sender<LevelsView>,
306        stats_out: watch::Sender<crate::stats::StatsSnapshot>,
307        identities: PeerIdentities,
308    ) -> Result<Self> {
309        let channels = CallChannels {
310            view,
311            connection,
312            levels,
313        };
314        // `send_replace`, not `send`, on every watch publish in this module: `send`
315        // silently drops the value with zero live receivers, and a headless caller
316        // (bot / e2e scenario) subscribes `watch_connection`/`watch_call` lazily —
317        // it must still see the latest state on first subscribe. Same footgun as
318        // `HoundClient::logout`/`load_communities` (see the comments there).
319        channels
320            .connection
321            .send_replace(ConnectionState::Connecting);
322
323        // Cross-reconnect state: shared cells the capture loops re-subscribe to on
324        // every establish so device/mute/camera selection survives a reconnect.
325        let devices = DeviceControl {
326            mic: watch::channel(DeviceChoice::Default).0,
327            camera: watch::channel(DeviceChoice::Default).0,
328            speaker: watch::channel(DeviceChoice::Default).0,
329        };
330        let params = ConnectionParams {
331            rest: rest.clone(),
332            base_url,
333            space_id,
334            sink,
335            channels: channels.clone(),
336            identities,
337            devices: devices.clone(),
338            mute: Arc::new(MuteFlags::default()),
339            volumes: Arc::new(StreamVolumes::default()),
340            camera_enabled: Arc::new(AtomicBool::new(camera_enabled)),
341            #[cfg(feature = "synthetic")]
342            capture_source,
343            video_paused: watch::channel(false).0,
344            tile_quality: watch::channel(HashMap::new()).0,
345            screen_quality: Arc::new(ArcSwap::from_pointee(ScreenQuality::default())),
346            // One registry for the whole call (survives reconnects — it counts them).
347            stats: CallStats::new(),
348            stats_out,
349        };
350
351        // First establish: a failure here is surfaced (no half-wired session held).
352        let initial = match establish_live(&params).await {
353            Ok(live) => live,
354            Err(e) => {
355                channels.connection.send_replace(ConnectionState::Failed {
356                    reason: e.to_string(),
357                });
358                return Err(e);
359            }
360        };
361        channels.connection.send_replace(ConnectionState::Connected);
362
363        // The cross-reconnect command seams. `control_outbound` is seeded from the
364        // first session; the supervisor swaps in the fresh sender on each re-establish.
365        let control_outbound: SharedControlOutbound =
366            Arc::new(ArcSwap::from_pointee(initial.control_outbound.clone()));
367        let mute = params.mute.clone();
368        let volumes = params.volumes.clone();
369        let camera_enabled = params.camera_enabled.clone();
370        let video_paused = params.video_paused.clone();
371        let tile_quality = params.tile_quality.clone();
372        let screen_quality = params.screen_quality.clone();
373        let cancel = Arc::new(Cancel::default());
374        let (screen_tx, screen_rx) = mpsc::unbounded_channel::<ScreenCommand>();
375
376        let supervisor = spawn_supervisor(
377            params,
378            initial,
379            control_outbound.clone(),
380            cancel.clone(),
381            screen_tx.clone(),
382            screen_rx,
383        );
384
385        Ok(Self {
386            space_id,
387            rest,
388            channels,
389            control_outbound,
390            devices,
391            mute,
392            volumes,
393            camera_enabled,
394            video_paused,
395            tile_quality,
396            screen_quality,
397            cancel,
398            screen_tx,
399            supervisor: Some(supervisor),
400        })
401    }

403    /// Mute or unmute the local mic. Gates the encoder locally (no audio is sent)
404    /// AND tells the control plane so the roster reflects it for everyone. Routes to
405    /// the CURRENT session via the shared outbound (survives a reconnect).
406    pub fn set_muted(&self, muted: bool) {
407        self.mute.mic.store(muted, Ordering::Relaxed);
408        // Tell the server about our own mic track's mute. We find it from the
409        // current view; if not yet published, the next publish carries no mute and
410        // the local gate still applies, so this is best-effort.
411        if let Some(track) = self.own_track(MediaKind::Audio, Source::Microphone) {
412            let _ = self
413                .control_outbound
414                .load()
415                .send(ClientMessage::SetMuted { track, muted });
416        }
417    }

419    /// Enable/disable the local camera. The capture task watches this flag and
420    /// starts/stops capturing (no re-join needed); the flag is shared across
421    /// reconnects, so the camera state survives a heal.
422    ///
423    /// Also announces/withdraws the camera track on the control plane: the
424    /// establish-time publish only fires when the camera is already on at join, so a
425    /// mid-call enable must `Publish` here — otherwise capture + self-preview run but
426    /// the call view has no camera track and peers never learn of it. A freshly
427    /// published track is un-muted, so the announce alone makes the self tile render
428    /// video.
429    pub fn set_camera_enabled(&self, enabled: bool) {
430        if self.camera_enabled.swap(enabled, Ordering::Relaxed) == enabled {
431            return; // unchanged — don't churn the control plane
432        }
433        let outbound = self.control_outbound.load();
434        if enabled {
435            let _ = outbound.send(ClientMessage::Publish {
436                track: TrackDecl {
437                    kind: MediaKind::Video,
438                    source: Source::Camera,
439                    codec: CtlCodec::H264,
440                    low: false,
441                },
442            });
443        } else if let Some(track) = self.own_track(MediaKind::Video, Source::Camera) {
444            let _ = outbound.send(ClientMessage::Unpublish { track });
445        }
446    }

448    /// Pause/resume *inbound* video. Paused, the subscribe loop drops every remote
449    /// video subscription (camera + screen); audio keeps flowing. Driven by app
450    /// foreground/visibility so a backgrounded call stops paying to receive +
451    /// render a screenshare nobody is looking at. Idempotent; survives reconnects.
452    pub fn set_video_paused(&self, paused: bool) {
453        self.video_paused.send_if_modified(|cur| {
454            if *cur == paused {
455                false
456            } else {
457                *cur = paused;
458                true
459            }
460        });
461    }

463    /// Report a remote video tile's rendered size + visibility, which the subscribe
464    /// loop maps to a quality demand (small/hidden ⇒ the low layer, large ⇒ source)
465    /// and folds across reconnects. Called by the UI per tile as it lays out / scrolls.
466    pub fn set_tile_quality(
467        &self,
468        peer: u32,
469        source: Source,
470        width: u32,
471        height: u32,
472        visible: bool,
473    ) {
474        let req = hound_media::tile_quality_request(width, height, visible);
475        let key = (PeerId(peer), source);
476        self.tile_quality.send_if_modified(|map| {
477            if map.get(&key) == Some(&req) {
478                false
479            } else {
480                tracing::info!(
481                    peer,
482                    ?source,
483                    width,
484                    height,
485                    visible,
486                    ?req,
487                    "simulcast[recv]: tile demand"
488                );
489                map.insert(key, req);
490                true
491            }
492        });
493    }

518    /// Start sharing the screen source identified by `source_id` (a token from
519    /// [`list_screen_sources`]). The command is routed to the supervisor (which OWNS
520    /// the running share so it can republish it after a reconnect); we await its open
521    /// result on a oneshot so a permission/unsupported error surfaces to the caller.
522    /// Replaces any current share. A *runtime* capture end (window closed, OS "stop
523    /// sharing") just stops the track.
524    pub async fn screen_share_start(&self, source_id: String) -> Result<()> {
525        let source = parse_screen_source(&source_id)?;
526        let (reply, reply_rx) = oneshot::channel();
527        self.screen_tx
528            .send(ScreenCommand::Start { source, reply })
529            .map_err(|_| HoundError::InvalidState("call ended (supervisor gone)".into()))?;
530        reply_rx
531            .await
532            .map_err(|_| HoundError::InvalidState("call ended before screen share opened".into()))?
533    }

550    /// Set one remote audio stream's playback volume (`0.0` = muted, `1.0` = unity,
551    /// clamped to ≤ 2.0) — the tile right-click volume/mute control. Applied by the
552    /// stream's decode pipeline before the mix, so it affects only this stream and
553    /// takes effect within one audio frame (~20 ms). Local-only (the sender is not
554    /// told); survives reconnects and layer switches.
555    pub fn set_stream_volume(&self, peer: u32, source: Source, volume: f32) {
556        tracing::info!(peer, ?source, volume, "call: stream volume set");
557        self.volumes.set(peer, source, volume);
558    }
559
560    /// Toggle whether an active screen share broadcasts its app audio (the
561    /// broadcaster's "share audio" control). Gates the publish loop before encode —
562    /// a muted share costs zero CPU and zero wire data (the DTX-aware receivers
563    /// read the silence as a quiet source, not a fault). Survives reconnects and
564    /// share restarts; a share started later honours the current setting.
565    pub fn set_screen_audio_enabled(&self, enabled: bool) {
566        tracing::info!(enabled, "call: screen share audio toggled");
567        self.mute.screen_audio.store(!enabled, Ordering::Relaxed);
568    }
569
570    /// Best-effort `voice/leave` broadcast, then cancel + await the supervisor (which
571    /// tears down the live session). Idempotent with drop.
572    pub async fn leave(mut self) {
573        // Tell the live session's control plane we're leaving (best-effort).
574        let _ = self.control_outbound.load().send(ClientMessage::Bye);
575        let space_id = self.space_id;
576        let _ = tokio::time::timeout(
577            std::time::Duration::from_secs(2),
578            self.rest.voice_leave(space_id),
579        )
580        .await;
581        // Cancel the supervisor (wakes a backoff sleep promptly) and let it tear the
582        // live session down; bound the wait so a wedged teardown can't hang leave.
583        self.cancel.cancel();
584        if let Some(supervisor) = self.supervisor.take() {
585            let _ = tokio::time::timeout(std::time::Duration::from_secs(3), supervisor).await;
586        }
587        self.channels.connection.send_replace(ConnectionState::Idle);
588        self.channels.view.send_replace(CallView::default());
589        self.channels.levels.send_replace(LevelsView::default());
590    }
591
592    /// Find one of our own tracks of the given kind/source in the current view.
593    fn own_track(&self, kind: MediaKind, source: Source) -> Option<hound_control::TrackId> {
594        let view = self.channels.view.borrow();
595        let me = view.self_id?;
596        view.tracks
597            .iter()
598            .find(|t| t.peer == me && t.kind == kind && t.source == source)
599            .map(|t| hound_control::TrackId(t.id))
600    }
601}

603impl Drop for CallSession {
604    fn drop(&mut self) {
605        // Safety net for an owner that dropped without calling `leave`: cancel the
606        // supervisor (it tears down the live session: capture stop + tasks abort +
607        // conn close) and abort the supervisor task itself.
608        self.cancel.cancel();
609        if let Some(supervisor) = self.supervisor.take() {
610            supervisor.abort();
611        }
612    }
613}

615// ─── establish_live: build one full control+MoQ session ─────────────────────────
616
617/// Build one full control+MoQ [`LiveSession`]. Each establish re-mints the relay
618/// token by calling `voice/join`, so a token-expiry SFU drop heals via reconnect —
619/// that re-mint is what keeps a call alive past the relay token's TTL. The capture
620/// loops subscribe to `params.devices.*` (so device selection survives a reconnect)
621/// and read `params.mute`/`params.camera_enabled`. Seeds the identity map from the
622/// fresh `voice/join` each time.
623pub(super) async fn establish_live(params: &ConnectionParams) -> Result<LiveSession> {
624    let rest = &params.rest;
625    let space_id = params.space_id;
626    let channels = &params.channels;
627    let identities = &params.identities;
628    let sink = &params.sink;
629
630    // 1. REST voice/join → relay url/token/peers/sfu node id (re-minted every establish).
631    let join = rest.voice_join(space_id).await?;
632    let Some(sfu_node_id) = join.sfu_node_id.clone() else {
633        // Dev backend with no SFU configured: nothing to dial. A typed error
634        // rather than a silent no-op so the caller knows media won't flow.
635        return Err(HoundError::Media(
636            "voice/join returned no sfu_node_id (no SFU configured)".into(),
637        ));
638    };
639
640    // Seed the peer_id → user_id map from the join roster: peers already in the
641    // channel carry their sfu_peer_id, so their identity is known before the first
642    // control publish. (Self is added once Welcome assigns our peer id below.)
643    {
644        let mut map = HashMap::new();
645        for p in &join.peers {
646            if let Some(pid) = p.sfu_peer_id {
647                map.insert(pid as u32, p.user_id);
648            }
649        }
650        identities.store(Arc::new(map));
651    }
652
653    // 2. Connect the control WS and drive the SessionView → CallView watch.
654    let token = rest
655        .token()
656        .ok_or_else(|| HoundError::Auth("cannot connect control plane: no session token".into()))?;
657    let control = ControlClient::connect(
658        &params.base_url,
659        space_id,
660        &token,
661        channels.clone(),
662        identities.clone(),
663    )
664    .await?;
665    let control_outbound = control.outbound.clone();
666    let control_closed = control.closed;
667    let keyframe_tx = control.keyframe_tx.clone();
668    let feedback_tx = control.feedback_tx.clone();
669    let quality_tx = control.quality_tx.clone();
670    let mut view_rx = control.view_rx.clone();
671
672    // Wait for the server's Welcome to learn our peer id (with a timeout, so a
673    // wedged handshake fails the establish rather than hanging the reconnect ladder).
674    let self_id = tokio::time::timeout(
675        std::time::Duration::from_secs(15),
676        wait_for_self_id(&mut view_rx),
677    )
678    .await
679    .map_err(|_| HoundError::Media("control Welcome timed out".into()))??;
680
681    // This client's own authenticated identity — used to build SELF broadcast
682    // paths (`broadcast_path`) under the SFU's peer-namespace convention.
683    let self_user_id = join.profile.user_id;
684
685    // Now that Welcome assigned our (fresh) peer id, add self to the identity map and
686    // refresh the published participants (the reader already sent peers).
687    identities.rcu(|m| {
688        let mut m = HashMap::clone(m);
689        m.insert(self_id.0, self_user_id);
690        m
691    });
692    channels.view.send_modify(|cv| {
693        cv.participants = build_participants(&cv.peers, &identities.load());
694    });
695
696    // 3. Dial the MoQ relay over a fresh iroh endpoint + connection. Keep the
697    //    endpoint (stored on the LiveSession) alive for the whole call — see
698    //    `dial_relay`.
699    let (endpoint, conn) = dial_relay(&join.relay_url, &sfu_node_id).await?;
700    let relay = Arc::new(
701        MoqRelaySession::connect_relay(conn.clone(), &join.relay_token)
702            .await
703            .map_err(|e| HoundError::Media(format!("MoQ relay handshake: {e}")))?,
704    );
705
706    // Shared session receive clock — every capture loop and the playout bus
707    // stamp against the SAME clock so A/V sync and the bus pop schedule align.
708    let clock = CaptureClock::start();
709
710    // 4. Audio infrastructure: one mixed-playout bus for all remote peers. The
711    //    speaker selection is shared across reconnects (the bus subscribes to it).
712    let apm = Engine::new().audio_processor().map(Arc::new);
713    // A synthetic capture source (a headless bot / CI) also gets a speaker-free
714    // playout bus: opening a real DAC is wrong (no hardware) and two real RT mix
715    // engines contending on one host give false audio-starvation verdicts.
716    #[cfg(feature = "synthetic")]
717    let playout_bus = if params.capture_source != CaptureSource::Default {
718        Engine::new().start_synthetic_playout_bus(
719            params.devices.speaker.subscribe(),
720            apm,
721            clock,
722            PLAYOUT_TARGET_US,
723        )
724    } else {
725        Engine::new().start_playout_bus(
726            params.devices.speaker.subscribe(),
727            apm,
728            clock,
729            PLAYOUT_TARGET_US,
730        )
731    };
732    #[cfg(not(feature = "synthetic"))]
733    let playout_bus = Engine::new().start_playout_bus(
734        params.devices.speaker.subscribe(),
735        apm,
736        clock,
737        PLAYOUT_TARGET_US,
738    );
739
740    let capture_stop = Arc::new(AtomicBool::new(false));
741    let mut tasks: Vec<JoinHandle<()>> = Vec::new();
742
743    // 4b. RTC debug stats: clear the per-establish counters (the registry itself
744    //     survives reconnects — it keeps the reconnect count), record this session's
745    //     identity, mark the transport connected, and start the 1 Hz emitter that
746    //     samples the registry into the `watch_stats` observable. Torn down +
747    //     respawned per establish like the other tasks.
748    params.stats.reset_session();
749    params
750        .stats
751        .set_session_identity(self_id.0, &join.relay_url, &sfu_node_id);
752    params.stats.set_transport_state(TransportState::Connected);
753    tasks.push(crate::stats::spawn_emitter(
754        params.stats_out.clone(),
755        params.stats.clone(),
756        playout_bus.clone(),
757    ));
758
759    // 5. Publish mic (Opus). Announce on the control plane, then run the
760    //    capture→encode→publish task (subscribes to the SHARED mic device watch).
761    control_outbound
762        .send(ClientMessage::Publish {
763            track: TrackDecl {
764                kind: MediaKind::Audio,
765                source: Source::Microphone,
766                codec: CtlCodec::Opus,
767                low: false,
768            },
769        })
770        .map_err(|_| HoundError::Media("control plane closed before mic publish".into()))?;
771    tasks.push(spawn_mic_publish(
772        relay.clone(),
773        space_id,
774        self_user_id,
775        self_id,
776        clock,
777        params.devices.mic.subscribe(),
778        params.mute.clone(),
779        channels.levels.clone(),
780        capture_stop.clone(),
781        params.stats.clone(),
782        #[cfg(feature = "synthetic")]
783        params.capture_source,
784    ));
785
786    // 6. Publish camera (H.264) if enabled. Best-effort: a missing camera does not
787    //    fail the establish (audio-only is valid). Only announce the video track when
788    //    the camera is on. The shared `camera_enabled` flag survives a reconnect.
789    if params.camera_enabled.load(Ordering::Relaxed) {
790        control_outbound
791            .send(ClientMessage::Publish {
792                track: TrackDecl {
793                    kind: MediaKind::Video,
794                    source: Source::Camera,
795                    codec: CtlCodec::H264,
796                    low: false,
797                },
798            })
799            .map_err(|_| HoundError::Media("control plane closed before cam publish".into()))?;
800    }
801    tasks.push(spawn_cam_publish(
802        relay.clone(),
803        space_id,
804        self_user_id,
805        self_id,
806        clock,
807        params.devices.camera.subscribe(),
808        params.camera_enabled.clone(),
809        sink.clone(),
810        control_outbound.clone(),
811        capture_stop.clone(),
812        conn.clone(),
813        view_rx.clone(),
814        keyframe_tx.subscribe(),
815        feedback_tx.subscribe(),
816        quality_tx.subscribe(),
817        params.stats.clone(),
818        #[cfg(feature = "synthetic")]
819        params.capture_source,
820    ));
821
822    // 7. Subscribe loop: reconcile remote tracks → decode → sink / mix bus. On a
823    //    reconnect this re-diffs the fresh roster and re-subscribes every remote.
824    tasks.push(spawn_subscribe_loop(
825        view_rx.clone(),
826        self_id,
827        space_id,
828        relay.clone(),
829        control_outbound.clone(),
830        sink.clone(),
831        playout_bus,
832        channels.levels.clone(),
833        clock,
834        params.stats.clone(),
835        params.video_paused.subscribe(),
836        params.tile_quality.subscribe(),
837        params.volumes.clone(),
838    ));
839
840    // 8. voice/confirm with our (fresh) peer id (best-effort) + heartbeat. No
841    //    relay-token refresh task: a reconnect re-mints the token via `voice/join`,
842    //    so a token-expiry SFU drop heals through the supervisor.
843    {
844        let rest = rest.clone();
845        let peer = self_id.0 as u16;
846        call_runtime().spawn(async move {
847            let _ = rest.voice_confirm(space_id, Some(peer)).await;
848        });
849    }
850    tasks.push(spawn_heartbeat(rest.clone(), space_id));
851
852    Ok(LiveSession {
853        relay,
854        rtt_conn: conn.clone(),
855        endpoint,
856        conn,
857        control_outbound,
858        self_id,
859        self_user_id,
860        clock,
861        view_rx,
862        keyframe_tx,
863        feedback_tx,
864        quality_tx,
865        control_closed,
866        tasks,
867        capture_stop,
868    })
869}

871// ─── The auto-reconnect supervisor ──────────────────────────────────────────────
872
873/// Spawn the auto-reconnect supervisor. It owns the live session + a [`Backoff`] +
874/// the screen-share state, monitors every layer, and on a failure tears the broken
875/// session fully down and rebuilds it with exponential backoff + jitter — re-minting
876/// via `voice/join` each attempt (which also refreshes the relay token). It gives up
877/// only on a FATAL (auth/permission) error or an explicit `leave` (cancel); transient
878/// errors retry FOREVER. State transitions are published on `channels.connection`.
879fn spawn_supervisor(
880    params: ConnectionParams,
881    initial: LiveSession,
882    control_outbound: SharedControlOutbound,
883    cancel: Arc<Cancel>,
884    screen_tx: mpsc::UnboundedSender<ScreenCommand>,
885    mut screen_rx: mpsc::UnboundedReceiver<ScreenCommand>,
886) -> JoinHandle<()> {
887    call_runtime().spawn(async move {
888        let mut live = initial;
889        let mut backoff = Backoff::new();
890
891        // Screen-share state owned by the supervisor: the running pipeline (if any)
892        // and the DESIRED source (cross-reconnect — republished on each fresh session).
893        // `notify` lets the `!Send` capture thread report its OWN exit (ended / resized
894        // / transient retry) back into this command loop, where it routes to a clean
895        // stop or an in-place restart.
896        let mut screen_pipeline: Option<ScreenShareHandle> = None;
897        let mut desired_source: Option<hound_engine::types::ScreenSource> = None;
898        let notify: ScreenExitNotify = {
899            let tx = screen_tx.clone();
900            Arc::new(move |exit| {
901                let _ = tx.send(match exit {
902                    ScreenCaptureExit::Ended => ScreenCommand::Ended,
903                    ScreenCaptureExit::Resized => ScreenCommand::Resized,
904                    ScreenCaptureExit::Retry => ScreenCommand::Retry,
905                });
906            })
907        };
908
909        loop {
910            // Monitor the current live session for a layer failure, a cancel, OR a
911            // screen-share command.
912            let cause = tokio::select! {
913                biased;
914                _ = cancel.notify.notified() => {
915                    if let Some(p) = screen_pipeline.take() { p.stop().await; }
916                    live.teardown().await;
917                    tracing::info!("call: supervisor cancelled (leave) — session torn down");
918                    return;
919                }
920                cmd = screen_rx.recv() => {
921                    // `None` = the command channel closed (CallSession dropped without
922                    // a clean leave) — keep monitoring; cancel/drop tears down.
923                    if let Some(cmd) = cmd {
924                        handle_screen_command(
925                            cmd,
926                            &mut screen_pipeline,
927                            &mut desired_source,
928                            &live,
929                            &params,
930                            &notify,
931                        )
932                        .await;
933                    }
934                    continue;
935                }
936                cause = live.monitor() => cause,
937            };
938
939            if cancel.is_cancelled() {
940                if let Some(p) = screen_pipeline.take() {
941                    p.stop().await;
942                }
943                live.teardown().await;
944                return;
945            }
946
947            tracing::warn!(%cause, "call: layer failed — reconnecting");
948            params
949                .stats
950                .set_transport_state(TransportState::Reconnecting);
951            params.stats.inc_reconnect_attempts();
952            let _ = params
953                .channels
954                .connection
955                .send_replace(ConnectionState::Reconnecting { attempt: 0 });
956            // The broken session's tasks/threads are fully stopped before we rebuild
957            // (no leaked threads, no stale broadcast path under the old PeerId). The
958            // screen pipeline rode the OLD session's relay identity, so tear it down
959            // too — it is republished against the fresh session below.
960            if let Some(p) = screen_pipeline.take() {
961                p.stop().await;
962            }
963            live.teardown().await;
964
965            match reconnect_loop(&params, &cancel, &mut backoff, cause).await {
966                ReconnectOutcome::Reconnected(new_live) => {
967                    live = *new_live;
968                    backoff.reset();
969                    // Publish the fresh session's outbound so commands reach it.
970                    control_outbound.store(Arc::new(live.control_outbound.clone()));
971                    params
972                        .channels
973                        .connection
974                        .send_replace(ConnectionState::Connected);
975                    tracing::info!("call: reconnected");
976
977                    // Republish the desired screen share against the fresh session.
978                    if let Some(source) = desired_source.clone() {
979                        match spawn_screen_publish(
980                            live.relay.clone(),
981                            params.space_id,
982                            live.self_user_id,
983                            live.self_id,
984                            live.clock,
985                            source,
986                            params.sink.clone(),
987                            live.control_outbound.clone(),
988                            live.rtt_conn.clone(),
989                            live.view_rx.clone(),
990                            live.keyframe_tx.subscribe(),
991                            live.feedback_tx.subscribe(),
992                            live.quality_tx.subscribe(),
993                            params.screen_quality.clone(),
994                            notify.clone(),
995                            params.stats.clone(),
996                            params.mute.clone(),
997                        )
998                        .await
999                        {
1000                            Ok(p) => {
1001                                screen_pipeline = Some(p);
1002                                tracing::info!("call: screen share republished after reconnect");
1003                            }
1004                            Err(e) => {
1005                                // Source vanished mid-outage / permission revoked: drop
1006                                // the desire (a clean end; the UI sees the track gone).
1007                                tracing::warn!(%e, "call: screen republish failed — ending share");
1008                                desired_source = None;
1009                            }
1010                        }
1011                    }
1012                }
1013                ReconnectOutcome::Cancelled => {
1014                    tracing::info!("call: supervisor cancelled during reconnect");
1015                    return;
1016                }
1017                ReconnectOutcome::Fatal(err) => {
1018                    params.stats.set_transport_state(TransportState::Failed);
1019                    params
1020                        .channels
1021                        .connection
1022                        .send_replace(ConnectionState::Failed {
1023                            reason: err.to_string(),
1024                        });
1025                    tracing::error!(%err, "call: reconnect gave up (fatal) — call failed");
1026                    return;
1027                }
1028            }
1029        }
1030    })
1031}

1033/// Handle one [`ScreenCommand`] inside the supervisor loop, against the CURRENT live
1034/// session. `Start` replaces any running share (stop-old-first + unpublish), starts a
1035/// fresh pipeline, records the desired source for cross-reconnect republish, and
1036/// replies on the oneshot. `Stop` and `Ended` both tear the pipeline down + unpublish
1037/// the tracks + drop the desire (a clean user/OS stop). `Resized`/`Retry` RESTART the
1038/// same share in place (the share must NOT end on a resize or a transient glitch).
1039async fn handle_screen_command(
1040    cmd: ScreenCommand,
1041    screen_pipeline: &mut Option<ScreenShareHandle>,
1042    desired_source: &mut Option<hound_engine::types::ScreenSource>,
1043    live: &LiveSession,
1044    params: &ConnectionParams,
1045    notify: &ScreenExitNotify,
1046) {
1047    match cmd {
1048        ScreenCommand::Start { source, reply } => {
1049            // Replace: stop any running share + unpublish its tracks first.
1050            if let Some(prev) = screen_pipeline.take() {
1051                prev.stop().await;
1052                unpublish_screen_tracks(live, params);
1053            }
1054            match spawn_screen_publish(
1055                live.relay.clone(),
1056                params.space_id,
1057                live.self_user_id,
1058                live.self_id,
1059                live.clock,
1060                source.clone(),
1061                params.sink.clone(),
1062                live.control_outbound.clone(),
1063                live.rtt_conn.clone(),
1064                live.view_rx.clone(),
1065                live.keyframe_tx.subscribe(),
1066                live.feedback_tx.subscribe(),
1067                live.quality_tx.subscribe(),
1068                params.screen_quality.clone(),
1069                notify.clone(),
1070                params.stats.clone(),
1071                params.mute.clone(),
1072            )
1073            .await
1074            {
1075                Ok(p) => {
1076                    *screen_pipeline = Some(p);
1077                    *desired_source = Some(source);
1078                    let _ = reply.send(Ok(()));
1079                }
1080                Err(e) => {
1081                    *desired_source = None;
1082                    let _ = reply.send(Err(e));
1083                }
1084            }
1085        }
1086        // A user/OS stop (explicit `Stop`) or a self-reported clean capture `Ended`:
1087        // tear the pipeline down, unpublish, and drop the desire so a reconnect does
1088        // NOT republish it. `Ended` is de-duped — if the supervisor already cleared the
1089        // pipeline (e.g. a `Stop` raced the notify), this is a harmless no-op.
1090        ScreenCommand::Stop | ScreenCommand::Ended => {
1091            *desired_source = None;
1092            if let Some(p) = screen_pipeline.take() {
1093                p.stop().await;
1094                unpublish_screen_tracks(live, params);
1095            }
1096        }
1097        // Same source, restart in place. `Resized` = the window changed geometry (the
1098        // capture's pixel size is fixed at open, so we re-open at the new geometry);
1099        // `Retry` = a transient capture fault/panic. Both reuse the one restart, which
1100        // re-opens cleanly; a restart whose re-open fails (the source is genuinely gone)
1101        // degrades to a clean end.
1102        ScreenCommand::Resized => {
1103            restart_screen_share(
1104                "new source geometry",
1105                screen_pipeline,
1106                desired_source,
1107                live,
1108                params,
1109                notify,
1110            )
1111            .await;
1112        }
1113        ScreenCommand::Retry => {
1114            restart_screen_share(
1115                "transient capture fault",
1116                screen_pipeline,
1117                desired_source,
1118                live,
1119                params,
1120                notify,
1121            )
1122            .await;
1123        }
1124    }
1125}
1126
1127/// Tear down the running screen pipeline and start it again against the SAME source
1128/// (recorded in `desired_source`). Shared by the `Resized` and `Retry` exits: receivers
1129/// see the screen track replaced, the desire is untouched on success, and a restart
1130/// whose `spawn_screen_publish` fails — the source truly vanished — degrades to a clean
1131/// end (drops the desire so a later reconnect won't republish it).
1132async fn restart_screen_share(
1133    reason: &str,
1134    screen_pipeline: &mut Option<ScreenShareHandle>,
1135    desired_source: &mut Option<hound_engine::types::ScreenSource>,
1136    live: &LiveSession,
1137    params: &ConnectionParams,
1138    notify: &ScreenExitNotify,
1139) {
1140    let Some(source) = desired_source.clone() else {
1141        return; // nothing to restart (already stopped)
1142    };
1143    if let Some(p) = screen_pipeline.take() {
1144        p.stop().await;
1145        unpublish_screen_tracks(live, params);
1146    }
1147    match spawn_screen_publish(
1148        live.relay.clone(),
1149        params.space_id,
1150        live.self_user_id,
1151        live.self_id,
1152        live.clock,
1153        source,
1154        params.sink.clone(),
1155        live.control_outbound.clone(),
1156        live.rtt_conn.clone(),
1157        live.view_rx.clone(),
1158        live.keyframe_tx.subscribe(),
1159        live.feedback_tx.subscribe(),
1160        live.quality_tx.subscribe(),
1161        params.screen_quality.clone(),
1162        notify.clone(),
1163        params.stats.clone(),
1164        params.mute.clone(),
1165    )
1166    .await
1167    {
1168        Ok(p) => {
1169            *screen_pipeline = Some(p);
1170            tracing::info!(reason, "call: screen share restarted");
1171        }
1172        Err(e) => {
1173            // The re-open failed — the source is genuinely gone. Degrade to a clean end.
1174            tracing::warn!(reason, %e, "call: screen restart failed — ending share");
1175            *desired_source = None;
1176        }
1177    }
1178}
