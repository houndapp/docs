1//! Outbound camera: the `!Send` capture+encode thread (device hot-swap,
2//! self-preview tee, adaptive encode) and the async publish task running the 1 s
3//! BWE adaptation tick. The capture side reads its decisions off
4//! [`adapt::AdaptShared`]; the codec is negotiated against the roster and
5//! re-driven mid-call by the re-negotiator. Self-preview keeps running even
6//! while the ladder pauses video (AudioOnly), so the local tile never freezes.

10// ─── Publish: camera (capture → H.264 → MoQ + self-preview) ─────────────────────
11
12/// Spawn the camera publish task. When `enabled`, a blocking thread opens the
13/// engine `Camera`, pushes each frame's I420 to the self-preview sink (`"self"`),
14/// H.264-encodes it, and queues the access units for the async publish half. When
15/// disabled, capture stops (the device is released). Self-preview works even if
16/// the publish/subscribe never gets a remote peer.
17///
18/// Sender-side BWE adaptation: a 1 s tick folds `rtt_conn`'s
19/// RTT + the worst receiver-reported loss (from the per-track feedback listener) into
20/// a target the capture/encode loop applies (bitrate retarget, fps gate, downscale to
21/// the rung, AudioOnly pause), and the keyframe listener arms forced IDRs on PLI. The
22/// adaptation is per-establish — rebuilt fresh on each reconnect (correct: the rate
23/// controller starts clean against the new connection).
24#[allow(clippy::too_many_arguments)]
25pub(super) fn spawn_cam_publish(
26    relay: Arc<MoqRelaySession>,
27    space_id: Uuid,
28    self_id: PeerId,
29    clock: CaptureClock,
30    device_rx: watch::Receiver<DeviceChoice<CameraId>>,
31    enabled: Arc<AtomicBool>,
32    sink: Arc<dyn VideoSink>,
33    stop: Arc<AtomicBool>,
34    rtt_conn: iroh::endpoint::Connection,
35    view_rx: watch::Receiver<SessionView>,
36    keyframe_rx: broadcast::Receiver<TrackId>,
37    feedback_rx: broadcast::Receiver<MediaFeedback>,
38    quality_rx: broadcast::Receiver<QualityReport>,
39    stats: Arc<CallStats>,
40) -> JoinHandle<()> {
41    let (enc_tx, mut enc_rx) = tokio::sync::mpsc::channel::<WireFrame>(8);
42
43    // The shared adaptation seam between the async tick (below) and the capture
44    // thread. The keyframe/feedback listeners write into it too; the encode loop reads.
45    let adapt = Arc::new(AdaptShared::default());
46
47    // Negotiate the camera codec against the live roster (HW-encode ∩ all-peers
48    // decode) and seed the slot BEFORE the capture thread starts so its first encoder
49    // is built in the right codec. `encode_caps` is probed once and reused by the
50    // re-negotiator below.
51    let encode_caps = Engine::new().encode_caps();
52    let codec = negotiate_codec(encode_caps, &view_rx.borrow());
53    adapt
54        .target_codec
55        .store(codec_rank(codec), Ordering::Relaxed);
56
57    let capture_stop = stop.clone();
58    let adapt_for_capture = adapt.clone();
59    let stats_capture = stats.clone();
60    std::thread::Builder::new()
61        .name("call-cam-capture-encode".into())
62        .spawn(move || {
63            cam_capture_encode_loop(
64                capture_stop,
65                clock,
66                device_rx,
67                enabled,
68                sink,
69                enc_tx,
70                self_id.0 as u16,
71                adapt_for_capture,
72                stats_capture,
73            )
74        })
75        .ok();
76
77    // Producer-side PLI relay + receiver→producer BWE feedback, scoped to OUR camera
78    // track. Their tasks ride the cam publish task (returned), so they tear down with
79    // it on leave/reconnect.
80    let keyframe_task = spawn_keyframe_listener(
81        keyframe_rx,
82        view_rx.clone(),
83        self_id,
84        Source::Camera,
85        adapt.clone(),
86        None,
87    );
88    let feedback_task = spawn_feedback_listener(
89        feedback_rx,
90        view_rx.clone(),
91        self_id,
92        Source::Camera,
93        adapt.clone(),
94    );
95    // Demand-driven simulcast: aggregate receiver quality requests into the layer
96    // planner (drives the low-layer supervisor). Dark until `SIMULCAST_ACTUATE`.
97    let layers = Arc::new(LayerControl::new(CAM_BITRATE.bps() as u64));
98    let quality_task =
99        spawn_quality_listener(quality_rx, view_rx.clone(), self_id, Source::Camera, layers);
100    // Re-negotiate the codec as the roster changes mid-call (a less-capable peer
101    // joins → downgrade so they get video; leaves → climb back).
102    let renegotiate_task =
103        spawn_codec_renegotiator(view_rx, encode_caps, adapt.clone(), codec, "cam");
104
105    let path = broadcast_path(space_id, self_id, CAM_TRACK);
106    call_runtime().spawn(async move {
107        // Abort the listeners when this task ends (clean return or `leave`-time abort).
108        let _keyframe_task = AbortOnDrop(keyframe_task);
109        let _feedback_task = AbortOnDrop(feedback_task);
110        let _quality_task = AbortOnDrop(quality_task);
111        let _renegotiate_task = AbortOnDrop(renegotiate_task);
112
113        let mut publisher = match relay.publish_track(&path, CAM_TRACK, VIDEO_PRIORITY) {
114            Ok(p) => p,
115            Err(e) => {
116                tracing::error!(%e, "cam: failed to publish track");
117                return;
118            }
119        };
120        // RTC-debug: the camera track is announced in the negotiated codec.
121        stats.set_video_enabled(true);
122        stats.set_video_codec(video_codec_to_ctl(codec).label());
123
124        // Sender-side adaptation core (sans-IO), seeded at the encoder's static launch
125        // bitrate. The 1 s tick folds backpressure + RTT + remote loss into a decision
126        // the capture thread reads off `adapt`; ticks run from the start, forwarding
127        // waits on the live gate (see `drive_video_publish`).
128        let mut adaptation = VideoAdaptation::standard(CAM_BITRATE.bps() as u64);
129        drive_video_publish(
130            &mut publisher,
131            &mut enc_rx,
132            &adapt,
133            &mut adaptation,
134            &rtt_conn,
135            &stats,
136            false,
137            "cam",
138            |_| {},
139        )
140        .await;
141    })
142}

144/// The camera capture+encode loop. Mirrors `crates/frb/src/camera.rs` for the
145/// self-preview push, and additionally H.264-encodes for publish. Honors the
146/// `enabled` flag (released device when off) and hot-swaps on `device_rx` change.
147/// Reads `adapt` each frame: retargets the encoder bitrate, downsamples to the rung
148/// fps, downscales to the rung geometry, serves a forced IDR, and skips encode/publish
149/// while paused (AudioOnly under severe congestion). Self-preview keeps running even
150/// while paused.
151#[allow(clippy::too_many_arguments)]
152fn cam_capture_encode_loop(
153    stop: Arc<AtomicBool>,
154    clock: CaptureClock,
155    mut device_rx: watch::Receiver<DeviceChoice<CameraId>>,
156    enabled: Arc<AtomicBool>,
157    sink: Arc<dyn VideoSink>,
158    enc_tx: tokio::sync::mpsc::Sender<WireFrame>,
159    sender: u16,
160    adapt: Arc<AdaptShared>,
161    stats: Arc<CallStats>,
162) {
163    let mut choice = device_rx.borrow_and_update().clone();
164    let mut camera = None;
165    let mut encoder: Option<hound_engine::api::VideoEncoder> = None;
166    let mut frame_seq = 0u32;
167    // The negotiated codec target we last (re)built the encoder for; the mid-call
168    // rebuild keys off this changing, not off the encoder's actual codec, so a target
169    // the device can't build doesn't re-trigger every frame.
170    let mut last_target = codec_from_rank(adapt.target_codec.load(Ordering::Relaxed));
171    // The encoder bitrate we last applied, so a steady target costs no syscall.
172    let mut applied_bps = CAM_BITRATE.bps();
173    // Downsamples capture (CAM_FPS) to the chosen rung's fps.
174    let mut fps_gate = hound_media::FrameRateGate::new();
175
176    while !stop.load(Ordering::Relaxed) {
177        // Camera disabled: release the device and idle.
178        if !enabled.load(Ordering::Relaxed) {
179            if camera.is_some() {
180                drop(camera.take());
181                drop(encoder.take());
182            }
183            std::thread::sleep(std::time::Duration::from_millis(100));
184            continue;
185        }
186
187        if device_rx.has_changed().unwrap_or(false) {
188            choice = device_rx.borrow_and_update().clone();
189            drop(camera.take());
190            drop(encoder.take());
191        }
192
193        // Open (or re-open) the camera + encoder.
194        if camera.is_none() {
195            match Engine::new().open_camera(&choice, CAM_DIMS, CAM_FPS, clock) {
196                Ok(c) => camera = Some(c),
197                Err(_) => {
198                    std::thread::sleep(std::time::Duration::from_millis(300));
199                    continue;
200                }
201            }
202            // Build in the negotiated codec (HW-encode ∩ all-peers decode), seeded by
203            // `spawn_cam_publish` and re-driven by the mid-call re-negotiator,
204            // descending the ladder if it won't open.
205            let want = codec_from_rank(adapt.target_codec.load(Ordering::Relaxed));
206            encoder =
207                open_encoder_laddered(want, CAM_DIMS, CAM_FPS, CAM_BITRATE, CAM_KEYFRAME_CADENCE);
208            last_target = want;
209            if let Some(enc) = encoder.as_ref() {
210                use hound_engine::backend::VideoEncode;
211                tracing::info!(
212                    target: "video_encode",
213                    track = "cam",
214                    codec = ?enc.codec(),
215                    backend = enc.backend(),
216                    width = CAM_DIMS.width(),
217                    height = CAM_DIMS.height(),
218                    "video: encode path selected"
219                );
220            }
221            // Re-baseline so the first frame re-applies the live BWE target, and
222            // open at a fresh IDR for any subscriber that attached meanwhile.
223            applied_bps = CAM_BITRATE.bps();
224            adapt.force_keyframe.store(true, Ordering::Relaxed);
225        }
226
227        let Some(cap) = camera.as_mut() else {
228            continue;
229        };
230        let pulled = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| cap.next_frame()));
231        let frame = match pulled {
232            Ok(Ok(f)) => f,
233            _ => {
234                drop(camera.take());
235                drop(encoder.take());
236                std::thread::sleep(std::time::Duration::from_millis(200));
237                continue;
238            }
239        };
240
241        // RTC-debug: one raw camera frame pulled from the device.
242        stats.video_captured();
243
244        // Self-preview keyed "self" (runs even while paused, so the local preview
245        // never freezes mid-call). Surface-first: a surface-backed camera frame is
246        // rendered zero-copy; a scaled/I420 frame falls back to bytes. The capture
247        // geometry gauge is what the self tile sizes its texture from (stable,
248        // unlike the BWE-driven encode rung) — kept fresh per frame so a camera
249        // hot-swap with a different geometry follows automatically.
250        stats.set_video_capture_dims(frame.dims.width(), frame.dims.height());
251        push_preview(sink.as_ref(), SELF_KEY, &frame);
252
253        // AudioOnly: the ladder dropped video under severe congestion. Keep pulling the
254        // device (resume is instant) but skip encode + publish entirely.
255        if adapt.paused.load(Ordering::Relaxed) {
256            continue;
257        }
258
259        // Mid-call codec switch (a peer joined/left mid-call): rebuild the encoder in
260        // the re-negotiated codec at the live bitrate; its first frame is a fresh IDR.
261        let want = codec_from_rank(adapt.target_codec.load(Ordering::Relaxed));
262        if encoder.is_some() && want != last_target {
263            last_target = want;
264            if let Some(e) = open_encoder_laddered(
265                want,
266                CAM_DIMS,
267                CAM_FPS,
268                Bitrate::from_bps(applied_bps),
269                CAM_KEYFRAME_CADENCE,
270            ) {
271                tracing::info!(track = "cam", to = ?want, "call: switching encoder codec mid-call");
272                encoder = Some(e);
273                adapt.force_keyframe.store(true, Ordering::Relaxed);
274            } else {
275                tracing::warn!(track = "cam", codec = ?want, "call: mid-call codec switch found no usable encoder; keeping current");
276            }
277        }
278
279        let Some(enc) = encoder.as_mut() else {
280            continue;
281        };
282        encode_and_publish_adaptive(
283            &frame,
284            enc,
285            &adapt,
286            &mut applied_bps,
287            &mut fps_gate,
288            CAM_DIMS,
289            |wire| enc_tx.try_send(wire).is_ok(),
290            &mut frame_seq,
291            sender,
292            TrackKind::CamVideo,
293            &stats,
294        );
295    }
