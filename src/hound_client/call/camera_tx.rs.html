1//! Outbound camera: the `!Send` capture+encode thread (device hot-swap,
2//! self-preview tee, adaptive encode) and the async publish task running the 1 s
3//! BWE adaptation tick. The capture side reads its decisions off
4//! [`adapt::AdaptShared`]; the codec is negotiated against the roster and
5//! re-driven mid-call by the re-negotiator. Self-preview keeps running even
6//! while the ladder pauses video (AudioOnly), so the local tile never freezes.

10// ─── Publish: camera (capture → H.264 → MoQ + self-preview) ─────────────────────

12/// Spawn the camera publish task. When `enabled`, a blocking thread opens the
13/// engine `Camera`, pushes each frame's I420 to the self-preview sink (`"self"`),
14/// H.264-encodes it, and queues the access units for the async publish half. When
15/// disabled, capture stops (the device is released). Self-preview works even if
16/// the publish/subscribe never gets a remote peer.
17///
18/// Sender-side BWE adaptation: a 1 s tick folds `rtt_conn`'s
19/// RTT + the worst receiver-reported loss (from the per-track feedback listener) into
20/// a target the capture/encode loop applies (bitrate retarget, fps gate, downscale to
21/// the rung, AudioOnly pause), and the keyframe listener arms forced IDRs on PLI. The
22/// adaptation is per-establish — rebuilt fresh on each reconnect (correct: the rate
23/// controller starts clean against the new connection).
24#[allow(clippy::too_many_arguments)]
25pub(super) fn spawn_cam_publish(
26    relay: Arc<MoqRelaySession>,
27    space_id: Uuid,
28    self_id: PeerId,
29    clock: CaptureClock,
30    device_rx: watch::Receiver<DeviceChoice<CameraId>>,
31    enabled: Arc<AtomicBool>,
32    sink: Arc<dyn VideoSink>,
33    control_outbound: tokio::sync::mpsc::UnboundedSender<ClientMessage>,
34    stop: Arc<AtomicBool>,
35    rtt_conn: iroh::endpoint::Connection,
36    view_rx: watch::Receiver<SessionView>,
37    keyframe_rx: broadcast::Receiver<TrackId>,
38    feedback_rx: broadcast::Receiver<MediaFeedback>,
39    quality_rx: broadcast::Receiver<QualityReport>,
40    stats: Arc<CallStats>,
41    #[cfg(feature = "synthetic")] capture_source: CaptureSource,
42) -> JoinHandle<()> {
43    let (enc_tx, mut enc_rx) = tokio::sync::mpsc::channel::<WireFrame>(8);
44
45    // The shared adaptation seam between the async tick (below) and the capture
46    // thread. The keyframe/feedback listeners write into it too; the encode loop reads.
47    let adapt = Arc::new(AdaptShared::default());
48
49    // Negotiate the camera codec against the live roster (HW-encode ∩ all-peers
50    // decode) and seed the slot BEFORE the capture thread starts so its first encoder
51    // is built in the right codec. `encode_caps` is probed once and reused by the
52    // re-negotiator below.
53    let encode_caps = Engine::new().encode_caps();
54    let codec = negotiate_codec(encode_caps, &view_rx.borrow());
55    adapt
56        .target_codec
57        .store(codec_rank(codec), Ordering::Relaxed);
58
59    // Demand-driven simulcast control for the camera source + the LOW-layer channel
60    // the capture thread feeds. `low_active` is the capture thread's gate; it stays
61    // `None` (low path inert) until a receiver demands the low layer AND actuation is
62    // on. Created here so it can be handed to the capture thread below.
63    let layers = Arc::new(LayerControl::new(CAM_BITRATE.bps() as u64));
64    let low_active = layers.low_active.subscribe();
65    let (low_vid_tx, low_vid_rx) = tokio::sync::mpsc::channel::<WireFrame>(8);
66    // A low-track PLI sets this; the capture thread forces a low IDR so a receiver
67    // subscribing to the low layer resyncs promptly (not on the slow safety-net IDR).
68    let low_keyframe = Arc::new(AtomicBool::new(false));
69
70    let capture_stop = stop.clone();
71    let adapt_for_capture = adapt.clone();
72    let stats_capture = stats.clone();
73    let low_keyframe_capture = low_keyframe.clone();
74    std::thread::Builder::new()
75        .name("call-cam-capture-encode".into())
76        .spawn(move || {
77            cam_capture_encode_loop(
78                capture_stop,
79                clock,
80                device_rx,
81                enabled,
82                sink,
83                enc_tx,
84                self_id.0 as u16,
85                adapt_for_capture,
86                stats_capture,
87                low_active,
88                low_vid_tx,
89                low_keyframe_capture,
90                #[cfg(feature = "synthetic")]
91                capture_source,
92            )
93        })
94        .ok();
95
96    // Producer-side PLI relay + receiver→producer BWE feedback, scoped to OUR camera
97    // track. Their tasks ride the cam publish task (returned), so they tear down with
98    // it on leave/reconnect.
99    let keyframe_task = spawn_keyframe_listener(
100        keyframe_rx,
101        view_rx.clone(),
102        self_id,
103        Source::Camera,
104        adapt.clone(),
105        Some(low_keyframe.clone()),
106    );
107    let feedback_task = spawn_feedback_listener(
108        feedback_rx,
109        view_rx.clone(),
110        self_id,
111        Source::Camera,
112        adapt.clone(),
113        layers.clone(),
114    );
115    // Demand-driven simulcast: aggregate receiver quality requests for the camera
116    // track into the layer planner (drives the low layer's `low_active` gate). Dark
117    // until `SIMULCAST_ACTUATE`.
118    let quality_task =
119        spawn_quality_listener(quality_rx, view_rx.clone(), self_id, Source::Camera, layers);
120    // LOW-layer publish: drains the capture thread's low encode and publishes it on
121    // the `cam_l0` track, announcing it on the first frame (so it's invisible to
122    // receivers until actually producing).
123    let low_publish_task = spawn_low_video_publish(
124        relay.clone(),
125        broadcast_path(space_id, self_id, "cam_l0"),
126        "cam_l0",
127        Source::Camera,
128        video_codec_to_ctl(codec),
129        low_vid_rx,
130        control_outbound.clone(),
131    );
132    // Re-negotiate the codec as the roster changes mid-call (a less-capable peer
133    // joins → downgrade so they get video; leaves → climb back).
134    let renegotiate_task =
135        spawn_codec_renegotiator(view_rx, encode_caps, adapt.clone(), codec, "cam");
136
137    let path = broadcast_path(space_id, self_id, CAM_TRACK);
138    call_runtime().spawn(async move {
139        // Abort the listeners when this task ends (clean return or `leave`-time abort).
140        let _keyframe_task = AbortOnDrop(keyframe_task);
141        let _feedback_task = AbortOnDrop(feedback_task);
142        let _quality_task = AbortOnDrop(quality_task);
143        let _low_publish_task = AbortOnDrop(low_publish_task);
144        let _renegotiate_task = AbortOnDrop(renegotiate_task);
145
146        let mut publisher = match relay.publish_track(&path, CAM_TRACK, VIDEO_PRIORITY) {
147            Ok(p) => p,
148            Err(e) => {
149                tracing::error!(%e, "cam: failed to publish track");
150                return;
151            }
152        };
153        // RTC-debug: the camera track is announced in the negotiated codec.
154        stats.set_video_enabled(true);
155        stats.set_video_codec(video_codec_to_ctl(codec).label());
156
157        // Sender-side adaptation core (sans-IO), seeded at the encoder's static launch
158        // bitrate. The 1 s tick folds backpressure + RTT + remote loss into a decision
159        // the capture thread reads off `adapt`; ticks run from the start, forwarding
160        // waits on the live gate (see `drive_video_publish`).
161        let mut adaptation = VideoAdaptation::standard(CAM_BITRATE.bps() as u64);
162        // The encoder-rebuild geometry dwell, seeded at the capture's own native
163        // geometry (CAM_DIMS) — exactly what the capture thread's encoder is already
164        // built at before any tick runs, so a healthy call at the top rung never
165        // rebuilds. See `hound_media::GeometryDwell`.
166        let mut geometry = hound_media::GeometryDwell::new((CAM_DIMS.width(), CAM_DIMS.height()));
167        drive_video_publish(
168            &mut publisher,
169            &mut enc_rx,
170            &adapt,
171            &mut adaptation,
172            &mut geometry,
173            &rtt_conn,
174            &stats,
175            false,
176            "cam",
177            |_| {},
178        )
179        .await;
180    })
181}

183/// Open the camera per `capture_source`: real hardware at `CAM_DIMS`@`CAM_FPS`,
184/// or (behind `synthetic`) the engine's camera-free test pattern. The one seam
185/// `cam_capture_encode_loop`'s (re)open routes through.
186fn open_cam(
187    choice: &DeviceChoice<CameraId>,
188    clock: CaptureClock,
189    #[cfg(feature = "synthetic")] capture_source: CaptureSource,
190) -> std::result::Result<hound_engine::api::Camera, hound_engine::error::CaptureError> {
191    #[cfg(feature = "synthetic")]
192    if capture_source != CaptureSource::Default {
193        let testcard = capture_source == CaptureSource::SyntheticTestcard;
194        return Engine::new().open_synthetic_camera(CAM_DIMS, CAM_FPS, clock, testcard);
195    }
196    Engine::new().open_camera(choice, CAM_DIMS, CAM_FPS, clock)
197}

199/// The camera capture+encode loop. Mirrors `crates/frb/src/camera.rs` for the
200/// self-preview push, and additionally H.264-encodes for publish. Honors the
201/// `enabled` flag (released device when off) and hot-swaps on `device_rx` change.
202/// Reads `adapt` each frame: retargets the encoder bitrate, downsamples to the rung
203/// fps, downscales to the rung geometry, serves a forced IDR, and skips encode/publish
204/// while paused (AudioOnly under severe congestion). Self-preview keeps running even
205/// while paused.
206#[allow(clippy::too_many_arguments)]
207fn cam_capture_encode_loop(
208    stop: Arc<AtomicBool>,
209    clock: CaptureClock,
210    mut device_rx: watch::Receiver<DeviceChoice<CameraId>>,
211    enabled: Arc<AtomicBool>,
212    sink: Arc<dyn VideoSink>,
213    enc_tx: tokio::sync::mpsc::Sender<WireFrame>,
214    sender: u16,
215    adapt: Arc<AdaptShared>,
216    stats: Arc<CallStats>,
217    // Demand-driven LOW simulcast layer: `Some(bps)` ⇒ also downscale + encode this
218    // capture at `LOW_CAM_DIMS` and forward on `low_vid_tx` (the `cam_l0` track).
219    // Stays `None` (inert, source path byte-identical) until a receiver demands it
220    // AND `SIMULCAST_ACTUATE` is on.
221    low_active: watch::Receiver<Option<u64>>,
222    low_vid_tx: tokio::sync::mpsc::Sender<WireFrame>,
223    low_keyframe: Arc<AtomicBool>,
224    #[cfg(feature = "synthetic")] capture_source: CaptureSource,
225) {
226    let mut choice = device_rx.borrow_and_update().clone();
227    let mut camera = None;
228    let mut encoder: Option<hound_engine::api::VideoEncoder> = None;
229    let mut frame_seq = 0u32;
230    // The negotiated codec target we last (re)built the encoder for; the mid-call
231    // rebuild keys off this changing, not off the encoder's actual codec, so a target
232    // the device can't build doesn't re-trigger every frame.
233    let mut last_target = codec_from_rank(adapt.target_codec.load(Ordering::Relaxed));
234    // The geometry the live `encoder` was actually built at — starts at CAM_DIMS
235    // (native capture) and only moves when the adaptation's dwelled rung geometry
236    // (`adapt.encoder_w/h`) settles on something different. Passed to
237    // `encode_and_publish_adaptive` so it always scales the frame to match whatever
238    // the encoder really is.
239    let mut enc_dims = CAM_DIMS;
240    // The encoder bitrate we last applied, so a steady target costs no syscall.
241    let mut applied_bps = CAM_BITRATE.bps();
242    // Downsamples capture (CAM_FPS) to the chosen rung's fps.
243    let mut fps_gate = hound_media::FrameRateGate::new();
244
245    // LOW simulcast layer state — mirrors the source's, at `LOW_CAM_DIMS`. Built
246    // lazily the first frame `low_active` is `Some`, torn down when it goes `None`,
247    // rebuilt on a codec switch (tracks the same `target_codec` as the source).
248    let mut low_encoder: Option<hound_engine::api::VideoEncoder> = None;
249    let mut low_last_target: Option<VideoCodec> = None;
250    let mut low_applied_bps = 0u32;
251    let mut low_frame_seq = 0u32;
252    let mut low_fps_gate = hound_media::FrameRateGate::new();
253    let low_adapt = Arc::new(AdaptShared::default());
254    low_adapt.target_fps.store(CAM_FPS.get(), Ordering::Relaxed);
255
256    while !stop.load(Ordering::Relaxed) {
257        // Camera disabled: release the device and idle.
258        if !enabled.load(Ordering::Relaxed) {
259            if camera.is_some() {
260                drop(camera.take());
261                drop(encoder.take());
262            }
263            std::thread::sleep(std::time::Duration::from_millis(100));
264            continue;
265        }
266
267        if device_rx.has_changed().unwrap_or(false) {
268            choice = device_rx.borrow_and_update().clone();
269            drop(camera.take());
270            drop(encoder.take());
271        }
272
273        // Open (or re-open) the camera + encoder.
274        if camera.is_none() {
275            match open_cam(
276                &choice,
277                clock,
278                #[cfg(feature = "synthetic")]
279                capture_source,
280            ) {
281                Ok(c) => camera = Some(c),
282                Err(_) => {
283                    std::thread::sleep(std::time::Duration::from_millis(300));
284                    continue;
285                }
286            }
287            // Build in the negotiated codec (HW-encode ∩ all-peers decode), seeded by
288            // `spawn_cam_publish` and re-driven by the mid-call re-negotiator,
289            // descending the ladder if it won't open.
290            let want = codec_from_rank(adapt.target_codec.load(Ordering::Relaxed));
291            encoder =
292                open_encoder_laddered(want, CAM_DIMS, CAM_FPS, CAM_BITRATE, CAM_KEYFRAME_CADENCE);
293            last_target = want;
294            if let Some(enc) = encoder.as_ref() {
295                use hound_engine::backend::VideoEncode;
296                tracing::info!(
297                    target: "video_encode",
298                    track = "cam",
299                    codec = ?enc.codec(),
300                    backend = enc.backend(),
301                    width = CAM_DIMS.width(),
302                    height = CAM_DIMS.height(),
303                    "video: encode path selected"
304                );
305            }
306            // Re-baseline so the first frame re-applies the live BWE target, and
307            // open at a fresh IDR for any subscriber that attached meanwhile. A
308            // (re)open always starts at native geometry — the geometry dwell only
309            // degrades it once a fresh adaptation decision settles.
310            applied_bps = CAM_BITRATE.bps();
311            enc_dims = CAM_DIMS;
312            adapt.force_keyframe.store(true, Ordering::Relaxed);
313        }
314
315        let Some(cap) = camera.as_mut() else {
316            continue;
317        };
318        let pulled = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| cap.next_frame()));
319        let frame = match pulled {
320            Ok(Ok(f)) => f,
321            _ => {
322                drop(camera.take());
323                drop(encoder.take());
324                std::thread::sleep(std::time::Duration::from_millis(200));
325                continue;
326            }
327        };
328
329        // RTC-debug: one raw camera frame pulled from the device.
330        stats.video_captured();
331
332        // Self-preview keyed "self" (runs even while paused, so the local preview
333        // never freezes mid-call). Surface-first: a surface-backed camera frame is
334        // rendered zero-copy; a scaled/I420 frame falls back to bytes. The capture
335        // geometry gauge is what the self tile sizes its texture from (stable,
336        // unlike the BWE-driven encode rung) — kept fresh per frame so a camera
337        // hot-swap with a different geometry follows automatically.
338        stats.set_video_capture_dims(frame.dims.width(), frame.dims.height());
339        push_preview(sink.as_ref(), SELF_KEY, &frame);
340
341        // AudioOnly: the ladder dropped video under severe congestion. Keep pulling the
342        // device (resume is instant) but skip encode + publish entirely.
343        if adapt.paused.load(Ordering::Relaxed) {
344            continue;
345        }
346
347        // Mid-call codec switch (a peer joined/left mid-call) AND/OR mid-call
348        // resolution-rung change (the adaptation tick's DWELLED geometry —
349        // `adapt.encoder_w/h`, see `hound_media::GeometryDwell` — settled on
350        // something different from what's built): rebuild the encoder to match
351        // either, at the live bitrate. `0` on the geometry atomics means "no
352        // adaptation decision yet" (before the first tick), so fall back to the
353        // current `enc_dims` rather than a bogus 0×0 rebuild. Either change forces
354        // a fresh IDR — a decoder can't switch codec OR resolution mid-GOP, so
355        // every receiver must resync.
356        let want_codec = codec_from_rank(adapt.target_codec.load(Ordering::Relaxed));
357        let (want_w, want_h) = (
358            adapt.encoder_w.load(Ordering::Relaxed),
359            adapt.encoder_h.load(Ordering::Relaxed),
360        );
361        let want_dims = if want_w > 0 && want_h > 0 {
362            Dimensions::even_clamped(want_w, want_h)
363        } else {
364            enc_dims
365        };
366        if encoder.is_some() && (want_codec != last_target || want_dims != enc_dims) {
367            last_target = want_codec;
368            if let Some(e) = open_encoder_laddered(
369                want_codec,
370                want_dims,
371                CAM_FPS,
372                Bitrate::from_bps(applied_bps),
373                CAM_KEYFRAME_CADENCE,
374            ) {
375                tracing::info!(
376                    track = "cam",
377                    codec = ?want_codec,
378                    from = %enc_dims,
379                    to = %want_dims,
380                    "call: rebuilding encoder mid-call (codec/geometry)"
381                );
382                encoder = Some(e);
383                enc_dims = want_dims;
384                adapt.force_keyframe.store(true, Ordering::Relaxed);
385            } else {
386                tracing::warn!(track = "cam", codec = ?want_codec, "call: mid-call encoder rebuild found no usable encoder; keeping current");
387            }
388        }
389
390        let Some(enc) = encoder.as_mut() else {
391            continue;
392        };
393        encode_and_publish_adaptive(
394            &frame,
395            enc,
396            &adapt,
397            &mut applied_bps,
398            &mut fps_gate,
399            enc_dims,
400            |wire| enc_tx.try_send(wire).is_ok(),
401            &mut frame_seq,
402            sender,
403            TrackKind::CamVideo,
404            &stats,
405        );
406
407        // LOW simulcast layer: when a constrained / small-tile receiver demands it,
408        // downscale this same capture frame and encode + forward it on the cam_l0
409        // track. Inert while `low_active` is `None` (dark until SIMULCAST_ACTUATE),
410        // so the source path above is byte-identical to a no-simulcast build.
411        match *low_active.borrow() {
412            Some(low_bps) => {
413                low_adapt.target_bps.store(low_bps, Ordering::Relaxed);
414                // A low-track PLI (new/recovering low subscriber) → force a low IDR.
415                if low_keyframe.swap(false, Ordering::Relaxed) {
416                    low_adapt.force_keyframe.store(true, Ordering::Relaxed);
417                }
418                // (Re)build the low encoder at LOW geometry, tracking the source codec.
419                let low_want = codec_from_rank(adapt.target_codec.load(Ordering::Relaxed));
420                if low_encoder.is_none() || low_last_target != Some(low_want) {
421                    low_last_target = Some(low_want);
422                    low_encoder = open_encoder_laddered(
423                        low_want,
424                        LOW_CAM_DIMS,
425                        CAM_FPS,
426                        Bitrate::from_bps(low_bps.min(u32::MAX as u64) as u32),
427                        CAM_KEYFRAME_CADENCE,
428                    );
429                    if low_encoder.is_some() {
430                        low_adapt.force_keyframe.store(true, Ordering::Relaxed);
431                    }
432                }
433                if let Some(low_enc) = low_encoder.as_mut()
434                    && let Some(low_frame) = frame.scaled_to(LOW_CAM_DIMS)
435                {
436                    encode_and_publish_adaptive(
437                        &low_frame,
438                        low_enc,
439                        &low_adapt,
440                        &mut low_applied_bps,
441                        &mut low_fps_gate,
442                        LOW_CAM_DIMS,
443                        |wire| low_vid_tx.try_send(wire).is_ok(),
444                        &mut low_frame_seq,
445                        sender,
446                        TrackKind::CamVideoLow,
447                        &stats,
448                    );
449                }
450            }
451            // Demand gone — release the low encoder (resumes instantly on re-demand).
452            None => {
453                low_encoder = None;
454            }
455        }
456    }
