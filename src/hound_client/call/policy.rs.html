1//! Pure, unit-tested call policy — the sans-IO decision layer. Inputs (sizes,
2//! instants, samples, roster caps) in, decisions out: codec negotiation + the
3//! mid-call [`CodecSwitch`], spatial-layer [`LayerHysteresis`], [`RestartPacer`],
4//! [`ResizeWatch`], mic route warmup / dead-silence detection, per-stream playback
5//! gain, and the audio gap/cadence math. Nothing here touches sockets, clocks, or

10/// The LOW simulcast layer's encode geometry, derived from the SOURCE so it **preserves
11/// aspect ratio** — a fixed 16:9 rung (e.g. `640×360`) would stretch any non-16:9
12/// share (a near-square window gets squished horizontally). Scale the
13/// source so its long edge is [`LOW_SCREEN_LONG_EDGE`], even-clamped (the encoder needs
14/// even extents): a 16:9 source still yields 640×360, a 1052×1080 window yields ~624×640,
15/// a portrait phone-mirror its own tall geometry — all at the source aspect, so
16/// `scaled_to` (which stretches to the exact target) downscales without distortion.
17pub(super) fn low_screen_dims(source: Dimensions) -> Dimensions {
18    let (w, h) = (source.width().max(1), source.height().max(1));
19    let long = w.max(h);
20    if long <= LOW_SCREEN_LONG_EDGE {
21        return Dimensions::even_clamped(w, h); // already small — don't upscale
22    }
23    let scale = LOW_SCREEN_LONG_EDGE as f64 / long as f64;
24    Dimensions::even_clamped(
25        (w as f64 * scale).round() as u32,
26        (h as f64 * scale).round() as u32,
27    )
28}

30/// How long the screen capture must be IDLE (no real frames delivered) before the
31/// loop starts probing whether the source still exists. Static content is
32/// legitimately idle, so probing starts only after a quiet spell.
33const SOURCE_LIVENESS_IDLE: std::time::Duration = std::time::Duration::from_secs(3);
34/// Minimum spacing between source-liveness probes (each is a blocking ~tens-of-ms
35/// shareable-content fetch on the otherwise-idle capture thread).
36const SOURCE_LIVENESS_INTERVAL: std::time::Duration = std::time::Duration::from_secs(2);
37
38/// Whether an idle capture loop should probe source liveness now. Pure — the
39/// throttle policy in one testable place: probe only after a real-frame drought of
40/// [`SOURCE_LIVENESS_IDLE`], at most every [`SOURCE_LIVENESS_INTERVAL`].
41pub(super) fn source_probe_due(idle_for: Duration, since_last_probe: Duration) -> bool {
42    idle_for >= SOURCE_LIVENESS_IDLE && since_last_probe >= SOURCE_LIVENESS_INTERVAL
43}

45/// Map the control-plane codec onto the wire codec byte. The two enums are kept in
46/// sync deliberately (see `hound_control::Codec` docs). Used by the publish path's
47/// codec-negotiation seam (and pinned by a unit test); kept here as the single
48/// source of truth even though the camera codec is currently hardcoded to H.264.
49#[allow(dead_code)]
50fn wire_codec(c: CtlCodec) -> WireCodec {
51    match c {
52        CtlCodec::Av1 => WireCodec::Av1,
53        CtlCodec::H264 => WireCodec::H264,
54        CtlCodec::H265 => WireCodec::H265,
55        CtlCodec::Opus => WireCodec::Opus,
56    }
57}

69/// Bitrate scale per codec for EQUAL perceptual quality, relative to H.264. A more
70/// efficient codec reaches the same quality at fewer bits, so we bank the efficiency
71/// as lower bandwidth — and the per-packet receive CPU that dominates a screenshare —
72/// instead of a sharper picture at the same bitrate. The BWE rung stays the quality
73/// target (an H.264-equivalent ceiling); this scales what the encoder actually emits.
74/// Conservative vs the codecs' headline gains (HEVC ~40-50%, AV1 ~50%+, more still on
75/// screen content) so quality never regresses.
76pub(super) fn codec_bitrate_factor(c: VideoCodec) -> f64 {
77    match c {
78        VideoCodec::H264 => 1.0,
79        VideoCodec::H265 => 0.65,
80        VideoCodec::Av1 => 0.5,
81    }
82}

84/// Pick the video codec to PUBLISH: the highest rung on the encode ladder
85/// (AV1 → H.265 → H.264, [`hound_engine::types::ENCODE_PRIORITY`]) this device can
86/// **hardware-encode** AND **every current remote peer can decode** (so the
87/// forward-no-transcode relay never hands a peer a codec it can't play), falling back
88/// to H.264. Read from the live roster: a less-capable peer shrinks
89/// `common_remote_decode_caps`, an all-modern call gets HEVC/AV1. `encode_caps` is
90/// probed once by the caller (the probe spins a throwaway VT session) and reused — at
91/// publish time and by the mid-call re-negotiator on every roster change.
92pub(super) fn negotiate_codec(
93    encode_caps: hound_engine::types::VideoCodecCaps,
94    view: &SessionView,
95) -> VideoCodec {
96    let common = view.common_remote_decode_caps();
97    let common_caps = hound_engine::types::VideoCodecCaps {
98        av1: common.av1,
99        h265: common.h265,
100        h264: common.h264,
101    };
102    hound_engine::types::pick_encode_codec(encode_caps, common_caps)
103}

105/// Encode a [`VideoCodec`] for the `AtomicU8` codec slot shared with the encode loop.
106/// The value doubles as a priority rank (higher = richer), so the re-negotiator can
107/// tell a downgrade from an upgrade with a plain comparison.
108pub(super) fn codec_rank(c: VideoCodec) -> u8 {
109    match c {
110        VideoCodec::Av1 => 2,
111        VideoCodec::H265 => 1,
112        VideoCodec::H264 => 0,
113    }
114}
115
116pub(super) fn codec_from_rank(v: u8) -> VideoCodec {
117    match v {
118        2 => VideoCodec::Av1,
119        1 => VideoCodec::H265,
120        _ => VideoCodec::H264,
121    }
122}

126pub(super) const CODEC_UPGRADE_DEBOUNCE: Duration = Duration::from_secs(5);

128/// Decides when a live encoder should switch codecs as the roster changes. Pure
129/// (no IO) so the policy is unit-tested. **Downgrades land immediately** — a peer
130/// that can't decode the current codec has no video until we drop to one it can —
131/// while **upgrades are debounced** (a peer leaving lets us climb, but there's no
132/// rush, and rushing invites thrash if they rejoin).
133pub(super) struct CodecSwitch {
134    current: VideoCodec,
135    /// A richer codec seen but not yet committed, and when it was first seen.
136    pending: Option<(VideoCodec, Instant)>,
137}
138
139impl CodecSwitch {
140    pub(super) fn new(initial: VideoCodec) -> Self {
141        Self {
142            current: initial,
143            pending: None,
144        }
145    }
146
147    /// Feed the freshly-negotiated codec for the current roster. Returns `Some(c)`
148    /// when the encoder should switch to `c` now, else `None` (hold).
149    pub(super) fn observe(
150        &mut self,
151        want: VideoCodec,
152        now: Instant,
153        debounce: Duration,
154    ) -> Option<VideoCodec> {
155        if want == self.current {
156            self.pending = None;
157            return None;
158        }
159        if codec_rank(want) < codec_rank(self.current) {
160            // Downgrade: commit immediately (correctness — a peer can't decode us).
161            self.pending = None;
162            self.current = want;
163            return Some(want);
164        }
165        // Upgrade: commit only once `want` has held for `debounce`.
166        match self.pending {
167            Some((p, since)) if p == want && now.duration_since(since) >= debounce => {
168                self.pending = None;
169                self.current = want;
170                Some(want)
171            }
172            Some((p, _)) if p == want => None,
173            _ => {
174                self.pending = Some((want, now));
175                None
176            }
177        }
178    }
179
180    /// When the encode loop should re-poll even without a roster change (to let a
181    /// debounced upgrade fire). `None` when nothing is pending.
182    pub(super) fn pending_deadline(&self, debounce: Duration) -> Option<Instant> {
183        self.pending.map(|(_, since)| since + debounce)
184    }
185}

187/// The resize watch's running state inside the capture loop, factored out PURE so the
188/// debounce classification is unit-testable without a live capture source. Tracks the
189/// baseline content size and a pending (changed-but-not-yet-held) candidate.
190#[derive(Default)]
191pub(super) struct ResizeWatch {
192    baseline: Option<(u32, u32)>,
193    pending: Option<((u32, u32), std::time::Instant)>,
194}
195
196impl ResizeWatch {
197    /// Feed one observed content size at `now`. Returns `true` when a changed size has
198    /// HELD for at least `debounce` (the loop should exit with [`ScreenCaptureExit::Resized`]).
199    /// The first size seen becomes the baseline; a size back at baseline clears any pending.
200    pub(super) fn observe(
201        &mut self,
202        cur: (u32, u32),
203        now: std::time::Instant,
204        debounce: std::time::Duration,
205    ) -> bool {
206        match self.baseline {
207            None => {
208                self.baseline = Some(cur);
209                false
210            }
211            Some(base) if cur != base => match self.pending {
212                Some((size, since)) if size == cur => now.duration_since(since) >= debounce,
213                _ => {
214                    self.pending = Some((cur, now));
215                    false
216                }
217            },
218            // Back at baseline — a transient blip that didn't hold; clear the candidate.
219            Some(_) => {
220                self.pending = None;
221                false
222            }
223        }
224    }
225}

227/// Per-remote-audio-stream playback gain, keyed `(peer, source)` — the tile
228/// right-click volume/mute control. Each audio pipeline holds its stream's gain
229/// CELL (an `AtomicU32` of `f32` bits, loaded lock-free per frame); the map only
230/// locks on the ~UI-rate [`Self::set`] / pipeline-start [`Self::cell`] paths.
231/// Lives on [`ConnectionParams`] so a volume survives reconnects and resubscribes
232/// (the cell is looked up by key, not by pipeline instance).
233#[derive(Default)]
234pub(super) struct StreamVolumes {
235    map: std::sync::Mutex<HashMap<(u32, Source), Arc<std::sync::atomic::AtomicU32>>>,
236}
237
238impl StreamVolumes {
239    /// Get (or create at unity gain) the gain cell for one stream.
240    pub(super) fn cell(&self, peer: u32, source: Source) -> Arc<std::sync::atomic::AtomicU32> {
241        self.map
242            .lock()
243            .unwrap_or_else(|e| e.into_inner())
244            .entry((peer, source))
245            .or_insert_with(|| Arc::new(std::sync::atomic::AtomicU32::new(1.0f32.to_bits())))
246            .clone()
247    }
248
249    /// Set one stream's playback gain. `0.0` = muted, `1.0` = unity; clamped to
250    /// `[0.0, 2.0]` (a modest boost ceiling — beyond that is clipping territory).
251    pub(super) fn set(&self, peer: u32, source: Source, volume: f32) {
252        let v = if volume.is_finite() {
253            volume.clamp(0.0, 2.0)
254        } else {
255            1.0
256        };
257        self.cell(peer, source)
258            .store(v.to_bits(), Ordering::Relaxed);
259    }
260}

262/// Scale decoded PCM by `gain`, saturating at the i16 rails (a >1 boost must clip
263/// gracefully, never wrap). Unity is the caller's fast path — call only when
264/// `gain != 1.0`. Pure; unit-tested.
265pub(super) fn apply_gain(samples: &mut [i16], gain: f32) {
266    for s in samples.iter_mut() {
267        *s = (*s as f32 * gain).clamp(i16::MIN as f32, i16::MAX as f32) as i16;
268    }
269}

271/// One-shot mic route warmup: a non-48 kHz capture device (a Bluetooth headset in
272/// HFP mode) may open onto an SCO route that is not carrying audio yet — the
273/// stream runs, delivering silence, and no error ever fires. Reopening once,
274/// [`hound_engine::api::ROUTE_WARMUP_DELAY`] after open, lands on the settled
275/// route; 48 kHz devices never arm. The playout side has the same policy inside
276/// the mix bus; this is the capture-side half.
277pub(super) struct MicRouteWarmup {
278    pub(super) due: Option<Instant>,
279}
280
281impl MicRouteWarmup {
282    /// Arm (or disarm) for a just-opened device at `device_rate`.
283    pub(super) fn arm(device_rate: u32, now: Instant) -> Self {
284        Self {
285            due: (device_rate != 48_000).then(|| now + hound_engine::api::ROUTE_WARMUP_DELAY),
286        }
287    }
288
289    /// True exactly once, when the warmup deadline passes.
290    pub(super) fn fire(&mut self, now: Instant) -> bool {
291        match self.due {
292            Some(t) if now >= t => {
293                self.due = None;
294                true
295            }
296            _ => false,
297        }
298    }
299}

301/// Consecutive exactly-zero mic frames before warning once that the input is
302/// delivering pure silence — a dead capture route (BT/SCO, revoked permission, a
303/// silent virtual device) produces bit-exact zeros, which a live room's noise
304/// floor never does. ~3 s at 20 ms frames.
305const MIC_SILENCE_WARN_FRAMES: u32 = 150;
306
307/// Detects a dead-silent capture stream (see [`MIC_SILENCE_WARN_FRAMES`]).
308/// [`Self::observe`] returns `true` exactly once per silent streak, at the
309/// threshold; any non-zero sample resets the streak (and re-arms the warning).
310#[derive(Default)]
311pub(super) struct MicSilenceWatch {
312    zero_frames: u32,
313    warned: bool,
314}
315
316impl MicSilenceWatch {
317    pub(super) fn observe(&mut self, samples: &[i16]) -> bool {
318        if samples.iter().all(|&s| s == 0) {
319            self.zero_frames = self.zero_frames.saturating_add(1);
320            if self.zero_frames >= MIC_SILENCE_WARN_FRAMES && !self.warned {
321                self.warned = true;
322                return true;
323            }
324        } else {
325            self.zero_frames = 0;
326            self.warned = false;
327        }
328        false
329    }
330}

332/// How long a tile's demanded spatial layer must HOLD before we switch its video
333/// subscription between the source and low tracks. A switch tears down + reopens a
334/// moq subscription on the connection **shared with audio**; doing it on every
335/// transient tile resize (the Flutter view relays out after a (re)join, firing a
336/// burst of sizes that flap source→low→source within a second) churns the
337/// connection's stream state and disrupts AUDIO delivery (an "audio is sacred"
338/// violation). Holding the demand until it is durable collapses that burst to at
339/// most one switch. ~1.5 s is below human patience for a focus-to-sharpen yet well
340/// above any relayout flap.
341pub(super) const LAYER_SWITCH_DEBOUNCE: Duration = Duration::from_millis(1500);
342
343/// Minimum dwell on a layer after an actuated switch before the NEXT switch may
344/// commit. The debounce alone only filters flaps shorter than itself; a demand
345/// that alternates slower than the debounce (a stage↔filmstrip focus toggle
346/// re-laying tiles out every couple of seconds) would pass it and thrash the
347/// subscription anyway. The dwell bounds actuated churn to one switch per
348/// window per stream regardless of the flap period; a held demand still
349/// actuates the moment the dwell expires.
350pub(super) const LAYER_SWITCH_DWELL: Duration = Duration::from_secs(6);
351
352/// Per-(peer, source) hysteresis for the spatial-layer subscription decision. Pure
353/// (no IO) so the policy is unit-tested in isolation. [`Self::observe`] takes the raw
354/// per-tile demand and returns the **committed** demand the subscribe loop should act
355/// on: a change that would switch the subscribed *track* (source ⇄ low) commits only
356/// once the new layer has held for `debounce` AND the previous switch is at least
357/// `dwell` old; a change that does NOT switch tracks (a different cap *value* while
358/// already on the low layer, or no change) passes through immediately. This is what
359/// makes a rejoin's relayout flap — or a slow focus-toggle alternation — a bounded
360/// trickle instead of a resubscribe storm on the shared connection.
361#[derive(Debug)]
362pub(super) struct LayerHysteresis {
363    /// The demand we've committed to (what the loop subscribes / reports upstream).
364    committed: hound_media::QualityRequest,
365    /// A demand on the OTHER layer axis, seen but not yet committed, and when first seen.
366    pending: Option<(hound_media::QualityRequest, Instant)>,
367    /// When the last track switch actuated (gates the next one by the dwell).
368    last_switch: Option<Instant>,
369}
370
371/// Whether a quality request selects the low spatial layer (a capped tile) vs the
372/// source track. The track-switch axis the hysteresis debounces.
373fn wants_low_layer(req: hound_media::QualityRequest) -> bool {
374    matches!(req, hound_media::QualityRequest::Capped(_))
375}
376
377impl LayerHysteresis {
378    pub(super) fn new() -> Self {
379        Self {
380            committed: hound_media::QualityRequest::Source,
381            pending: None,
382            last_switch: None,
383        }
384    }
385
386    /// Whether the post-switch dwell has elapsed (a first switch is always free).
387    fn dwell_over(&self, now: Instant, dwell: Duration) -> bool {
388        self.last_switch
389            .is_none_or(|t| now.duration_since(t) >= dwell)
390    }
391
392    /// Feed the freshly-computed per-tile demand; returns the demand to act on now.
393    pub(super) fn observe(
394        &mut self,
395        want: hound_media::QualityRequest,
396        now: Instant,
397        debounce: Duration,
398        dwell: Duration,
399    ) -> hound_media::QualityRequest {
400        // A change that stays on the same layer axis (no track switch) is free —
401        // commit it at once. This covers "no change" and a new cap *value* while
402        // already on the low layer (only the reported cap moves, not the subscription).
403        if wants_low_layer(want) == wants_low_layer(self.committed) {
404            self.committed = want;
405            self.pending = None;
406            return self.committed;
407        }
408        // A track switch (source ⇄ low): commit only once `want`'s layer has held for
409        // `debounce` and the previous switch is `dwell` old. A flap back to the
410        // committed layer clears the pending switch via the same-axis branch above.
411        match self.pending {
412            Some((p, since))
413                if wants_low_layer(p) == wants_low_layer(want)
414                    && now.duration_since(since) >= debounce
415                    && self.dwell_over(now, dwell) =>
416            {
417                self.committed = want;
418                self.pending = None;
419                self.last_switch = Some(now);
420            }
421            Some((p, since)) if wants_low_layer(p) == wants_low_layer(want) => {
422                // Still waiting; keep the latest target value but hold the timer.
423                self.pending = Some((want, since));
424            }
425            _ => self.pending = Some((want, now)),
426        }
427        self.committed
428    }
429
430    /// When the loop should re-evaluate even without an input change, to let a pending
431    /// switch actuate — the later of the debounce and the dwell expiring. `None` when
432    /// nothing is pending.
433    pub(super) fn pending_deadline(&self, debounce: Duration, dwell: Duration) -> Option<Instant> {
434        self.pending.map(|(_, since)| {
435            let due = since + debounce;
436            match self.last_switch {
437                Some(t) => due.max(t + dwell),
438                None => due,
439            }
440        })
441    }
442}

444/// Base restart pace for a reaped pipeline, and the first rung of the escalation
445/// ladder: an UNHEALTHY death (died fast, or never delivered a frame) waits this
446/// long, doubling per consecutive unhealthy death up to [`RESTART_PACE_MAX`].
447const RESTART_PACE: Duration = Duration::from_millis(1000);
448/// Ladder ceiling: a persistently-dead track (producer gone dark, relay routing
449/// wedged) keeps retrying forever — resubscribes are how it eventually heals —
450/// but never faster than this once the ladder tops out.
451const RESTART_PACE_MAX: Duration = Duration::from_secs(20);
452
453/// Pure restart-pacing policy for reaped pipelines (unit-tested). A HEALTHY death
454/// — the pipeline delivered frames and lived ≥ [`RESTART_PACE`] — restarts
455/// immediately and resets the track's ladder (a live stream hit a transient
456/// fault; recover fast). An UNHEALTHY death (instant kill, or a subscription
457/// that never delivered anything) climbs an exponential ladder, so a track the
458/// relay/producer keeps wedging retries forever at a decaying, bounded rate
459/// instead of churning the shared connection. [`Self::ready_at`] is what the
460/// reconcile consults.
461#[derive(Default)]
462pub(super) struct RestartPacer {
463    /// Per-track consecutive unhealthy strikes + the earliest allowed restart.
464    ladders: HashMap<u32, (u32, Instant)>,
465}
466
467impl RestartPacer {
468    /// Record a pipeline death. `delivered` = it produced at least one frame
469    /// downstream; `lived` is how long it ran; `now` is the death time.
470    pub(super) fn on_death(
471        &mut self,
472        track_id: u32,
473        delivered: bool,
474        lived: Duration,
475        now: Instant,
476    ) {
477        if delivered && lived >= RESTART_PACE {
478            self.ladders.remove(&track_id);
479            return;
480        }
481        let strikes = self
482            .ladders
483            .get(&track_id)
484            .map_or(0, |&(s, _)| s.saturating_add(1));
485        let pace = RESTART_PACE
486            .saturating_mul(1u32 << strikes.min(16))
487            .min(RESTART_PACE_MAX);
488        self.ladders.insert(track_id, (strikes, now + pace));
489    }
490
491    /// When `track_id` may restart: `None` = immediately; `Some(t)` = defer to `t`.
492    /// Strikes persist past an elapsed deadline — only a healthy run clears them —
493    /// so a track that dies undelivered on every retry keeps climbing the ladder.
494    pub(super) fn ready_at(&self, track_id: u32, now: Instant) -> Option<Instant> {
495        match self.ladders.get(&track_id) {
496            Some(&(_, t)) if t > now => Some(t),
497            _ => None,
498        }
499    }
500}

502/// Minimum spacing between the recovery keyframe requests a video pipeline sends
503/// outside the subscribe-time nag — on a mid-stream stall or after a
504/// backpressure drop broke decode continuity. The producer coalesces duplicates
505/// through its own throttle; this just keeps a long producer pause from turning
506/// into a control-plane drip.
507const RECOVERY_PLI_MIN_INTERVAL: Duration = Duration::from_secs(2);
508
509/// Rate-limits a pipeline's recovery keyframe requests ([`RECOVERY_PLI_MIN_INTERVAL`]).
510/// Pure; unit-tested.
511#[derive(Default)]
512pub(super) struct PliThrottle {
513    last: Option<Instant>,
514}
515
516impl PliThrottle {
517    /// True when a request may fire now (and records it as fired).
518    pub(super) fn fire(&mut self, now: Instant) -> bool {
519        if self
520            .last
521            .is_some_and(|t| now.duration_since(t) < RECOVERY_PLI_MIN_INTERVAL)
522        {
523            return false;
524        }
525        self.last = Some(now);
526        true
527    }
528}

530/// Max consecutive frames concealed with Opus PLC on a timestamp gap. A gap of up to
531/// this many frames (~100 ms at 20 ms/frame) is plausibly recoverable brief loss worth
532/// concealing; beyond that the gap is the sender being silent (DTX emits no frames) or
533/// a capture discontinuity — fabricating extrapolated audio there is wrong (it mislabels
534/// silence as loss and splices garbage into an active stream). Silence is played across
535/// the larger gap instead.
536pub(super) const MAX_PLC_FRAMES: i64 = 5;
537
538/// Frames genuinely missing in a `gap_us` jump between consecutive in-order frame
539/// timestamps (frame size `frame_us`). Tolerant of sub-frame capture-timing jitter:
540/// SCO/resampled sources never space frames at exactly 20 ms, so a gap under ~1.5
541/// frames is the next frame arriving a little late, NOT a loss. Counting it would
542/// fabricate one spurious PLC frame per real frame (concealment dwarfing real audio).
543/// Pure + unit-tested. The caller separately caps the result at [`MAX_PLC_FRAMES`].
544pub(super) fn frames_missing(gap_us: i64, frame_us: i64) -> i64 {
545    if frame_us <= 0 || gap_us <= 0 {
546        return 0;
547    }
548    let frames = gap_us as f64 / frame_us as f64;
549    if frames < 1.5 {
550        0
551    } else {
552        (frames.round() as i64 - 1).max(0)
553    }
554}

557const CADENCE_WINDOW_FRAMES: u64 = 250;
558
559/// Per-stream receive-cadence diagnostic (a retained debugging tool). Accumulates over a
560/// window and, on completion, reports the signals that explain an active-stream gap the
561/// smoothed jitter number hides — see the call site. Pure bookkeeping; no IO.
562#[derive(Default)]
563pub(super) struct CadenceDiag {
564    real: u64,
565    pub(super) plc: u64,
566    pub(super) big_gaps: u64,
567    max_gap_us: i64,
568    first_ts: Option<i64>,
569    first_now: i64,
570}
571
572impl CadenceDiag {
573    /// Fold a timestamp gap between consecutive in-order frames.
574    pub(super) fn note_gap(&mut self, gap_us: i64) {
575        if gap_us > self.max_gap_us {
576            self.max_gap_us = gap_us;
577        }
578    }
579
580    /// Record one decoded frame at sender timestamp `ts` / local time `now`. Returns
581    /// `Some((real, plc, big_gaps, max_gap_ms, skew_ppm))` once per window, then resets.
582    /// `skew_ppm` > 0 ⇒ the local clock advanced faster than the sender's timestamps
583    /// over the window (receiver consumes faster than sender produces → buffer drains).
584    pub(super) fn observe(&mut self, ts: i64, now: i64) -> Option<(u64, u64, u64, i64, i64)> {
585        if self.first_ts.is_none() {
586            self.first_ts = Some(ts);
587            self.first_now = now;
588        }
589        self.real += 1;
590        if self.real < CADENCE_WINDOW_FRAMES {
591            return None;
592        }
593        let ts_elapsed = ts - self.first_ts.unwrap_or(ts);
594        let now_elapsed = now - self.first_now;
595        let skew_ppm = if ts_elapsed > 0 {
596            ((now_elapsed - ts_elapsed) as f64 / ts_elapsed as f64 * 1_000_000.0) as i64
597        } else {
598            0
599        };
600        let out = (
601            self.real,
602            self.plc,
603            self.big_gaps,
604            self.max_gap_us / 1000,
605            skew_ppm,
606        );
607        *self = CadenceDiag::default();
608        Some(out)
609    }
610}

848        p.on_death(7, true, Duration::from_secs(30), now);

1149            committed = next;
1150        }
1151        assert!(
1152            switches <= 24_000 / DWELL.as_millis() as u64 + 1,
1153            "alternation must actuate at most ~once per dwell, got {switches}"
1154        );
1155        assert!(switches >= 1, "a genuinely held demand still actuates");
1156    }

1158    // ─── Mid-call re-negotiation harness ────────────────────────────────────────
1159    //
1160    // Drives the FULL chain — roster (`SessionView`) → `common_remote_decode_caps`
1161    // → `negotiate_codec` (gated by local HW-encode caps) → `CodecSwitch` — over
1162    // simulated peer join/leave sequences, asserting the two reliability invariants
1163    // a live mid-stream codec switch must hold: bounded recovery (a peer that can't
1164    // decode us forces a decodable codec immediately) and no thrash (a flapping peer
1165    // never storms the encoder). The actual VideoToolbox rebuild is hardware and out
1166    // of scope here; this pins the decision layer that drives it.

1176    /// A fresh roster: self (`id` 1, full caps) plus the given remote peers.
1177    fn roster(remotes: &[(u32, DecodeCaps)]) -> SessionView {
1178        let mut peers = vec![PeerInfo {
1179            id: PeerId(1),
1180            decode_caps: DecodeCaps::ALL,
1181        }];
1182        peers.extend(remotes.iter().map(|&(id, c)| PeerInfo {
1183            id: PeerId(id),
1184            decode_caps: c,
1185        }));
1186        let mut v = SessionView::new();
1187        v.apply(&ServerMessage::Welcome {
1188            protocol_version: PROTOCOL_VERSION,
1189            self_id: PeerId(1),
1190            peers,
1191            tracks: vec![],
1192        });
1193        v
1194    }

1266    #[test]
1267    fn renegotiate_does_not_storm_when_peer_flaps() {
1268        let enc = full_encode();
1269        let mut view = roster(&[(2, DecodeCaps::ALL)]);
1270        let mut sw = CodecSwitch::new(negotiate_codec(enc, &view)); // AV1
1271        let t0 = Instant::now();
1272
1273        // An H.264-only peer joins/leaves every 1 s for 12 s — far faster than the
1274        // 5 s upgrade debounce. Downgrades are immediate (correctness); upgrades must
1275        // never commit mid-flap, so the encoder switches exactly ONCE (the first
1276        // downgrade) rather than thrashing every second.
1277        let mut switches = 0;
1278        for i in 0..12u64 {
1279            if i % 2 == 0 {
1280                view.apply(&ServerMessage::PeerJoined {
1281                    peer: PeerInfo {
1282                        id: PeerId(3),
1283                        decode_caps: dcaps(false, false, true),
1284                    },
1285                });
1286            } else {
1287                view.apply(&ServerMessage::PeerLeft { peer: PeerId(3) });
1288            }
1289            let now = t0 + Duration::from_secs(i);
1290            if sw.observe(negotiate_codec(enc, &view), now, DB).is_some() {
1291                switches += 1;
1292            }
1293        }
1294        assert_eq!(switches, 1, "a flapping peer must not thrash the encoder");
1295    }

1307/// The screen-share resize-debounce classification, tested PURE — no live
1308/// capture source. Pins exactly when a window-resize HOLDS long enough to be a
1309/// `Resized` exit (vs a transient drag-resize blip) and that returning to baseline
1310/// cancels the candidate.
1311#[cfg(test)]
1312mod resize_watch_tests {
1313    use super::*;
1314    use std::time::{Duration, Instant};
1315
1316    const DEBOUNCE: Duration = SCREEN_RESIZE_DEBOUNCE;
1317
1318    /// A changed size that holds for the full debounce classifies as `Resized`; the
1319    /// first observed size only sets the baseline (never fires).
1320    #[test]
1321    fn changed_size_held_past_debounce_fires() {
1322        let mut w = ResizeWatch::default();
1323        let t0 = Instant::now();
1324        // First observation = baseline, never fires.
1325        assert!(!w.observe((1920, 1080), t0, DEBOUNCE));
1326        // A new size, but not yet held → pending, no fire.
1327        assert!(!w.observe((1280, 720), t0, DEBOUNCE));
1328        // Same new size just before the debounce elapses → still no fire.
1329        assert!(!w.observe(
1330            (1280, 720),
1331            t0 + DEBOUNCE - Duration::from_millis(1),
1332            DEBOUNCE
1333        ));
1334        // Same new size, now past the debounce → fire.
1335        assert!(w.observe((1280, 720), t0 + DEBOUNCE, DEBOUNCE));
1336    }

1338    /// A live drag-resize that bounces back to the baseline before the debounce does
1339    /// NOT fire — only a settled departure restarts the share.
1340    #[test]
1341    fn returning_to_baseline_cancels_the_candidate() {
1342        let mut w = ResizeWatch::default();
1343        let t0 = Instant::now();
1344        assert!(!w.observe((1920, 1080), t0, DEBOUNCE));
1345        // Mid-drag intermediate size.
1346        assert!(!w.observe((1600, 900), t0 + Duration::from_millis(100), DEBOUNCE));
1347        // Snaps back to baseline → pending cleared.
1348        assert!(!w.observe((1920, 1080), t0 + Duration::from_millis(200), DEBOUNCE));
1349        // Even well past the debounce, baseline never fires.
1350        assert!(!w.observe((1920, 1080), t0 + DEBOUNCE * 2, DEBOUNCE));
1351    }

1353    /// A drag that keeps changing size resets the debounce each time (the timer is
1354    /// per-candidate-size), so a continuously-dragging resize only fires once it settles.
