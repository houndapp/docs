1//! Inbound video: the async pull half (receive-side keyframe gate; the nag task
2//! lives with the subscribe loop) feeds a blocking decode thread that owns the
3//! `!Send` `VideoDecoder` and rebuilds it whenever the sender's per-packet codec
4//! byte changes — the sender's choice drives decode. Decoded frames reach the
5//! sink surface-first (zero-copy GPU handle) with materialized I420 as the
6//! universal fallback; [`push_preview`] is the borrow-based capture-side twin
7//! the publish loops use for self-preview.
8
9use super::*;
10
11/// Push one CAPTURED frame to the render sink for a self-preview tile, preferring
12/// the ZERO-COPY surface path: a surface-backed frame (SCK / AVFoundation NV12
13/// IOSurface) is handed to the renderer to sample directly — no NV12→I420 CPU
14/// conversion, no ~1.4–3 MB per-frame alloc+copy, no re-upload (~90 MB/s of
15/// conversion at a 1080p 30 fps share). Falls back to materialized I420
16/// for non-surface frames (scaled camera frames, other
17/// platforms) or a sink without a surface renderer. Borrows the frame — the
18/// caller still owns it for encode. (The REMOTE decode path keeps its own
19/// ownership-optimized version of this pattern: it consumes the frame, so its
20/// no-surface case can MOVE the bytes instead of copying.)
21pub(super) fn push_preview(sink: &dyn VideoSink, key: &str, frame: &hound_engine::types::RawFrame) {
22    let zero_copy =
23        RenderSurfaceHandle::for_frame(frame).is_some_and(|handle| sink.push_surface(key, handle));
24    if !zero_copy && let Some(i420) = frame.as_i420() {
25        sink.push_frame(key, i420.width(), i420.height(), i420.into_owned().data);
26    }
27}

29/// How long the video pull tolerates delivering NOTHING before treating the
30/// subscription as stalled. Above the static-share re-emit cadence
31/// ([`SCREEN_STATIC_REFRESH`], 1 s) so a healthy static share never trips it; a
32/// mid-GOP loss stall or a silent producer does within seconds instead of a
33/// retransmit tail.
34const VIDEO_STALL: Duration = Duration::from_secs(3);
35
36/// How long a fresh video subscription may sit with NO keyframe before the
37/// pipeline ends itself so the reconcile re-subscribes (restart-ladder paced).
38/// Past the nag window (5 × 2 s): a producer that ignored every re-request is
39/// wedged or its broadcast is a zombie from a same-path replacement, and only a
40/// fresh subscribe — a new announce cursor + control-plane `Subscribe` — heals
41/// those; waiting on the 60 s safety-net IDR is a frozen tile.
42const FIRST_KEYFRAME_DEADLINE: Duration = Duration::from_secs(12);
43
44/// Video decode pipeline. The MoQ subscriber pull is async (must run on the
45/// runtime), but the engine `VideoDecoder` (VideoToolbox/dav1d) is `!Send`, so it
46/// CANNOT live across an `.await`. The async half therefore only PULLS frames —
47/// applying the receive-side keyframe gate so we never feed a delta before the
48/// first keyframe — and forwards them to a dedicated blocking decode thread that
49/// owns the `!Send` decoder and pushes decoded I420 to the sink keyed by peer.
50///
51/// The pull is stall-bounded ([`VIDEO_STALL`]): a stuck group is abandoned and a
52/// throttled `RequestKeyframe` sent, so the producer rolls a fresh group and the
53/// tile resumes at live content instead of freezing behind a retransmit tail and
54/// then fast-forwarding. A subscription that never yields its first keyframe
55/// ends itself at [`FIRST_KEYFRAME_DEADLINE`] (the reap → paced-resubscribe path).
56#[allow(clippy::too_many_arguments)]
57pub(super) async fn video_pipeline(
58    mut subscriber: MoqTrackSubscriber,
59    peer: PeerId,
60    source: Source,
61    sink: Arc<dyn VideoSink>,
62    stop: Arc<AtomicBool>,
63    peer_stats: Arc<crate::stats::PeerStats>,
64    got_keyframe: Arc<AtomicBool>,
65    control_outbound: tokio::sync::mpsc::UnboundedSender<ClientMessage>,
66    track_id: u32,
67) {
68    use hound_transport::VideoPull;
69
70    let key = match source {
71        Source::Screen => screen_key(peer),
72        _ => cam_key(peer),
73    };
74    // Camera vs screen feed SEPARATE inbound-video stat buckets (so a peer sharing
75    // both does not report a summed fps/bitrate).
76    let vkind = inbound_video_kind(source);
77
78    // Drop-on-full channel from the async pull half to the blocking decode thread.
79    let (frame_tx, frame_rx) = std::sync::mpsc::sync_channel::<WireFrame>(8);
80    let decode_stop = stop.clone();
81    let decode_stats = peer_stats.clone();
82    let decode_thread = std::thread::Builder::new()
83        .name(format!("call-video-decode:{}", peer.0))
84        .spawn(move || video_decode_loop(frame_rx, key, vkind, sink, decode_stop, decode_stats))
85        .ok();
86
87    let started = Instant::now();
88    let mut pli = PliThrottle::default();
89
90    let mut awaiting_keyframe = true;
91    while !stop.load(Ordering::Relaxed) {
92        let frame = match subscriber.next_frame_stalled(VIDEO_STALL).await {
93            Ok(VideoPull::Frame(f)) => f,
94            Ok(VideoPull::Stalled) => {
95                if !got_keyframe.load(Ordering::Relaxed)
96                    && started.elapsed() >= FIRST_KEYFRAME_DEADLINE
97                {
98                    tracing::warn!(
99                        peer = peer.0,
100                        track = track_id,
101                        "video: no keyframe within the deadline — ending pipeline for a fresh subscribe"
102                    );
103                    break;
104                }
105                // Whatever the abandoned group still owed is gone — re-latch the
106                // gate and ask the producer to roll a fresh, decodable group.
107                awaiting_keyframe = true;
108                if pli.fire(Instant::now()) {
109                    let _ = control_outbound.send(ClientMessage::RequestKeyframe {
110                        track: hound_control::TrackId(track_id),
111                    });
112                    peer_stats.video_keyframe_requested(vkind);
113                }
114                continue;
115            }
116            Ok(VideoPull::End) | Err(_) => break,
117        };
118        // RTC-debug: one inbound encoded AU received for this peer's camera/screen video.
119        peer_stats.video_received(vkind, frame.data.len());
120        // Keyframe gate: drop deltas until the first keyframe so the decode thread
121        // never decodes against a missing reference. Passing the gate also stops
122        // the awaiting-keyframe re-request nag (see `start_pipeline`).
123        if awaiting_keyframe {
124            if !frame.keyframe {
125                continue;
126            }
127            awaiting_keyframe = false;
128            got_keyframe.store(true, Ordering::Relaxed);
129        }
130        // Backpressure → drop. A dropped frame breaks decode continuity for the
131        // rest of its GOP, so re-latch the gate (cheap delta skips instead of
132        // corrupt decodes) and request a keyframe to resync.
133        if frame_tx.try_send(frame).is_err() {
134            // If the decode thread is gone, end the pull.
135            if decode_thread.as_ref().is_none_or(|h| h.is_finished()) {
136                break;
137            }
138            awaiting_keyframe = true;
139            if pli.fire(Instant::now()) {
140                let _ = control_outbound.send(ClientMessage::RequestKeyframe {
141                    track: hound_control::TrackId(track_id),
142                });
143                peer_stats.video_keyframe_requested(vkind);
144            }
145        }
146    }
147    drop(frame_tx); // ends the decode loop
148    if let Some(h) = decode_thread {
149        let _ = h.join();
150    }
151}

153/// The blocking video decode loop: owns the `!Send` `VideoDecoder`, rebuilds it
154/// when the sender's codec changes (the sender's choice drives decode), and pushes
155/// each decoded I420 to the sink.
156fn video_decode_loop(
157    frame_rx: std::sync::mpsc::Receiver<WireFrame>,
158    key: String,
159    vkind: InboundVideoKind,
160    sink: Arc<dyn VideoSink>,
161    stop: Arc<AtomicBool>,
162    peer_stats: Arc<crate::stats::PeerStats>,
163) {
164    use hound_engine::types::VideoCodec;
165
166    let mut decoder: Option<VideoDecoder> = None;
167    let mut decoder_codec: Option<WireCodec> = None;
168    // The current decode path + codec (captured at build) and a ~5 s throughput
169    // window, so the log shows decoded geometry + ACTUAL decode fps + mean decode
170    // cost — a crawling software decode is visible at a glance.
171    let mut current_backend: &'static str = "?";
172    let mut current_codec = VideoCodec::H264;
173    let mut win_start: Option<std::time::Instant> = None;
174    let mut win_frames: u64 = 0;
175    let mut win_decode_us: u64 = 0;
176
177    while let Ok(frame) = frame_rx.recv() {
178        if stop.load(Ordering::Relaxed) {
179            break;
180        }
181        // (Re)build the decoder if the sender's codec changed.
182        if decoder_codec != Some(frame.codec) {
183            let vc = match frame.codec {
184                WireCodec::H264 => VideoCodec::H264,
185                WireCodec::H265 => VideoCodec::H265,
186                WireCodec::Av1 => VideoCodec::Av1,
187                WireCodec::Opus => continue, // not a video codec; ignore
188            };
189            match Engine::new().video_decoder(vc) {
190                Ok(d) => {
191                    // Surface the actual decode path + codec. A software fallback
192                    // (e.g. AV1 with no HW block) is CPU-heavy and can starve the
193                    // audio fill path, so make HW-vs-SW visible in the call log.
194                    current_backend = d.backend();
195                    current_codec = vc;
196                    win_start = None;
197                    win_frames = 0;
198                    win_decode_us = 0;
199                    tracing::info!(
200                        target: "video_decode",
201                        key = %key,
202                        codec = ?vc,
203                        backend = current_backend,
204                        "video: decode path selected"
205                    );
206                    decoder = Some(d);
207                    decoder_codec = Some(frame.codec);
208                    // Surface the live decode codec in the RTC-debug panel (also
209                    // tracks a mid-stream sender codec switch).
210                    peer_stats.set_inbound_video_codec(vkind, video_codec_to_ctl(vc).label());
211                }
212                Err(e) => {
213                    tracing::warn!(%e, "video: failed to build decoder");
214                    continue;
215                }
216            }
217        }
218        let Some(dec) = decoder.as_mut() else {
219            continue;
220        };
221        let decode_start = std::time::Instant::now();
222        match dec.decode(&frame.data) {
223            Ok(Some(raw)) => {
224                let decode_us = decode_start.elapsed().as_micros().min(i64::MAX as u128) as i64;
225                let (w, h) = (raw.width(), raw.height());
226                // Every decoder backend emits an OWNED I420 `RawFrame` — each `decode`
227                // call allocates a fresh conversion buffer (vt_decoder / vaapi / sw all
228                // output I420, asserted in their tests). So MOVE that buffer straight
229                // into the sink: the decoder's single conversion alloc is the only one
230                // on this path; cloning here (borrow-then-clone of an already-I420
231                // frame) would add a redundant ~1.4 MB copy PER FRAME PER PEER. A
232                // non-I420 frame (defensive) converts via `as_i420`, itself Owned ⇒
233                // also a move, never a clone.
234                // Zero-copy fast path: if the decoder kept the frame GPU-resident
235                // (VideoToolbox IOSurface; MFT D3D11 texture) and the sink can
236                // render a surface, hand it over directly — no NV12→I420
237                // readback, no re-upload.
238                let zero_copy = RenderSurfaceHandle::for_frame(&raw)
239                    .is_some_and(|handle| sink.push_surface(&key, handle));
240
241                if !zero_copy {
242                    // Fallback: materialize tightly-packed I420 and push bytes (software
243                    // decoders, no-surface frames, or a sink without a surface renderer).
244                    let data = if raw.surface.is_none()
245                        && raw.format == hound_engine::types::PixelFormat::I420
246                    {
247                        raw.data
248                    } else {
249                        match raw.as_i420() {
250                            Some(c) => c.into_owned().data,
251                            None => continue,
252                        }
253                    };
254                    sink.push_frame(&key, w, h, data);
255                }
256                // RTC-debug: one decoded frame (geometry + decode-time tail).
257                peer_stats.video_decoded(vkind, w, h, decode_us);
258
259                // ~5 s decode-throughput sample: decoded geometry + ACTUAL fps +
260                // mean decode cost on the current path (HW/SW visible above).
261                let now = std::time::Instant::now();
262                let start = *win_start.get_or_insert(now);
263                win_frames += 1;
264                win_decode_us += decode_us.max(0) as u64;
265                let elapsed = now.duration_since(start);
266                if elapsed >= std::time::Duration::from_secs(5) {
267                    let secs = elapsed.as_secs_f64().max(0.001);
268                    let fps = win_frames as f64 / secs;
269                    let mean_ms = (win_decode_us as f64 / win_frames as f64) / 1000.0;
270                    tracing::info!(
271                        target: "video_decode",
272                        key = %key,
273                        codec = ?current_codec,
274                        backend = current_backend,
275                        width = w,
276                        height = h,
277                        fps = format!("{fps:.1}"),
278                        decode_ms = format!("{mean_ms:.1}"),
279                        "video: decode throughput"
280                    );
281                    win_start = Some(now);
282                    win_frames = 0;
283                    win_decode_us = 0;
284                }
285            }
286            Ok(None) => {} // decoder buffering; keep feeding
287            Err(_) => {
288                // Decode error: rebuild on the next codec change / keyframe.
289                decoder = None;
290                decoder_codec = None;
291            }
292        }
293    }
