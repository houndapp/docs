1//! Inbound video: the async pull half (receive-side keyframe gate; the nag task
2//! lives with the subscribe loop) feeds a blocking decode thread that owns the
3//! `!Send` `VideoDecoder` and rebuilds it whenever the sender's per-packet codec
4//! byte changes — the sender's choice drives decode. Decoded frames reach the
5//! sink surface-first (zero-copy GPU handle) with materialized I420 as the
6//! universal fallback; [`push_preview`] is the borrow-based capture-side twin
7//! the publish loops use for self-preview.
8
9use super::*;
10
11/// Push one CAPTURED frame to the render sink for a self-preview tile, preferring
12/// the ZERO-COPY surface path: a surface-backed frame (SCK / AVFoundation NV12
13/// IOSurface) is handed to the renderer to sample directly — no NV12→I420 CPU
14/// conversion, no ~1.4–3 MB per-frame alloc+copy, no re-upload (~90 MB/s of
15/// conversion at a 1080p 30 fps share). Falls back to materialized I420
16/// for non-surface frames (scaled camera frames, other
17/// platforms) or a sink without a surface renderer. Borrows the frame — the
18/// caller still owns it for encode. (The REMOTE decode path keeps its own
19/// ownership-optimized version of this pattern: it consumes the frame, so its
20/// no-surface case can MOVE the bytes instead of copying.)
21pub(super) fn push_preview(sink: &dyn VideoSink, key: &str, frame: &hound_engine::types::RawFrame) {
22    let zero_copy =
23        RenderSurfaceHandle::for_frame(frame).is_some_and(|handle| sink.push_surface(key, handle));
24    if !zero_copy && let Some(i420) = frame.as_i420() {
25        sink.push_frame(key, i420.width(), i420.height(), i420.into_owned().data);
26    }
27}

29/// How long the video pull tolerates delivering NOTHING before treating the
30/// subscription as stalled. Above the static-share re-emit cadence
31/// ([`SCREEN_STATIC_REFRESH`], 1 s) so a healthy static share never trips it; a
32/// mid-GOP loss stall or a silent producer does within seconds instead of a
33/// retransmit tail.
34const VIDEO_STALL: Duration = Duration::from_secs(3);
35
36/// How long a fresh video subscription may sit with NO keyframe before the
37/// pipeline ends itself so the reconcile re-subscribes (restart-ladder paced).
38/// Past the nag window (5 × 2 s): a producer that ignored every re-request is
39/// wedged or its broadcast is a zombie from a same-path replacement, and only a
40/// fresh subscribe — a new announce cursor + control-plane `Subscribe` — heals
41/// those; waiting on the 60 s safety-net IDR is a frozen tile.
42const FIRST_KEYFRAME_DEADLINE: Duration = Duration::from_secs(12);
43
44/// Video decode pipeline. The MoQ subscriber pull is async (must run on the
45/// runtime), but the engine `VideoDecoder` (VideoToolbox/dav1d) is `!Send`, so it
46/// CANNOT live across an `.await`. The async half therefore only PULLS frames —
47/// applying the receive-side keyframe gate so we never feed a delta before the
48/// first keyframe — and forwards them to a dedicated blocking decode thread that
49/// owns the `!Send` decoder and pushes decoded I420 to the sink keyed by peer.
50///
51/// The pull is stall-bounded ([`VIDEO_STALL`]): a stuck group is abandoned and a
52/// throttled `RequestKeyframe` sent, so the producer rolls a fresh group and the
53/// tile resumes at live content instead of freezing behind a retransmit tail and
54/// then fast-forwarding. A subscription that never yields its first keyframe
55/// ends itself at [`FIRST_KEYFRAME_DEADLINE`] (the reap → paced-resubscribe path).
56#[allow(clippy::too_many_arguments)]
57pub(super) async fn video_pipeline(
58    mut subscriber: MoqTrackSubscriber,
59    peer: PeerId,
60    source: Source,
61    sink: Arc<dyn VideoSink>,
62    stop: Arc<AtomicBool>,
63    peer_stats: Arc<crate::stats::PeerStats>,
64    got_keyframe: Arc<AtomicBool>,
65    control_outbound: tokio::sync::mpsc::UnboundedSender<ClientMessage>,
66    track_id: u32,
67    downlink_cell: Arc<DownlinkCell>,
68    source_bps: u64,
69) {
70    use hound_transport::VideoPull;
71
72    let key = match source {
73        Source::Screen => screen_key(peer),
74        _ => cam_key(peer),
75    };
76    // Camera vs screen feed SEPARATE inbound-video stat buckets (so a peer sharing
77    // both does not report a summed fps/bitrate).
78    let vkind = inbound_video_kind(source);
79
80    // Drop-on-full channel from the async pull half to the blocking decode thread.
81    let (frame_tx, frame_rx) = std::sync::mpsc::sync_channel::<WireFrame>(8);
82    let decode_stop = stop.clone();
83    let decode_stats = peer_stats.clone();
84    let decode_thread = std::thread::Builder::new()
85        .name(format!("call-video-decode:{}", peer.0))
86        .spawn(move || video_decode_loop(frame_rx, key, vkind, sink, decode_stop, decode_stats))
87        .ok();
88
89    let started = Instant::now();
90    let mut pli = PliThrottle::default();
91    // The bandwidth half of demand-driven simulcast: folds each arriving frame's
92    // wire `frame_seq` into a windowed delivery ratio against `source_bps`, with
93    // the source/4 · source/2 hysteresis (see `hound_media::downlink`'s docs). A
94    // fresh instance per pipeline — the reset-on-restart contract falls out of
95    // that for free, matching the fresh `downlink_cell` this pipeline was given.
96    let mut downlink_demand =
97        hound_media::DownlinkDemand::new(hound_media::DownlinkConfig::default());
98
99    let mut awaiting_keyframe = true;
100    while !stop.load(Ordering::Relaxed) {
101        let frame = match subscriber.next_frame_stalled(VIDEO_STALL).await {
102            Ok(VideoPull::Frame(f)) => f,
103            Ok(VideoPull::Stalled) => {
104                if !got_keyframe.load(Ordering::Relaxed)
105                    && started.elapsed() >= FIRST_KEYFRAME_DEADLINE
106                {
107                    tracing::warn!(
108                        peer = peer.0,
109                        track = track_id,
110                        "video: no keyframe within the deadline — ending pipeline for a fresh subscribe"
111                    );
112                    break;
113                }
114                // Whatever the abandoned group still owed is gone — re-latch the
115                // gate and ask the producer to roll a fresh, decodable group.
116                awaiting_keyframe = true;
117                if pli.fire(Instant::now()) {
118                    let _ = control_outbound.send(ClientMessage::RequestKeyframe {
119                        track: hound_control::TrackId(track_id),
120                    });
121                    peer_stats.video_keyframe_requested(vkind);
122                }
123                continue;
124            }
125            Ok(VideoPull::End) | Err(_) => break,
126        };
127        // RTC-debug: one inbound encoded AU received for this peer's camera/screen video.
128        peer_stats.video_received(vkind, frame.data.len());
129        // Downlink demand: fold this arrival's wire `frame_seq` into the windowed
130        // delivery-ratio decider, independent of the keyframe gate below — a frame
131        // that arrives but gets dropped pre-decode (awaiting a keyframe) still
132        // reached us over the wire, so it is NOT downlink loss. `now_us` mirrors
133        // the elapsed-since-`started` convention the decode-throughput log below
134        // already uses for this same pipeline.
135        let now_us = started.elapsed().as_micros().min(i64::MAX as u128) as i64;
136        let req = downlink_demand.observe_frame(frame.frame_seq, now_us, source_bps);
137        downlink_cell.store(req);
138        // Keyframe gate: drop deltas until the first keyframe so the decode thread
139        // never decodes against a missing reference. Passing the gate also stops
140        // the awaiting-keyframe re-request nag (see `start_pipeline`).
141        if awaiting_keyframe {
142            if !frame.keyframe {
143                continue;
144            }
145            awaiting_keyframe = false;
146            got_keyframe.store(true, Ordering::Relaxed);
147        }
148        // Backpressure → drop. A dropped frame breaks decode continuity for the
149        // rest of its GOP, so re-latch the gate (cheap delta skips instead of
150        // corrupt decodes) and request a keyframe to resync.
151        if frame_tx.try_send(frame).is_err() {
152            // If the decode thread is gone, end the pull.
153            if decode_thread.as_ref().is_none_or(|h| h.is_finished()) {
154                break;
155            }
156            awaiting_keyframe = true;
157            if pli.fire(Instant::now()) {
158                let _ = control_outbound.send(ClientMessage::RequestKeyframe {
159                    track: hound_control::TrackId(track_id),
160                });
161                peer_stats.video_keyframe_requested(vkind);
162            }
163        }
164    }
165    drop(frame_tx); // ends the decode loop
166    if let Some(h) = decode_thread {
167        let _ = h.join();
168    }
169}

171/// The blocking video decode loop: owns the `!Send` `VideoDecoder`, rebuilds it
172/// when the sender's codec changes (the sender's choice drives decode), and pushes
173/// each decoded I420 to the sink.
174fn video_decode_loop(
175    frame_rx: std::sync::mpsc::Receiver<WireFrame>,
176    key: String,
177    vkind: InboundVideoKind,
178    sink: Arc<dyn VideoSink>,
179    stop: Arc<AtomicBool>,
180    peer_stats: Arc<crate::stats::PeerStats>,
181) {
182    use hound_engine::types::VideoCodec;
183
184    let mut decoder: Option<VideoDecoder> = None;
185    let mut decoder_codec: Option<WireCodec> = None;
186    // The current decode path + codec (captured at build) and a ~5 s throughput
187    // window, so the log shows decoded geometry + ACTUAL decode fps + mean decode
188    // cost — a crawling software decode is visible at a glance.
189    let mut current_backend: &'static str = "?";
190    let mut current_codec = VideoCodec::H264;
191    let mut win_start: Option<std::time::Instant> = None;
192    let mut win_frames: u64 = 0;
193    let mut win_decode_us: u64 = 0;
194
195    while let Ok(frame) = frame_rx.recv() {
196        if stop.load(Ordering::Relaxed) {
197            break;
198        }
199        // (Re)build the decoder if the sender's codec changed.
200        if decoder_codec != Some(frame.codec) {
201            let vc = match frame.codec {
202                WireCodec::H264 => VideoCodec::H264,
203                WireCodec::H265 => VideoCodec::H265,
204                WireCodec::Av1 => VideoCodec::Av1,
205                WireCodec::Opus => continue, // not a video codec; ignore
206            };
207            match Engine::new().video_decoder(vc) {
208                Ok(d) => {
209                    // Surface the actual decode path + codec. A software fallback
210                    // (e.g. AV1 with no HW block) is CPU-heavy and can starve the
211                    // audio fill path, so make HW-vs-SW visible in the call log.
212                    current_backend = d.backend();
213                    current_codec = vc;
214                    win_start = None;
215                    win_frames = 0;
216                    win_decode_us = 0;
217                    tracing::info!(
218                        target: "video_decode",
219                        key = %key,
220                        codec = ?vc,
221                        backend = current_backend,
222                        "video: decode path selected"
223                    );
224                    decoder = Some(d);
225                    decoder_codec = Some(frame.codec);
226                    // Surface the live decode codec in the RTC-debug panel (also
227                    // tracks a mid-stream sender codec switch).
228                    peer_stats.set_inbound_video_codec(vkind, video_codec_to_ctl(vc).label());
229                }
230                Err(e) => {
231                    tracing::warn!(%e, "video: failed to build decoder");
232                    continue;
233                }
234            }
235        }
236        let Some(dec) = decoder.as_mut() else {
237            continue;
238        };
239        let decode_start = std::time::Instant::now();
240        match dec.decode(&frame.data) {
241            Ok(Some(raw)) => {
242                let decode_us = decode_start.elapsed().as_micros().min(i64::MAX as u128) as i64;
243                let (w, h) = (raw.width(), raw.height());
244                // Every decoder backend emits an OWNED I420 `RawFrame` — each `decode`
245                // call allocates a fresh conversion buffer (vt_decoder / vaapi / sw all
246                // output I420, asserted in their tests). So MOVE that buffer straight
247                // into the sink: the decoder's single conversion alloc is the only one
248                // on this path; cloning here (borrow-then-clone of an already-I420
249                // frame) would add a redundant ~1.4 MB copy PER FRAME PER PEER. A
250                // non-I420 frame (defensive) converts via `as_i420`, itself Owned ⇒
251                // also a move, never a clone.
252                // Zero-copy fast path: if the decoder kept the frame GPU-resident
253                // (VideoToolbox IOSurface; MFT D3D11 texture) and the sink can
254                // render a surface, hand it over directly — no NV12→I420
255                // readback, no re-upload.
256                let zero_copy = RenderSurfaceHandle::for_frame(&raw)
257                    .is_some_and(|handle| sink.push_surface(&key, handle));
258
259                if !zero_copy {
260                    // Fallback: materialize tightly-packed I420 and push bytes (software
261                    // decoders, no-surface frames, or a sink without a surface renderer).
262                    let data = if raw.surface.is_none()
263                        && raw.format == hound_engine::types::PixelFormat::I420
264                    {
265                        raw.data
266                    } else {
267                        match raw.as_i420() {
268                            Some(c) => c.into_owned().data,
269                            None => continue,
270                        }
271                    };
272                    sink.push_frame(&key, w, h, data);
273                }
274                // RTC-debug: one decoded frame (geometry + decode-time tail).
275                peer_stats.video_decoded(vkind, w, h, decode_us);
276
277                // ~5 s decode-throughput sample: decoded geometry + ACTUAL fps +
278                // mean decode cost on the current path (HW/SW visible above).
279                let now = std::time::Instant::now();
280                let start = *win_start.get_or_insert(now);
281                win_frames += 1;
282                win_decode_us += decode_us.max(0) as u64;
283                let elapsed = now.duration_since(start);
284                if elapsed >= std::time::Duration::from_secs(5) {
285                    let secs = elapsed.as_secs_f64().max(0.001);
286                    let fps = win_frames as f64 / secs;
287                    let mean_ms = (win_decode_us as f64 / win_frames as f64) / 1000.0;
288                    tracing::info!(
289                        target: "video_decode",
290                        key = %key,
291                        codec = ?current_codec,
292                        backend = current_backend,
293                        width = w,
294                        height = h,
295                        fps = format!("{fps:.1}"),
296                        decode_ms = format!("{mean_ms:.1}"),
297                        "video: decode throughput"
298                    );
299                    win_start = Some(now);
300                    win_frames = 0;
301                    win_decode_us = 0;
302                }
303            }
304            Ok(None) => {} // decoder buffering; keep feeding
305            Err(_) => {
306                // Decode error: rebuild on the next codec change / keyframe.
307                decoder = None;
308                decoder_codec = None;
309            }
310        }
311    }
