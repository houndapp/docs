1//! The subscribe loop: reconcile remote tracks → decode → sink / mix bus. Diffs
2//! the `SessionView` roster into per-track pipelines ([`start_pipeline`]), picks
3//! source vs low layer per tile demand (debounced through the policy hysteresis),
4//! and self-heals: every pipeline carries a [`PipelineEndGuard`] whose Drop
5//! reports the death so the reconcile reaps + re-subscribes it (restart-paced).
6//! Invariant: audio subscribes with arrival-order drain and the smaller moq
7//! priority value — audio must strictly outrank video on the shared connection.
8
9use super::*;
10
11/// Lock-free single-slot handle for one video pipeline's downlink-driven demand:
12/// the video pull loop ([`video_pipeline`]) writes its latest
13/// [`hound_media::QualityRequest`] here as frames arrive (via
14/// [`hound_media::DownlinkDemand`]); the reconcile loop below reads it every tick
15/// and folds it with the tile demand via [`hound_media::most_constrained`]. A
16/// plain atomic (not a lock) so the per-frame write on the pull loop never
17/// blocks. `None` (audio, or a pipeline that hasn't decided yet) reads as
18/// [`hound_media::QualityRequest::Source`] — the optimistic default, matching
19/// [`hound_media::DownlinkDemand::new`].
20#[derive(Debug)]
21pub(super) struct DownlinkCell(AtomicU64);
22
23/// Sentinel encoding `QualityRequest::Source` in the packed atomic — a `Capped`
24/// bitrate this large never occurs (an encode rung tops out in the low millions).
25const SOURCE_SENTINEL: u64 = u64::MAX;
26
27impl DownlinkCell {
28    fn new() -> Self {
29        Self(AtomicU64::new(SOURCE_SENTINEL))
30    }
31
32    pub(super) fn store(&self, req: hound_media::QualityRequest) {
33        let v = match req {
34            hound_media::QualityRequest::Source => SOURCE_SENTINEL,
35            hound_media::QualityRequest::Capped(bps) => bps.min(SOURCE_SENTINEL - 1),
36        };
37        self.0.store(v, Ordering::Relaxed);
38    }
39
40    fn load(&self) -> hound_media::QualityRequest {
41        match self.0.load(Ordering::Relaxed) {
42            SOURCE_SENTINEL => hound_media::QualityRequest::Source,
43            bps => hound_media::QualityRequest::Capped(bps),
44        }
45    }
46}

48/// One active remote subscription: the decode task + a stop flag.
49struct Subscription {
50    stop: Arc<AtomicBool>,
51    task: JoinHandle<()>,
52    /// Generation stamp: a reap signal carries the epoch of the pipeline that died,
53    /// so a LATE signal (a stopped pipeline's guard firing after the loop already
54    /// re-subscribed the same track id) can never remove the healthy replacement.
55    epoch: u64,
56    /// When this pipeline started — a death soon after start is paced (see
57    /// [`RestartPacer`]) so a track the relay instantly rejects can't hot-loop.
58    started: Instant,
59    /// Whether the pipeline delivered anything downstream (a decoded audio frame /
60    /// a keyframe past the video gate). A death without delivery climbs the
61    /// restart ladder — the subscription itself is suspect, not just the moment.
62    delivered: Arc<AtomicBool>,
63    /// This pipeline's downlink-driven demand (video only — `None` for audio). A
64    /// FRESH cell per pipeline start, so a resubscribe / layer switch / reap
65    /// naturally resets the demand to `Source` (see [`DownlinkCell`]'s docs).
66    downlink: Option<Arc<DownlinkCell>>,
67}
68
69impl Subscription {
70    fn stop(self) {
71        self.stop.store(true, Ordering::Relaxed);
72        self.task.abort();
73    }
74}

76/// Fires `(track_id, epoch)` on the reap channel when the owning pipeline task ends
77/// — by ANY path: clean end-of-track, a transport error, a panic unwind, or an
78/// abort (tokio drops the future at an await point, running destructors). This is
79/// what lets the subscribe loop REAP a self-terminated pipeline: without it, a
80/// dead track would stay in `active` forever, every reconcile would skip it as
81/// "already subscribed", and its tile would freeze permanently.
82struct PipelineEndGuard {
83    track_id: u32,
84    epoch: u64,
85    tx: tokio::sync::mpsc::UnboundedSender<(u32, u64)>,
86}
87
88impl Drop for PipelineEndGuard {
89    fn drop(&mut self) {
90        let _ = self.tx.send((self.track_id, self.epoch));
91    }
92}

94/// Spawn the subscribe loop: watch the `SessionView`, and for every remote track
95/// start a per-track pipeline (subscribe → decode → sink for video, mix bus for
96/// audio); stop pipelines for tracks that vanish. Sends `Subscribe`/keyframe
97/// intent on the control plane as tracks appear.
98#[allow(clippy::too_many_arguments)]
99pub(super) fn spawn_subscribe_loop(
100    mut view_rx: watch::Receiver<SessionView>,
101    self_id: PeerId,
102    space_id: Uuid,
103    relay: Arc<MoqRelaySession>,
104    control_outbound: tokio::sync::mpsc::UnboundedSender<ClientMessage>,
105    sink: Arc<dyn VideoSink>,
106    playout_bus: Arc<PlayoutBus>,
107    levels: watch::Sender<LevelsView>,
108    clock: CaptureClock,
109    stats: Arc<CallStats>,
110    mut video_paused: watch::Receiver<bool>,
111    mut tile_quality: watch::Receiver<HashMap<(PeerId, Source), hound_media::QualityRequest>>,
112    volumes: Arc<StreamVolumes>,
113) -> JoinHandle<()> {
114    call_runtime().spawn(async move {
115        // Keyed by TrackId.0 so we can diff against the view on each change.
116        let mut active: HashMap<u32, Subscription> = HashMap::new();
117        // The cap we last reported per (peer, source), so a RequestQuality is sent
118        // only when our demand for that peer's video actually changes.
119        let mut last_sent: HashMap<(PeerId, Source), Option<u64>> = HashMap::new();
120        // Per-(peer, source) hysteresis so a transient tile-resize flap (notably a
121        // rejoin relayout) can't thrash the source⇄low subscription on the shared
122        // audio+video connection. Holds a demanded layer switch until it is durable.
123        let mut hysteresis: HashMap<(PeerId, Source), LayerHysteresis> = HashMap::new();
124        // Self-healing: every pipeline task carries a PipelineEndGuard that reports
125        // its (track_id, epoch) here on ANY exit, so a self-terminated pipeline is
126        // reaped from `active` and the reconcile re-subscribes it (paced).
127        let (ended_tx, mut ended_rx) = tokio::sync::mpsc::unbounded_channel::<(u32, u64)>();
128        let mut next_epoch: u64 = 0;
129        let mut pacer = RestartPacer::default();
130
131        loop {
132            // Soonest pending debounced layer switch — the loop wakes to actuate it
133            // even if no input changes in the meantime.
134            let mut next_switch_deadline: Option<Instant> = None;
135            // When video is paused (app backgrounded / call surface hidden) drop
136            // every inbound video sub — receiving a high-bitrate screenshare nobody
137            // is watching is wasted CPU + bandwidth. Audio always flows.
138            let drop_video = *video_paused.borrow();
139            let demand = tile_quality.borrow().clone();
140            // Snapshot the remote tracks we should be subscribed to. Audio is taken
141            // as-is; for each (peer, video source) we pick EITHER the source or the
142            // low track by our demand, and report that demand upstream so the producer
143            // spins up the low layer.
144            let desired: Vec<(u32, PeerId, MediaKind, Source, bool, Uuid)> = {
145                let view = view_rx.borrow();
146                let mut out: Vec<(u32, PeerId, MediaKind, Source, bool, Uuid)> = Vec::new();
147                for t in view.tracks().filter(|t| {
148                    t.peer != self_id
149                        && t.kind == MediaKind::Audio
150                        && matches!(t.source, Source::Microphone | Source::Screen)
151                }) {
152                    // The broadcast path is built from the peer's AUTHENTICATED
153                    // user_id (control-plane roster), never the realtime-derived
154                    // identity map. An old backend that hasn't stamped it yet
155                    // sends the nil default — skip rather than subscribe under a
156                    // wrong (or nobody's) path.
157                    let Some(user_id) =
158                        view.peer(t.peer).map(|p| p.user_id).filter(|u| !u.is_nil())
159                    else {
160                        tracing::warn!(
161                            peer = t.peer.0,
162                            track = t.id.0,
163                            "subscribe: remote peer has no known user_id (old backend?) — skipping"
164                        );
165                        continue;
166                    };
167                    out.push((t.id.0, t.peer, t.kind, t.source, false, user_id));
168                }
169                if !drop_video {
170                    // Group each peer's video into (source track, low track).
171                    let mut groups: HashMap<(PeerId, Source), (Option<u32>, Option<u32>)> =
172                        HashMap::new();
173                    for t in view.tracks().filter(|t| {
174                        t.peer != self_id
175                            && t.kind == MediaKind::Video
176                            && matches!(t.source, Source::Camera | Source::Screen)
177                    }) {
178                        let e = groups.entry((t.peer, t.source)).or_insert((None, None));
179                        if t.low {
180                            e.1 = Some(t.id.0);
181                        } else {
182                            e.0 = Some(t.id.0);
183                        }
184                    }
185                    for (&(peer, source), &(src_id, low_id)) in &groups {
186                        let Some(user_id) =
187                            view.peer(peer).map(|p| p.user_id).filter(|u| !u.is_nil())
188                        else {
189                            tracing::warn!(
190                                peer = peer.0,
191                                ?source,
192                                "subscribe: remote peer has no known user_id (old backend?) — \
193                                 skipping video"
194                            );
195                            continue;
196                        };
197                        let tile_req = demand
198                            .get(&(peer, source))
199                            .copied()
200                            .unwrap_or(hound_media::QualityRequest::Source);
201                        // This receiver's OWN downlink-driven demand for this video,
202                        // read from whichever track (source or low) is CURRENTLY
203                        // subscribed — the pull loop for that pipeline folds each
204                        // arriving frame's `frame_seq` into a `DownlinkDemand` and
205                        // writes the result here (see `DownlinkCell`). Not yet
206                        // subscribed / audio-only / no data yet all read as `Source`
207                        // (optimistic default), so a fresh pipeline never spuriously
208                        // caps.
209                        let bwe_req = [src_id, low_id]
210                            .into_iter()
211                            .flatten()
212                            .find_map(|id| active.get(&id))
213                            .and_then(|s| s.downlink.as_ref())
214                            .map(|c| c.load())
215                            .unwrap_or(hound_media::QualityRequest::Source);
216                        // The EFFECTIVE demand is the more constrained of the tile-size
217                        // demand and this receiver's own bandwidth demand — neither
218                        // signal alone should ever request MORE than the other allows.
219                        let raw_req = hound_media::most_constrained(tile_req, bwe_req);
220                        // Debounce the source⇄low decision so a transient tile-resize
221                        // flap (a rejoin relayout) can't thrash the subscription on the
222                        // shared audio+video connection — only a DURABLE layer change
223                        // actuates. A cap-value change on the same layer passes through.
224                        let now = Instant::now();
225                        let req = hysteresis
226                            .entry((peer, source))
227                            .or_insert_with(LayerHysteresis::new)
228                            .observe(raw_req, now, LAYER_SWITCH_DEBOUNCE, LAYER_SWITCH_DWELL);
229                        if let Some(deadline) = hysteresis.get(&(peer, source)).and_then(|h| {
230                            h.pending_deadline(LAYER_SWITCH_DEBOUNCE, LAYER_SWITCH_DWELL)
231                        }) {
232                            next_switch_deadline = Some(
233                                next_switch_deadline.map_or(deadline, |d: Instant| d.min(deadline)),
234                            );
235                        }
236                        // Report our cap upstream (against the SOURCE track) on change,
237                        // so the producer activates / tears down the shared low layer.
238                        if let Some(sid) = src_id {
239                            let cap_bps = match req {
240                                hound_media::QualityRequest::Capped(b) => Some(b),
241                                hound_media::QualityRequest::Source => None,
242                            };
243                            if last_sent.get(&(peer, source)).copied() != Some(cap_bps) {
244                                last_sent.insert((peer, source), cap_bps);
245                                tracing::info!(
246                                    peer = peer.0,
247                                    ?source,
248                                    ?cap_bps,
249                                    src_track = sid,
250                                    low_present = low_id.is_some(),
251                                    "simulcast[recv]: RequestQuality → producer"
252                                );
253                                let _ = control_outbound.send(ClientMessage::RequestQuality {
254                                    track: hound_control::TrackId(sid),
255                                    cap_bps,
256                                });
257                            }
258                        }
259                        // Subscribe the low track when we want it AND it exists (so a
260                        // demand for low before the producer publishes it falls back to
261                        // source until the low track appears, then switches).
262                        let pick = match req {
263                            hound_media::QualityRequest::Capped(_) => low_id.or(src_id),
264                            hound_media::QualityRequest::Source => src_id.or(low_id),
265                        };
266                        if let Some(id) = pick {
267                            out.push((
268                                id,
269                                peer,
270                                MediaKind::Video,
271                                source,
272                                Some(id) == low_id,
273                                user_id,
274                            ));
275                        }
276                    }
277                }
278                out
279            };
280
281            // Stop pipelines for tracks no longer present (or video we just paused).
282            let desired_ids: std::collections::HashSet<u32> =
283                desired.iter().map(|(id, ..)| *id).collect();
284            let gone: Vec<u32> = active
285                .keys()
286                .copied()
287                .filter(|id| !desired_ids.contains(id))
288                .collect();
289            for id in gone {
290                if let Some(sub) = active.remove(&id) {
291                    sub.stop();
292                    // Tell the relay to stop forwarding this track to us (dropping the
293                    // subscriber closes the moq sub; this also clears server routing).
294                    let _ = control_outbound.send(ClientMessage::Unsubscribe {
295                        track: hound_control::TrackId(id),
296                    });
297                }
298            }
299
300            // Start pipelines for new tracks.
301            for (track_id, peer, kind, source, low, user_id) in desired {
302                if active.contains_key(&track_id) {
303                    continue;
304                }
305                // A recently-died pipeline restarts PACED (bounded retry): defer this
306                // track and wake when its pace elapses.
307                if let Some(at) = pacer.ready_at(track_id, Instant::now()) {
308                    next_switch_deadline =
309                        Some(next_switch_deadline.map_or(at, |d: Instant| d.min(at)));
310                    continue;
311                }
312                // Tell the server we want this track (+ a keyframe to bootstrap video).
313                let _ = control_outbound.send(ClientMessage::Subscribe {
314                    track: hound_control::TrackId(track_id),
315                });
316                if kind == MediaKind::Video {
317                    tracing::info!(
318                        peer = peer.0,
319                        ?source,
320                        low,
321                        track = track_id,
322                        "simulcast[recv]: subscribing video track ({})",
323                        if low { "LOW" } else { "source" }
324                    );
325                    let _ = control_outbound.send(ClientMessage::RequestKeyframe {
326                        track: hound_control::TrackId(track_id),
327                    });
328                    // RTC-debug: a PLI we sent for this peer's camera vs screen track.
329                    stats
330                        .peer(peer.0)
331                        .video_keyframe_requested(inbound_video_kind(source));
332                }
333                let epoch = next_epoch;
334                next_epoch += 1;
335                let sub = start_pipeline(
336                    peer,
337                    user_id,
338                    kind,
339                    source,
340                    low,
341                    space_id,
342                    relay.clone(),
343                    sink.clone(),
344                    playout_bus.clone(),
345                    levels.clone(),
346                    clock,
347                    stats.clone(),
348                    control_outbound.clone(),
349                    track_id,
350                    volumes.clone(),
351                    PipelineEndGuard {
352                        track_id,
353                        epoch,
354                        tx: ended_tx.clone(),
355                    },
356                );
357                active.insert(track_id, sub);
358            }
359
360            // Re-reconcile when the roster changes, the pause flag flips, the tile
361            // demand changes, OR a debounced layer switch comes due (so a held switch
362            // actuates even if nothing else moves).
363            let switch_wait =
364                next_switch_deadline.map(|d| d.saturating_duration_since(Instant::now()));
365            tokio::select! {
366                r = view_rx.changed() => {
367                    if r.is_err() { break; }
368                }
369                r = video_paused.changed() => {
370                    if r.is_err() { break; }
371                }
372                r = tile_quality.changed() => {
373                    if r.is_err() { break; }
374                }
375                // A pipeline ended on its own (end-of-track / transport error): REAP
376                // it so the reconcile re-subscribes — leaving it in `active` would
377                // freeze that tile permanently (every reconcile would skip it as
378                // "already subscribed"). Epoch-guarded: a stale signal from a
379                // pipeline the loop itself stopped (and maybe already replaced)
380                // must not touch the healthy replacement.
381                Some((track_id, epoch)) = ended_rx.recv() => {
382                    if active.get(&track_id).is_some_and(|s| s.epoch == epoch) {
383                        let sub = active.remove(&track_id).expect("checked present");
384                        let delivered = sub.delivered.load(Ordering::Relaxed);
385                        pacer.on_death(track_id, delivered, sub.started.elapsed(), Instant::now());
386                        tracing::warn!(
387                            track = track_id,
388                            lived_ms = sub.started.elapsed().as_millis() as u64,
389                            delivered,
390                            "call: pipeline ended on its own — reaping + re-subscribing"
391                        );
392                    }
393                }
394                _ = async {
395                    match switch_wait {
396                        Some(d) => tokio::time::sleep(d).await,
397                        None => std::future::pending::<()>().await,
398                    }
399                } => {}
400            }
401        }
402
403        for (_, sub) in active.drain() {
404            sub.stop();
405        }
406    })
407}

409/// How often the receiver re-sends `RequestKeyframe` while a fresh video
410/// subscription is still awaiting its first keyframe, and how many times. The
411/// subscribe-time request can RACE the producer's PLI-forced IDR (the IDR lands in
412/// a group the moq live-join skips), leaving the receiver black indefinitely on a
413/// keyframe that never comes. With the screen encoder
414/// on an on-demand cadence (no frequent periodic IDR to paper over the race), this
415/// bounded re-request is the fast recovery: producer-side the keyframe throttle
416/// coalesces duplicates, so nagging is cheap. Once the attempts are exhausted the
417/// pipeline's [`FIRST_KEYFRAME_DEADLINE`] ends it — a fresh, ladder-paced
418/// resubscribe replaces a black tile that would otherwise wait on the safety-net IDR.
419const KEYFRAME_NAG_INTERVAL: Duration = Duration::from_secs(2);
420const KEYFRAME_NAG_ATTEMPTS: u32 = 5;
421
422/// Start one per-track decode pipeline. Audio → Opus decode → mix bus; video →
423/// H.264/etc decode → sink keyed `cam:{peer}` / `screen:{peer}`. For video,
424/// `control_outbound` + `track_id` drive the awaiting-keyframe re-request nag.
425#[allow(clippy::too_many_arguments)]
426fn start_pipeline(
427    peer: PeerId,
428    user_id: Uuid,
429    kind: MediaKind,
430    source: Source,
431    low: bool,
432    space_id: Uuid,
433    relay: Arc<MoqRelaySession>,
434    sink: Arc<dyn VideoSink>,
435    playout_bus: Arc<PlayoutBus>,
436    levels: watch::Sender<LevelsView>,
437    clock: CaptureClock,
438    stats: Arc<CallStats>,
439    control_outbound: tokio::sync::mpsc::UnboundedSender<ClientMessage>,
440    track_id: u32,
441    volumes: Arc<StreamVolumes>,
442    end_guard: PipelineEndGuard,
443) -> Subscription {
444    let epoch = end_guard.epoch;
445    let stop = Arc::new(AtomicBool::new(false));
446    let task_stop = stop.clone();
447    // First-delivery latch: audio sets it on the first decoded frame, video when
448    // the first keyframe passes the gate. Read at reap time to pick the restart
449    // pace (an undelivered death climbs the ladder).
450    let delivered = Arc::new(AtomicBool::new(false));
451    let task_delivered = delivered.clone();
452    // This pipeline's downlink-driven demand (video only). A FRESH cell every
453    // call — the whole reset-on-restart contract for `DownlinkDemand` is "a new
454    // pipeline gets a new cell", so a resubscribe / layer switch / reap can never
455    // carry a stale capped demand forward.
456    let downlink = (kind == MediaKind::Video).then(|| Arc::new(DownlinkCell::new()));
457    let task_downlink = downlink.clone();
458    // The per-peer inbound stats handle (cloned once; the pipeline increments on it
459    // lock-free per frame).
460    let peer_stats = stats.peer(peer.0);
461    // The moq track name carries the spatial layer: a constrained receiver subscribes
462    // to the `_l0` low track instead of the source.
463    let track_name = match (kind, source, low) {
464        (MediaKind::Audio, Source::Screen, _) => SCREEN_AUDIO_TRACK,
465        (MediaKind::Audio, _, _) => MIC_TRACK,
466        (MediaKind::Video, Source::Screen, false) => SCREEN_TRACK,
467        (MediaKind::Video, Source::Screen, true) => "screen_l0",
468        (MediaKind::Video, _, false) => CAM_TRACK,
469        (MediaKind::Video, _, true) => "cam_l0",
470    };
471    let path = broadcast_path(space_id, user_id, peer, track_name);
472
473    // Audio decode + fill runs on the RT-promoted audio runtime so video CPU can't
474    // starve it; video stays on the general call runtime.
475    let runtime = match kind {
476        MediaKind::Audio => audio_runtime(),
477        MediaKind::Video => call_runtime(),
478    };
479    let task = runtime.spawn(async move {
480        // Owned by the task so its Drop reports this pipeline's end to the reap
481        // channel on ANY exit path (clean end, error, panic, abort).
482        let _end_guard = end_guard;
483        let subscriber = match relay
484            .subscribe_track(
485                &path,
486                track_name,
487                peer.0 as u16,
488                priority_for(kind),
489                // Audio drains in arrival order (a reordered Opus frame is still playable);
490                // video priority-drops stale GOPs. This is what stops a screenshare from
491                // silently dropping reordered mic/app-audio frames on the shared connection.
492                kind == MediaKind::Audio,
493            )
494            .await
495        {
496            Ok(s) => s,
497            Err(e) => {
498                tracing::warn!(%e, peer = peer.0, "subscribe failed");
499                return;
500            }
501        };
502
503        match kind {
504            MediaKind::Audio => {
505                // This stream's playback-gain cell (the tile volume/mute control),
506                // looked up by key so a set volume survives resubscribes.
507                let gain = volumes.cell(peer.0, source);
508                audio_pipeline(
509                    subscriber,
510                    peer,
511                    source,
512                    playout_bus,
513                    levels,
514                    clock,
515                    task_stop,
516                    peer_stats,
517                    gain,
518                    task_delivered,
519                )
520                .await
521            }
522            MediaKind::Video => {
523                // Awaiting-keyframe re-request nag: a separate small task so the
524                // fast re-requests don't race the pipeline's own pull. Ends on
525                // first keyframe or pipeline stop; after the bounded attempts the
526                // pipeline's first-keyframe deadline takes over (it ends itself,
527                // and the reconcile re-subscribes on the restart ladder).
528                let nag = {
529                    let got = task_delivered.clone();
530                    let stop = task_stop.clone();
531                    let outbound = control_outbound.clone();
532                    call_runtime().spawn(async move {
533                        for _ in 0..KEYFRAME_NAG_ATTEMPTS {
534                            tokio::time::sleep(KEYFRAME_NAG_INTERVAL).await;
535                            if got.load(Ordering::Relaxed) || stop.load(Ordering::Relaxed) {
536                                return;
537                            }
538                            tracing::info!(
539                                peer = peer.0,
540                                track = track_id,
541                                "video: still awaiting first keyframe — re-requesting"
542                            );
543                            let _ = outbound.send(ClientMessage::RequestKeyframe {
544                                track: hound_control::TrackId(track_id),
545                            });
546                        }
547                    })
548                };
549                video_pipeline(
550                    subscriber,
551                    peer,
552                    source,
553                    sink,
554                    task_stop,
555                    peer_stats,
556                    task_delivered,
557                    control_outbound,
558                    track_id,
559                    // `.expect`: this arm is MediaKind::Video, where `downlink` is
560                    // always `Some` (see its construction above).
561                    task_downlink.expect("video pipeline always has a downlink cell"),
562                    source_top_bps(source),
563                )
564                .await;
565                nag.abort();
566            }
567        }
568    });
569
570    Subscription {
571        stop,
572        task,
573        epoch,
574        started: Instant::now(),
575        delivered,
576        downlink,
577    }
578}
579
580fn priority_for(kind: MediaKind) -> u8 {
581    match kind {
582        MediaKind::Audio => AUDIO_PRIORITY,
583        MediaKind::Video => VIDEO_PRIORITY,
584    }
585}

587/// The top-rung bitrate a video `source` is encoded at — the same per-source
588/// constant the producer's own [`hound_media::SimulcastPlanner`] assumes (see the
589/// `camera_tx`/`screen_tx` publish loops), so the receiver's downlink-demand
590/// thresholds (source/4, source/2) are computed against the SAME notion of
591/// "source" the producer is coalescing low-layer requests against.
592fn source_top_bps(source: Source) -> u64 {
593    match source {
594        Source::Camera => CAM_BITRATE.bps() as u64,
595        Source::Screen => SCREEN_BITRATE.bps() as u64,
596        Source::Microphone => unreachable!("video pipelines never carry Source::Microphone"),
597    }
598}
