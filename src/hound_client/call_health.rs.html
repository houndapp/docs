1//! # Steady-state audio-starvation monitor
2//!
3//! A warmup-anchored pass/fail gate over the same [`StatsSnapshot`] stream the audio
4//! debug report ([`crate::audio_report`]) diagnoses per-tick — the difference is this
5//! module answers "is the call healthy", not "why isn't it". Shared by any
6//! long-running call harness that needs a boolean verdict: the `hound-testkit`
7//! scenario suite's `InCall::expect_steady_audio` and `houndbot`'s CI smoke-test
8//! verdict both gate on [`sample_call_health`], so a call that passes one passes
9//! the other by construction.
10//!
11//! ## Why warmup-anchored
12//! Every peer's playout jitter buffer starts EMPTY and primes over its first few
13//! seconds — on the raw counters that priming is indistinguishable from starvation.
14//! Gating on absolute counts would fail every honest run. Instead: once `min_peers`
15//! have been continuously present for [`WARMUP`], the starve-signal total observed
16//! at that instant becomes the baseline; starvation is then any LATER window's total
17//! exceeding it.
18//!
19//! ## Never a blind sleep
20//! [`sample_call_health`] is watch-driven: it re-checks the observation deadline on
21//! every [`StatsSnapshot`] change (capped so a quiet stream still re-checks), never a
22//! fixed sleep-then-assert.
23//!
24//! ## The starvation signal
25//! Sums the exact counters [`crate::audio_report::diagnose`] attributes a glitch to —
26//! device-side output underruns/mix-starves, plus each peer's jitter-buffer
27//! underflow/late-drop/overflow-drop/loss — into one pass/fail total instead of
28//! naming a cause. Reach for the audio debug report when this fails and you need to
29//! know why.
30
31use std::time::Duration;
32
33use tokio::sync::watch;
34
35use crate::stats::StatsSnapshot;
36
37/// How long `min_peers` must have been continuously present before the
38/// starve-signal baseline is taken. See the module doc for why this can't be zero.
39pub const WARMUP: Duration = Duration::from_secs(5);
40
41/// Concealed-loss budget past warmup: the mix/jitter path may starve at most
42/// this many ticks beyond its baseline over the sampled window (each tick
43/// ≈10 ms, masked by loss concealment). The product's bar is "no audible
44/// degradation", not zero events — a lone tick is inaudible, while any
45/// systemic stall blows straight past this. Device-output underruns get NO
46/// budget: with a healthy (or paced-synthetic) sink they are uniformly zero,
47/// so any growth there is a real regression.
48pub const STEADY_STARVE_BUDGET: u64 = 2;
49
50/// One observation window's rollup. Peer/frame counts are high-water marks over the
51/// whole window (a peer that briefly joined/left, or a frame count that can only
52/// grow, are both captured correctly by tracking a running max rather than reading
53/// only the final sample); `starved` is the warmup-gated verdict.
54#[derive(Debug, Clone, Copy, Default)]
55pub struct CallHealthSample {
56    pub peers_seen: u32,
57    pub video_frames_rx: u64,
58    pub audio_frames_rx: u64,
59    pub starved: bool,
60}
61
62/// Watch `stats` for `duration`, gated on `min_peers` (see [`WARMUP`]). The caller
63/// supplies the receiver (e.g. `HoundClient::watch_stats()`) — this function has no
64/// dependency on any particular client/frontend, only the wire snapshot type.
65pub async fn sample_call_health(
66    mut stats: watch::Receiver<StatsSnapshot>,
67    min_peers: u32,
68    duration: Duration,
69) -> CallHealthSample {
70    let deadline = tokio::time::Instant::now() + duration;
71    let mut peers_up_at: Option<tokio::time::Instant> = None;
72    let mut starve_baseline: Option<(u64, u64)> = None;
73    let mut sample = CallHealthSample::default();
74
75    loop {
76        let snapshot = stats.borrow_and_update().clone();
77        let peers = snapshot.inbound.len() as u32;
78        let video: u64 = snapshot
79            .inbound
80            .iter()
81            .map(|p| p.video.frames_received)
82            .sum();
83        let audio: u64 = snapshot
84            .inbound
85            .iter()
86            .map(|p| p.audio.frames_received)
87            .sum();
88        let (output_total, mix_total) = starve_counters(&snapshot);
89
90        if peers >= min_peers && peers_up_at.is_none() {
91            peers_up_at = Some(tokio::time::Instant::now());
92        }
93        let starved = match starve_baseline {
94            None => {
95                if peers_up_at.is_some_and(|t| t.elapsed() >= WARMUP) {
96                    starve_baseline = Some((output_total, mix_total));
97                }
98                false
99            }
100            Some((output_base, mix_base)) => {
101                output_total > output_base || mix_total > mix_base + STEADY_STARVE_BUDGET
102            }
103        };
104
105        sample.peers_seen = sample.peers_seen.max(peers);
106        sample.video_frames_rx = sample.video_frames_rx.max(video);
107        sample.audio_frames_rx = sample.audio_frames_rx.max(audio);
108        sample.starved |= starved;
109
110        tracing::info!(peers, video, audio, starved, "call health sample");
111
112        let now = tokio::time::Instant::now();
113        if now >= deadline {
114            return sample;
115        }
116        let _ = tokio::time::timeout(
117            (deadline - now).min(Duration::from_millis(750)),
118            stats.changed(),
119        )
120        .await;
121    }
122}
123
124/// Every starvation signal [`sample_call_health`] gates on: device-side output
125/// underruns/mix-starves plus each peer's jitter-buffer underflow/late-drop/
126/// overflow-drop/loss counters. Cross-references
127/// [`crate::audio_report::diagnose`]'s per-stage rules — this sums the same
128/// counters into one pass/fail total rather than attributing a cause.
129/// (device-output underruns, mix/jitter starve events) — judged separately:
130/// output underruns get no budget, mix/jitter ticks get [`STEADY_STARVE_BUDGET`].
131fn starve_counters(s: &StatsSnapshot) -> (u64, u64) {
132    let mix = s.transport.playout_mix_starves
133        + s.inbound
134            .iter()
135            .map(|p| p.audio.underflow + p.audio.late_drops + p.audio.overflow_drops + p.audio.lost)
136            .sum::<u64>();
137    (s.transport.playout_output_underruns, mix)
138}

140#[cfg(test)]
141mod tests {
142    use super::*;
143    use crate::stats::InboundSnapshot;
144
145    /// Spawn a background task that waits `at` (paused-clock, auto-advanced) then
146    /// sends `snap`. Returns a keep-alive sender clone: the test must hold it for
147    /// the whole observation window — once EVERY sender handle drops, `changed()`
148    /// resolves instantly forever (a closed-channel signal, not a timer), which
149    /// would starve [`sample_call_health`]'s auto-advance-driven wait of a real
150    /// timer to park on.
151    fn send_after(
152        tx: watch::Sender<StatsSnapshot>,
153        at: Duration,
154        snap: StatsSnapshot,
155    ) -> watch::Sender<StatsSnapshot> {
156        let keep_alive = tx.clone();
157        tokio::spawn(async move {
158            tokio::time::sleep(at).await;
159            let _ = tx.send(snap);
160        });
161        keep_alive
162    }
163
164    /// Before `min_peers` ever shows up, no baseline is taken and nothing can be
165    /// flagged as starved — even if the starve counters are nonzero throughout.
166    /// `start_paused` auto-advances through [`sample_call_health`]'s internal
167    /// timeouts as soon as the runtime has nothing else to do, so this test
168    /// completes instantly despite the 2 s observation window.
169    #[tokio::test(start_paused = true)]
170    async fn no_peers_never_starves() {
171        let (_tx, rx) = watch::channel(StatsSnapshot::default());
172        let sample = sample_call_health(rx, 1, Duration::from_secs(2)).await;
173        assert_eq!(sample.peers_seen, 0);
174        assert!(!sample.starved);
175    }
176
177    /// A peer present for the whole window with zero starve signal never starves,
178    /// and its frame counts are captured as the high-water mark.
179    #[tokio::test(start_paused = true)]
180    async fn steady_peer_is_healthy() {
181        let (tx, rx) = watch::channel(healthy_snapshot(100, 200));
182        let _keep_alive = send_after(
183            tx,
184            WARMUP + Duration::from_secs(1),
185            healthy_snapshot(150, 260),
186        );
187
188        let sample = sample_call_health(rx, 1, WARMUP + Duration::from_secs(3)).await;
189        assert_eq!(sample.peers_seen, 1);
190        assert_eq!(sample.audio_frames_rx, 150);
191        assert_eq!(sample.video_frames_rx, 260);
192        assert!(!sample.starved);
193    }
194
195    /// A starve counter that only rises BEFORE warmup completes sets the baseline
196    /// but is not itself flagged (matches "anchored", not "any starvation ever").
197    #[tokio::test(start_paused = true)]
198    async fn pre_warmup_starvation_is_absorbed_into_the_baseline() {
199        let (tx, rx) = watch::channel(starved_snapshot(5));
200        // Same total as the baseline snapshot — no growth once warmup anchors it.
201        let _keep_alive = send_after(tx, WARMUP + Duration::from_millis(500), starved_snapshot(5));
202
203        let sample = sample_call_health(rx, 1, WARMUP + Duration::from_secs(3)).await;
204        assert!(!sample.starved);
205    }
206
207    /// Starvation growth AFTER the warmup baseline is anchored is flagged.
208    #[tokio::test(start_paused = true)]
209    async fn post_warmup_growth_is_starvation() {
210        let (tx, rx) = watch::channel(starved_snapshot(0));
211        // Grew past the baseline of 0.
212        let _keep_alive = send_after(tx, WARMUP + Duration::from_millis(500), starved_snapshot(5));
213
214        let sample = sample_call_health(rx, 1, WARMUP + Duration::from_secs(3)).await;
215        assert!(sample.starved);
216    }
217
218    #[tokio::test(start_paused = true)]
219    async fn growth_within_the_budget_is_not_starvation() {
220        let (tx, rx) = watch::channel(starved_snapshot(0));
221        let _keep_alive = send_after(
222            tx,
223            WARMUP + Duration::from_millis(500),
224            starved_snapshot(STEADY_STARVE_BUDGET),
225        );
226
227        let sample = sample_call_health(rx, 1, WARMUP + Duration::from_secs(3)).await;
228        assert!(!sample.starved, "growth == budget must stay healthy");
229    }
230
231    #[tokio::test(start_paused = true)]
232    async fn output_underruns_have_no_budget() {
233        let (tx, rx) = watch::channel(StatsSnapshot::default());
234        let mut one_underrun = StatsSnapshot::default();
235        one_underrun.transport.playout_output_underruns = 1;
236        one_underrun.inbound.push(InboundSnapshot {
237            peer_id: 1,
238            ..Default::default()
239        });
240        let _keep_alive = send_after(tx, WARMUP + Duration::from_millis(500), one_underrun);
241
242        let sample = sample_call_health(rx, 0, WARMUP + Duration::from_secs(3)).await;
243        assert!(sample.starved, "a single device underrun is a regression");
244    }

250            audio: crate::stats::InboundAudioSnapshot {

252                ..Default::default()
253            },
254            video: crate::stats::InboundVideoSnapshot {
255                frames_received: video_frames,
256                ..Default::default()
257            },
258            ..Default::default()
259        });
260        s
261    }
