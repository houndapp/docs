1//! The realtime plane: a socket.io client for the backend `/community` namespace
2//! (server is `socketioxide` 0.18 — Socket.IO v5 / Engine.IO v4, which
3//! [`rust_socketio`] 0.6 speaks). It connects with `{ token }` handshake auth, lets
4//! the caller subscribe/unsubscribe to per-space rooms, and decodes every server
5//! event into a typed [`RealtimeEvent`] delivered over a [`tokio::sync::mpsc`]
6//! channel — so the reducer consumes a single typed stream, never raw JSON.
7//!
8//! **Threading:** `rust_socketio`'s async handlers are `Fn` closures returning
9//! `BoxFuture<'static, ()>`, so they cannot borrow. Each registered handler captures
10//! a clone of the event sender and a parser fn; on an event it parses the payload and
11//! forwards a [`RealtimeEvent`] (or a [`RealtimeEvent::Decode`] marker on malformed
12//! input — loud, never silent). The socket.io v5 wire delivers an event's argument as
13//! the first element of a JSON array, surfaced by `rust_socketio` as
14//! [`Payload::Text`]`(Vec<Value>)`; we read `values[0]`.
15
16use rust_socketio::asynchronous::{Client, ClientBuilder};
17use rust_socketio::{Payload, TransportType};
18use serde::Deserialize;
19use tokio::sync::mpsc;
20use uuid::Uuid;
21
22use crate::error::{HoundError, Result};

24/// A decoded server-side realtime event. The reducer folds these into view-models.
25/// Every variant mirrors a `/community` namespace event the backend emits (see
26/// `services/backend/src/api/{community,space}.rs`).
27#[derive(Clone, Debug, PartialEq)]
28pub enum RealtimeEvent {
29    /// `message:create` — a new message in a subscribed space.
30    MessageCreated(MessageEvent),
31    /// `message:update` — an edit to an existing message.
32    MessageUpdated(MessageEvent),
33    /// `space:create` — a new space in a joined community.
34    SpaceCreated(SpaceEvent),
35    /// `space:remove` — a space was deleted.
36    SpaceRemoved { space_id: Uuid },
37    /// `voice:join` — a user joined a voice channel (SFU peer id known).
38    VoiceJoined(VoiceJoinEvent),
39    /// `voice:leave` — a user left a voice channel.
40    VoiceLeft(VoiceLeaveEvent),
41    /// `voice:screenshare` — a participant toggled screenshare.
42    VoiceScreenShare(VoiceScreenShareEvent),
43    /// `voice:state` — the full voice roster snapshot pushed on connect.
44    VoiceState(Vec<VoiceJoinEvent>),
45    /// A recognised event whose payload failed to parse — surfaced (not dropped) so a
46    /// contract drift is visible to the reducer/logs. Carries the event name + detail.
47    Decode { event: String, detail: String },
48    /// The socket connected (handshake + auth accepted).
49    Connected,
50    /// The socket disconnected (transport closed or server kicked us).
51    Disconnected,
52}

116/// A live connection to the `/community` namespace. Drop it (or call
117/// [`disconnect`](Self::disconnect)) to tear the socket down. Cheap to hold; all the
118/// work happens on `rust_socketio`'s own task.
119pub struct RealtimeClient {
120    socket: Client,
121}
122
123impl RealtimeClient {
124    /// Connect to `base_url`'s `/community` namespace with `token` as handshake auth
125    /// (`{ "token": "<jwt>" }`, matching the backend's `AuthPayload`). Returns the
126    /// live client plus the receiver end of the typed event stream.
127    ///
128    /// A live connect requires a running backend; this wires the builder, auth,
129    /// handlers, and `connect()` exactly per the `rust_socketio` async API. On
130    /// transport/handshake failure it returns [`HoundError::Realtime`].
131    pub async fn connect(
132        base_url: &str,
133        token: &str,
134    ) -> Result<(Self, mpsc::UnboundedReceiver<RealtimeEvent>)> {
135        let (tx, rx) = mpsc::unbounded_channel();
136
137        let mut builder = ClientBuilder::new(base_url.to_owned())
138            .namespace("/community")
139            .transport_type(TransportType::Websocket)
140            .auth(serde_json::json!({ "token": token }));
141
142        // Typed event handlers. Each forwards a parsed RealtimeEvent over `tx`.
143        builder = builder
144            .on(
145                "message:create",
146                handler(tx.clone(), |v| parse(v).map(RealtimeEvent::MessageCreated)),
147            )
148            .on(
149                "message:update",
150                handler(tx.clone(), |v| parse(v).map(RealtimeEvent::MessageUpdated)),
151            )
152            .on(
153                "space:create",
154                handler(tx.clone(), |v| parse(v).map(RealtimeEvent::SpaceCreated)),
155            )
156            .on(
157                "space:remove",
158                handler(tx.clone(), |v| {
159                    #[derive(Deserialize)]
160                    struct Removed {
161                        id: Uuid,
162                    }
163                    parse::<Removed>(v).map(|r| RealtimeEvent::SpaceRemoved { space_id: r.id })
164                }),
165            )
166            .on(
167                "voice:join",
168                handler(tx.clone(), |v| parse(v).map(RealtimeEvent::VoiceJoined)),
169            )
170            .on(
171                "voice:leave",
172                handler(tx.clone(), |v| parse(v).map(RealtimeEvent::VoiceLeft)),
173            )
174            .on(
175                "voice:screenshare",
176                handler(tx.clone(), |v| {
177                    parse(v).map(RealtimeEvent::VoiceScreenShare)
178                }),
179            )
180            .on(
181                "voice:state",
182                handler(tx.clone(), |v| parse(v).map(RealtimeEvent::VoiceState)),
183            );
184
185        // Connection lifecycle events (reserved names emitted by the engine).
186        builder = builder
187            .on("open", lifecycle(tx.clone(), RealtimeEvent::Connected))
188            .on("close", lifecycle(tx.clone(), RealtimeEvent::Disconnected))
189            .on("error", {
190                let tx = tx.clone();
191                move |payload: Payload, _: Client| {
192                    let tx = tx.clone();
193                    Box::pin(async move {
194                        let detail = match payload {
195                            Payload::Text(v) => format!("{v:?}"),
196                            #[allow(deprecated)]
197                            Payload::String(s) => s,
198                            Payload::Binary(_) => "binary error payload".to_owned(),
199                        };
200                        let _ = tx.send(RealtimeEvent::Decode {
201                            event: "error".to_owned(),
202                            detail,
203                        });
204                    })
205                }
206            });
207
208        let socket = builder
209            .connect()
210            .await
211            .map_err(|e| HoundError::Realtime(format!("connect failed: {e}")))?;
212
213        Ok((Self { socket }, rx))
214    }

216    /// Join a space room so this socket receives that space's `message:*` events.
217    /// Emits `space:subscribe` with the space UUID (matches the backend handler).
218    pub async fn subscribe_space(&self, space_id: Uuid) -> Result<()> {
219        self.socket
220            .emit("space:subscribe", serde_json::json!(space_id))
221            .await
222            .map_err(|e| HoundError::Realtime(format!("space:subscribe failed: {e}")))
223    }
224
225    /// Leave a space room (emits `space:unsubscribe`).
226    pub async fn unsubscribe_space(&self, space_id: Uuid) -> Result<()> {
227        self.socket
228            .emit("space:unsubscribe", serde_json::json!(space_id))
229            .await
230            .map_err(|e| HoundError::Realtime(format!("space:unsubscribe failed: {e}")))
231    }
232
233    /// Announce a screenshare toggle (emits `voice:screenshare`); the backend
234    /// re-broadcasts it to the community room.
235    pub async fn set_screenshare(&self, space_id: Uuid, enabled: bool) -> Result<()> {
236        self.socket
237            .emit(
238                "voice:screenshare",
239                serde_json::json!({ "space_id": space_id, "enabled": enabled }),
240            )
241            .await
242            .map_err(|e| HoundError::Realtime(format!("voice:screenshare failed: {e}")))
243    }
244
245    /// Tear down the connection.
246    pub async fn disconnect(self) -> Result<()> {
247        self.socket
248            .disconnect()
249            .await
250            .map_err(|e| HoundError::Realtime(format!("disconnect failed: {e}")))
251    }
252}

254/// Pull the event argument out of a socket.io payload. v5 delivers the emitted
255/// argument as the first element of a JSON array ([`Payload::Text`]); the deprecated
256/// [`Payload::String`] carries it as a raw JSON string. Returns the inner value.
257fn payload_value(payload: Payload) -> std::result::Result<serde_json::Value, String> {
258    match payload {
259        Payload::Text(mut values) => {
260            if values.is_empty() {
261                Err("empty event payload".to_owned())
262            } else {
263                Ok(values.swap_remove(0))
264            }
265        }
266        #[allow(deprecated)]
267        Payload::String(s) => {
268            serde_json::from_str(&s).map_err(|e| format!("invalid json string payload: {e}"))
269        }
270        Payload::Binary(_) => Err("unexpected binary payload".to_owned()),
271    }
272}
273
274/// Deserialize the event argument into `T`.
275fn parse<T: serde::de::DeserializeOwned>(
276    value: serde_json::Value,
277) -> std::result::Result<T, String> {
278    serde_json::from_value(value).map_err(|e| e.to_string())
279}

281/// Build a `rust_socketio` async handler that decodes the payload with `f` and
282/// forwards the resulting [`RealtimeEvent`], or a [`RealtimeEvent::Decode`] marker
283/// on failure. `f` takes the already-unwrapped event-argument JSON value.
284fn handler<F>(
285    tx: mpsc::UnboundedSender<RealtimeEvent>,
286    f: F,
287) -> impl Fn(Payload, Client) -> futures::future::BoxFuture<'static, ()> + Send + Sync + 'static
288where
289    F: Fn(serde_json::Value) -> std::result::Result<RealtimeEvent, String>
290        + Send
291        + Sync
292        + Clone
293        + 'static,
294{
295    move |payload: Payload, _: Client| {
296        let tx = tx.clone();
297        let f = f.clone();
298        Box::pin(async move {
299            let event = match payload_value(payload) {
300                Ok(value) => f(value).unwrap_or_else(|detail| RealtimeEvent::Decode {
301                    event: "payload".to_owned(),
302                    detail,
303                }),
304                Err(detail) => RealtimeEvent::Decode {
305                    event: "payload".to_owned(),
306                    detail,
307                },
308            };
309            let _ = tx.send(event);
310        })
311    }
312}
313
314/// A handler that ignores the payload and forwards a fixed lifecycle event.
315fn lifecycle(
316    tx: mpsc::UnboundedSender<RealtimeEvent>,
317    event: RealtimeEvent,
318) -> impl Fn(Payload, Client) -> futures::future::BoxFuture<'static, ()> + Send + Sync + 'static {
319    move |_: Payload, _: Client| {
320        let tx = tx.clone();
321        let event = event.clone();
322        Box::pin(async move {
323            let _ = tx.send(event);
324        })
325    }
326}

332    #[test]
333    fn payload_value_unwraps_v5_array() {
334        let p = Payload::Text(vec![serde_json::json!({ "hello": "world" })]);
335        let v = payload_value(p).unwrap();
336        assert_eq!(v["hello"], "world");
337    }
338
339    #[test]
340    fn payload_value_parses_string_variant() {
341        #[allow(deprecated)]
342        let p = Payload::String("{\"n\":1}".to_owned());
343        let v = payload_value(p).unwrap();
344        assert_eq!(v["n"], 1);
345    }
346
347    #[test]
348    fn payload_value_rejects_empty() {
349        assert!(payload_value(Payload::Text(vec![])).is_err());
350    }

352    #[test]
353    fn parses_message_event() {
354        let v = serde_json::json!({
355            "id": "018c5b6e-0000-7000-0000-000000000010",
356            "space_id": "018c5b6e-0000-7000-0000-000000000002",
357            "user_id": "018c5b6e-0000-7000-0000-000000000003",
358            "contents": "hi",
359            "edited_at": null,
360            "profile": { "user_id": "018c5b6e-0000-7000-0000-000000000003",
361                         "global_profile_id": "018c5b6e-0000-7000-0000-000000000004",
362                         "community_profile_id": null, "name": "Aoife", "picture": "p" },
363            "attachments": []
364        });
365        let m: MessageEvent = parse(v).unwrap();
366        assert_eq!(m.contents, "hi");
