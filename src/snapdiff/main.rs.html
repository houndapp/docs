1//! `snapdiff` — perceptual visual-regression comparison for the Layer-2
2//! snapshot system (see ui/integration_test/visual_test.dart + the harness in
3//! ui/integration_test/visual.sh). Pairs PNGs by name across a baselines dir
4//! and a current dir, scores each pair with `image_compare::rgb_hybrid_compare`,
5//! and emits a JSON summary (stdout) plus one self-contained HTML report.
6
7mod compare;
8mod report;
9
10use std::path::PathBuf;
11use std::process::ExitCode;
12
13use clap::{Args, Parser, Subcommand};
14
15#[derive(Parser)]
16#[command(
17    name = "snapdiff",
18    about = "Perceptual visual-regression diffing for PNG snapshot pairs"
19)]
20struct Cli {
21    #[command(subcommand)]
22    command: Command,
23}
24
25#[derive(Subcommand)]
26enum Command {
27    /// Compare a `current` snapshot directory against `baselines`.
28    Compare(CompareArgs),
29}
30
31#[derive(Args)]
32struct CompareArgs {
33    /// Accepted-baseline PNGs (may contain `<name>.mask.png` ignore masks).
34    #[arg(long)]
35    baselines: PathBuf,
36    /// This run's freshly captured PNGs.
37    #[arg(long)]
38    current: PathBuf,
39    /// Directory to write the report into (created if missing): a copy of
40    /// the built `crates/snapdiff/ui` SvelteKit app, a `data.js` payload,
41    /// and an `imgs/` dir of baseline/current/diff PNGs. Resolves the built
42    /// UI from `SNAPDIFF_UI_DIST` or `crates/snapdiff/ui/build` — see
43    /// `report::write_report`.
44    #[arg(long)]
45    report: PathBuf,
46    /// Similarity score (in `[0,1]`, `1.0` = identical) at/above which a pair
47    /// passes. Perceptual comparisons are inherently a little fuzzy (font
48    /// hinting / subpixel AA can shift a handful of pixels between runs on
49    /// the SAME machine); 0.995 tolerates that noise while still catching a
50    /// real layout/content regression.
51    #[arg(long, default_value_t = 0.995)]
52    threshold: f64,
53    /// Copy every current image over its baseline after comparing (accepts
54    /// `New` pairs and any diffs as the new baseline). Masks are untouched.
55    #[arg(long, default_value_t = false)]
56    update: bool,
57}
58
59fn main() -> ExitCode {
60    let cli = Cli::parse();
61    match cli.command {
62        Command::Compare(args) => run_compare(args),
63    }
64}
65
66fn run_compare(args: CompareArgs) -> ExitCode {
67    let outcomes = match compare::compare_all(&args.baselines, &args.current, args.threshold) {
68        Ok(o) => o,
69        Err(e) => {
70            eprintln!(
71                "snapdiff: failed to scan {:?}/{:?}: {e}",
72                args.baselines, args.current
73            );
74            return ExitCode::from(2);
75        }
76    };
77
78    if args.update
79        && let Err(e) = compare::accept_current(&outcomes, &args.baselines, &args.current)
80    {
81        eprintln!("snapdiff: --update failed: {e}");
82        return ExitCode::from(2);
83    }
84
85    if let Err(e) = report::write_report(&outcomes, &args.report) {
86        eprintln!("snapdiff: failed to write report to {:?}: {e}", args.report);
87        return ExitCode::from(2);
88    }
89
90    // Exit-code gate bundles Fail/Missing/Error (see `Status::is_failure`);
91    // `New` alone must NOT fail the run — see the JSON `new` field for that
92    // signal instead (an "N pairs are pending baseline acceptance" nudge).
93    let any_failure = outcomes.iter().any(|o| o.status.is_failure());
94
95    let summary = Summary::from(&outcomes[..]);
96    match serde_json::to_string(&summary) {
97        Ok(json) => println!("{json}"),
98        Err(e) => eprintln!("snapdiff: failed to serialize summary: {e}"),
99    }
100
101    if any_failure {
102        ExitCode::from(1)
103    } else {
104        ExitCode::SUCCESS
105    }
106}
107
108#[derive(serde::Serialize)]
109struct Summary {
110    total: usize,
111    passed: usize,
112    failed: usize,
113    new: usize,
114    missing: usize,
115    errored: usize,
116    pairs: Vec<compare::PairSummary>,
117}
118
119impl From<&[compare::PairOutcome]> for Summary {
120    fn from(outcomes: &[compare::PairOutcome]) -> Self {
121        use compare::Status;
122        let mut s = Summary {
123            total: outcomes.len(),
124            passed: 0,
125            failed: 0,
126            new: 0,
127            missing: 0,
128            errored: 0,
129            pairs: outcomes.iter().map(compare::PairSummary::from).collect(),
130        };
131        for o in outcomes {
132            match o.status {
133                Status::Pass => s.passed += 1,
134                Status::Fail => s.failed += 1,
135                Status::New => s.new += 1,
136                Status::Missing => s.missing += 1,
137                Status::Error => s.errored += 1,
138            }
139        }
140        s
141    }
142}
