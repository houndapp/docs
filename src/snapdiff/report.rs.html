1//! Report DATA, not markup: the report is a static SvelteKit SPA
2//! (`tools/snapdiff/ui/`, built with `pnpm build`) that renders
3//! `window.__SNAPDIFF__` at runtime. This module's job is just: serialize
4//! the outcomes into that payload (`data.js`), copy the built UI's assets
5//! into the report directory, and copy each pair's baseline/current/diff
6//! PNGs into `imgs/<name>/` as plain files — relative `<img src>` paths, no
7//! base64, so opening the report is one `<img>` fetch per image (no CORS to
8//! fight; only the JS module-script CORS restriction under `file://` needed
9//! solving, and that's handled entirely on the UI side — see
10//! `ui/svelte.config.js`'s `bundleStrategy: 'inline'`).
11

19/// One pair's report-payload projection: mirrors
20/// `ui/src/lib/payload.ts`'s `PairPayload` — keep the two in sync by hand.
21#[derive(Debug, Clone, Serialize)]
22struct PairPayload {
23    name: String,
24    status: Status,
25    score: Option<f64>,
26    threshold: f64,
27    masked: bool,
28    message: Option<String>,
29    baseline_img: Option<String>,
30    current_img: Option<String>,
31    diff_img: Option<String>,
32}
33
34#[derive(Debug, Serialize)]
35struct ReportPayload {
36    pairs: Vec<PairPayload>,
37}
38
39/// Default built-UI location: `tools/snapdiff/ui/build`, resolved relative
40/// to THIS crate's own manifest dir (a compile-time constant) rather than
41/// the process's current working directory, so `snapdiff` finds it
42/// regardless of where it's invoked from.
43fn default_ui_dist() -> PathBuf {
44    PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("ui/build")
45}
46
47/// The per-pair image subdirectory: the pair name minus its `.png` suffix
48/// (e.g. `login.png` -> `login`) — pair names are already unique, so this
49/// alone is enough to avoid collisions, no further sanitization needed.
50fn image_subdir(name: &str) -> &str {
51    name.strip_suffix(".png").unwrap_or(name)
52}
53
54/// Write the full report directory: the built UI's assets, `data.js`, and
55/// `imgs/`. Resolves the UI dist dir from `SNAPDIFF_UI_DIST` if set, else
56/// [`default_ui_dist`].
57pub fn write_report(outcomes: &[PairOutcome], report_dir: &Path) -> Result<(), String> {
58    let ui_dist = env::var_os("SNAPDIFF_UI_DIST")
59        .map(PathBuf::from)
60        .unwrap_or_else(default_ui_dist);
61    write_report_from(outcomes, report_dir, &ui_dist)
62}
63
64/// The testable core of [`write_report`] — takes the UI dist dir explicitly
65/// so tests can point it at a throwaway fixture instead of a real build.
66fn write_report_from(
67    outcomes: &[PairOutcome],
68    report_dir: &Path,
69    ui_dist: &Path,
70) -> Result<(), String> {
71    if !ui_dist.is_dir() {
72        return Err(format!(
73            "snapdiff UI dist not found at {}. Build it first: \
74             cd tools/snapdiff/ui && pnpm install --frozen-lockfile && pnpm build. \
75             Or point SNAPDIFF_UI_DIST at an already-built dist.",
76            ui_dist.display()
77        ));
78    }
79
80    fs::create_dir_all(report_dir)
81        .map_err(|e| format!("failed to create report dir {}: {e}", report_dir.display()))?;
82    copy_dir_recursive(ui_dist, report_dir)
83        .map_err(|e| format!("failed to copy UI assets from {}: {e}", ui_dist.display()))?;
84
85    let imgs_dir = report_dir.join("imgs");
86    let mut pairs = Vec::with_capacity(outcomes.len());
87    for o in outcomes {
88        let subdir = image_subdir(&o.name);
89        let dir = imgs_dir.join(subdir);
90        let baseline_img = write_img(&dir, "baseline.png", o.baseline_png.as_deref())
91            .map_err(|e| format!("failed to write baseline image for {}: {e}", o.name))?
92            .map(|_| format!("imgs/{subdir}/baseline.png"));
93        let current_img = write_img(&dir, "current.png", o.current_png.as_deref())
94            .map_err(|e| format!("failed to write current image for {}: {e}", o.name))?
95            .map(|_| format!("imgs/{subdir}/current.png"));
96        let diff_img = write_img(&dir, "diff.png", o.diff_png.as_deref())
97            .map_err(|e| format!("failed to write diff image for {}: {e}", o.name))?
98            .map(|_| format!("imgs/{subdir}/diff.png"));
99
100        pairs.push(PairPayload {
101            name: o.name.clone(),
102            status: o.status,
103            score: o.score,
104            threshold: o.threshold,
105            masked: o.masked,
106            message: o.message.clone(),
107            baseline_img,
108            current_img,
109            diff_img,
110        });
111    }
112
113    let json = serde_json::to_string(&ReportPayload { pairs })
114        .map_err(|e| format!("failed to serialize report payload: {e}"))?;
115    // A classic (non-module) assignment — see ui/src/app.html's doc comment
116    // for why this specifically must not be `type="module"`.
117    let data_js = format!("window.__SNAPDIFF__ = {json};\n");
118    fs::write(report_dir.join("data.js"), data_js)
119        .map_err(|e| format!("failed to write data.js: {e}"))?;
120
121    Ok(())
122}

124/// Write `bytes` (if present) to `dir/filename`, creating `dir` first.
125/// Returns `Ok(None)` (not an error) when `bytes` is `None` — a pair legally
126/// has no baseline/current/diff image (e.g. `Missing`/`New`).
127fn write_img(dir: &Path, filename: &str, bytes: Option<&[u8]>) -> std::io::Result<Option<()>> {
128    let Some(bytes) = bytes else {
129        return Ok(None);
130    };
131    fs::create_dir_all(dir)?;
132    fs::write(dir.join(filename), bytes)?;
133    Ok(Some(()))
134}
135
136/// Plain recursive file copy (no crate dependency for this — it's a dozen
137/// lines and pulling in e.g. `fs_extra` for just this would be overkill).
138/// Skips anything that isn't a regular file or directory (i.e. symlinks —
139/// a Vite build shouldn't produce any, so this is a safety net, not a
140/// feature).
141fn copy_dir_recursive(from: &Path, to: &Path) -> std::io::Result<()> {
142    for entry in fs::read_dir(from)? {
143        let entry = entry?;
144        let file_type = entry.file_type()?;
145        let dest = to.join(entry.file_name());
146        if file_type.is_dir() {
147            fs::create_dir_all(&dest)?;
148            copy_dir_recursive(&entry.path(), &dest)?;
149        } else if file_type.is_file() {
150            fs::copy(entry.path(), &dest)?;
151        }
152    }
153    Ok(())
154}

161    fn outcome(name: &str, status: Status) -> PairOutcome {
162        PairOutcome {
163            name: name.to_owned(),
164            status,
165            score: Some(0.98),
166            threshold: 0.995,
167            masked: false,
168            message: None,
169            baseline_png: Some(vec![1, 2, 3]),
170            current_png: Some(vec![4, 5, 6]),
171            diff_png: Some(vec![7, 8, 9]),
172        }
173    }

175    /// A fake built-UI dist: `write_report_from` only needs to copy
176    /// whatever's there, so this doesn't need a real SvelteKit build to
177    /// exercise the copy+payload logic.
178    fn fake_ui_dist() -> TempDir {
179        let dir = TempDir::new().unwrap();
180        fs::write(dir.path().join("index.html"), "<html></html>").unwrap();
181        fs::create_dir_all(dir.path().join("_app/immutable")).unwrap();
182        fs::write(dir.path().join("_app/immutable/entry.js"), "console.log(1)").unwrap();
183        dir
184    }

186    #[test]
187    fn errors_clearly_when_ui_dist_is_missing() {
188        let tmp = TempDir::new().unwrap();
189        let missing_ui = tmp.path().join("nonexistent-ui-dist");
190        let report_dir = tmp.path().join("report");
191        let err = write_report_from(&[], &report_dir, &missing_ui).unwrap_err();
192        assert!(err.contains("UI dist not found"), "{err}");

199        let tmp = TempDir::new().unwrap();

213    #[test]
214    fn writes_data_js_with_the_full_payload() {
215        let ui_dist = fake_ui_dist();
216        let tmp = TempDir::new().unwrap();
217        let report_dir = tmp.path().join("report");
218        let outcomes = vec![
219            outcome("login.png", Status::Pass),
220            outcome("home.png", Status::Fail),
221        ];
222        write_report_from(&outcomes, &report_dir, ui_dist.path()).unwrap();
223
224        let data_js = fs::read_to_string(report_dir.join("data.js")).unwrap();
225        assert!(data_js.starts_with("window.__SNAPDIFF__ = "));
226        assert!(!data_js.contains("type=\"module\""));
227
228        let json_str = data_js
229            .strip_prefix("window.__SNAPDIFF__ = ")
230            .unwrap()
231            .trim_end()
232            .trim_end_matches(';');
233        let payload: serde_json::Value = serde_json::from_str(json_str).unwrap();
234        let pairs = payload["pairs"].as_array().unwrap();
235        assert_eq!(pairs.len(), 2);
236        let login = pairs.iter().find(|p| p["name"] == "login.png").unwrap();
237        assert_eq!(login["status"], "pass");
238        assert_eq!(login["baseline_img"], "imgs/login/baseline.png");
239        assert_eq!(login["current_img"], "imgs/login/current.png");
240        assert_eq!(login["diff_img"], "imgs/login/diff.png");
241    }

243    #[test]

257        assert_eq!(
258            fs::read(report_dir.join("imgs/call-1-peer/diff.png")).unwrap(),
259            vec![7, 8, 9]
260        );
261
262        let data_js = fs::read_to_string(report_dir.join("data.js")).unwrap();
263        assert!(data_js.contains("\"baseline_img\":null"));
264    }
