1//! # SFU service endpoint
2//!
3//! This module owns the iroh `Endpoint` and the `accept()` loop. It is thin
4//! glue: every accepted connection is handed to
5//! [`moq_relay::handle_relay_connection`], which authenticates the relay
6//! token, admits the session against [`moq_relay::SessionLimits`], scopes the
7//! connection to its space, and drives the moq-net session. All forwarding
8//! logic lives in [`crate::moq_relay`]; this module just binds the endpoint
9//! and dispatches connections to it.
10

12use std::sync::atomic::AtomicUsize;

15
16use crate::moq_relay::{self, IngressLimits, RELAY_ALPN, RelayConfig, SessionRegistry};

20/// The SFU server.
21///
22/// Holds the iroh endpoint and the shared MoQ relay origin. Call [`Sfu::run`]
23/// to start accepting connections.
24pub struct Sfu {
25    pub endpoint: Endpoint,
26    relay_hmac_secret: String,
27    /// Shared root MoQ origin every accepted `moql` connection is scoped
28    /// against, so moq-net fans broadcasts out between same-space
29    /// publishers/subscribers — exactly as a standalone [`moq_relay::Relay`]
30    /// does, but on the endpoint/node-id the backend registers as
31    /// `sfu_node_id`.
32    relay_origin: moq_relay::OriginProducer,
33    /// Global + per-user session admission accounting (see
34    /// [`SessionRegistry`]). A session is only ever admitted here — and so
35    /// only ever counted toward [`Sfu::active_sessions`] — AFTER
36    /// [`moq_relay::handle_connection`]'s auth gate succeeds; see that
37    /// function's doc for why unauthenticated connections must never touch
38    /// this counter.
39    registry: SessionRegistry,
40    /// Auth handshake deadline — see [`moq_relay::AUTH_DEADLINE`].
41    auth_deadline: std::time::Duration,
42    /// Per-connection ingress-watchdog budget — see [`IngressLimits`].
43    ingress: IngressLimits,
44}
45
46/// Whether the SIGTERM drain wait should stop polling and let the process
47/// exit: every session has ended, or the deadline has elapsed (a stuck peer
48/// must not block a rollout forever). Factored out of the wait loop so the
49/// exit decision is a pure, directly-testable function.
50pub fn drain_complete(
51    active_sessions: usize,
52    elapsed: std::time::Duration,
53    deadline: std::time::Duration,
54) -> bool {
55    active_sessions == 0 || elapsed >= deadline
56}
57
58impl Sfu {
59    /// Bind a new iroh endpoint and return an `Sfu` ready to accept connections.
60    ///
61    /// `relay_url` is the iroh relay server used for NAT traversal. `secret_key`
62    /// is the iroh secret key for this node (determines the `EndpointId`).
63    /// `relay_hmac_secret` is the shared HMAC secret used to verify relay tokens.
64    /// `bind_addr` is the **exact** IPv4 socket the QUIC endpoint binds to.
65    /// `config` carries the auth deadline and session admission caps — see
66    /// [`RelayConfig`].
67    ///
68    /// ## Why a specific bind address (not `0.0.0.0`)
69    ///
70    /// The SFU runs with `hostNetwork: true`, so the node has many interface
71    /// addresses: its public IP plus a pile of private/cluster ones (CNI,
72    /// docker bridge, the Kubernetes node InternalIP). iroh advertises the
73    /// addresses it can reach a peer on; **bound to `0.0.0.0` it enumerates and
74    /// advertises every interface address**, so clients spend the call probing
75    /// unreachable private IPs — paths open, fail validation, get abandoned, and
76    /// the churn destabilises the connection (the reliable control stream often
77    /// never lands). Bound to a **specific** address, iroh advertises only that
78    /// one (see `iroh::socket::update_direct_addresses`). So the caller resolves
79    /// the node's public IPv4 and passes `public_ip:LISTEN_PORT` here, giving
80    /// clients exactly one reachable address to dial. The port is fixed (not
81    /// ephemeral) so it matches the single UDP port opened on the node.
82    ///
83    /// If the public IP can't be determined the caller may pass `0.0.0.0`, which
84    /// falls back to enumerating every interface address (correct, just noisy).
85    pub async fn bind(
86        relay_url: RelayUrl,
87        secret_key: SecretKey,
88        relay_hmac_secret: String,
89        bind_addr: std::net::SocketAddrV4,
90        config: RelayConfig,
91    ) -> eyre::Result<Self> {
92        let endpoint = Endpoint::builder(iroh::endpoint::presets::N0)
93            .secret_key(secret_key)
94            .relay_mode(RelayMode::Custom(iroh::RelayMap::from(relay_url)))
95            .alpns(vec![RELAY_ALPN.to_vec()])
96            .clear_ip_transports()
97            .bind_addr(std::net::SocketAddr::V4(bind_addr))
98            .map_err(|e| eyre::eyre!("invalid SFU bind address {bind_addr}: {e}"))?
99            .bind()
100            .await?;
101
102        Ok(Self::from_endpoint(endpoint, relay_hmac_secret, config))
103    }
104
105    /// Build an `Sfu` around an already-bound iroh [`Endpoint`].
106    ///
107    /// [`Sfu::bind`] is the production constructor: it builds an endpoint wired
108    /// to the iroh relay for NAT traversal. Tests need a *hermetic* endpoint
109    /// (relay disabled, direct loopback addressing) so they can run without
110    /// network access — they construct that endpoint themselves and hand it
111    /// here. Keeping the accept loop in [`Sfu::run`] identical for both paths is
112    /// the whole point: the e2e tests exercise the same connection-handling glue
113    /// that runs in production.
114    ///
115    /// The caller is responsible for having set [`RELAY_ALPN`] on the endpoint.
116    pub fn from_endpoint(
117        endpoint: Endpoint,
118        relay_hmac_secret: String,
119        config: RelayConfig,
120    ) -> Self {
121        Self {
122            endpoint,
123            relay_hmac_secret,
124            relay_origin: moq_relay::new_root_origin(),
125            registry: SessionRegistry::new(config.limits),
126            auth_deadline: config.auth_deadline,
127            ingress: config.ingress,
128        }
129    }
130
131    /// Shared handle on the active-session counter, for the drain wait loop in
132    /// `main` to poll without needing `&Sfu` (the endpoint's accept loop and the
133    /// SIGTERM handler run concurrently). Counts only AUTHENTICATED sessions
134    /// (admitted after [`moq_relay::handle_connection`]'s auth gate) — a
135    /// connection stuck at the token handshake, rejected, or over a session cap
136    /// never bumps this, so the drain wait only ever waits on real, authenticated
137    /// calls.
138    pub fn active_sessions(&self) -> Arc<AtomicUsize> {
139        self.registry.active_sessions()
140    }
141
142    /// Accept connections in a loop until the endpoint closes.
143    ///
144    /// Each accepted connection is handed off to
145    /// [`moq_relay::handle_relay_connection`] in its own tokio task. This
146    /// function only returns on error or endpoint close.
147    pub async fn run(&self) -> eyre::Result<()> {
148        tracing::info!(
149            node_id = %self.endpoint.id(),
150            relay_alpn = %String::from_utf8_lossy(RELAY_ALPN),
151            "SFU accepting connections (MoQ relay)"
152        );
153
154        loop {
155            let Some(incoming) = self.endpoint.accept().await else {
156                tracing::info!("SFU endpoint closed — stopping accept loop");
157                break;
158            };
159
160            let conn = match incoming.await {
161                Ok(c) => c,
162                Err(e) => {
163                    tracing::warn!("incoming connection failed: {e}");
164                    continue;
165                }
166            };
167
168            let origin = self.relay_origin.clone();
169            let secret = self.relay_hmac_secret.clone();
170            let registry = self.registry.clone();
171            let auth_deadline = self.auth_deadline;
172            let ingress = self.ingress;
173            tokio::spawn(async move {
174                if let Err(e) = moq_relay::handle_relay_connection(
175                    conn,
176                    origin,
177                    secret,
178                    registry,
179                    auth_deadline,
180                    ingress,
181                )
182                .await
183                {
184                    tracing::warn!("relay connection handler error: {e}");
185                }
186            });
187        }
188
189        Ok(())
190    }
191
192    /// The iroh `EndpointId` for this SFU node.
193    ///
194    /// Return this in the backend's `VoiceJoinResponse.sfu_node_id` so that
195    /// clients know which node to dial.
196    pub fn node_id(&self) -> EndpointId {
197        self.endpoint.id()
198    }
199}

201#[cfg(test)]
202mod tests {
203    use std::time::Duration;
204
205    use super::drain_complete;
206
207    #[test]
208    fn drain_completes_once_sessions_reach_zero() {
209        assert!(drain_complete(0, Duration::ZERO, Duration::from_secs(300)));
210        assert!(drain_complete(
211            0,
212            Duration::from_secs(299),
213            Duration::from_secs(300)
214        ));
215    }
216
217    #[test]
218    fn drain_waits_while_sessions_remain_under_deadline() {
219        assert!(!drain_complete(
220            3,
221            Duration::from_secs(10),
222            Duration::from_secs(300)
223        ));
224    }
225
226    #[test]
227    fn drain_completes_at_deadline_even_with_sessions_remaining() {
228        assert!(drain_complete(
229            5,
230            Duration::from_secs(300),
231            Duration::from_secs(300)
232        ));
233        assert!(drain_complete(
234            5,
235            Duration::from_secs(301),
236            Duration::from_secs(300)
237        ));
238    }
