1//! # SFU service endpoint
2//!
3//! This module owns the iroh `Endpoint` and the `accept()` loop. It is thin
4//! glue: every accepted connection is handed to
5//! [`moq_relay::handle_relay_connection`], which authenticates the relay
6//! token, scopes the connection to its space, and drives the moq-net session.
7//! All forwarding logic lives in [`crate::moq_relay`]; this module just binds
8//! the endpoint and dispatches connections to it.
9
10use iroh::{Endpoint, EndpointId, RelayMode, RelayUrl, SecretKey};
11
12use crate::moq_relay::{self, RELAY_ALPN};
13
14// ─── Sfu ─────────────────────────────────────────────────────────────────────
15
16/// The SFU server.
17///
18/// Holds the iroh endpoint and the shared MoQ relay origin. Call [`Sfu::run`]
19/// to start accepting connections.
20pub struct Sfu {
21    pub endpoint: Endpoint,
22    relay_hmac_secret: String,
23    /// Shared root MoQ origin every accepted `moql` connection is scoped
24    /// against, so moq-net fans broadcasts out between same-space
25    /// publishers/subscribers — exactly as a standalone [`moq_relay::Relay`]
26    /// does, but on the endpoint/node-id the backend registers as
27    /// `sfu_node_id`.
28    relay_origin: moq_relay::OriginProducer,
29}
30
31impl Sfu {
32    /// Bind a new iroh endpoint and return an `Sfu` ready to accept connections.
33    ///
34    /// `relay_url` is the iroh relay server used for NAT traversal. `secret_key`
35    /// is the iroh secret key for this node (determines the `EndpointId`).
36    /// `relay_hmac_secret` is the shared HMAC secret used to verify relay tokens.
37    /// `bind_addr` is the **exact** IPv4 socket the QUIC endpoint binds to.
38    ///
39    /// ## Why a specific bind address (not `0.0.0.0`)
40    ///
41    /// The SFU runs with `hostNetwork: true`, so the node has many interface
42    /// addresses: its public IP plus a pile of private/cluster ones (CNI,
43    /// docker bridge, the Kubernetes node InternalIP). iroh advertises the
44    /// addresses it can reach a peer on; **bound to `0.0.0.0` it enumerates and
45    /// advertises every interface address**, so clients spend the call probing
46    /// unreachable private IPs — paths open, fail validation, get abandoned, and
47    /// the churn destabilises the connection (the reliable control stream often
48    /// never lands). Bound to a **specific** address, iroh advertises only that
49    /// one (see `iroh::socket::update_direct_addresses`). So the caller resolves
50    /// the node's public IPv4 and passes `public_ip:LISTEN_PORT` here, giving
51    /// clients exactly one reachable address to dial. The port is fixed (not
52    /// ephemeral) so it matches the single UDP port opened on the node.
53    ///
54    /// If the public IP can't be determined the caller may pass `0.0.0.0`, which
55    /// falls back to enumerating every interface address (correct, just noisy).
56    pub async fn bind(
57        relay_url: RelayUrl,
58        secret_key: SecretKey,
59        relay_hmac_secret: String,
60        bind_addr: std::net::SocketAddrV4,
61    ) -> eyre::Result<Self> {
62        let endpoint = Endpoint::builder(iroh::endpoint::presets::N0)
63            .secret_key(secret_key)
64            .relay_mode(RelayMode::Custom(iroh::RelayMap::from(relay_url)))
65            .alpns(vec![RELAY_ALPN.to_vec()])
66            .clear_ip_transports()
67            .bind_addr(std::net::SocketAddr::V4(bind_addr))
68            .map_err(|e| eyre::eyre!("invalid SFU bind address {bind_addr}: {e}"))?
69            .bind()
70            .await?;
71
72        Ok(Self::from_endpoint(endpoint, relay_hmac_secret))
73    }
74
75    /// Build an `Sfu` around an already-bound iroh [`Endpoint`].
76    ///
77    /// [`Sfu::bind`] is the production constructor: it builds an endpoint wired
78    /// to the iroh relay for NAT traversal. Tests need a *hermetic* endpoint
79    /// (relay disabled, direct loopback addressing) so they can run without
80    /// network access — they construct that endpoint themselves and hand it
81    /// here. Keeping the accept loop in [`Sfu::run`] identical for both paths is
82    /// the whole point: the e2e tests exercise the same connection-handling glue
83    /// that runs in production.
84    ///
85    /// The caller is responsible for having set [`RELAY_ALPN`] on the endpoint.
86    pub fn from_endpoint(endpoint: Endpoint, relay_hmac_secret: String) -> Self {
87        Self {
88            endpoint,
89            relay_hmac_secret,
90            relay_origin: moq_relay::new_root_origin(),
91        }
92    }
93
94    /// Accept connections in a loop until the endpoint closes.
95    ///
96    /// Each accepted connection is handed off to
97    /// [`moq_relay::handle_relay_connection`] in its own tokio task. This
98    /// function only returns on error or endpoint close.
99    pub async fn run(&self) -> eyre::Result<()> {
100        tracing::info!(
101            node_id = %self.endpoint.id(),
102            relay_alpn = %String::from_utf8_lossy(RELAY_ALPN),
103            "SFU accepting connections (MoQ relay)"
104        );
105
106        loop {
107            let Some(incoming) = self.endpoint.accept().await else {
108                tracing::info!("SFU endpoint closed — stopping accept loop");
109                break;
110            };
111
112            let conn = match incoming.await {
113                Ok(c) => c,
114                Err(e) => {
115                    tracing::warn!("incoming connection failed: {e}");
116                    continue;
117                }
118            };
119
120            let origin = self.relay_origin.clone();
121            let secret = self.relay_hmac_secret.clone();
122            tokio::spawn(async move {
123                if let Err(e) = moq_relay::handle_relay_connection(conn, origin, secret).await {
124                    tracing::warn!("relay connection handler error: {e}");
125                }
126            });
127        }
128
129        Ok(())
130    }
131
132    /// The iroh `EndpointId` for this SFU node.
133    ///
134    /// Return this in the backend's `VoiceJoinResponse.sfu_node_id` so that
135    /// clients know which node to dial.
136    pub fn node_id(&self) -> EndpointId {
137        self.endpoint.id()
138    }
139}
