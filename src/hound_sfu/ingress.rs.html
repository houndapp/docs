1//! Sustained-rate ingress budget: a pure, synchronous policy answering "has
2//! this connection's ingress rate stayed over the cap across the whole
3//! sliding window?" from periodic cumulative-byte samples.
4//!
5//! A single high sample (a keyframe burst, a momentary retransmit storm)
6//! must not trip the budget — only a rate sustained across the *entire*
7//! configured window does. [`crate::moq_relay`]'s per-connection watchdog
8//! drives this by feeding it `conn.stats().udp_rx.bytes` (cumulative QUIC
9//! ingress bytes) on a timer; this type itself stays free of any iroh/tokio
10//! dependency so the policy is unit-testable with synthetic timestamps.
11
12use std::collections::VecDeque;
13use std::time::{Duration, Instant};
14
15/// Sliding-window sustained ingress-rate budget.
16///
17/// Feed cumulative byte counts with [`Self::sample`] (or [`Self::sample_only`]
18/// if the caller wants to check [`Self::observed_rate`] separately, e.g. for
19/// an ops log line even when under budget). A `cap_bytes_per_sec` of `0`
20/// disables enforcement — [`Self::is_over_cap`] always reports `None`.
21#[derive(Debug, Clone)]
22pub struct IngressBudget {
23    cap_bytes_per_sec: u64,
24    window: Duration,
25    /// Wall-clock time of the very first sample — the warm-up clock. The
26    /// budget cannot judge a "sustained" rate until at least one full window
27    /// of history has been collected, so nothing trips before then: a fresh
28    /// connection (or a burst that hasn't persisted a full window) reads as
29    /// "not enough data" rather than "over cap".
30    started_at: Option<Instant>,
31    /// `(sample time, cumulative bytes)`, oldest first, pruned to `window`.
32    samples: VecDeque<(Instant, u64)>,
33}
34
35impl IngressBudget {
36    /// `cap_bytes_per_sec == 0` disables the budget entirely. `window` is the
37    /// sustained-rate averaging span: a spike shorter than `window` is
38    /// diluted by the surrounding low-rate samples and self-corrects; a rate
39    /// held for the whole window trips.
40    pub fn new(cap_bytes_per_sec: u64, window: Duration) -> Self {
41        Self {
42            cap_bytes_per_sec,
43            window,
44            started_at: None,
45            samples: VecDeque::new(),
46        }
47    }
48
49    /// Feed one `(cumulative_bytes, now)` observation and report whether the
50    /// sustained rate is over the cap after this sample. `Some(rate)` is the
51    /// observed bytes/sec at the moment of the trip, for the caller's log
52    /// line. Equivalent to [`Self::sample_only`] followed by
53    /// [`Self::is_over_cap`].
54    pub fn sample(&mut self, cumulative_bytes: u64, now: Instant) -> Option<u64> {
55        self.sample_only(cumulative_bytes, now);
56        self.is_over_cap()
57    }
58
59    /// Feed one observation without checking the cap. Useful when a caller
60    /// wants [`Self::observed_rate`] regardless of whether it trips (e.g. an
61    /// ops log line that reports the rate even when it's fine).
62    pub fn sample_only(&mut self, cumulative_bytes: u64, now: Instant) {
63        self.started_at.get_or_insert(now);
64        self.samples.push_back((now, cumulative_bytes));
65        while let Some(&(t, _)) = self.samples.front() {
66            if now.saturating_duration_since(t) > self.window {
67                self.samples.pop_front();
68            } else {
69                break;
70            }
71        }
72    }
73
74    /// Sustained bytes/sec across the samples currently retained within
75    /// `window`. `None` during warm-up (fewer than a full `window` of
76    /// wall-clock history collected since the first sample) or before any
77    /// sample has been fed — never a false "zero rate" read for a fresh
78    /// connection.
79    pub fn observed_rate(&self) -> Option<u64> {
80        let started_at = self.started_at?;
81        let &(newest_t, newest_bytes) = self.samples.back()?;
82        if newest_t.saturating_duration_since(started_at) < self.window {
83            return None;
84        }
85        let &(oldest_t, oldest_bytes) = self.samples.front()?;
86        let elapsed = newest_t.saturating_duration_since(oldest_t);
87        if elapsed.is_zero() {
88            return None;
89        }
90        let delta = newest_bytes.saturating_sub(oldest_bytes);
91        Some((delta as f64 / elapsed.as_secs_f64()).round() as u64)
92    }
93
94    /// `Some(rate)` when the cap is enabled (nonzero) and
95    /// [`Self::observed_rate`] exceeds it; `None` when disabled, still
96    /// warming up, or within budget.
97    pub fn is_over_cap(&self) -> Option<u64> {
98        if self.cap_bytes_per_sec == 0 {
99            return None;
100        }
101        self.observed_rate()
102            .filter(|rate| *rate > self.cap_bytes_per_sec)
103    }
104}

106#[cfg(test)]
107mod tests {
108    use super::*;
109
110    const WINDOW: Duration = Duration::from_secs(30);
111    const INTERVAL: Duration = Duration::from_secs(5);
112    const CAP: u64 = 1_000_000; // 1 MB/s
113
114    /// Push `count` samples `INTERVAL` apart starting at `start`, each adding
115    /// `bytes_per_tick` to a running cumulative total that begins at
116    /// `start_total`. Returns `(time of the last sample, its total)` so a
117    /// test can chain a second phase onto the first.
118    fn push_ticks(
119        budget: &mut IngressBudget,
120        start: Instant,
121        start_total: u64,
122        bytes_per_tick: u64,
123        count: u32,
124    ) -> (Instant, u64) {
125        let mut total = start_total;
126        let mut t = start;
127        for _ in 0..count {
128            total += bytes_per_tick;
129            budget.sample_only(total, t);
130            t += INTERVAL;
131        }
132        (t - INTERVAL, total)
133    }
134
135    #[test]
136    fn warmup_never_trips_even_at_extreme_rate() {
137        let mut budget = IngressBudget::new(CAP, WINDOW);
138        let t0 = Instant::now();
139        // Two samples, 5s apart — nowhere near a full 30s window of
140        // history — at a rate wildly over the cap.
141        push_ticks(&mut budget, t0, 0, CAP * 100, 2);
142        assert_eq!(
143            budget.observed_rate(),
144            None,
145            "no rate before warm-up completes"
146        );
147        assert_eq!(
148            budget.is_over_cap(),
149            None,
150            "warm-up must never trip regardless of the fed rate"
151        );
152    }
153
154    #[test]
155    fn burst_shorter_than_the_window_does_not_trip() {
156        let mut budget = IngressBudget::new(CAP, WINDOW);
157        let t0 = Instant::now();
158        // Steady low rate (20 KB/s) for a full window — completes warm-up,
159        // and must not trip on its own.
160        let low_per_tick = 100_000;
161        let (t_after_baseline, total) = push_ticks(&mut budget, t0, 0, low_per_tick, 7);
162        assert_eq!(budget.is_over_cap(), None, "steady low rate must not trip");
163
164        // One single-tick spike well over the instantaneous cap...
165        let spike = 8_000_000; // 1.6 MB/s for this one 5s tick alone
166        let t_spike = t_after_baseline + INTERVAL;
167        budget.sample_only(total + spike, t_spike);
168
169        // ...but blended across the whole 30s window (six ticks of low rate,
170        // one spike) the sustained average still sits under the cap.
171        assert_eq!(
172            budget.is_over_cap(),
173            None,
174            "a spike shorter than the window must self-correct, not trip"
175        );
176    }
177
178    #[test]
179    fn sustained_over_rate_trips() {
180        let mut budget = IngressBudget::new(CAP, WINDOW);
181        let t0 = Instant::now();
182        // 1.4 MB/s sustained (40% over the 1 MB/s cap), held for a full
183        // window (7 ticks spans the 30s window at a 5s sample interval).
184        let bytes_per_tick = 1_400_000 * INTERVAL.as_secs();
185        push_ticks(&mut budget, t0, 0, bytes_per_tick, 7);
186
187        let over = budget.is_over_cap();
188        assert!(
189            over.is_some(),
190            "a rate sustained across the whole window must trip"
191        );
192        assert!(over.unwrap() > CAP, "reported rate must exceed the cap");
193    }
194
195    #[test]
196    fn rate_returning_to_normal_clears_the_trip() {
197        let mut budget = IngressBudget::new(CAP, WINDOW);
198        let t0 = Instant::now();
199        let high_per_tick = 1_400_000 * INTERVAL.as_secs();
200        let (t_after_high, total) = push_ticks(&mut budget, t0, 0, high_per_tick, 7);
201        assert!(
202            budget.is_over_cap().is_some(),
203            "sustained high rate must trip first"
204        );
205
206        // Feed a full extra window's worth of low-rate ticks so every
207        // high-rate sample ages out of the retained window.
208        let low_per_tick = 100_000;
209        push_ticks(&mut budget, t_after_high + INTERVAL, total, low_per_tick, 7);
210        assert_eq!(
211            budget.is_over_cap(),
212            None,
213            "once the window is entirely low-rate again the trip must clear"
214        );
215    }
216
217    #[test]
218    fn disabled_cap_never_trips() {
219        let mut budget = IngressBudget::new(0, WINDOW);
220        let t0 = Instant::now();
221        push_ticks(&mut budget, t0, 0, u64::MAX / 16, 7);
222        assert_eq!(
223            budget.is_over_cap(),
224            None,
225            "a zero cap disables the budget regardless of the observed rate"
226        );
227    }
