1//! Mailpit polling: find the magic-link email the backend sent and pull out its
2//! sign-in code. Moved here (from `tools/houndbot`) so both `houndbot` and the
3//! scenario suite share one implementation.
4//!
5//! Wire shapes verified empirically against a running `axllent/mailpit`
6//! (v1 API): `GET /api/v1/search?query=to:"<email>"` returns
7//! `{"messages": [{"ID": "...", ...}], ...}` (search results are metadata
8//! only — no body); `GET /api/v1/message/{ID}` returns the full message,
9//! whose `Text` field is the plain-text part. The backend's magic-link email
10//! is `multipart/alternative` (`services/backend/src/api/auth/magic.rs`), so
11//! `Text` is exactly the part carrying the unadorned code line (the `HTML`
12//! part wraps the code in `<b>`, which would need stripping — `Text` avoids
13//! that entirely).
14
15use std::time::Duration;
16
17use regex::Regex;
18use serde::Deserialize;
19
20/// Anchors on the plain-text body's code line (`services/backend/src/api/auth/magic.rs`):
21/// `"Or enter this code in the app: {code}"`. The same email's link also embeds
22/// `?code=...`, but this prefix is unambiguous and doesn't require URL parsing.
23/// Alphabet excludes 0/O/1/I, so `[A-Z2-9]{12}` matches the code exactly.
24const CODE_LINE_RE: &str = r"Or enter this code in the app: ([A-Z2-9]{12})";
25
26/// Extract the magic-link code from an email's plain-text body. `None` if the
27/// anchor line is missing (wrong email, truncated snippet, malformed body).
28pub fn extract_code(text: &str) -> Option<String> {
29    let re = Regex::new(CODE_LINE_RE).expect("CODE_LINE_RE is a valid static pattern");
30    re.captures(text).map(|c| c[1].to_owned())
31}
32
33#[derive(Debug, Deserialize)]
34struct SearchResponse {
35    messages: Vec<SearchHit>,
36}
37
38#[derive(Debug, Deserialize)]
39struct SearchHit {
40    #[serde(rename = "ID")]
41    id: String,
42}
43
44#[derive(Debug, Deserialize)]
45struct MessageBody {
46    #[serde(rename = "Text")]
47    text: String,
48}
49
50/// Poll Mailpit for the newest email sent to `email` and extract its magic-link
51/// code, retrying at ~1s intervals until `timeout` elapses.
52pub async fn wait_for_code(
53    client: &reqwest::Client,
54    mailpit_url: &str,
55    email: &str,
56    timeout: Duration,
57) -> eyre::Result<String> {
58    let base = mailpit_url.trim_end_matches('/');
59    let deadline = tokio::time::Instant::now() + timeout;
60    loop {
61        if let Some(code) = try_fetch_code(client, base, email).await? {
62            return Ok(code);
63        }
64        if tokio::time::Instant::now() >= deadline {
65            eyre::bail!(
66                "no magic-link code appeared in mailpit ({mailpit_url}) for {email} within {timeout:?}"
67            );
68        }
69        tokio::time::sleep(Duration::from_secs(1)).await;
70    }
71}
72
73/// One search-and-fetch pass. Mailpit returns search hits newest-first; we still
74/// scan every hit defensively rather than assume ordering.
75async fn try_fetch_code(
76    client: &reqwest::Client,
77    base: &str,
78    email: &str,
79) -> eyre::Result<Option<String>> {
80    let search: SearchResponse = client
81        .get(format!("{base}/api/v1/search"))
82        .query(&[("query", format!("to:\"{email}\""))])
83        .send()
84        .await?
85        .error_for_status()?
86        .json()
87        .await?;
88
89    for hit in &search.messages {
90        let body: MessageBody = client
91            .get(format!("{base}/api/v1/message/{}", hit.id))
92            .send()
93            .await?
94            .error_for_status()?
95            .json()
96            .await?;
97        if let Some(code) = extract_code(&body.text) {
98            return Ok(Some(code));
99        }
100    }
101    Ok(None)
102}

104#[cfg(test)]
105mod tests {
106    use super::*;
107
108    #[test]
109    fn extracts_the_code_from_the_real_email_template() {
110        let text = "Sign in to Hound\n\n\
111                     Click to sign in on this device:\nhttps://example.com/auth?code=A7B9CDEF2345&port=1234\n\n\
112                     Or enter this code in the app: A7B9CDEF2345\n\n\
113                     This code expires in 10 minutes. If you didn't request this, you can ignore this email.";
114        assert_eq!(extract_code(text), Some("A7B9CDEF2345".to_owned()));
115    }
116
117    #[test]
118    fn anchors_on_the_explicit_line_not_the_link_query_param() {
119        // The link's `?code=` carries a DIFFERENT value than the real code in
120        // this fixture — proves we anchor on the labeled line, not the link.
121        let text = "Click here: https://example.com/auth?code=ZZZZZZZZZZZZ&port=1\n\
122                     Or enter this code in the app: A7B9CDEF2345\n";
123        assert_eq!(extract_code(text), Some("A7B9CDEF2345".to_owned()));
124    }
125
126    #[test]
127    fn returns_none_when_the_anchor_line_is_missing() {
128        assert_eq!(extract_code("no code here"), None);
129        assert_eq!(extract_code(""), None);
130    }
131
132    #[test]
133    fn returns_none_on_a_short_or_malformed_code() {
134        let text = "Or enter this code in the app: SHORT";
135        assert_eq!(extract_code(text), None);
136    }
137
138    #[test]
139    fn rejects_excluded_alphabet_characters() {
140        // 0/O/1/I never appear in a real code; a line carrying one shouldn't match.
141        let text = "Or enter this code in the app: A0B9CDEF234O";
142        assert_eq!(extract_code(text), None);
143    }
144
145    #[test]
146    fn matches_the_first_valid_code_line_in_a_multi_line_body() {
147        let text = "preamble\nOr enter this code in the app: ABCDEFGHJKLM\ntrailer";
148        assert_eq!(extract_code(text), Some("ABCDEFGHJKLM".to_owned()));
149    }
150}
