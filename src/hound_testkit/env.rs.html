1//! `Env`: the shared per-process scenario harness — backend/Mailpit endpoints, a
2//! lazily-bootstrapped default community (`general`/`voice` space pair), and
3//! per-run-unique participant emails.
4//!
5//! The bootstrap (admin sign-up → community create → instance provision) is a
6//! Rust port of what `crates/bot/e2e.sh` used to do by hand with curl/jq; the
7//! shell script now owns only the compose stack (see its module doc).
8//!
9//! **Bootstrap under nextest's process-per-test model:** every test gets its
10//! own process, so "exactly once per process" (the `static OnceCell` below)
11//! mostly just protects a single non-nextest `cargo test` binary running many
12//! `#[tokio::test]`s. Across *processes* sharing one backend, only the
13//! literal first-ever registrant holds `ManageCommunities`
14//! (`services/backend/src/api/community.rs`) — so at most one racing process
15//! can actually create+provision the shared community; every other process's
16//! bootstrap polls `GET /instance` until that lands. See [`bootstrap`].
17

30/// How long a losing bootstrap process waits for the winner to provision the
31/// instance before giving up.
32const PROVISION_POLL_TIMEOUT: Duration = Duration::from_secs(30);

34static BOOTSTRAP: OnceCell<Bootstrap> = OnceCell::const_new();

36/// Install a stderr tracing subscriber for the test process so `hound-client`
37/// call-plane warnings (control WS drops, reconnect attempts) land in
38/// nextest's captured per-test output — the difference between a diagnosable
39/// e2e failure and a bare timeout. Idempotent; `RUST_LOG` overrides the
40/// default filter.
41fn init_test_tracing() {
42    use tracing_subscriber::EnvFilter;
43    let filter = EnvFilter::try_from_default_env()
44        .unwrap_or_else(|_| "warn,hound_client=info,hound_testkit=info".into());
45    let _ = tracing_subscriber::fmt()
46        .with_env_filter(filter)
47        .with_writer(std::io::stderr)
48        .with_test_writer()
49        .try_init();
50}

58/// The shared scenario harness for one test process. Cheap to clone.
59#[derive(Clone)]
60pub struct Env {
61    inner: Arc<Inner>,
62}
63
64struct Inner {
65    backend_url: String,
66    mailpit_url: String,
67    run_id: String,
68    community_id: Uuid,
69    voice_space: Uuid,
70    text_space: Uuid,
71    user_seq: AtomicU32,
72}
73
74impl Env {
75    /// Read `BACKEND_URL`/`MAILPIT_URL` (defaulting to the compose stack's dev
76    /// ports — the stack is provided externally, e.g. by `mise run e2e:call`)
77    /// and lazily bootstrap the shared default community. Safe to call from
78    /// every test; the bootstrap network calls happen at most once.
79    pub async fn from_env() -> eyre::Result<Self> {
80        init_test_tracing();
81        let backend_url =
82            std::env::var("BACKEND_URL").unwrap_or_else(|_| "http://localhost:8080".into());
83        let mailpit_url =
84            std::env::var("MAILPIT_URL").unwrap_or_else(|_| "http://localhost:8025".into());
85
86        let b = BOOTSTRAP
87            .get_or_try_init(|| bootstrap(backend_url.clone(), mailpit_url.clone()))
88            .await?;
89
90        Ok(Self {
91            inner: Arc::new(Inner {
92                backend_url,
93                mailpit_url,
94                run_id: Uuid::new_v4().simple().to_string()[..8].to_owned(),
95                community_id: b.community_id,
96                voice_space: b.voice_space,
97                text_space: b.text_space,
98                user_seq: AtomicU32::new(0),
99            }),
100        })
101    }
102
103    pub fn backend_url(&self) -> &str {
104        &self.inner.backend_url
105    }
106
107    pub fn mailpit_url(&self) -> &str {
108        &self.inner.mailpit_url
109    }
110
111    /// The shared default community's id (for a test that wants to assert
112    /// against `watch_community_tree()` directly).
113    pub fn community_id(&self) -> Uuid {
114        self.inner.community_id
115    }
116
117    /// The shared default community's `voice` space.
118    pub fn voice_space(&self) -> Space {
119        Space::new(self.inner.voice_space, SpaceKind::Voice)
120    }
121
122    /// The shared default community's `general` (text) space.
123    pub fn text_space(&self) -> Space {
124        Space::new(self.inner.text_space, SpaceKind::Text)
125    }
126
127    /// Register/sign in a fresh participant: a per-run-unique email (so every
128    /// call gets a brand-new account, auto-joining the shared default
129    /// community via instance provisioning). `name` is both the email
130    /// local-part prefix and the display name.
131    pub async fn user(&self, name: &str) -> eyre::Result<Participant> {
132        let seq = self.inner.user_seq.fetch_add(1, Ordering::Relaxed);
133        let email = format!("{name}+{}-{seq}@e2e.local", self.inner.run_id);
134        let client = auth::sign_in_magic_link(
135            &self.inner.backend_url,
136            &self.inner.mailpit_url,
137            &email,
138            name,
139        )
140        .await?;
141        Ok(Participant::new(client, name.to_owned()))
142    }
143}

145/// Bootstrap the shared community. See the module doc for the cross-process
146/// race handling.
147///
148/// Uniform shape regardless of who wins which race: (1) sign up a throwaway
149/// account; (2) if the instance isn't provisioned yet, race to create a
150/// community + provision it (best-effort — losing either is fine, since
151/// `POST /instance/provision` is first-wins-atomic on the backend: at most
152/// one caller ever gets back a live `default_community`, everyone else a 409);
153/// (3) poll `/instance` to the settled default community — a bounded wait for
154/// whichever process actually won, not a workaround for backend-side
155/// eventual consistency; (4) resolve its space pair with a guaranteed-member
156/// fresh account ([`resolve_as_member`]). Never trusts "our" community or
157/// "our" membership — only the final `/instance` read, which is exactly what
158/// every subsequently-registered participant auto-joins.
159async fn bootstrap(backend_url: String, mailpit_url: String) -> eyre::Result<Bootstrap> {
160    let http = reqwest::Client::new();
161    let token = admin_sign_up(&http, &backend_url, &mailpit_url, "bootstrap").await?;
162
163    let instance = get_instance(&http, &backend_url, &token).await?;
164    if !instance.provisioned {
165        // Race to be the winner. `try_create_community` is `None` on 403 (we
166        // don't hold `ManageCommunities`); a lost provision race is settled by
167        // the poll below either way.
168        if let Some(community_id) = try_create_community(&http, &backend_url, &token).await? {
169            let _ = try_provision(&http, &backend_url, &token, community_id).await?;
170        }
171    }
172
173    let community_id = settled_default_community(&http, &backend_url, &token).await?;
174    resolve_as_member(&http, &backend_url, &mailpit_url, community_id).await
175}

177/// Poll `/instance` until provisioned, returning `default_community`. Bounded
178/// by [`PROVISION_POLL_TIMEOUT`] — this waits out a genuine cross-process
179/// race (some *other* test process is still mid-bootstrap), not a
180/// workaround: the backend's provision write is atomic, so the first
181/// `provisioned=true` read already carries the settled value.
182async fn settled_default_community(
183    http: &reqwest::Client,
184    backend_url: &str,
185    token: &str,
186) -> eyre::Result<Uuid> {
187    let deadline = tokio::time::Instant::now() + PROVISION_POLL_TIMEOUT;
188    loop {
189        let instance = get_instance(http, backend_url, token).await?;
190        if instance.provisioned {
191            return instance.default_community.ok_or_else(|| {
192                eyre::eyre!("instance is provisioned but has no default_community")
193            });
194        }
195        if tokio::time::Instant::now() >= deadline {
196            eyre::bail!(
197                "instance still not provisioned after {PROVISION_POLL_TIMEOUT:?} — no bootstrap process ever won the ManageCommunities race"
198            );
199        }
200        tokio::time::sleep(Duration::from_secs(1)).await;
201    }
202}

204/// Sign up a throwaway bootstrap account (per-call-unique email built from
205/// `prefix`) over raw REST, returning its access token. Mirrors `hound-testkit`'s
206/// own `auth::sign_in_magic_link`, but returns the bearer token directly
207/// (needed for the admin-only `/community` and `/instance` calls below, which
208/// have no typed `hound-client` surface). Magic start → Mailpit code → verify
209/// → register, once through — the backend's first-registration path is
210/// race-safe (`services/backend/src/api/auth.rs`), so concurrent bootstrap
211/// processes registering at the same time never need a retry here.
212async fn admin_sign_up(
213    http: &reqwest::Client,
214    backend_url: &str,
215    mailpit_url: &str,
216    prefix: &str,
217) -> eyre::Result<String> {
218    let email = format!("{prefix}+{}@e2e.local", Uuid::new_v4().simple());
219    http.post(format!("{backend_url}/auth/magic/start"))
220        .json(&json!({ "email": email }))
221        .send()
222        .await?
223        .error_for_status()?;
224    let code =
225        crate::mailpit::wait_for_code(http, mailpit_url, &email, auth::MAIL_WAIT_TIMEOUT).await?;
226    let verify: serde_json::Value = http
227        .post(format!("{backend_url}/auth/magic/verify"))
228        .json(&json!({ "code": code }))
229        .send()
230        .await?
231        .error_for_status()?
232        .json()
233        .await?;
234    match verify["status"].as_str() {
235        Some("ok") => verify["session"]["access_token"]
236            .as_str()
237            .map(str::to_owned)
238            .ok_or_else(|| eyre::eyre!("magic/verify ok with no access_token: {verify}")),
239        Some("unknown") => {
240            let link_token = verify["link_token"]
241                .as_str()
242                .ok_or_else(|| eyre::eyre!("magic/verify unknown with no link_token: {verify}"))?;
243            let reg: serde_json::Value = http
244                .post(format!("{backend_url}/auth/magic/register"))
245                .json(&json!({ "link_token": link_token, "display_name": "bootstrap" }))
246                .send()
247                .await?
248                .error_for_status()?
249                .json()
250                .await?;
251            reg["access_token"]
252                .as_str()
253                .map(str::to_owned)
254                .ok_or_else(|| eyre::eyre!("magic/register with no access_token: {reg}"))
255        }
256        other => eyre::bail!("unexpected magic/verify status: {other:?} ({verify})"),
257    }
258}

266async fn get_instance(
267    http: &reqwest::Client,
268    backend_url: &str,
269    token: &str,
270) -> eyre::Result<InstanceModel> {
271    Ok(http
272        .get(format!("{backend_url}/instance"))
273        .bearer_auth(token)
274        .send()
275        .await?
276        .error_for_status()?
277        .json()
278        .await?)
279}

281/// `POST /community` — requires `ManageCommunities`, which only the literal
282/// first-ever registrant on this DB holds. `Ok(None)` on a 403 (we're not
283/// that account); other non-success statuses are hard errors.
284async fn try_create_community(
285    http: &reqwest::Client,
286    backend_url: &str,
287    token: &str,
288) -> eyre::Result<Option<Uuid>> {
289    let resp = http
290        .post(format!("{backend_url}/community"))
291        .bearer_auth(token)
292        .json(&json!({ "name": "e2e" }))
293        .send()
294        .await?;
295    if resp.status() == reqwest::StatusCode::FORBIDDEN {
296        return Ok(None);
297    }
298    let community: serde_json::Value = resp.error_for_status()?.json().await?;
299    Ok(Some(serde_json::from_value(community["id"].clone())?))
300}
301
302/// `POST /instance/provision`. `Ok(true)` on success, `Ok(false)` on a 409
303/// (already provisioned — a race we lost).
304async fn try_provision(
305    http: &reqwest::Client,
306    backend_url: &str,
307    token: &str,
308    community_id: Uuid,
309) -> eyre::Result<bool> {
310    let resp = http
311        .post(format!("{backend_url}/instance/provision"))
312        .bearer_auth(token)
313        .json(&json!({ "default_community": community_id }))
314        .send()
315        .await?;
316    if resp.status() == reqwest::StatusCode::CONFLICT {
317        return Ok(false);
318    }
319    resp.error_for_status()?;
320    Ok(true)
321}

323/// Find the `general`(text)/`voice` pair among `community_id`'s spaces —
324/// auto-created alongside every community
325/// (`services/backend/src/api/community.rs::provision_default_layout`).
326async fn fetch_voice_and_text_spaces(
327    http: &reqwest::Client,
328    backend_url: &str,
329    token: &str,
330    community_id: Uuid,
331) -> eyre::Result<(Uuid, Uuid)> {
332    #[derive(serde::Deserialize)]
333    struct SpaceRow {
334        id: Uuid,
335        kind: String,
336    }
337    #[derive(serde::Deserialize)]
338    struct Page {
339        contents: Vec<SpaceRow>,
340    }
341    let page: Page = http
342        .get(format!(
343            "{backend_url}/space?id={community_id}&page=0&per_page=100"
344        ))
345        .bearer_auth(token)
346        .send()
347        .await?
348        .error_for_status()?
349        .json()
350        .await?;
351    let find = |kind: &str| {
352        page.contents
353            .iter()
354            .find(|s| s.kind == kind)
355            .map(|s| s.id)
356            .ok_or_else(|| eyre::eyre!("community {community_id} has no {kind} space"))
357    };
358    Ok((find("Voice")?, find("Text")?))
359}

361/// Resolve the provisioned default community's space pair with a
362/// guaranteed-member token: a fresh account registered now (i.e. strictly
363/// post-provisioning) auto-joins the default community — necessary because
364/// space listing is membership-gated (`filter_accessible_spaces` — a
365/// non-member sees an empty page, not an error), and whether our own
366/// bootstrap account auto-joined depends on registration/provision ordering.
367async fn resolve_as_member(
368    http: &reqwest::Client,
369    backend_url: &str,
370    mailpit_url: &str,
371    community_id: Uuid,
372) -> eyre::Result<Bootstrap> {
373    let member = admin_sign_up(http, backend_url, mailpit_url, "bootstrap-member").await?;
374    let (voice_space, text_space) =
375        fetch_voice_and_text_spaces(http, backend_url, &member, community_id).await?;
376    Ok(Bootstrap {
377        community_id,
378        voice_space,
379        text_space,
380    })
