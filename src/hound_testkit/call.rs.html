1//! `InCall`: a joined participant's call handle. Every `expect_*` is
2//! deadline-bounded and watch-driven — it polls a `tokio::sync::watch`
3//! channel to a condition (never a blind `sleep`-then-assert), and fails with
4//! the last-observed state on timeout.
5
6use std::time::Duration;
7
8use hound_client::CallView;
9use hound_client::stats::StatsSnapshot;
10use tokio::sync::watch;
11
12use crate::participant::Participant;
13
14/// Default deadline for an `expect_*` call when the caller passes `None`.
15const DEFAULT_DEADLINE: Duration = Duration::from_secs(20);
16
17/// A joined call. Dropping this without calling [`InCall::leave`] leaves the
18/// call running server-side until the participant's `HoundClient` is dropped
19/// or another `join` supersedes it — tests should always `leave()` explicitly.
20///
21/// Holds the `watch_call`/`watch_stats` receivers from *before* the join
22/// (see [`crate::participant::Participant::join`]): a receiver that exists
23/// from the start can never miss a publish, whatever the sender's
24/// `send`-vs-`send_replace` discipline.
25pub struct InCall {
26    participant: Participant,
27    call_rx: watch::Receiver<CallView>,
28    stats_rx: watch::Receiver<StatsSnapshot>,
29}
30
31impl InCall {
32    pub(crate) fn new(
33        participant: Participant,
34        call_rx: watch::Receiver<CallView>,
35        stats_rx: watch::Receiver<StatsSnapshot>,
36    ) -> Self {
37        Self {
38            participant,
39            call_rx,
40            stats_rx,
41        }
42    }
43
44    pub fn participant(&self) -> &Participant {
45        &self.participant
46    }

48    /// Wait until the call roster has at least `n` peers (including self).
49    pub async fn expect_peer_count(
50        &self,
51        n: usize,
52        deadline: Option<Duration>,
53    ) -> eyre::Result<()> {
54        let mut rx = self.call_rx.clone();
55        wait_for(
56            deadline,
57            &mut rx,
58            |v: &CallView| v.peers.len() >= n,
59            |v| format!("peers={:?}", v.peers),
60        )
61        .await
62    }
63
64    /// Wait until total inbound video frames (summed across every remote
65    /// peer) reach `min_frames` — proof that a peer's media actually arrived
66    /// and decoded, not just that the roster shows them.
67    pub async fn expect_peer_media(
68        &self,
69        min_frames: u64,
70        deadline: Option<Duration>,
71    ) -> eyre::Result<()> {
72        let mut rx = self.stats_rx.clone();
73        wait_for(
74            deadline,
75            &mut rx,
76            |s: &StatsSnapshot| inbound_video_total(s) >= min_frames,
77            |s| format!("inbound_video_frames={}", inbound_video_total(s)),
78        )
79        .await
80    }
81
82    /// Wait for inbound audio frames (summed across every remote peer) to
83    /// grow by at least `min_delta` from the count observed when this call is
84    /// made — the "media is still flowing" half of the mute/unmute assertion.
85    pub async fn expect_audio_growth(
86        &self,
87        min_delta: u64,
88        deadline: Option<Duration>,
89    ) -> eyre::Result<()> {
90        let mut rx = self.stats_rx.clone();
91        let start = inbound_audio_total(&rx.borrow_and_update());
92        wait_for(
93            deadline,
94            &mut rx,
95            move |s: &StatsSnapshot| inbound_audio_total(s).saturating_sub(start) >= min_delta,
96            move |s| {
97                format!(
98                    "inbound_audio_frames={} (start {start})",
99                    inbound_audio_total(s)
100                )
101            },
102        )
103        .await
104    }
105
106    /// Wait `quiet_for`, asserting inbound audio frames never grew — the
107    /// "muted means silence" half of the mute assertion. A negative
108    /// expectation, but still watch-driven: it wakes on every stats tick
109    /// rather than blind-sleeping the whole window before checking once.
110    pub async fn expect_no_audio_growth(&self, quiet_for: Duration) -> eyre::Result<()> {
111        let mut rx = self.stats_rx.clone();
112        let start = inbound_audio_total(&rx.borrow_and_update());
113        let deadline = tokio::time::Instant::now() + quiet_for;
114        loop {
115            let now = tokio::time::Instant::now();
116            if now >= deadline {
117                return Ok(());
118            }
119            if tokio::time::timeout(deadline - now, rx.changed())
120                .await
121                .is_err()
122            {
123                return Ok(());
124            }
125            let current = inbound_audio_total(&rx.borrow());
126            if current > start {
127                eyre::bail!(
128                    "expected no inbound audio growth for {quiet_for:?} while muted, \
129                     but frames grew {start} -> {current}"
130                );
131            }
132        }
133    }
134
135    /// The warmup-baseline audio-health gate — see
136    /// [`hound_client::call_health`] for the shared monitor (the same one
137    /// `houndbot` gates its verdict on). Observes for `window`, anchoring the
138    /// starve-counter baseline once `min_peers` have been present for
139    /// [`hound_client::call_health::WARMUP`], then fails if any starvation
140    /// counter grew past that baseline.
141    pub async fn expect_steady_audio(&self, min_peers: u32, window: Duration) -> eyre::Result<()> {
142        let sample = crate::sample::sample_call(self.participant.client(), min_peers, window).await;
143        if sample.starved {
144            eyre::bail!(
145                "audio starvation detected over {window:?} \
146                 (peers_seen={}, audio_frames_rx={})",
147                sample.peers_seen,
148                sample.audio_frames_rx
149            );
150        }
151        Ok(())
152    }

154    pub async fn mute(&self) -> eyre::Result<()> {
155        self.participant
156            .client()
157            .call_set_muted(true)
158            .await
159            .map_err(|e| eyre::eyre!("mute failed: {e}"))
160    }
161
162    pub async fn unmute(&self) -> eyre::Result<()> {
163        self.participant
164            .client()
165            .call_set_muted(false)
166            .await
167            .map_err(|e| eyre::eyre!("unmute failed: {e}"))
168    }
169
170    /// Leave the call.
171    pub async fn leave(self) -> eyre::Result<()> {
172        self.participant
173            .client()
174            .call_leave()
175            .await
176            .map_err(|e| eyre::eyre!("leave failed: {e}"))
177    }
178}

180fn inbound_video_total(s: &StatsSnapshot) -> u64 {
181    s.inbound.iter().map(|p| p.video.frames_received).sum()
182}
183
184fn inbound_audio_total(s: &StatsSnapshot) -> u64 {
185    s.inbound.iter().map(|p| p.audio.frames_received).sum()
186}

188/// Shared polling core for a watch-driven `expect_*`: check `pred` on every
189/// change (and once up front), bail with `describe`'s last-observed state on
190/// timeout. `deadline` defaults to [`DEFAULT_DEADLINE`].
191async fn wait_for<T>(
192    deadline: Option<Duration>,
193    rx: &mut watch::Receiver<T>,
194    pred: impl Fn(&T) -> bool,
195    describe: impl Fn(&T) -> String,
196) -> eyre::Result<()> {
197    let deadline = deadline.unwrap_or(DEFAULT_DEADLINE);
198    let wait = async {
199        loop {
200            if pred(&rx.borrow_and_update()) {
201                return;
202            }
203            if rx.changed().await.is_err() {
204                return; // sender dropped — the final check below settles it
205            }
206        }
207    };
208    if tokio::time::timeout(deadline, wait).await.is_ok() && pred(&rx.borrow()) {
209        return Ok(());
210    }
211    eyre::bail!(
212        "timed out after {deadline:?} waiting for condition; last observed: {}",
213        describe(&rx.borrow())
214    )
215}
