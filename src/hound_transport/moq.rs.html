1//! # MoQ media transport over an iroh `Endpoint`
2//!
3//! The IO shell that moves encoded media over a **MoQ session** (`moq-net`)
4//! carried on a raw-QUIC **iroh** connection (`web-transport-iroh`). MoQ's
5//! group/object model owns framing and (for video) loss recovery, so the custom
6//! packetizer/reassembler in [`hound_media`] is not on the video path.
7//!
8//! ## Sans-IO boundary
9//!
10//! This shell only does IO. The sans-IO [`hound_media`] cores it sits in front of —
11//! jitter/playout ([`hound_media::PlayoutBuffer`]), the A/V clock
12//! ([`hound_media::CaptureClock`]), the BWE→encoder loop
13//! ([`hound_media::RateController`]/[`hound_media::QualityLadder`]) and AEC — are
14//! never touched here. The publisher takes encoded [`hound_media::Frame`]s (exactly
15//! what the encoder produces) and the subscriber hands back reassembled
16//! [`hound_media::Frame`]s (exactly what jitter/playout expects). The clock is
17//! injected as `now_us`.
18//!
19//! ## Frame ↔ group/object mapping
20//!
21//! ```text
22//! one media flow   →  one TRACK              (the subscribe unit)
23//! one GOP          →  one GROUP              (its own QUIC uni-stream)
24//! one encoded frame →  one OBJECT (frame)    (opaque payload, byte-for-byte)
25//! ```
26//!
27//! * **keyframe → NEW group boundary.** A keyframe finishes the open group and
28//!   `create_group`s the next sequence; the keyframe payload is the first object
29//!   in the new group.
30//! * **delta → object appended to the open keyframe group** (`write_frame`). A
31//!   delta before any keyframe is rejected — there is no decodable group to
32//!   anchor it.
33//! * **object payload is forwarded verbatim** — nothing decodes or rewrites it.
34//!
35//! ## Delivery mode & why it kills the keyframe-after-deltas freeze
36//!
37//! moq-net serves **each group on its own QUIC uni-stream**, so:
38//!
39//! * **within a GOP:** ordered + reliable (QUIC stream); a lost delta is repaired
40//!   by QUIC stream retransmit — we do **not** hand-roll NACK for video.
41//! * **across GOPs:** independent streams; an old GOP's stalled stream can never
42//!   head-of-line-block a newer one. The subscriber drains with
43//!   [`TrackConsumer::next_group`], which **discards a late group at the
44//!   boundary** instead of waiting — freeze prevention expressed as transport
45//!   semantics rather than app-level NACK.
46//!
47//! Audio takes the same group/object path but rolls groups by COUNT, not
48//! keyframe: successive Opus frames ride one ordered group, rolled every few
49//! frames (see `AUDIO_GROUP_OBJECTS`).
50//!
51//! ## Live-join gate (correctness, not timing)
52//!
53//! moq-net is **live** pub/sub: a subscriber joins at the *latest* group and does
54//! not replay groups produced before its per-track SUBSCRIBE landed. The session
55//! handshake completing does **not** imply the SUBSCRIBE arrived, so the
56//! publisher gates first-frame production on [`MoqPublisher::wait_subscribed`]
57//! ([`TrackProducer::used`]) — which resolves deterministically once a consumer
58//! attaches — guaranteeing the opening keyframe (group 0) is delivered with no
59//! `sleep`/yield timing dependency.

82/// Errors from the MoQ IO shell. Every moq-net/iroh failure is surfaced; the
83/// caller decides whether to retry the session or fail the call.
84#[derive(Debug, thiserror::Error)]
85pub enum MoqError {
86    /// The underlying iroh connection failed.
87    #[error("iroh: {0}")]
88    Iroh(String),
89    /// A moq-net session/broadcast/track/group operation failed.
90    #[error("moq: {0}")]
91    Moq(String),
92    /// A delta frame arrived before any keyframe — no decodable group to anchor
93    /// it. The publisher rejects rather than guessing a group boundary.
94    #[error("delta frame before any keyframe (no open group)")]
95    DeltaBeforeKeyframe,
96    /// A reassembled object carried a track/codec byte that is not a known
97    /// [`hound_media::TrackKind`]/[`hound_media::Codec`]. Dropped, never guessed.
98    #[error("unknown track/codec descriptor on object")]
99    UnknownDescriptor,
100    /// The relay rejected the relay-token handshake (bad/expired/missing token,
101    /// or a namespace the token's space does not authorise).
102    #[error("relay rejected the connection (auth)")]
103    RelayAuthRejected,
104}

106// ─── Relay-token handshake (client side) ───────────────────────────────────────
107//
108// When connecting to the MoQ relay (`services/sfu::moq_relay`) the
109// client must present its relay token BEFORE the MoQ handshake. The relay reads
110// the token off the first bi-stream, verifies the backend's HMAC scheme, and
111// replies one byte (1 = ok, 0 = reject). The client gates its MoQ handshake on
112// reading the ok byte, so the relay's token bi-stream is always accepted before
113// moq-net opens its own control stream (QUIC accepts bi-streams in open order).
114//
115// Wire (client → relay): u16 LE length + token bytes. (relay → client): 1 byte.
116//
117// Direct peer-to-peer pub/sub ([`MoqPublisher::connect`] /
118// [`MoqSubscriber::accept`]) does NOT use this — only the relay path does.
119
120const RELAY_AUTH_OK: u8 = 1;
121
122/// Open the relay-token control bi-stream, send `token`, and await the relay's
123/// one-byte verdict. Returns `Ok(())` only if the relay accepted the token.
124async fn relay_auth_handshake(conn: &Connection, token: &str) -> Result<(), MoqError> {
125    let (mut send, mut recv) = conn
126        .open_bi()
127        .await
128        .map_err(|e| MoqError::Iroh(format!("open token stream: {e}")))?;
129
130    let token_bytes = token.as_bytes();
131    let len = u16::try_from(token_bytes.len())
132        .map_err(|_| MoqError::Moq("relay token too long".into()))?;
133    send.write_all(&len.to_le_bytes())
134        .await
135        .map_err(|e| MoqError::Iroh(format!("write token len: {e}")))?;
136    send.write_all(token_bytes)
137        .await
138        .map_err(|e| MoqError::Iroh(format!("write token: {e}")))?;
139    send.finish()
140        .map_err(|e| MoqError::Iroh(format!("finish token stream: {e}")))?;
141
142    let mut verdict = [0u8; 1];
143    recv.read_exact(&mut verdict)
144        .await
145        .map_err(|e| MoqError::Iroh(format!("read auth verdict: {e}")))?;
146    if verdict[0] == RELAY_AUTH_OK {
147        Ok(())
148    } else {
149        Err(MoqError::RelayAuthRejected)
150    }
151}

153// ─── Object descriptor ────────────────────────────────────────────────────────
154//
155// A MoQ object payload is opaque to the relay, but the *endpoints* still need to
156// recover the `hound_media::Frame` envelope (which track, which codec, the
157// keyframe flag, the capture timestamp) that the encoder stamped and the
158// jitter/playout core consumes. We prepend a tiny fixed descriptor to the object
159// payload and forward the encoded bytes after it verbatim. The relay never reads it.
160//
161// Layout (little-endian), OBJ_HEADER_LEN = 16 bytes, then the encoded payload:
162//   [0]       track     (TrackKind as u8)
163//   [1]       codec     (Codec as u8)
164//   [2]       flags     (bit0 = keyframe)
165//   [3]       reserved  (0)
166//   [4..8]    frame_seq (u32)
167//   [8..16]   timestamp_us (i64)
168//   [16..]    encoded frame bytes (forwarded verbatim)
169//
170// Kept separate from `hound_media::HEADER_LEN` because this is the MoQ-object
171// envelope, not the datagram `PacketHeader`.
172
173const OBJ_HEADER_LEN: usize = 16;
174
175fn encode_object(frame: &Frame) -> Bytes {
176    let mut buf = Vec::with_capacity(OBJ_HEADER_LEN + frame.data.len());
177    buf.push(frame.track as u8);
178    buf.push(frame.codec as u8);
179    buf.push(if frame.keyframe { 1 } else { 0 });
180    buf.push(0); // reserved/pad
181    buf.extend_from_slice(&frame.frame_seq.to_le_bytes());
182    buf.extend_from_slice(&frame.timestamp_us.to_le_bytes());
183    buf.extend_from_slice(&frame.data);
184    Bytes::from(buf)
185}
186
187fn decode_object(sender: u16, bytes: &[u8]) -> Result<Frame, MoqError> {
188    if bytes.len() < OBJ_HEADER_LEN {
189        return Err(MoqError::UnknownDescriptor);
190    }
191    let track = TrackKind::from_u8(bytes[0]).map_err(|_| MoqError::UnknownDescriptor)?;
192    let codec = Codec::from_u8(bytes[1]).map_err(|_| MoqError::UnknownDescriptor)?;
193    let keyframe = bytes[2] & 1 == 1;
194    let frame_seq = u32::from_le_bytes([bytes[4], bytes[5], bytes[6], bytes[7]]);
195    let timestamp_us = i64::from_le_bytes([
196        bytes[8], bytes[9], bytes[10], bytes[11], bytes[12], bytes[13], bytes[14], bytes[15],
197    ]);
198    Ok(Frame {
199        sender,
200        track,
201        codec,
202        keyframe,
203        frame_seq,
204        timestamp_us,
205        data: bytes[OBJ_HEADER_LEN..].to_vec(),
206    })
207}

211/// MoQ publisher over an iroh connection. Owns the moq-net session (held alive
212/// for the track's lifetime), the broadcast producer, and the open track
213/// producer. Hand it encoded [`Frame`]s via [`MoqPublisher::publish_frame`]; it
214/// maps each onto the GOP group / object model in §2.
215pub struct MoqPublisher {
216    _session: moq_net::Session,
217    _broadcast: BroadcastProducer,
218    track: TrackProducer,
219    /// The currently open group (a GOP for video, a rolled audio window for audio).
220    /// `None` until the first frame opens one.
221    open_group: Option<moq_net::GroupProducer>,
222    /// Sequence of the NEXT group to open (advanced per group roll).
223    next_group_seq: u64,
224    /// Objects written to the open group — drives the audio count-roll
225    /// ([`gop_publish_audio_frame`]). Unused by the video keyframe-driven path.
226    objects_in_group: usize,
227}
228
229impl MoqPublisher {
230    /// Wrap an established iroh `Connection` as a MoQ session and announce one
231    /// broadcast with one track. Returns once the moq-net client handshake
232    /// completes. (The peer's SUBSCRIBE may not have landed yet — gate the first
233    /// frame on [`Self::wait_subscribed`].)
234    pub async fn connect(
235        conn: Connection,
236        broadcast_path: &str,
237        track_name: &str,
238    ) -> Result<Self, MoqError> {
239        Self::connect_inner(conn, broadcast_path, track_name).await
240    }
241
242    /// Like [`Self::connect`] but for the **MoQ relay** path (`services/sfu::moq_relay`): perform the relay-token auth handshake first,
243    /// then announce. The relay scopes this publisher's announces to the space
244    /// in `relay_token`, so `broadcast_path` must live under `space/{space_id}`
245    /// (the [`crate::moq`] namespace convention) or the relay drops it.
246    ///
247    /// The token is presented on a dedicated bi-stream and the relay's accept
248    /// verdict is awaited *before* the MoQ handshake — so a rejected token fails
249    /// fast with [`MoqError::RelayAuthRejected`] and no broadcast is announced.
250    pub async fn connect_relay(
251        conn: Connection,
252        relay_token: &str,
253        broadcast_path: &str,
254        track_name: &str,
255    ) -> Result<Self, MoqError> {
256        relay_auth_handshake(&conn, relay_token).await?;
257        Self::connect_inner(conn, broadcast_path, track_name).await
258    }
259
260    /// Shared body of [`Self::connect`] / [`Self::connect_relay`]: wrap the conn
261    /// as a raw-QUIC MoQ session, announce one broadcast + track, run the moq-net
262    /// `Client` handshake. The caller has already done any auth handshake.
263    async fn connect_inner(
264        conn: Connection,
265        broadcast_path: &str,
266        track_name: &str,
267    ) -> Result<Self, MoqError> {
268        // Raw-QUIC mode: speak MoQ directly over iroh, no HTTP/3.
269        let wt = web_transport_iroh::Session::raw(conn);
270
271        let origin = Origin::random().produce();
272        let mut broadcast = origin.create_broadcast(broadcast_path).ok_or_else(|| {
273            MoqError::Moq(format!("broadcast path not allowed: {broadcast_path}"))
274        })?;
275        let track = broadcast
276            .create_track(Track::new(track_name))
277            .map_err(|e| MoqError::Moq(format!("create_track: {e}")))?;
278
279        let session = moq_net::Client::new()
280            .with_publish(origin.consume())
281            .connect(wt)
282            .await
283            .map_err(|e| MoqError::Moq(format!("client connect: {e}")))?;
284
285        Ok(Self {
286            _session: session,
287            _broadcast: broadcast,
288            track,
289            open_group: None,
290            next_group_seq: 0,
291            objects_in_group: 0,
292        })
293    }
294
295    /// Block until at least one subscriber has attached to the track. Awaiting
296    /// this before producing the first keyframe guarantees the opening group is
297    /// delivered (moq-net live-join gate; see module docs). Deterministic — no
298    /// `sleep`/yield.
299    pub async fn wait_subscribed(&self) -> Result<(), MoqError> {
300        self.track
301            .used()
302            .await
303            .map_err(|e| MoqError::Moq(format!("wait_subscribed: {e}")))
304    }
305
306    /// Map one encoded [`Frame`] onto the MoQ group/object model and publish it.
307    ///
308    /// * **keyframe** → finish the open group (if any) and `create_group` at the
309    ///   next sequence; the keyframe is the first object in that group.
310    /// * **delta** → append an object to the open keyframe group. A delta before
311    ///   any keyframe is [`MoqError::DeltaBeforeKeyframe`].
312    ///
313    /// The encoded payload is forwarded byte-for-byte (an [`OBJ_HEADER_LEN`]-byte
314    /// `hound_media::Frame` envelope is prepended so the subscriber can rebuild
315    /// the `Frame`; the relay never reads it). Per §2 a GOP is one group: the
316    /// keyframe opens it and deltas ride the same ordered stream until the next
317    /// keyframe rolls the boundary.
318    pub fn publish_frame(&mut self, frame: &Frame) -> Result<(), MoqError> {
319        gop_publish_frame(
320            &mut self.track,
321            &mut self.open_group,
322            &mut self.next_group_seq,
323            frame,
324        )
325    }
326
327    /// Publish one AUDIO frame as an object in a count-rolled ordered group
328    /// ([`gop_publish_audio_frame`]) — the ordered-substream model that keeps
329    /// successive Opus frames on one in-order QUIC stream instead of one stream/frame.
330    pub fn publish_audio_frame(&mut self, frame: &Frame) -> Result<(), MoqError> {
331        gop_publish_audio_frame(
332            &mut self.track,
333            &mut self.open_group,
334            &mut self.next_group_seq,
335            &mut self.objects_in_group,
336            frame,
337        )
338    }
339
340    /// Finish the open group and signal end-of-track so the subscriber's drain
341    /// loop terminates cleanly. Call when the media flow ends.
342    pub fn finish(&mut self) -> Result<(), MoqError> {
343        gop_finish(&mut self.track, &mut self.open_group)
344    }
345}

347/// Map one encoded [`Frame`] onto the MoQ GOP/object model on `track`, advancing
348/// `open_group`/`next_group_seq`. The single GOP-publish code path shared by the
349/// per-track [`MoqPublisher`] and the multiplexed [`MoqTrackPublisher`] — see
350/// [`MoqPublisher::publish_frame`] for the keyframe-rolls-the-group contract.
351fn gop_publish_frame(
352    track: &mut TrackProducer,
353    open_group: &mut Option<moq_net::GroupProducer>,
354    next_group_seq: &mut u64,
355    frame: &Frame,
356) -> Result<(), MoqError> {
357    let object = encode_object(frame);
358    if frame.keyframe {
359        // Roll the GOP boundary: a stale group is dropped *whole* at its boundary
360        // by the subscriber, so a fresh keyframe is never queued behind an older
361        // group's retransmits.
362        if let Some(mut prev) = open_group.take() {
363            prev.finish()
364                .map_err(|e| MoqError::Moq(format!("finish prev group: {e}")))?;
365        }
366        let mut group = track
367            .create_group(Group::from(*next_group_seq))
368            .map_err(|e| MoqError::Moq(format!("create_group: {e}")))?;
369        group
370            .write_frame(object)
371            .map_err(|e| MoqError::Moq(format!("write keyframe object: {e}")))?;
372        *next_group_seq += 1;
373        *open_group = Some(group);
374    } else {
375        let group = open_group.as_mut().ok_or(MoqError::DeltaBeforeKeyframe)?;
376        group
377            .write_frame(object)
378            .map_err(|e| MoqError::Moq(format!("write delta object: {e}")))?;
379    }
380    Ok(())
381}

383/// Audio objects per group before rolling the group boundary (the **ordered audio
384/// substream** model). Audio is sequential and stateful — not GOP-structured like
385/// video — so instead of one group (= one QUIC uni-stream) PER FRAME (which gives ~50
386/// unordered racing streams/s whose cross-stream reordering corrupts playout),
387/// successive Opus frames are appended as objects to ONE group so QUIC delivers them
388/// **in order**. The group is rolled every `AUDIO_GROUP_OBJECTS` frames because a single
389/// never-rolled group is not viable in moq-net 0.1.9 (the 1024-object / 32 MB eviction
390/// cap → fatal `CacheFull`; late-join only attaches at a group boundary; and a lost
391/// object head-of-line-blocks the whole group until retransmit).
392///
393/// The window size is the **HoL blast radius**: under connection congestion (e.g. a
394/// high-bitrate screenshare surge sharing the link) a lost audio group's stream stalls
395/// until QUIC retransmit, so a whole window of audio arrives too late and is dropped —
396/// one audible hole the size of the window. At 20 ms/frame, 4 objects ≈ 80 ms: within
397/// the jitter buffer's absorption (the receiver's reorder buffer mops up the extra
398/// cross-group boundaries), while still cutting stream churn ~4× vs per-frame and
399/// staying far under the eviction cap.
400const AUDIO_GROUP_OBJECTS: usize = 4;
401
402/// Publish one audio frame as an object in a COUNT-ROLLED group (the ordered-substream
403/// model — see [`AUDIO_GROUP_OBJECTS`]). The first frame and every `AUDIO_GROUP_OBJECTS`-th
404/// frame opens a fresh group (new ordered QUIC stream); the rest append to the open
405/// group, so a window of frames rides one in-order stream. The frame's `keyframe` bit is
406/// IGNORED — every Opus frame is independently decodable, so audio grouping is driven by
407/// count, not by a keyframe flag. Shared by [`MoqPublisher`] and [`MoqTrackPublisher`].
408fn gop_publish_audio_frame(
409    track: &mut TrackProducer,
410    open_group: &mut Option<moq_net::GroupProducer>,
411    next_group_seq: &mut u64,
412    objects_in_group: &mut usize,
413    frame: &Frame,
414) -> Result<(), MoqError> {
415    let object = encode_object(frame);
416    // Roll on the first frame (no open group) or once the window is full.
417    if open_group.is_none() || *objects_in_group >= AUDIO_GROUP_OBJECTS {
418        if let Some(mut prev) = open_group.take() {
419            prev.finish()
420                .map_err(|e| MoqError::Moq(format!("finish prev audio group: {e}")))?;
421        }
422        let mut group = track
423            .create_group(Group::from(*next_group_seq))
424            .map_err(|e| MoqError::Moq(format!("create audio group: {e}")))?;
425        group
426            .write_frame(object)
427            .map_err(|e| MoqError::Moq(format!("write audio group-open object: {e}")))?;
428        *next_group_seq += 1;
429        *objects_in_group = 1;
430        *open_group = Some(group);
431    } else {
432        let group = open_group
433            .as_mut()
434            .expect("the roll branch covers the None case");
435        group
436            .write_frame(object)
437            .map_err(|e| MoqError::Moq(format!("write audio object: {e}")))?;
438        *objects_in_group += 1;
439    }
440    Ok(())
441}

443/// Finish the open GOP group (if any) then the track, so the subscriber's drain
444/// terminates cleanly. Shared by [`MoqPublisher`] and [`MoqTrackPublisher`].
445fn gop_finish(
446    track: &mut TrackProducer,
447    open_group: &mut Option<moq_net::GroupProducer>,
448) -> Result<(), MoqError> {
449    if let Some(mut group) = open_group.take() {
450        group
451            .finish()
452            .map_err(|e| MoqError::Moq(format!("finish open group: {e}")))?;
453    }
454    track
455        .finish()
456        .map_err(|e| MoqError::Moq(format!("finish track: {e}")))
457}

459// ─── Subscriber ──────────────────────────────────────────────────────────────
460
461/// MoQ subscriber over an iroh connection. Accepts the moq-net session, waits
462/// for the publisher's broadcast announce, and subscribes to a track. Drains
463/// groups/objects and reassembles [`Frame`]s for the jitter/playout core.
464pub struct MoqSubscriber {
465    _session: moq_net::Session,
466    track: TrackConsumer,
467    /// The group (GOP) currently being drained object-by-object, if any.
468    current: Option<moq_net::GroupConsumer>,
469    /// Stamped onto every reassembled [`Frame`] (the object envelope does not
470    /// carry the sender id; the remote peer's identity is known at subscribe
471    /// time, above this layer).
472    sender: u16,
473    /// Drain in arrival order (audio) vs priority-drop stale groups (video). See
474    /// [`drain_next_frame`].
475    reorder_ok: bool,
476}
477
478impl MoqSubscriber {
479    /// Accept the MoQ session on this side, wait for the publisher's broadcast
480    /// announce, and subscribe to `track_name`. `sender` is the remote peer id to
481    /// stamp on reassembled frames (identity is resolved above this layer).
482    pub async fn accept(
483        conn: Connection,
484        broadcast_path: &str,
485        track_name: &str,
486        sender: u16,
487    ) -> Result<Self, MoqError> {
488        let wt = web_transport_iroh::Session::raw(conn);
489
490        let origin = Origin::random().produce();
491        let mut announced = origin.consume();
492
493        let session = moq_net::Server::new()
494            .with_consume(origin)
495            .accept(wt)
496            .await
497            .map_err(|e| MoqError::Moq(format!("server accept: {e}")))?;
498
499        // Wait for the publisher's broadcast announce for our path.
500        let broadcast = loop {
501            match announced.announced().await {
502                Some((path, Some(bc))) if path.as_str() == broadcast_path => break bc,
503                Some((_, Some(_))) => continue, // a different broadcast; ignore
504                Some((_, None)) => continue,    // an unannounce; keep waiting
505                None => return Err(MoqError::Moq("origin closed before announce".into())),
506            }
507        };
508
509        let track = broadcast
510            .subscribe_track(&Track::new(track_name))
511            .map_err(|e| MoqError::Moq(format!("subscribe_track: {e}")))?;
512
513        Ok(Self {
514            _session: session,
515            track,
516            current: None,
517            sender,
518            // Test-only path (production audio uses MoqTrackSubscriber); keep the
519            // video-style monotonic drain.
520            reorder_ok: false,
521        })
522    }
523
524    /// Receive the next reassembled [`Frame`], or `None` at end-of-track.
525    ///
526    /// Drains one group (GOP) at a time with [`TrackConsumer::next_group`] — the
527    /// **priority-drop** discipline: a late group (sequence ≤ one already
528    /// returned) is discarded at the boundary instead of head-of-line-blocking a
529    /// newer GOP. Within a group, objects are read in order (QUIC-reliable). Each
530    /// object's [`OBJ_HEADER_LEN`]-byte envelope is decoded back into a `Frame`
531    /// (track/codec/keyframe/timestamp) and the encoded payload handed out
532    /// verbatim to jitter/playout.
533    ///
534    /// `now_us` is accepted for symmetry with the clock-injected cores and for
535    /// future per-subscriber pacing; today groups are read as they arrive.
536    pub async fn next_frame(&mut self, _now_us: i64) -> Result<Option<Frame>, MoqError> {
537        drain_next_frame(
538            &mut self.track,
539            &mut self.current,
540            self.sender,
541            self.reorder_ok,
542        )
543        .await
544    }
545
546    /// Stall-bounded pull (see [`drain_next_frame_stalled`]); video-only.
547    pub async fn next_frame_stalled(
548        &mut self,
549        stall: std::time::Duration,
550    ) -> Result<VideoPull, MoqError> {
551        debug_assert!(!self.reorder_ok, "stall-bounded pull is video-only");
552        drain_next_frame_stalled(&mut self.track, &mut self.current, self.sender, stall).await
553    }
554}

556/// The outcome of a stall-bounded video pull ([`MoqTrackSubscriber::next_frame_stalled`]).
557pub enum VideoPull {
558    /// One reassembled frame.
559    Frame(Frame),
560    /// Nothing arrived within the stall bound. If a group was mid-drain it has
561    /// been ABANDONED (its remaining objects are skipped), so the next delivery
562    /// resumes at a fresh group boundary — the caller should request a keyframe
563    /// so the producer rolls one.
564    Stalled,
565    /// End of track.
566    End,
567}
568
569/// Shared group/object drain used by [`MoqSubscriber`] and
570/// [`MoqRelaySubscriber`] — the only difference between the two is how the
571/// `TrackConsumer` was obtained (direct `Server::accept` vs. `Client::connect`
572/// to the relay); the priority-drop drain discipline is identical.
573///
574/// Drains one group (GOP) at a time via [`TrackConsumer::next_group`]
575/// (priority-drop: a stale group is discarded at its boundary rather than
576/// head-of-line-blocking a newer keyframe); within a group, objects are read in
577/// order. Each object's [`OBJ_HEADER_LEN`]-byte envelope is decoded back into a
578/// `Frame` and the encoded payload handed out verbatim.
579///
580/// ## Known limitation — within-GOP head-of-line stall under heavy loss
581///
582/// `next_group`'s priority-drop only fires at a *group boundary*: it discards a
583/// stale group instead of head-of-line-blocking a newer keyframe, but only once
584/// the current group has drained. A subscriber already part-way through group K
585/// can still **stall inside K** if an early object of K is lost — QUIC
586/// retransmits it, but moq-net prioritises the *newest* streams, so under
587/// continuous production K's retransmit can be starved behind newer GOPs and the
588/// inner `GroupConsumer::read_frame` blocks. Recovery is then bounded by QUIC
589/// eventually retransmitting K's missing object, which under a sustained,
590/// contiguous downstream loss burst can be slow. Preemptively abandoning the
591/// stalled GOP to jump to the next keyframe does not reliably help: when the
592/// burst also drops the *newer* GOP's opener, `next_group` has nothing newer to
593/// offer either. The fault-injection e2e
594/// (`services/sfu/tests/moq_relay_faults.rs`) characterises the recoverable
595/// regime; see its module docs.
596///
597/// `reorder_ok = true` drains via [`TrackConsumer::recv_group`] (ARRIVAL order, never
598/// discards) instead of [`TrackConsumer::next_group`] (monotonic, drops a group whose
599/// sequence ≤ one already returned). Set it for AUDIO: audio groups are small and
600/// frequent, and on the shared connection a high-bitrate screenshare reorders those
601/// tiny audio streams, so `next_group`'s staleness rule would silently drop reordered
602/// ones — fatal, since a slightly-reordered Opus frame is perfectly playable (the
603/// jitter buffer reorders by timestamp and the decoder skips truly-stale frames). Video
604/// keeps `next_group`: a stale GOP genuinely should be dropped for a newer keyframe.
605async fn drain_next_frame(
606    track: &mut TrackConsumer,
607    current: &mut Option<moq_net::GroupConsumer>,
608    sender: u16,
609    reorder_ok: bool,
610) -> Result<Option<Frame>, MoqError> {
611    loop {
612        // Finish draining the currently open group's objects first.
613        if let Some(group) = current.as_mut() {
614            match group.read_frame().await {
615                Ok(Some(object)) => return decode_object(sender, &object).map(Some),
616                Ok(None) => {
617                    *current = None; // group exhausted; advance
618                }
619                Err(e) => {
620                    // A group-level reset is a NORMAL boundary, not a pipeline
621                    // error: when a new keyframe group appears, the relay's
622                    // per-subscriber forwarding priority-drops the stale group
623                    // by resetting its stream ("remote error: code=0" /
624                    // "cancelled" mid-read). Treating that as fatal would tear
625                    // the session down on every PLI-forced keyframe roll.
626                    // Abandon the group and pull the next one; only
627                    // `next_group` / session-level errors below propagate.
628                    tracing::debug!("moq: group abandoned mid-read (boundary): {e}");
629                    *current = None;
630                }
631            }
632        }
633
634        // No open group: pull the next one. Audio takes EVERY group in arrival order
635        // (`recv_group`); video priority-drops stale groups (`next_group`).
636        let next = if reorder_ok {
637            track.recv_group().await
638        } else {
639            track.next_group().await
640        }
641        .map_err(|e| MoqError::Moq(format!("next_group: {e}")))?;
642        match next {
643            Some(group) => *current = Some(group),
644            None => return Ok(None), // track finished
645        }
646    }
647}
648
649/// [`drain_next_frame`] bounded by a stall budget, for VIDEO tracks.
650///
651/// With on-demand keyframes a GOP (= one group) can span many seconds, and a
652/// lost object inside the group being drained head-of-line-blocks every frame
653/// behind it until QUIC retransmits — the tile freezes, then fast-forwards.
654/// This pull instead treats "no object within `stall`" as a stuck group:
655/// abandon it (skip its remaining objects) and report [`VideoPull::Stalled`] so
656/// the caller can request a keyframe; the producer's IDR opens a fresh group
657/// that `next_group` jumps to, bounding the freeze at ~`stall` instead of a
658/// retransmit tail.
659///
660/// Cancel-safe: the moq-net consumers are poll-based state machines whose
661/// cursors advance only on `Ready`, so the timed-out future is dropped without
662/// losing a frame — and a mid-object timeout only ever discards the group it
663/// abandons. Never use for audio (audio must drain every group).
664async fn drain_next_frame_stalled(
665    track: &mut TrackConsumer,
666    current: &mut Option<moq_net::GroupConsumer>,
667    sender: u16,
668    stall: std::time::Duration,
669) -> Result<VideoPull, MoqError> {
670    loop {
671        if let Some(group) = current.as_mut() {
672            match tokio::time::timeout(stall, group.read_frame()).await {
673                Ok(Ok(Some(object))) => {
674                    return decode_object(sender, &object).map(VideoPull::Frame);
675                }
676                Ok(Ok(None)) => {
677                    *current = None; // group exhausted; advance
678                }
679                Ok(Err(e)) => {
680                    // Group reset at a keyframe roll — a normal boundary (see
681                    // `drain_next_frame`); advance to the next group.
682                    tracing::debug!("moq: group abandoned mid-read (boundary): {e}");
683                    *current = None;
684                }
685                Err(_) => {
686                    tracing::debug!("moq: group stalled mid-drain — abandoning for next keyframe");
687                    *current = None;
688                    return Ok(VideoPull::Stalled);
689                }
690            }
691            continue;
692        }
693        match tokio::time::timeout(stall, track.next_group()).await {
694            Ok(r) => match r.map_err(|e| MoqError::Moq(format!("next_group: {e}")))? {
695                Some(group) => *current = Some(group),
696                None => return Ok(VideoPull::End),
697            },
698            Err(_) => return Ok(VideoPull::Stalled),
699        }
700    }
701}

703// ─── Relay subscriber ──────────────────────────────────────────────────────────

705/// MoQ subscriber over the **relay** path (`services/sfu::moq_relay`).
706///
707/// Unlike [`MoqSubscriber`] (which is the moq-net *server* for direct
708/// peer-to-peer), this subscriber **dials** the relay: it is a moq-net `Client`,
709/// presents its relay token first, then consumes broadcasts the relay announces
710/// (the relay re-announces every publisher's broadcast to subscribers of the
711/// same space). The drain discipline is identical — it shares
712/// [`drain_next_frame`].
713///
714/// The relay scopes this subscriber to the space in `relay_token`, so
715/// `broadcast_path` must live under `space/{space_id}` or the relay never
716/// announces it to this connection.
717pub struct MoqRelaySubscriber {
718    _session: moq_net::Session,
719    track: TrackConsumer,
720    current: Option<moq_net::GroupConsumer>,
721    sender: u16,
722    reorder_ok: bool,
723}
724
725impl MoqRelaySubscriber {
726    /// Dial the relay: relay-token auth handshake, MoQ `Client` handshake
727    /// consuming into a local origin, wait for the relay to announce
728    /// `broadcast_path`, then subscribe to `track_name`. `sender` is the remote
729    /// peer id stamped on reassembled frames.
730    pub async fn connect_relay(
731        conn: Connection,
732        relay_token: &str,
733        broadcast_path: &str,
734        track_name: &str,
735        sender: u16,
736    ) -> Result<Self, MoqError> {
737        relay_auth_handshake(&conn, relay_token).await?;
738
739        let wt = web_transport_iroh::Session::raw(conn);
740
741        // Local origin the relay's announces are inserted INTO (the client
742        // `with_consume` half). We then watch it for `broadcast_path`.
743        let origin = Origin::random().produce();
744        let mut announced = origin.consume();
745
746        let session = moq_net::Client::new()
747            .with_consume(origin)
748            .connect(wt)
749            .await
750            .map_err(|e| MoqError::Moq(format!("relay client connect: {e}")))?;
751
752        // Wait for the relay to announce the publisher's broadcast for our path.
753        let broadcast = loop {
754            match announced.announced().await {
755                Some((path, Some(bc))) if path.as_str() == broadcast_path => break bc,
756                Some((_, Some(_))) => continue, // a different broadcast; ignore
757                Some((_, None)) => continue,    // an unannounce; keep waiting
758                None => {
759                    return Err(MoqError::Moq("relay origin closed before announce".into()));
760                }
761            }
762        };
763
764        let track = broadcast
765            .subscribe_track(&Track::new(track_name))
766            .map_err(|e| MoqError::Moq(format!("subscribe_track: {e}")))?;
767
768        Ok(Self {
769            _session: session,
770            track,
771            current: None,
772            sender,
773            // Test-only path (production audio uses MoqTrackSubscriber); keep the
774            // video-style monotonic drain.
775            reorder_ok: false,
776        })
777    }
778
779    /// Receive the next reassembled [`Frame`], or `None` at end-of-track. See
780    /// [`drain_next_frame`] for the priority-drop drain discipline.
781    pub async fn next_frame(&mut self, _now_us: i64) -> Result<Option<Frame>, MoqError> {
782        drain_next_frame(
783            &mut self.track,
784            &mut self.current,
785            self.sender,
786            self.reorder_ok,
787        )
788        .await
789    }
790}

792// ─── Multiplexed relay session ──────────────────────────────────────────────────
793
794/// One MoQ relay session multiplexing MANY tracks — publish AND subscribe — over a
795/// SINGLE iroh/QUIC connection.
796///
797/// ## Why
798///
799/// One connection (one moq-net session, one relay-token handshake, one QUIC
800/// congestion-control context) per *track* per *peer* would run ~20 connections
801/// from one client in a call with a handful of peers each sending mic + camera +
802/// a screen share. That wastes a full QUIC handshake + auth round trip
803/// per track, fragments congestion control into many independent controllers that
804/// can't see each other's RTT/loss, and multiplies the relay's per-connection
805/// bookkeeping. moq-net already multiplexes any number of broadcasts/tracks over one
806/// session (the relay's [`crate`]-side handler wires a single session with both a
807/// publish and a consume half over the same scoped origin — see
808/// `services/sfu::moq_relay`), so the client does too.
809///
810/// ## Model
811///
812/// One [`moq_net::Client`] session is wired with BOTH halves:
813/// * `with_publish` serves OUR local broadcasts out to the relay — new tracks are
814///   announced live by [`Self::publish_track`] creating a broadcast on the held
815///   [`OriginProducer`];
816/// * `with_consume` ingests the relay's announces into a local origin —
817///   [`Self::subscribe_track`] resolves a specific track by path via
818///   [`OriginConsumer::announced_broadcast`] (each call gets its own announcement
819///   cursor, so concurrent subscribes to different tracks don't race).
820///
821/// The per-track handles ([`MoqTrackPublisher`]/[`MoqTrackSubscriber`]) carry the
822/// same GOP/object framing and priority-drop drain as the single-track types; they
823/// MUST NOT outlive the session (it owns the connection). Dropping a track handle
824/// unannounces / unsubscribes just that track; dropping the session tears down the
825/// connection and every track on it.
826pub struct MoqRelaySession {
827    _session: moq_net::Session,
828    /// Local origin: broadcasts WE publish are created here and served out via the
829    /// session's publish half. Held alive so our announces persist for the call.
830    publish_origin: OriginProducer,
831    /// The relay's announces are ingested here; cloned per subscribe so each track
832    /// resolves its broadcast on an independent cursor.
833    consume: OriginConsumer,
834}
835
836impl MoqRelaySession {
837    /// Dial the relay, present the relay token, and run one MoQ `Client` handshake
838    /// that both publishes and consumes. Tracks are added afterwards with
839    /// [`Self::publish_track`] / [`Self::subscribe_track`]; this performs the
840    /// connection + auth + handshake exactly ONCE for all of them.
841    pub async fn connect_relay(conn: Connection, relay_token: &str) -> Result<Self, MoqError> {
842        relay_auth_handshake(&conn, relay_token).await?;
843
844        let wt = web_transport_iroh::Session::raw(conn);
845
846        // Publish half: our local broadcasts live here and are served out.
847        let publish_origin = Origin::random().produce();
848        // Consume half: the relay's announces are ingested into this producer; we
849        // keep a consumer handle to resolve tracks by path and clone per subscribe.
850        let consume_origin = Origin::random().produce();
851        let consume = consume_origin.consume();
852
853        let session = moq_net::Client::new()
854            .with_publish(publish_origin.consume())
855            .with_consume(consume_origin)
856            .connect(wt)
857            .await
858            .map_err(|e| MoqError::Moq(format!("relay session connect: {e}")))?;
859
860        Ok(Self {
861            _session: session,
862            publish_origin,
863            consume,
864        })
865    }
866
867    /// Announce `broadcast_path` with one `track_name` on this session and return a
868    /// per-track publisher. The announce is live — gate first-frame production on
869    /// [`MoqTrackPublisher::wait_subscribed`] exactly as with [`MoqPublisher`].
870    /// `broadcast_path` must live under the token's `space/{id}` scope or the relay
871    /// drops the announce.
872    ///
873    /// `priority` is the moq-net track delivery priority (higher preempts lower when
874    /// the uplink is bandwidth-constrained, so the caller can make audio outrank
875    /// video). It is carried on the publisher's outgoing streams.
876    pub fn publish_track(
877        &self,
878        broadcast_path: &str,
879        track_name: &str,
880        priority: u8,
881    ) -> Result<MoqTrackPublisher, MoqError> {
882        let mut broadcast = self
883            .publish_origin
884            .create_broadcast(broadcast_path)
885            .ok_or_else(|| {
886                MoqError::Moq(format!("broadcast path not allowed: {broadcast_path}"))
887            })?;
888        let track = broadcast
889            .create_track(Track {
890                name: track_name.into(),
891                priority,
892            })
893            .map_err(|e| MoqError::Moq(format!("create_track: {e}")))?;
894        Ok(MoqTrackPublisher {
895            _broadcast: broadcast,
896            track,
897            open_group: None,
898            next_group_seq: 0,
899            objects_in_group: 0,
900        })
901    }
902
903    /// Wait for the relay to announce `broadcast_path`, subscribe to `track_name`,
904    /// and return a per-track subscriber. `sender` is the remote peer id stamped on
905    /// reassembled frames (identity is resolved above this layer). Resolving the
906    /// broadcast uses its own announcement cursor, so many concurrent subscribes to
907    /// different tracks over this one session don't interfere.
908    ///
909    /// `priority` is the moq-net subscription priority sent to the relay: it orders
910    /// THIS subscription against the subscriber's other subscriptions when the relay
911    /// forwards downlink under contention (higher preempts lower), so the caller can
912    /// keep audio ahead of video on a constrained downlink.
913    /// `reorder_ok` selects the drain discipline (see [`drain_next_frame`]): `true` for
914    /// AUDIO tracks (deliver every group in arrival order — a reordered Opus frame is
915    /// still playable), `false` for VIDEO (priority-drop stale GOPs for a newer keyframe).
916    pub async fn subscribe_track(
917        &self,
918        broadcast_path: &str,
919        track_name: &str,
920        sender: u16,
921        priority: u8,
922        reorder_ok: bool,
923    ) -> Result<MoqTrackSubscriber, MoqError> {
924        let broadcast = self
925            .consume
926            .announced_broadcast(broadcast_path)
927            .await
928            .ok_or_else(|| MoqError::Moq(format!("relay never announced {broadcast_path}")))?;
929        let track = broadcast
930            .subscribe_track(&Track {
931                name: track_name.into(),
932                priority,
933            })
934            .map_err(|e| MoqError::Moq(format!("subscribe_track: {e}")))?;
935        Ok(MoqTrackSubscriber {
936            track,
937            current: None,
938            sender,
939            reorder_ok,
940        })
941    }
942
943    /// Wait until `broadcast_path` is no longer announced on this session — the
944    /// **awaited-teardown** primitive for same-path re-acquire.
945    ///
946    /// moq-net withdraws a dropped broadcast's announce asynchronously (a spawned
947    /// `closed()`-watcher), and parks a same-path re-announce as a silent *backup*
948    /// until the old broadcast's withdrawal lands. A publisher that re-announces the
949    /// same path too early therefore announces nothing, and a subscriber that
950    /// resolves the path meanwhile gets the closed (zombie) broadcast. Await this on
951    /// the SUBSCRIBING session before re-publishing/re-subscribing a path: the
952    /// withdrawal observed here has propagated end-to-end (publisher origin → relay
953    /// → this session), so a subsequent re-announce is fresh by construction.
954    ///
955    /// Resolves immediately if the path is not currently announced. `Err` means the
956    /// session closed while waiting (its origin will never announce again).
957    pub async fn unannounced(&self, broadcast_path: &str) -> Result<(), MoqError> {
958        // Cursor before the liveness check: an unannounce arriving after the check
959        // queues an event on it, so check-then-wait can't miss the edge.
960        let mut cursor = self.consume.consume();
961        while cursor.get_broadcast(broadcast_path).is_some() {
962            if cursor.announced().await.is_none() {
963                return Err(MoqError::Moq(
964                    "session closed while awaiting unannounce".into(),
965                ));
966            }
967        }
968        Ok(())
969    }
970}

1078// These guard the only place the IO shell parses untrusted bytes off the wire:
1079// the 16-byte object envelope (`encode_object` / `decode_object`). The relay
1080// forwards opaque payloads, so a malformed or malicious envelope must be dropped,
1081// never panic the drain task and never be silently mis-decoded into a bogus
1082// `Frame`. The drain loop runs on a moq-net background task; a panic here would
1083// be a denial-of-service on the whole session.

1085#[cfg(test)]
1086mod envelope_tests {

1101    /// Round-trip: every envelope field survives encode→decode byte-for-byte, the
1102    /// sender is stamped from the decode argument (not the wire), and the encoded
1103    /// payload is forwarded verbatim. This is the forward-no-transcode contract at
1104    /// the envelope layer.
1105    #[test]
1106    fn encode_decode_round_trips_all_fields() {
1107        for track in [
1108            TrackKind::CamVideo,
1109            TrackKind::CamAudio,
1110            TrackKind::ScrVideo,
1111            TrackKind::ScrAudio,
1112        ] {
1113            for codec in [Codec::Av1, Codec::H264, Codec::H265, Codec::Opus] {
1114                for keyframe in [true, false] {
1115                    let payload: Vec<u8> = (0u16..300).map(|b| b as u8).collect();
1116                    let frame = sample(track, codec, keyframe, payload.clone());
1117                    let bytes = encode_object(&frame);
1118                    let decoded = decode_object(99, &bytes).expect("decode");
1119                    assert_eq!(decoded.sender, 99, "sender stamped from arg, not wire");
1120                    assert_eq!(decoded.track, track);
1121                    assert_eq!(decoded.codec, codec);
1122                    assert_eq!(decoded.keyframe, keyframe);
1123                    assert_eq!(decoded.frame_seq, frame.frame_seq);
1124                    assert_eq!(decoded.timestamp_us, frame.timestamp_us);
1125                    assert_eq!(decoded.data, payload, "payload forwarded verbatim");
1126                }
1127            }
1128        }
1129    }

1198    /// The keyframe flag is bit0 only; high bits in the flags byte are ignored
1199    /// (forward-compatible) and do not corrupt the keyframe decode.
1200    #[test]
1201    fn keyframe_flag_reads_only_bit0() {
1202        let mut bytes =
1203            encode_object(&sample(TrackKind::CamVideo, Codec::Av1, false, vec![9])).to_vec();
1204        // Set every high bit but leave bit0 clear → still a delta.
1205        bytes[2] = 0b1111_1110;
1206        assert!(!decode_object(0, &bytes).unwrap().keyframe);
1207        // Set bit0 (plus noise) → keyframe.
1208        bytes[2] = 0b1010_1011;
1209        assert!(decode_object(0, &bytes).unwrap().keyframe);
1210    }
